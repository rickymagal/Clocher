119def _extract_expected_lines(obj: Dict[str, Any]) -> List[str]:
120 prompts = obj.get("prompts", [])
121 if not isinstance(prompts, list):
122 raise SystemExit("error: engine JSON missing prompts list")
123
124 lines: List[str] = []
125 for p in prompts:
126 if not isinstance(p, dict):
127 continue
128 prompt_id = p.get("prompt_id")
129 if not isinstance(prompt_id, str):
130 raise SystemExit("error: prompt_id missing in engine JSON")
131
132 rounds = p.get("rounds", [])
133 tokens: Optional[List[int]] = None
134 if isinstance(rounds, list) and rounds:
135 r0 = rounds[0]
136 if isinstance(r0, dict):
137 tok = r0.get("tokens")
138 if isinstance(tok, list):
139 tokens = [int(x) for x in tok]
140 if tokens is None:
141 tok = p.get("tokens")
142 if isinstance(tok, list):
143 tokens = [int(x) for x in tok]
144
145 if tokens is None or not tokens:
146 raise SystemExit(f"error: missing tokens for prompt_id {prompt_id}")
147
148 token_csv = ",".join(str(x) for x in tokens)
149 lines.append(f"{prompt_id} {token_csv}")
150
151 return lines
