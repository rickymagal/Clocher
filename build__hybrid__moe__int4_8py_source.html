2"""
3build_hybrid_moe_int4.py
4
5Repack an existing IEBIN tensor map + weights binary into a new aligned binary,
6and emit a compatible tensor map that points to the new weights bin.
7
8This tool is intentionally tolerant of older tensor-map JSON formats that do not
9include a top-level "weights_bin" field. In that case, it defaults to
10"model.ie.bin" inside --model-dir (or --in-bin if provided).
11
12Typical usage (matches your invocation):
13
14 MODEL_DIR=".../models/gpt-oss-20b"
15 python3 tools/build_hybrid_moe_int4.py \
16 --model-dir "$MODEL_DIR" \
17 --out-bin "$MODEL_DIR/model.q4.bin" \
18 --out-map "$MODEL_DIR/model.ie.compat.json" \
19 --align 256
20
21Notes:
22- This script does not perform quantization. It repacks whatever tensors already
23 exist in the input weights bin (BF16 weights, INT4 blocks/scales, etc.).
24- The engine selects the INT4 artifact pair via --precision int4, which maps to
25 model.ie.compat.json + model.q4.bin.
26"""

53def _align_up(x: int, a: int) -> int:

56 r = x % a

119def _parse_tensors(m: Dict[str, Any]) -> List[TensorRec]:
120 ts = m.get("tensors")
121 if not isinstance(ts, list) or not ts:
122 _die("tensor map missing/invalid tensors[]")
123
124 out: List[TensorRec] = []
125 for t in ts:
126 if not isinstance(t, dict):
127 _die("tensor map tensors[] contains non-object")
128 name = t.get("name")
129 dtype = t.get("dtype")
130 shape = t.get("shape")
131 offset = t.get("offset")
132 nbytes = t.get("nbytes")
133 if not isinstance(name, str) or not name:
134 _die("tensor missing/invalid name")
135 if not isinstance(dtype, str) or not dtype:
136 _die(f"{name}: tensor missing/invalid dtype")
137 if not isinstance(shape, list) or not all(isinstance(x, int) for x in shape):
138 _die(f"{name}: tensor missing/invalid shape")
139 if not isinstance(offset, int) or offset < 0:
140 _die(f"{name}: tensor missing/invalid offset")
141 if not isinstance(nbytes, int) or nbytes <= 0:
142 _die(f"{name}: tensor missing/invalid nbytes")
143
144 out.append(TensorRec(
145 name=name,
146 dtype=dtype,
147 shape=shape,
148 offset=offset,
149 nbytes=nbytes,
150 raw=t,
151 ))
152
153 # Sort by file offset so we can stream copies.
154 out.sort(key=lambda r: r.offset)
155 return out

158def _copy_range(fin, fout, src_off: int, nbytes: int) -> None:
159 fin.seek(src_off, os.SEEK_SET)
160 remaining = nbytes
161 buf_sz = 4 * 1024 * 1024
162 while remaining > 0:
163 take = buf_sz if remaining > buf_sz else remaining
164 chunk = fin.read(take)
165 if not chunk or len(chunk) != take:
166 _die(f"short read at offset={src_off} (wanted {nbytes} bytes)")
167 fout.write(chunk)
168 remaining -= take

171def _repack(model_dir: str,
172 in_map_path: str,
173 in_bin_path: str,
174 out_bin_path: str,
175 out_map_path: str,
176 align: int) -> None:
177 m = _load_json(in_map_path)
178 tensors = _parse_tensors(m)
179
180 os.makedirs(os.path.dirname(out_bin_path) or ".", exist_ok=True)
181
182 # Stream-copy each tensor into the new bin with alignment.
183 new_tensors: List[Dict[str, Any]] = []
184 cur = 0
185
186 with open(in_bin_path, "rb") as fin, open(out_bin_path, "wb") as fout:
187 for r in tensors:
188 cur = _align_up(cur, align)
189 dst_off = cur
190
191 # Ensure output file is positioned correctly.
192 fout.seek(dst_off, os.SEEK_SET)
193
194 _copy_range(fin, fout, r.offset, r.nbytes)
195
196 cur = dst_off + r.nbytes
197
198 t2 = dict(r.raw)
199 t2["offset"] = dst_off
200 t2["nbytes"] = r.nbytes
201 new_tensors.append(t2)
202
203 out_map = dict(m)
204 out_map["weights_bin"] = os.path.basename(out_bin_path)
205 out_map["tensors"] = new_tensors
206
207 _write_json(out_map_path, out_map)
208
209 out_sz = os.path.getsize(out_bin_path)
210 print(f"[int4-artifacts] in_map={in_map_path}", file=sys.stderr)
211 print(f"[int4-artifacts] in_bin={in_bin_path}", file=sys.stderr)
212 print(f"[int4-artifacts] out_bin={out_bin_path} ({out_sz} bytes)", file=sys.stderr)
213 print(f"[int4-artifacts] out_map={out_map_path}", file=sys.stderr)
