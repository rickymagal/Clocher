2"""
3Build a runtime tokenizer pack ("TKPK" v1) from a tiktoken encoding.
4
5Pack format (little-endian):
6 u8 magic[4] = "TKPK"
7 u32 version = 1
8 u32 n_mergeable
9 u32 n_special
10 u32 pat_len
11 u8 pat_bytes[pat_len] (UTF-8)
12
13 Repeat n_mergeable times:
14 u16 len
15 u8 bytes[len] (raw bytes key)
16 u32 token_id
17
18 Repeat n_special times:
19 u16 len
20 u8 utf8[len] (UTF-8 special string)

51def extract_from_encoding(enc: Any) -> Tuple[Dict[bytes, int], Dict[str, int], str]:
52 """
53 Tries multiple internal layouts across tiktoken versions.
54
55 Returns:
56 mergeable_ranks: Dict[bytes, int]
57 special_tokens: Dict[str, int]
58 pat_str: str
59 """
60 mergeable = getattr(enc, "_mergeable_ranks", None)
61 special = getattr(enc, "_special_tokens", None)
62 pat = getattr(enc, "_pat_str", None)
63
64 if mergeable is None or special is None or pat is None:
65 core = getattr(enc, "_core_bpe", None)
66 if core is not None:
67 if mergeable is None:
68 mergeable = getattr(core, "mergeable_ranks", None)
69 if special is None:
70 special = getattr(core, "special_tokens", None)
71 if pat is None:
72 pat = getattr(core, "pat_str", None)
73
74 if mergeable is None or special is None or pat is None:
75 die(
76 "could not extract encoding internals "
77 "(mergeable_ranks/special_tokens/pat_str missing)."
78 )
79
80 if not isinstance(mergeable, dict):
81 die(f"mergeable_ranks is not a dict (type={type(mergeable)})")
82 if not isinstance(special, dict):
83 die(f"special_tokens is not a dict (type={type(special)})")
84 if not isinstance(pat, str):
85 die(f"pat_str is not a str (type={type(pat)})")
86
87 mergeable_ranks: Dict[bytes, int] = {bytes(k): int(v) for k, v in mergeable.items()}
88 special_tokens: Dict[str, int] = {str(k): int(v) for k, v in special.items()}
89 pat_str: str = pat
90 return mergeable_ranks, special_tokens, pat_str

105def build_tokpack(encoding_name: str, out_path: str) -> None:
106 enc = get_encoding_obj(encoding_name)
107 mergeable_ranks, special_tokens, pat_str = extract_from_encoding(enc)
108
109 pat_bytes = pat_str.encode("utf-8")
110
111 merge_items = sorted(mergeable_ranks.items(), key=lambda kv: (kv[1], kv[0]))
112 spec_items = sorted(special_tokens.items(), key=lambda kv: (kv[1], kv[0]))
113
114 tmp_path = out_path + ".tmp"
115 os.makedirs(os.path.dirname(out_path) or ".", exist_ok=True)
116
117 with open(tmp_path, "wb") as f:
118 f.write(b"TKPK")
119 write_u32(f, 1) # version
120 write_u32(f, len(merge_items))
121 write_u32(f, len(spec_items))
122 write_u32(f, len(pat_bytes))
123 f.write(pat_bytes)
124
125 for b, tid in merge_items:
126 if len(b) > 0xFFFF:
127 die(f"mergeable bytes too long: len={len(b)} token_id={tid}")
128 write_u16(f, len(b))
129 f.write(b)
130 write_u32(f, tid)
131
132 for s, tid in spec_items:
133 sb = s.encode("utf-8")
134 if len(sb) > 0xFFFF:
135 die(f"special string too long: len={len(sb)} token_id={tid}")
136 write_u16(f, len(sb))
137 f.write(sb)
138 write_u32(f, tid)
139
140 os.replace(tmp_path, out_path)
141
142 max_id_merge = max(mergeable_ranks.values()) if mergeable_ranks else -1
143 max_id_spec = max(special_tokens.values()) if special_tokens else -1
144 max_id = max(max_id_merge, max_id_spec)
145
146 print(f"ok: wrote {out_path}")
147 print(f" encoding : {encoding_name}")
148 print(f" mergeable_ranks: {len(merge_items)}")
149 print(f" special_tokens : {len(spec_items)}")
150 print(f" pat_len : {len(pat_bytes)}")
151 print(f" max_token_id : {max_id}")
152
153 for k in ("<|start|>", "<|end|>", "<|message|>"):
154 if k in special_tokens:
155 print(f" special {k:12s}: {special_tokens[k]}")
156 else:
157 print(f" special {k:12s}: (missing)")
