57static void ie_set_affinity(const char *policy, unsigned tidx, unsigned nth, int use_affinity) {
58#ifdef __linux__
59 if (!use_affinity || !policy) return;
60
61 cpu_set_t set;
62 CPU_ZERO(&set);
63
64 long ncpu = sysconf(_SC_NPROCESSORS_ONLN);
65 if (ncpu <= 0) return;
66
67 unsigned cpu = tidx % (unsigned)ncpu; /* default "auto" */
68
69 if (strcmp(policy, "compact") == 0) {
70 cpu = tidx % (unsigned)ncpu; /* fill CPUs from 0.. */
71 } else if (strcmp(policy, "scatter") == 0) {
72 /* spread across the machine */
73 unsigned stride = (unsigned)(ncpu / (nth ? nth : 1)) + 1u;
74 cpu = (unsigned)((tidx * stride) % (unsigned)ncpu);
75 } /* else "auto": keep default */
76
77 CPU_SET(cpu, &set);
78 (void)pthread_setaffinity_np(pthread_self(), sizeof(set), &set);
79#else
80 (void)policy; (void)tidx; (void)nth; (void)use_affinity;
81#endif
82}

115ie_threadpool_t* ie_tp_create(unsigned nth, const char *affinity) {
116 if (nth < 1) nth = 1;
117
118 struct ie_threadpool *tp = (struct ie_threadpool*)calloc(1, sizeof(*tp));
119 if (!tp) return NULL;
120
121 tp->nth = nth;
122 tp->affinity = (affinity && *affinity) ? affinity : "auto";
123
124#ifdef __linux__
125 const char *env = getenv("IE_TP_USE_AFFINITY");
126 tp->use_affinity = (env && strcmp(env, "1") == 0) ? 1 : 0;
127#else
128 tp->use_affinity = 0;
129#endif
130
131 /* Spawn long-lived idle workers (not strictly required for the baseline). */
132 tp->threads = (pthread_t*)calloc(nth, sizeof(pthread_t));
133 if (!tp->threads) { free(tp); return NULL; }
134
135 for (unsigned i = 0; i < nth; ++i) {
136 ie_worker_arg_t *wa = (ie_worker_arg_t*)malloc(sizeof(*wa));
137 if (!wa) continue;
138 wa->tp = tp; wa->tidx = i;
139 if (pthread_create(&tp->threads[i], NULL, ie_worker_main, wa) != 0) {
140 free(wa);
141 }
142 }
143 return (ie_threadpool_t*)tp;
144}

194void ie_tp_parallel_for(ie_threadpool_t *h,
195 unsigned n,
196 unsigned grainsize,
197 ie_tp_for_body_fn body_fn,
198 void *ctx) {
199 if (n == 0 || !body_fn) return;
200
201 /* Preferred worker count and affinity policy for mapping. */
202 unsigned nth_pref = 1;
203 const char *aff = "auto";
204 int use_aff = 0;
205
206 if (h) {
207 struct ie_threadpool *tp = (struct ie_threadpool*)h;
208 nth_pref = tp->nth ? tp->nth : 1u;
209 aff = tp->affinity ? tp->affinity : "auto";
210 use_aff = tp->use_affinity;
211 }
212
213 if (nth_pref <= 1) {
214 body_fn(ctx, 0u, n);
215 return;
216 }
217
218 /* Determine chunk size and total number of chunks. */
219 unsigned min_chunk = (grainsize > 0 ? grainsize : (n + nth_pref - 1) / nth_pref);
220 if (min_chunk == 0) min_chunk = 1;
221
222 unsigned nchunks = (n + min_chunk - 1) / min_chunk;
223
224 /* Launch one task thread per chunk to guarantee full coverage. */
225 pthread_t *ths = (pthread_t*)calloc(nchunks, sizeof(pthread_t));
226 if (!ths) {
227 /* Fallback: single-threaded if allocation fails. */
228 body_fn(ctx, 0u, n);
229 return;
230 }
231
232 for (unsigned c = 0; c < nchunks; ++c) {
233 unsigned start = c * min_chunk;
234 unsigned end = start + min_chunk;
235 if (start >= n) { ths[c] = (pthread_t)0; continue; }
236 if (end > n) end = n;
237
238 ie_task_arg_t *ta = (ie_task_arg_t*)malloc(sizeof(*ta));
239 if (!ta) { ths[c] = (pthread_t)0; continue; }
240
241 /* Map chunk index to a virtual worker index for affinity purposes. */
242 ta->affinity = aff;
243 ta->use_affinity = use_aff;
244 ta->nth_map = nth_pref;
245 ta->tidx_map = (nth_pref ? (c % nth_pref) : 0u);
246
247 ta->start = start;
248 ta->end = end;
249 ta->body = body_fn;
250 ta->ctx = ctx;
251
252 if (pthread_create(&ths[c], NULL, ie_task_main, ta) != 0) {
253 free(ta);
254 ths[c] = (pthread_t)0;
255 }
256 }
257
258 for (unsigned c = 0; c < nchunks; ++c) {
259 if (ths[c]) pthread_join(ths[c], NULL);
260 }
261 free(ths);
262}
