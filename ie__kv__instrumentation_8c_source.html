45static size_t env_size(const char *name, size_t defv) {
46 const char *s = getenv(name);
47 if (!s || !*s) return defv;
48 char *end = NULL;

61static int kv_contains(uint32_t token) {
62 for (size_t i = 0; i < g_kv.count; ++i) {
63 if (g_kv.slots[i] == token) return 1;
64 }
65 return 0;

73static void kv_insert_fifo(uint32_t token) {
74 if (g_kv.capacity == 0) return;
75 if (g_kv.count < g_kv.capacity) {
76 g_kv.slots[g_kv.count++] = token;
77 } else {
78 /* Evict oldest at head, write new token, advance head */
79 g_kv.slots[g_kv.head] = token;
80 g_kv.head = (g_kv.head + 1) % g_kv.capacity;
81 }
82}

87int ie_kv_begin_round(void) {
88 /* Reset counters and (re)allocate set as needed. */
89 const size_t cap = env_size("IE_KV_CAP", 4096); /* default: 4K tokens */
90 if (g_kv.slots && g_kv.capacity != cap) {
91 free(g_kv.slots);
92 g_kv.slots = NULL;
93 }
94 if (!g_kv.slots) {
95 g_kv.slots = (uint32_t *)malloc(sizeof(uint32_t) * cap);
96 if (!g_kv.slots) {
97 g_kv.capacity = 0;
98 g_kv.count = g_kv.head = 0;
99 g_kv.hits = g_kv.misses = 0;
100 g_kv.ready = 0;
101 return -1;
102 }
103 }
104 g_kv.capacity = cap;
105 g_kv.count = 0;
106 g_kv.head = 0;
107 g_kv.hits = 0;
108 g_kv.misses = 0;
109 g_kv.ready = 1;
110 return 0;
111}

113void ie_kv_on_token(uint32_t token) {
114 if (!g_kv.ready || g_kv.capacity == 0) return;
115 if (kv_contains(token)) {
116 g_kv.hits++;
117 return;
118 }
119 g_kv.misses++;
120 kv_insert_fifo(token);
121}

123void ie_kv_finish_round(uint64_t total_tokens,
124 uint64_t *out_hits,
125 uint64_t *out_misses) {
126 assert(out_hits != NULL);
127 assert(out_misses != NULL);
128 if (!g_kv.ready) {
129 /* Not initialized: conservative fallback. */
130 *out_hits = 0;
131 *out_misses = total_tokens;
132 return;
133 }
134 /* Fallback if no tokens were observed but caller claims otherwise. */
135 uint64_t hits = g_kv.hits;
136 uint64_t misses = g_kv.misses;
137 if ((hits + misses) == 0 && total_tokens > 0) {
138 misses = total_tokens;
139 }
140
141 *out_hits = hits;
142 *out_misses = misses;
143
144 /* Free the set to keep memory footprint low across short-lived runs. */
145 free(g_kv.slots);
146 g_kv.slots = NULL;
147 g_kv.capacity = g_kv.count = g_kv.head = 0;
148 g_kv.hits = g_kv.misses = 0;
149 g_kv.ready = 0;
150}
