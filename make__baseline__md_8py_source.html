18def newest_report_dir() -> Path:
19 candidates = [p for p in REPORTS_DIR.glob("*") if p.is_dir()]

24def read_summary(d: Path) -> dict:

28def read_samples(d: Path) -> list[dict]:

35def main() -> int:
