162int ie_tok_encode(const ie_vocab_t *v,
163 const char *text,
164 uint32_t *ids,
165 uint32_t *out_count) {
166 (void)v; /* vocab_size is not required for hashing but reserved for future use */
167 if (!text || !out_count) return -1;
168
169 const unsigned char *s = (const unsigned char*)text;
170 uint32_t needed = 0;
171
172 /* First pass: count tokens. */
173 size_t i = 0;
174 const size_t L = strlen(text);
175 while (i < L) {
176 /* Skip leading whitespace between tokens. */
177 while (i < L && is_space_byte(s[i])) ++i;
178 if (i >= L) break;
179
180 /* Start of token. */
181 size_t start = i;
182 while (i < L && !is_space_byte(s[i])) ++i;
183 size_t tok_len = i - start;
184 if (tok_len > 0) {
185 ++needed;
186 }
187 }
188
189 if (!ids) {
190 *out_count = needed;
191 return 0;
192 }
193
194 /* Second pass: produce IDs; respect capacity in *out_count. */
195 uint32_t cap = *out_count;
196 uint32_t wrote = 0;
197 i = 0;
198 while (i < L && wrote < cap) {
199 while (i < L && is_space_byte(s[i])) ++i;
200 if (i >= L) break;
201 size_t start = i;
202 while (i < L && !is_space_byte(s[i])) ++i;
203 size_t tok_len = i - start;
204 if (tok_len > 0) {
205 uint32_t h = fnv1a_32(s + start, tok_len);
206 ids[wrote++] = clamp_to_pos31(h);
207 }
208 }
209
210 *out_count = wrote;
211 /* If capacity was insufficient, signal failure. Tests pre-size correctly. */
212 return (wrote == needed) ? 0 : -2;
213}

231int ie_tok_decode(const ie_vocab_t *v,
232 const uint32_t *ids,
233 uint32_t count,
234 char *out,
235 size_t out_sz) {
236 (void)v;
237 if (!out || out_sz == 0) return -1;
238
239 /* Handle empty sequence. */
240 if (!ids || count == 0) {
241 if (out_sz > 0) out[0] = '\0';
242 return 0;
243 }
244
245 /* Conservative sizing: worst case "T4294967295 " per token + NUL. */
246 const size_t worst_per = sizeof("T4294967295 ") - 1; /* 12 bytes */
247 size_t worst = (size_t)count * worst_per + 1;
248 if (out_sz < worst) {
249 /* Strict contract for tests: return error if too small. */
250 return -2;
251 }
252
253 char *p = out;
254 size_t rem = out_sz;
255
256 for (uint32_t i = 0; i < count; ++i) {
257 int n = (i == 0)
258 ? snprintf(p, rem, "T%u", ids[i])
259 : snprintf(p, rem, " T%u", ids[i]);
260
261 if (n < 0 || (size_t)n >= rem) {
262 return -2;
263 }
264 p += (size_t)n;
265 rem -= (size_t)n;
266 }
267
268 /* Ensure NUL-termination. */
269 if (rem == 0) return -2;
270 *p = '\0';
271 return 0;
