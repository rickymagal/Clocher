51static uint32_t fnv1a32(const char *s) {
52 uint32_t h = 2166136261u;
53 if (!s) return h ^ 0xA5A5u;
54 const unsigned char *p = (const unsigned char *)s;
55 while (*p) {
56 h ^= (uint32_t)(*p++);
57 h *= 16777619u;
58 }
59 return h ? h : 0x9E3779B1u;
60}

68static uint64_t xorshift64star(uint64_t *state) {
69 uint64_t x = (*state == 0 ? 0x106689D45497fdb5ULL : *state);

91ie_status_t ie_engine_create(const ie_engine_params_t *p, ie_engine_t **out) {
92 if (!out) return 1;

95 if (!e) return 1;

98 /* Copy by value to preserve ABI compatibility if the header evolves. */
99 e->cfg = *p;
100 } else {
101 memset(&e->cfg, 0, sizeof(e->cfg));
102 }
103
104 /* Initialize metrics snapshot to safe defaults. */
105 memset(&e->last, 0, sizeof(e->last));
106

108 e->seed = 0x9E3779B97F4A7C15ULL;

112 e->seed ^= (uint64_t)fnv1a32(e->cfg.prefetch) << 24;
113
114 *out = e;
115 return IE_OK;

142ie_status_t ie_engine_generate(ie_engine_t *h,
143 const char *prompt,
144 size_t max_new_tokens,
145 uint32_t *out_tokens,
146 uint32_t *out_count) {
147 if (!h || !out_count || !prompt) return 1;
148
149 if (max_new_tokens == 0) {
150 *out_count = 0;
151 h->last.tps_true = 0.0;
152 h->last.latency_p50_ms = 0.0;
153 h->last.latency_p95_ms = 0.0;
154 h->last.rss_peak_mb = 0u;
155 /* KV counters are maintained globally by ie_kv_instrumentation */
156 return IE_OK;
157 }
158 if (!out_tokens) return 1;
159
160 /* Per-call deterministic PRNG seeded by engine seed XOR prompt hash. */
161 uint64_t rng = h->seed ^ (uint64_t)fnv1a32(prompt);
162
163 uint32_t produced = 0;
164 for (size_t i = 0; i < max_new_tokens; ++i) {
165 const uint64_t r = xorshift64star(&rng);
166 const uint32_t tok = (uint32_t)(r % 50000u); /* Compact ID space for tests. */
167 out_tokens[i] = tok;
168 ie_kv_on_token(tok); /* count inside timing window */
169 ++produced;
170 }
171 *out_count = produced;
172
173 /* Metrics snapshot placeholders (harness fills time-related values). */
174 h->last.tps_true = 0.0;
175 h->last.latency_p50_ms = 0.0;
176 h->last.latency_p95_ms = 0.0;
177 h->last.rss_peak_mb = 0u;
178
179 return IE_OK;
180}
