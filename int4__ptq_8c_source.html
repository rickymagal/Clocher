22static inline int clamp_q4(float x) {
23 /* round to nearest even is fine; typical compilers use roundf. */
24 float r = roundf(x);
25 if (r > 7.0f) return 7;
26 if (r < -7.0f) return -7;
27 return (int)r;
28}

40static inline uint8_t pack_q4_pair(int q0, int q1) {
41 uint8_t n0 = (uint8_t)(q0 + 8); /* 1..15 (0 reserved for -8, not used) */
42 uint8_t n1 = (uint8_t)(q1 + 8);
43 return (uint8_t)((n1 << 4) | (n0 & 0x0F));
44}

53static inline void unpack_q4_pair(uint8_t byte, int *out_q0, int *out_q1) {
54 int n0 = (int)(byte & 0x0F);
55 int n1 = (int)((byte >> 4) & 0x0F);
56 *out_q0 = n0 - 8;
57 *out_q1 = n1 - 8;
58}

99int ie_int4_quantize_per_tensor(const float *src,
100 size_t rows,
101 size_t cols,
102 uint8_t *dst_packed,
103 float *out_scale) {
104 if (!src || !dst_packed || !out_scale || rows == 0 || cols == 0)
105 return IE_INT4_STATUS_BADARG;
106
107 float absmax = 0.0f;
108 int st = ie_int4_absmax(src, rows, cols, &absmax);
109 if (st != IE_INT4_STATUS_OK) return st;
110
111 float scale = (absmax > 0.0f) ? (absmax / 7.0f) : 1.0f;
112 if (!isfinite(scale) || scale <= 0.0f) return IE_INT4_STATUS_NUMERIC;
113 *out_scale = scale;
114
115 const size_t rb = ie_int4_rowbytes(cols);
116 for (size_t r = 0; r < rows; ++r) {
117 const float *row = src + r * cols;
118 uint8_t *out = dst_packed + r * rb;
119
120 size_t c = 0;
121 for (; c + 1 < cols; c += 2) {
122 int q0 = clamp_q4(row[c] / scale);
123 int q1 = clamp_q4(row[c + 1] / scale);
124 out[c / 2] = pack_q4_pair(q0, q1);
125 }
126 if (c < cols) {
127 /* Odd tail: encode last even column and set high nibble to zero code. */
128 int q0 = clamp_q4(row[c] / scale);
129 out[c / 2] = pack_q4_pair(q0, 0 /* high nibble = 0 -> encoded as 8 after bias */);
130 }
131 }
132
133 return IE_INT4_STATUS_OK;
134}

240int ie_int4_error_metrics(const float *ref,
241 const float *test,
242 size_t n,
243 double *out_mse,
244 double *out_cosine) {
245 if (!ref || !test || !out_mse || !out_cosine || n == 0) return IE_INT4_STATUS_BADARG;
246
247 long double se = 0.0L;
248 long double dot = 0.0L;
249 long double nr = 0.0L;
250 long double nt = 0.0L;
251
252 for (size_t i = 0; i < n; ++i) {
253 long double r = ref[i];
254 long double t = test[i];
255 long double d = r - t;
256 se += d * d;
257 dot += r * t;
258 nr += r * r;
259 nt += t * t;
260 }
261
262 double mse = (double)(se / (long double)n);
263 double denom = sqrt((double)nr) * sqrt((double)nt);
264 double cosine = (denom > 0.0) ? (double)(dot / denom) : 0.0;
265
266 if (bad_float3(mse, cosine, denom)) return IE_INT4_STATUS_NUMERIC;
267
268 *out_mse = mse;
269 *out_cosine = cosine;
270 return IE_INT4_STATUS_OK;
271}
