80static size_t view_materialized_nbytes(const ie_weight_view_t *v) {
81 if (!v) return 0;
82

88 return 0;

92 /* Packed int4: defaults_nbytes is the packed byte length of the tensor. */

94 }
95

110static uint64_t hash_view(const ie_weight_view_t *v) {
111 size_t n = view_materialized_nbytes(v);
112 if (n == 0) return 0;
113
114 void *buf = malloc(n);
115 assert(buf && "OOM in test");

117 size_t written = 0;
118
119 if (v->kind == IE_WVIEW_DIRECT) {
120 assert(v->data != NULL);
121 assert(v->nbytes == n);
122 memcpy(buf, v->data, n);
123 written = n;
124 } else {
125 written = ie_weights_dedup_materialize(v, buf, n);
126 }
127
128 assert(written == n && "materialization failed");
129 uint64_t h = hash_bytes(buf, n);
