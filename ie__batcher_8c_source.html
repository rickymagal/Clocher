100static int pop_next(ie_batcher_t *b, size_t *out_idx) {
101 int ret = 0;
102 pthread_mutex_lock(&b->mu);
103 if (b->produced < b->n_prompts) {
104 *out_idx = b->produced++;
105 ret = 1;
106 if (b->produced == b->n_prompts) {
107 b->producer_done = 1;
108 }
109 }
110 pthread_mutex_unlock(&b->mu);
111 return ret;
112}

123static void enqueue_item(ie_batcher_t *b, const ie_batcher_item_t *it) {
124 pthread_mutex_lock(&b->mu);
125 while (b->size == b->cap && !b->stop) {
126 pthread_cond_wait(&b->cv_not_full, &b->mu);
127 }
128 if (!b->stop) {
129 b->ring[b->w] = *it; /* struct copy; moves pointer ownership */
130 b->w = (b->w + 1) % b->cap;
131 b->size++;
132 pthread_cond_signal(&b->cv_not_empty);
133 } else {
134 if (it->ids) free((void*)it->ids);
135 }
136 pthread_mutex_unlock(&b->mu);
137}

144static void *worker_main(void *arg) {
145 ie_batcher_t *b = (ie_batcher_t*)arg;
146
147 for (;;) {
148 size_t idx = 0;
149 if (!pop_next(b, &idx)) break;
150
151 ie_batcher_item_t it;
152 memset(&it, 0, sizeof(it));
153 it.prompt = b->prompts[idx];
154
155 uint32_t *ids = NULL;
156 size_t n = 0;
157 int st = b->tokenize(it.prompt, &ids, &n, b->user_ctx);
158 it.ids = ids;
159 it.n_ids = n;
160 it.status = st;
161
162 enqueue_item(b, &it);
163 }
164 return NULL;
165}

177 void *user_ctx) {
178 (void)n_workers; /* determinism: always 1 producer thread */
179
180 if (!prompts || n_prompts == 0 || !tokenize_cb) return NULL;
181 if (queue_capacity == 0) queue_capacity = microbatch ? microbatch : 8;
182
183 ie_batcher_t *b = (ie_batcher_t*)xcalloc(1, sizeof(*b));
184 if (!b) return NULL;
185
186 b->prompts = prompts;
187 b->n_prompts = n_prompts;
188 b->microbatch = microbatch ? microbatch : 1;
189 b->cap = queue_capacity;
190 b->ring = (ie_batcher_item_t*)xcalloc(b->cap, sizeof(*b->ring));
191 if (!b->ring) { free(b); return NULL; }
192
193 pthread_mutex_init(&b->mu, NULL);
194 pthread_cond_init(&b->cv_not_full, NULL);
195 pthread_cond_init(&b->cv_not_empty, NULL);
196
197 b->tokenize = tokenize_cb;
198 b->user_ctx = user_ctx;
199 b->n_workers = 1;
200 b->threads = (pthread_t*)xcalloc(b->n_workers, sizeof(pthread_t));
201 if (!b->threads) {
202 free(b->ring);
203 free(b);
204 return NULL;
205 }
206
207 if (pthread_create(&b->threads[0], NULL, worker_main, b) != 0) {
208 free(b->threads);
209 free(b->ring);
210 free(b);
211 return NULL;
212 }
213
214 return b;

217int ie_batcher_get(ie_batcher_t *b, ie_batcher_view_t *out_view) {
218 if (!b || !out_view) return 0;
219
220 pthread_mutex_lock(&b->mu);
221 while (b->size == 0 && !b->producer_done && !b->stop) {
222 pthread_cond_wait(&b->cv_not_empty, &b->mu);
223 }
224
225 if (b->size == 0 && (b->producer_done || b->stop)) {
226 out_view->items = NULL;
227 out_view->count = 0;
228 out_view->done = 1;
229 pthread_mutex_unlock(&b->mu);
230 return 0;
231 }
232
233 size_t avail = b->size;
234 size_t first = b->r;
235 size_t can_take = b->microbatch;
236 if (can_take > avail) can_take = avail;
237
238 /* Return a contiguous tail segment only (no wrap across r->end). */
239 size_t tail = b->cap - first;
240 if (can_take > tail) can_take = tail;
241
242 out_view->items = &b->ring[first];
243 out_view->count = can_take;
244 out_view->done = (b->producer_done && (b->size == can_take));
245
246 pthread_mutex_unlock(&b->mu);
247 return 1;
248}

250void ie_batcher_advance(ie_batcher_t *b, size_t n_consumed) {
251 if (!b || n_consumed == 0) return;
252
253 pthread_mutex_lock(&b->mu);
254
255 for (size_t i = 0; i < n_consumed; ++i) {
256 size_t idx = (b->r + i) % b->cap;
257 clear_item(&b->ring[idx]);
258 }
259
260 b->r = (b->r + n_consumed) % b->cap;
261 b->size -= n_consumed;
262
263 pthread_cond_signal(&b->cv_not_full);
264 pthread_mutex_unlock(&b->mu);
265}

267void ie_batcher_destroy(ie_batcher_t *b) {
268 if (!b) return;
269
270 pthread_mutex_lock(&b->mu);
271 b->stop = 1;
272 pthread_cond_broadcast(&b->cv_not_full);
273 pthread_cond_broadcast(&b->cv_not_empty);
274 pthread_mutex_unlock(&b->mu);
275
276 for (unsigned i = 0; i < b->n_workers; ++i) {
277 if (b->threads[i]) pthread_join(b->threads[i], NULL);
278 }
279
280 /* Free any remaining items in the ring. */
281 for (size_t i = 0; i < b->cap; ++i) clear_item(&b->ring[i]);
282
283 pthread_cond_destroy(&b->cv_not_full);
284 pthread_cond_destroy(&b->cv_not_empty);
285 pthread_mutex_destroy(&b->mu);
286
287 free(b->threads);
288 free(b->ring);
289 free(b);
290}
