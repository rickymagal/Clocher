4"""
5Generate docs/PERFORMANCE.md from strict benchmark JSON logs.
6
7Behavior:
8- Accepts CPU and/or GPU JSONL files (each produced by scripts/true_tps_strict.sh).
9- If only one side is provided, it still renders both sections using any existing
10 counterpart JSON found under build/ (best effort).
11- Aggregations and the comparative table use ONLY the last 3 runs for each device.
12- "Best true TPS" is the best aggregate TPS across the currently included CPU/GPU sections.
13- "Run Parameters & Conditions" come from CLI overrides first, then CPU summary, then GPU summary.
14- Spatial metrics include MB/token, total bytes touched, coverage vs model.bin, and effective bandwidth.
15- System & Model Info is discovered locally (Linux-friendly).
16
17Inputs (per device JSONL):
18- Per-run rows: include tokens_generated, wall_time_s, tps_true, latency_p50_ms, latency_p95_ms, rss_peak_mb, kv_hits, kv_misses.
19- Final summary row: includes aggregated fields and configuration (threads, precision, prompts, model_dir, etc.).
20"""

41def _read_json_lines(path: str) -> List[Dict[str, Any]]:
42 out: List[Dict[str, Any]] = []
43 with open(path, "r", encoding="utf-8") as f:
44 for line in f:
45 s = line.strip()
46 if not s:
47 continue
48 try:
49 out.append(json.loads(s))

51 # Ignore non-JSON lines from the harness

55def _partition_runs_and_summary(objs: List[Dict[str, Any]]) -> Tuple[List[Run], Optional[Dict[str, Any]]]:
56 runs: List[Run] = []
57 summary: Optional[Dict[str, Any]] = None
58 for o in objs:
59 is_summary = any(k in o for k in ("threads", "precision", "model_dir", "prompts", "runs", "rss_peak_mb_max"))
60 is_run = all(k in o for k in ("tokens_generated", "wall_time_s", "tps_true"))
61 if is_summary:
62 summary = o
63 elif is_run:
64 runs.append(o)
65 return runs, summary

102def _discover_cpu_model() -> Optional[str]:
103 try:
104 with open("/proc/cpuinfo", "r", encoding="utf-8") as f:
105 m = re.search(r"model name\s*:\s*(.+)", f.read())
106 if m:
107 return m.group(1).strip()
108 except Exception:
109 pass
110 return platform.processor() or platform.machine() or None
111
112def _discover_mem_total_gb() -> Optional[float]:
113 try:
114 with open("/proc/meminfo", "r", encoding="utf-8") as f:
115 m = re.search(r"MemTotal:\s+(\d+)\s+kB", f.read())
116 if m:
117 kb = float(m.group(1))
118 return kb * 1024.0 / 1_000_000_000.0
119 except Exception:
120 pass
121 return None

123def _discover_os() -> Optional[str]:
124 try:
125 with open("/etc/os-release", "r", encoding="utf-8") as f:
126 kv = dict(
127 (line.split("=", 1)[0], line.split("=", 1)[1].strip().strip('"'))
128 for line in f if "=" in line
129 )
130 return kv.get("PRETTY_NAME") or platform.platform()
131 except Exception:
132 return platform.platform()
133
134def _discover_kernel() -> Optional[str]:
135 try:
136 u = platform.uname()
137 return f"{u.release}-{u.machine}"
138 except Exception:
139 return platform.version()

141def _discover_git() -> Optional[str]:
142 try:
143 sha = subprocess.check_output(["git", "rev-parse", "--short", "HEAD"], stderr=subprocess.DEVNULL).decode().strip()
144 dirty = subprocess.call(["git", "diff", "--quiet", "--ignore-submodules"], stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
145 return f"{sha} {'DIRTY' if dirty != 0 else ''}".strip()
146 except Exception:
147 return None

168def _take_last3(runs: List[Run]) -> List[Run]:
169 return runs[-3:] if len(runs) > 3 else runs

171def _agg(runs: List[Run]) -> Agg:
172 tokens = [int(r.get("tokens_generated", 0)) for r in runs]
173 wall = [float(r.get("wall_time_s", 0.0)) for r in runs]
174 tps = [float(r.get("tps_true", 0.0)) for r in runs]
175 p50 = [float(r.get("latency_p50_ms", r.get("p50_ms", 0.0))) for r in runs if ("latency_p50_ms" in r or "p50_ms" in r)]
176 p95 = [float(r.get("latency_p95_ms", r.get("p95_ms", 0.0))) for r in runs if ("latency_p95_ms" in r or "p95_ms" in r)]
177 rss = [float(r.get("rss_peak_mb", 0.0)) for r in runs if "rss_peak_mb" in r]
178 kvh = [int(r.get("kv_hits", 0)) for r in runs]
179 kvm = [int(r.get("kv_misses", 0)) for r in runs]
180
181 tokens_sum = _sum_i(tokens)
182 wall_sum = _sum_f(wall)
183 tps_true = (float(tokens_sum) / wall_sum) if wall_sum > 0 else None
184
185 return {
186 "runs": len(runs),
187 "tokens_sum": tokens_sum,
188 "wall_sum": wall_sum,
189 "tps_true": tps_true,
190 "p50_mean": _mean(p50),
191 "p95_mean": _mean(p95),
192 "rss_mean": _mean(rss),
193 "rss_max": max(rss) if rss else None,
194 "kv_hits": _sum_i(kvh),
195 "kv_misses": _sum_i(kvm),
196 }

198def _mb_per_token(bpt: Optional[int]) -> str:
199 if bpt is None:
200 return "n/a"
201 return f"{_fmt_float(bpt / 1_000_000.0, 1)} MB/token"

203def _bytes_touched(tokens_sum: int, bpt: Optional[int]) -> Optional[float]:
204 if not bpt or tokens_sum <= 0:
205 return None
206 return float(tokens_sum) * float(bpt)

208def _bandwidth_gbps(bytes_touched: Optional[float], wall_sum: Optional[float]) -> Optional[float]:
209 if bytes_touched is None or wall_sum is None or wall_sum <= 0:
210 return None
211 return _bytes_to_gb(bytes_touched) / wall_sum

215def _render_device(title: str, agg: Agg, bpt: Optional[int], model_size: Optional[int]) -> str:
216 bt = _bytes_touched(agg["tokens_sum"], bpt)
217 bw = _bandwidth_gbps(bt, agg["wall_sum"])
218 coverage = "n/a"
219 if bpt and model_size and model_size > 0:
220 coverage = f"{(float(bpt)/float(model_size)):.3f}"
221 return f"""## {title} — Summary (latest benchmark)
222- Runs: **{agg['runs']}**
223- Tokens generated (Σ): **{agg['tokens_sum']}**
224- Wall time (Σ): **{_fmt_float(agg['wall_sum'], 3, 's')}**
225- True TPS (Σ tokens / Σ time): **{_fmt_float(agg['tps_true'], 3)}**
226
227## Latency
228- p50 (mean across runs): **{_fmt_float(agg['p50_mean'], 3, 'ms')}**
229- p95 (mean across runs): **{_fmt_float(agg['p95_mean'], 3, 'ms')}**
230
231## Spatial Complexity (Memory & Cache)
232- RSS peak (mean): **{_fmt_float(agg['rss_mean'], 3, 'MB')}**
233- RSS peak (max): **{_fmt_float(agg['rss_max'], 3, 'MB')}**
234- KV cache: **{agg['kv_hits']} hits / {agg['kv_misses']} misses**
235- IE_BYTES_PER_TOKEN: **{_mb_per_token(bpt)}**
236- Bytes touched (Σ): **{_fmt_float(_bytes_to_gb(bt), 1, 'GB')}**
237- Working-set coverage (bytes_per_token / model.bin): **{coverage}**
238- Effective bandwidth: **{_fmt_float(bw, 2, 'GB/s')}**
239"""

274def _render_table(cpu_runs: List[Run], gpu_runs: List[Run]) -> str:
275 lines = []
276 lines.append("## Comparative Runs\n")
277 lines.append("| Device | Run | Tokens | Wall (s) | TPS | p50 (ms) | p95 (ms) | RSS peak (MB) | KV hits | KV misses |")
278 lines.append("|:------:|----:|-------:|---------:|----:|---------:|---------:|--------------:|--------:|----------:|")
279 def row(dv: str, r: Run, idx: int) -> str:
280 return (
281 f"| {dv} | {idx} | "
282 f"{int(r.get('tokens_generated', 0))} | "
283 f"{_fmt_float(float(r.get('wall_time_s', 0.0)), 3)} | "
284 f"{_fmt_float(float(r.get('tps_true', 0.0)), 3)} | "
285 f"{_fmt_float(float(r.get('latency_p50_ms', r.get('p50_ms', 0.0))), 3)} | "
286 f"{_fmt_float(float(r.get('latency_p95_ms', r.get('p95_ms', 0.0))), 3)} | "
287 f"{_fmt_float(float(r.get('rss_peak_mb', 0.0)), 3)} | "
288 f"{int(r.get('kv_hits', 0))} | {int(r.get('kv_misses', 0))} |"
289 )
290 for i, r in enumerate(cpu_runs, 1):
291 lines.append(row("CPU", r, i))
292 for i, r in enumerate(gpu_runs, 1):
293 lines.append(row("GPU", r, i))
294 return "\n".join(lines) + "\n"

297 def g(k: str) -> Any:
298 if cli.get(k) is not None:
299 return cli[k]
