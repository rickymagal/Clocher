79static uint32_t kib_to_mib_ceil(uint64_t kib) {
80 if (kib == 0) return 0u;
81 /* ceil(kib/1024) = (kib + 1023) >> 10 */
82 uint64_t mib = (kib + 1023u) >> 10;
83 return (mib > 0xFFFFFFFFu) ? 0xFFFFFFFFu : (uint32_t)mib;
84}

89static int parse_status_kb_line(const char* line, const char* key, uint64_t* out_kb) {
90 if (!line || !key || !out_kb) return 0;
91 size_t klen = strlen(key);
92 if (strncmp(line, key, klen) != 0) return 0;
93
94 const char* p = line + klen;
95 while (*p == ' ' || *p == '\t') ++p;
96
97 uint64_t val = 0;
98 int matched = 0;
99 while (*p >= '0' && *p <= '9') {
100 matched = 1;
101 val = val * 10 + (uint64_t)(*p - '0');
102 ++p;
103 }
104 if (!matched) return 0;
105
106 *out_kb = val; /* procfs reports kB */
107 return 1;
108}

118static int read_proc_kb_single(const char *path, const char *key, uint64_t *out_kb) {
119#if defined(__linux__)
120 FILE *f = fopen(path, "r");
121 if (!f) return -errno ? -errno : -1;
122 char buf[512];
123 while (fgets(buf, sizeof buf, f)) {
124 uint64_t kb = 0;
125 if (parse_status_kb_line(buf, key, &kb)) {
126 fclose(f);
127 *out_kb = kb;
128 return 0;
129 }
130 }
131 fclose(f);
132 return -ENOENT;
133#else
134 (void)path; (void)key; (void)out_kb;
135 return -ENOSYS;
136#endif
137}

163void ie_metrics_snapshot(ie_metrics_t* out, int reset_after) {
164 if (!out) return;
165 const uint64_t hits = atomic_load_explicit(&g_kv_hits, memory_order_relaxed);
166 const uint64_t miss = atomic_load_explicit(&g_kv_miss, memory_order_relaxed);
167
168 out->kv_hits = hits;
169 out->kv_misses = miss;
170
171 if (reset_after) {
172 atomic_store_explicit(&g_kv_hits, 0, memory_order_relaxed);
173 atomic_store_explicit(&g_kv_miss, 0, memory_order_relaxed);

181uint32_t ie_metrics_sample_rss_peak(void) {
182 uint32_t best_mib = 0;
183
184#if defined(__linux__)
185 /* 1) Prefer VmHWM (kB) — process peak resident set size. */
186 {
187 uint64_t kb = 0;
188 int rc = read_proc_kb_single("/proc/self/status", "VmHWM:", &kb);
189 if (rc == 0) {
190 uint32_t mib = kib_to_mib_ceil(kb);
191 if (mib > best_mib) best_mib = mib;
192 dbg("[RSS] VmHWM: %llu kB -> %u MiB\n",
193 (unsigned long long)kb, mib);
194 } else {
195 dbg("[RSS] VmHWM not found (rc=%d)\n", rc);
196 }
197 }
198
199 /* 2) Fallback: current VmRSS (kB) as lower bound if HWM is unavailable. */
200 if (best_mib == 0) {
201 uint64_t kb = 0;
202 int rc = read_proc_kb_single("/proc/self/status", "VmRSS:", &kb);
203 if (rc == 0) {
204 uint32_t mib = kib_to_mib_ceil(kb);
205 if (mib > best_mib) best_mib = mib;
206 dbg("[RSS] VmRSS: %llu kB -> %u MiB\n",
207 (unsigned long long)kb, mib);
208 } else {
209 dbg("[RSS] VmRSS not found (rc=%d)\n", rc);
210 }
211 }
212
213 /* 3) Try smaps_rollup Rss: (kB) — often more accurate with mmaps. */
214 if (best_mib == 0) {
215 uint64_t kb = 0;
216 int rc = read_proc_kb_single("/proc/self/smaps_rollup", "Rss:", &kb);
217 if (rc == 0) {
218 uint32_t mib = kib_to_mib_ceil(kb);
219 if (mib > best_mib) best_mib = mib;
220 dbg("[RSS] smaps_rollup Rss: %llu kB -> %u MiB\n",
221 (unsigned long long)kb, mib);
222 } else {
223 dbg("[RSS] smaps_rollup Rss not found (rc=%d)\n", rc);
224 }
225 }
226
227 /* 4) Final fallback: getrusage (Linux ru_maxrss is kB). */
228 if (best_mib == 0) {
229 struct rusage ru;
230 if (getrusage(RUSAGE_SELF, &ru) == 0) {
231 uint64_t kb = (uint64_t)ru.ru_maxrss;
232 uint32_t mib = kib_to_mib_ceil(kb);
233 if (mib > best_mib) best_mib = mib;
234 dbg("[RSS] getrusage ru_maxrss: %llu kB -> %u MiB\n",
235 (unsigned long long)kb, mib);
236 } else {
237 dbg("[RSS] getrusage failed: errno=%d\n", errno);
238 }
239 }
240
241#elif defined(__APPLE__)
242 /* macOS primary: mach_task_basic_info (bytes). */
243 {
244 mach_task_basic_info info;
245 mach_msg_type_number_t count = MACH_TASK_BASIC_INFO_COUNT;
246 kern_return_t kr = task_info(mach_task_self(), MACH_TASK_BASIC_INFO,
247 (task_info_t)&info, &count);
248 if (kr == KERN_SUCCESS) {
249 /* resident_size_max is peak if supported; otherwise 0. */
250 uint64_t bytes = (uint64_t)info.resident_size_max;
251 if (bytes == 0) bytes = (uint64_t)info.resident_size; /* at least current */
252 uint32_t mib = (uint32_t)((bytes + (1024u*1024u - 1)) / (1024u*1024u));
253 if (mib > best_mib) best_mib = mib;
254 dbg("[RSS] mach_task_basic_info: %llu bytes -> %u MiB\n",
255 (unsigned long long)bytes, mib);
256 } else {
257 dbg("[RSS] mach task_info failed: kr=%d\n", (int)kr);
258 }
259 }
260
261 /* Fallback: getrusage (on macOS ru_maxrss is bytes). */
262 if (best_mib == 0) {
263 struct rusage ru;
264 if (getrusage(RUSAGE_SELF, &ru) == 0) {
265 uint64_t bytes = (uint64_t)ru.ru_maxrss;
266 uint32_t mib = (uint32_t)((bytes + (1024u*1024u - 1)) / (1024u*1024u));
267 if (mib > best_mib) best_mib = mib;
268 dbg("[RSS] getrusage ru_maxrss(bytes): %llu -> %u MiB\n",
269 (unsigned long long)bytes, mib);
270 }
271 }
272#else
273 /* Other OS: try getrusage and assume kB (conservative). */
274 {
275 struct rusage ru;
276 if (getrusage(RUSAGE_SELF, &ru) == 0) {
277 uint64_t kb = (uint64_t)ru.ru_maxrss;
278 uint32_t mib = kib_to_mib_ceil(kb);
279 if (mib > best_mib) best_mib = mib;
280 dbg("[RSS] generic getrusage ru_maxrss: %llu kB -> %u MiB\n",
281 (unsigned long long)kb, mib);
282 }
283 }
284#endif
285
286 return best_mib;
287}
