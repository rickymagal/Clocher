96static uint32_t ie_kib_to_mib_ceil(uint64_t kib) {
97 if (kib == 0) return 0u;
98 /* ceil(kib/1024) = (kib + 1023) >> 10 */
99 const uint64_t mib = (kib + 1023u) >> 10;
100 return (mib > 0xFFFFFFFFu) ? 0xFFFFFFFFu : (uint32_t)mib;
101}

135static int ie_parse_status_u64_line(const char *line, const char *key, uint64_t *out_val) {
136 if (!line || !key || !out_val) return 0;
137
138 const size_t klen = strlen(key);
139 if (strncmp(line, key, klen) != 0) return 0;
140
141 const char *p = line + klen;
142 while (*p == ' ' || *p == '\t') ++p;
143
144 uint64_t val = 0;
145 int matched = 0;
146 while (*p >= '0' && *p <= '9') {
147 matched = 1;
148 val = (val * 10u) + (uint64_t)(*p - '0');
149 ++p;
150 }
151 if (!matched) return 0;
152
153 *out_val = val;
154 return 1;
155}

169static int ie_read_proc_u64_single(const char *path, const char *key, uint64_t *out_val) {
170#if defined(__linux__)
171 FILE *f = fopen(path, "r");
172 if (!f) return (errno ? -errno : -1);
173
174 char buf[512];
175 while (fgets(buf, (int)sizeof buf, f)) {
176 uint64_t v = 0;
177 if (ie_parse_status_u64_line(buf, key, &v)) {
178 fclose(f);
179 *out_val = v;
180 return 0;
181 }
182 }
183
184 fclose(f);
185 return -ENOENT;
186#else
187 (void)path;
188 (void)key;
189 (void)out_val;
190 return -ENOSYS;
191#endif
192}

219 if (!out) return;
220

222 const uint64_t miss = atomic_load_explicit(&g_kv_miss, memory_order_relaxed);

237uint32_t ie_metrics_sample_rss_peak(void) {
238 uint32_t best_mib = 0;
239
240#if defined(__linux__)
241 /* 1) Prefer VmHWM (KiB): process peak RSS. */
242 {
243 uint64_t kb = 0;
244 const int rc = ie_read_proc_u64_single("/proc/self/status", "VmHWM:", &kb);
245 if (rc == 0) {
246 const uint32_t mib = ie_kib_to_mib_ceil(kb);
247 if (mib > best_mib) best_mib = mib;
248 ie_dbg_rss("[RSS] VmHWM: %llu kB -> %u MiB\n", (unsigned long long)kb, mib);
249 } else {
250 ie_dbg_rss("[RSS] VmHWM not found (rc=%d)\n", rc);
251 }
252 }
253
254 /* 2) Fallback: VmRSS (KiB): current RSS. */
255 if (best_mib == 0) {
256 uint64_t kb = 0;
257 const int rc = ie_read_proc_u64_single("/proc/self/status", "VmRSS:", &kb);
258 if (rc == 0) {
259 const uint32_t mib = ie_kib_to_mib_ceil(kb);
260 if (mib > best_mib) best_mib = mib;
261 ie_dbg_rss("[RSS] VmRSS: %llu kB -> %u MiB\n", (unsigned long long)kb, mib);
262 } else {
263 ie_dbg_rss("[RSS] VmRSS not found (rc=%d)\n", rc);
264 }
265 }
266
267 /* 3) Try smaps_rollup Rss: (KiB). */
268 if (best_mib == 0) {
269 uint64_t kb = 0;
270 const int rc = ie_read_proc_u64_single("/proc/self/smaps_rollup", "Rss:", &kb);
271 if (rc == 0) {
272 const uint32_t mib = ie_kib_to_mib_ceil(kb);
273 if (mib > best_mib) best_mib = mib;
274 ie_dbg_rss("[RSS] smaps_rollup Rss: %llu kB -> %u MiB\n", (unsigned long long)kb, mib);
275 } else {
276 ie_dbg_rss("[RSS] smaps_rollup Rss not found (rc=%d)\n", rc);
277 }
278 }
279
280 /* 4) Final fallback: getrusage (Linux ru_maxrss is KiB). */
281 if (best_mib == 0) {
282 struct rusage ru;
283 if (getrusage(RUSAGE_SELF, &ru) == 0) {
284 const uint64_t kb = (uint64_t)ru.ru_maxrss;
285 const uint32_t mib = ie_kib_to_mib_ceil(kb);
286 if (mib > best_mib) best_mib = mib;
287 ie_dbg_rss("[RSS] getrusage ru_maxrss: %llu kB -> %u MiB\n",
288 (unsigned long long)kb, mib);
289 } else {
290 ie_dbg_rss("[RSS] getrusage failed: errno=%d\n", errno);
291 }
292 }
293
294#elif defined(__APPLE__)
295 /* macOS primary: mach_task_basic_info (bytes). */
296 {
297 mach_task_basic_info info;
298 mach_msg_type_number_t count = MACH_TASK_BASIC_INFO_COUNT;
299 const kern_return_t kr =
300 task_info(mach_task_self(), MACH_TASK_BASIC_INFO, (task_info_t)&info, &count);
301
302 if (kr == KERN_SUCCESS) {
303 uint64_t bytes = (uint64_t)info.resident_size_max;
304 if (bytes == 0) bytes = (uint64_t)info.resident_size;
305
306 const uint32_t mib = ie_bytes_to_mib_ceil(bytes);
307 if (mib > best_mib) best_mib = mib;
308
309 ie_dbg_rss("[RSS] mach_task_basic_info: %llu bytes -> %u MiB\n",
310 (unsigned long long)bytes, mib);
311 } else {
312 ie_dbg_rss("[RSS] mach task_info failed: kr=%d\n", (int)kr);
313 }
314 }
315
316 /* Fallback: getrusage (macOS ru_maxrss is bytes). */
317 if (best_mib == 0) {
318 struct rusage ru;
319 if (getrusage(RUSAGE_SELF, &ru) == 0) {
320 const uint64_t bytes = (uint64_t)ru.ru_maxrss;
321 const uint32_t mib = ie_bytes_to_mib_ceil(bytes);
322 if (mib > best_mib) best_mib = mib;
323
324 ie_dbg_rss("[RSS] getrusage ru_maxrss(bytes): %llu -> %u MiB\n",
325 (unsigned long long)bytes, mib);
326 }
327 }
328
329#else
330 /* Other OS: getrusage; interpret ru_maxrss as KiB (best-effort). */
331 {
332 struct rusage ru;
333 if (getrusage(RUSAGE_SELF, &ru) == 0) {
334 const uint64_t kb = (uint64_t)ru.ru_maxrss;
335 const uint32_t mib = ie_kib_to_mib_ceil(kb);
336 if (mib > best_mib) best_mib = mib;
337
338 ie_dbg_rss("[RSS] generic getrusage ru_maxrss: %llu kB -> %u MiB\n",
339 (unsigned long long)kb, mib);
340 }
341 }
342#endif
343
344 return best_mib;
345}
