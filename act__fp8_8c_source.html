36/* ---- generic encode/decode scaffolding ----
37 * We implement a simple normalized encode path:
38 * - take absolute float, handle zero and non-finite
39 * - extract exponent with frexpf: x = m * 2^e, m in [0.5, 1)
40 * - convert to target exponent/mantissa with rounding-to-nearest-even
41 * - clamp to range; optional Inf/NaN support for E5M2
42 */
43
44static inline uint8_t ie_pack_fp8_e4m3(float x) {
45 /* Finite-only E4M3 with bias 7. */
46 if (x == 0.0f) return 0u;
47 uint32_t u = ie_u32_from_f(x);
48 uint8_t sign = (u >> 31) & 0x1;
49
50 if (ie_is_nan_u32(u)) {
51 /* Treat NaN as max finite with payload dropped. */
52 const uint8_t payload = 0x1; /* minimal mantissa to signal "non-zero" */
53 return (uint8_t)((sign << 7) | (0xE << 3) | (payload & 0x7));
54 }
55 if (ie_is_inf_u32(u)) {
56 /* Saturate to max finite. */
57 return (uint8_t)((sign << 7) | (0xE << 3) | 0x7);
58 }
59
60 float ax = fabsf(x);
61 int e2;
62 float m = frexpf(ax, &e2); /* ax = m * 2^e2, m in [0.5,1) */
63
64 /* Convert to 1.xxx with mantissa in [1,2) by shifting: */
65 float m_shift = m * 2.0f; /* [1,2) */
66 int e = e2 - 1; /* adjust because of the shift */
67
68 const int bias = 7;
69 int exp_fp8 = e + bias;
70
71 /* Round mantissa to 3 bits on [1,2): mant = 1 + frac; q = round(frac * 2^3) */
72 float frac = m_shift - 1.0f;
73 float qf = nearbyintf(frac * 8.0f); /* 2^3 = 8 */
74 int qm = (int)qf;
75
76 if (qm == 8) { /* carry into exponent */
77 qm = 0;
78 exp_fp8 += 1;
79 }
80
81 if (exp_fp8 <= 0) {
82 /* Underflow to zero (we skip subnormals for simplicity/perf). */
83 return (uint8_t)(sign << 7);
84 }
85 if (exp_fp8 >= 0xE) {
86 /* 0xF would be reserved; clamp to max finite exponent (0xE) and mant=7 */
87 return (uint8_t)((sign << 7) | (0xE << 3) | 0x7);
88 }
89
90 return (uint8_t)((sign << 7) | ((uint8_t)exp_fp8 << 3) | ((uint8_t)qm & 0x7));
91}

93static inline float ie_unpack_fp8_e4m3(uint8_t v) {
94 if (v == 0u) return 0.0f;
95 uint8_t sign = (v >> 7) & 0x1;
96 uint8_t exp = (v >> 3) & 0xF;
97 uint8_t man = (v & 0x7);
98
99 if (exp == 0) {
100 /* We do not encode subnormals; interpret as +/- 0. */
101 return sign ? -0.0f : 0.0f;
102 }
103 /* Finite-only: exp==0xF unused; exp==0xE is largest finite. */
104 const int bias = 7;
105 int e = ((int)exp) - bias;
106
107 float frac = (float)man / 8.0f; /* 2^3 */
108 float val = (1.0f + frac) * ldexpf(1.0f, e); /* (1 + frac) * 2^e */
109 return sign ? -val : val;
110}

112static inline uint8_t ie_pack_fp8_e5m2(float x) {
113 /* E5M2 with bias 15; supports Inf/NaN. */
114 uint32_t u = ie_u32_from_f(x);
115 uint8_t sign = (u >> 31) & 0x1;
116
117 if (x == 0.0f) return (uint8_t)(sign << 7);
118 if (ie_is_nan_u32(u)) {
119 return (uint8_t)((sign << 7) | (0x1F << 2) | 0x1); /* quiet NaN surrogate */
120 }
121 if (ie_is_inf_u32(u)) {
122 return (uint8_t)((sign << 7) | (0x1F << 2) | 0x0); /* infinity */
123 }
124
125 float ax = fabsf(x);
126 int e2;
127 float m = frexpf(ax, &e2); /* ax = m * 2^e2, m in [0.5,1) */
128 float m_shift = m * 2.0f; /* [1,2) */
129 int e = e2 - 1;
130
131 const int bias = 15;
132 int exp_fp8 = e + bias;
133
134 /* 2 mantissa bits on [1,2): */
135 float frac = m_shift - 1.0f;
136 float qf = nearbyintf(frac * 4.0f); /* 2^2 = 4 */
137 int qm = (int)qf;
138
139 if (qm == 4) { /* carry */
140 qm = 0;
141 exp_fp8 += 1;
142 }
143
144 if (exp_fp8 <= 0) {
145 /* Underflow -> flush to zero (skip subnormals). */
146 return (uint8_t)(sign << 7);
147 }
148 if (exp_fp8 >= 0x1F) {
149 /* Overflow -> infinity. */
150 return (uint8_t)((sign << 7) | (0x1F << 2));
151 }
152
153 return (uint8_t)((sign << 7) | ((uint8_t)exp_fp8 << 2) | ((uint8_t)qm & 0x3));
154}

156static inline float ie_unpack_fp8_e5m2(uint8_t v) {
157 uint8_t sign = (v >> 7) & 0x1;
158 uint8_t exp = (v >> 2) & 0x1F;
159 uint8_t man = (v & 0x3);
160
161 if (exp == 0) {
162 /* We do not produce subnormals; interpret as +/- 0. */
163 return sign ? -0.0f : 0.0f;
164 }
165 if (exp == 0x1F) {
166 if (man == 0) return sign ? -INFINITY : INFINITY;
167 /* NaN: return a quiet NaN. */
168 volatile float qnan = NAN;
169 return qnan;
170 }
171
172 const int bias = 15;
173 int e = ((int)exp) - bias;
174
175 float frac = (float)man / 4.0f; /* 2^2 */
176 float val = (1.0f + frac) * ldexpf(1.0f, e);
177 return sign ? -val : val;
178}
