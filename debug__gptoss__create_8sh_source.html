29model_dir = os.environ.get("MODEL_DIR", "models/gpt-oss-20b")

34with open(ie_json, "r", encoding="utf-8") as f:
35 j = json.load(f)
36
37tensors = j.get("tensors", [])
38if not isinstance(tensors, list) or not tensors:
39 raise SystemExit("model.ie.json has no tensors[]")
40
41by = {}
42for t in tensors:
43 if isinstance(t, dict) and isinstance(t.get("name"), str):
44 by[t["name"]] = t
45
46def get_int_field(t, *keys):
47 for k in keys:
48 v = t.get(k)
49 if v is not None:
50 return int(v)
51 return None
52
53def get_shape(name):
54 t = by.get(name)
55 if not t:
56 return None
57 sh = t.get("shape")
58 if isinstance(sh, list) and all(isinstance(x, int) for x in sh) and len(sh) >= 1:
59 return sh
60 return None
61
62def pick_existing(*cands):
63 for c in cands:
64 if c in by:
65 return c
66 return ""

68# ---- tensor_map.json
69out = []
70for name, t in by.items():
71 off = get_int_field(t, "offset")
72 nb = get_int_field(t, "size_bytes", "nbytes", "byte_size")
73 if off is None or nb is None:
74 continue
75 out.append({"name": name, "offset": off, "size_bytes": nb})
76
77if not out:
78 raise SystemExit("No tensors had (name, offset, size_bytes).")

80with open(tmap_json, "w", encoding="utf-8") as f:

85# ---- core dims
86embed = pick_existing("model.embed_tokens.weight", "transformer.wte.weight", "tok_embeddings.weight")
87if not embed:
88 raise SystemExit("Could not find embed_tokens weight.")
89
90esh = get_shape(embed)

92 raise SystemExit(f"Embed tensor has no usable shape: {embed} shape={esh}")
93
94vocab_size, d_model = esh[0], esh[1]
95
96# layers count
97layer_ids = []
98pat = re.compile(r"^model\.layers\.(\d+)\.")
99for name in by.keys():
100 m = pat.match(name)
101 if m:
102 layer_ids.append(int(m.group(1)))
103n_layers = (max(layer_ids) + 1) if layer_ids else 0
104if n_layers <= 0:
105 raise SystemExit("Could not detect n_layers from tensor names.")
106
107# attention proj dims from shapes (preferred) else bytes fallback is unreliable across dtypes
108q_name = pick_existing(
109 "model.layers.0.self_attn.q_proj.weight",
110 "model.layers.0.self_attn.q_proj.qweight",
111 "model.layers.0.attn.q_proj.weight",
112)
113k_name = pick_existing(
114 "model.layers.0.self_attn.k_proj.weight",
115 "model.layers.0.self_attn.k_proj.qweight",
116 "model.layers.0.attn.k_proj.weight",
117)
118v_name = pick_existing(
119 "model.layers.0.self_attn.v_proj.weight",
120 "model.layers.0.self_attn.v_proj.qweight",
121 "model.layers.0.attn.v_proj.weight",
122)
123
124if not (q_name and k_name and v_name):
125 # If naming differs, print helpful candidates and fail.
126 cands = [n for n in by.keys() if n.startswith("model.layers.0") and "attn" in n and n.endswith("weight")]
127 cands = sorted(cands)[:50]
128 raise SystemExit("Could not find q/k/v proj weights. First 50 attn weights:\n" + "\n".join(cands))
129
130def out_dim_from_proj(name):
131 sh = get_shape(name)
132 if sh and len(sh) == 2:
133 a, b = sh
134 # one dim should match d_model (input features); the other is output
135 if a == d_model and b != d_model:
136 return b
137 if b == d_model and a != d_model:
138 return a
139 # if neither matches, still return first dim as a fallback
140 return sh[0]
141 # bytes fallback (very shaky). Fail instead of guessing wrong.
142 raise SystemExit(f"Proj tensor missing shape; cannot infer dims safely: {name}")
143
144q_dim = out_dim_from_proj(q_name)
145kv_dim = out_dim_from_proj(k_name)
146kv_dim2= out_dim_from_proj(v_name)
147if kv_dim2 != kv_dim:
148 print(f"warn: k_proj out_dim={kv_dim} v_proj out_dim={kv_dim2} (using k_proj)")
149
150# head geometry: choose a head_dim that divides both q_dim and kv_dim, prefer 64 then 128 then 32, else max <= 256.
151prefs = [64, 128, 32, 80, 96, 48, 40, 24, 16]
152head_dim = None
153for hd in prefs:
154 if q_dim % hd == 0 and kv_dim % hd == 0:
155 head_dim = hd
156 break
157if head_dim is None:
158 # fallback: largest common divisor <= 256
159 for hd in range(256, 7, -1):
160 if q_dim % hd == 0 and kv_dim % hd == 0:
161 head_dim = hd
162 break
163if head_dim is None:
164 raise SystemExit(f"Could not find a consistent head_dim for q_dim={q_dim} kv_dim={kv_dim}")
165
166n_heads = q_dim // head_dim
167n_kv_heads = kv_dim // head_dim
168
169# d_ff from MLP shapes
170mlp_candidates = [
171 "model.layers.0.mlp.gate_proj.weight",
172 "model.layers.0.mlp.up_proj.weight",
173 "model.layers.0.mlp.fc1.weight",
174 "model.layers.0.mlp.w1.weight",
175 "model.layers.0.mlp.gate.weight",
176]
177mlp_found = ""
178for nm in mlp_candidates:
179 if nm in by:
180 mlp_found = nm
181 break
182if not mlp_found:
183 # try fuzzy
184 for nm in by.keys():
185 if nm.startswith("model.layers.0") and ".mlp." in nm and nm.endswith(".weight"):
186 if any(k in nm for k in ("gate", "up", "fc1", "w1")):
187 mlp_found = nm
188 break
189if not mlp_found:
190 cands = [n for n in by.keys() if n.startswith("model.layers.0") and ".mlp." in n and n.endswith(".weight")]
191 cands = sorted(cands)[:80]
192 raise SystemExit("Could not find an MLP proj weight to infer d_ff. First 80 mlp weights:\n" + "\n".join(cands))
193
194mlp_sh = get_shape(mlp_found)
195if not mlp_sh or len(mlp_sh) != 2:
196 raise SystemExit(f"MLP tensor has no usable shape: {mlp_found} shape={mlp_sh}")
197
198a, b = mlp_sh
199if a == d_model and b != d_model:
200 d_ff = b
201elif b == d_model and a != d_model:
202 d_ff = a
203else:
204 d_ff = max(a, b)
205
206# experts count (best-effort)
207n_experts = None
208for nm in by.keys():
209 if nm.startswith("model.layers.0") and ("experts" in nm or "moe" in nm):
210 # very rough: look for "experts.<id>."
211 m = re.search(r"\.experts\.(\d+)\.", nm)
212 if m:
213 n_experts = max(n_experts or 0, int(m.group(1)) + 1)
214
215hp = {
216 "d_model": int(d_model),
217 "vocab_size": int(vocab_size),
218 "n_layers": int(n_layers),
219 "n_heads": int(n_heads),
220 "n_kv_heads": int(n_kv_heads),
221 "d_head": int(head_dim),
222 "d_ff": int(d_ff),
223 "q_dim": int(q_dim),
224 "kv_dim": int(kv_dim),
225}
226if n_experts is not None:
227 hp["n_experts_layer0"] = int(n_experts)
228
229with open(hp_json, "w", encoding="utf-8") as f:
230 json.dump(hp, f, indent=2)
