28static IE_TARGET_AVX2 inline float ie_hsum256_ps(__m256 v) {
29 __m128 lo = _mm256_castps256_ps128(v);
30 __m128 hi = _mm256_extractf128_ps(v, 1);
31 __m128 sum = _mm_add_ps(lo, hi);
32
33 __m128 shuf = _mm_movehdup_ps(sum);
34 sum = _mm_add_ps(sum, shuf);
35 shuf = _mm_movehl_ps(shuf, sum);
36 sum = _mm_add_ss(sum, shuf);
37
38 return _mm_cvtss_f32(sum);
39}

59void ie_vec_bf16_to_f32_avx2_impl(const uint16_t *in, float *out, size_t n) {
60 if (!in || !out || n == 0u) return;
61
62 size_t i = 0;
63 for (; i + 8u <= n; i += 8u) {
64 __m128i v16 = _mm_loadu_si128((const __m128i *)(const void *)(in + i));
65 __m256 vf = ie_bf16x8_to_f32(v16);
66 _mm256_storeu_ps(out + i, vf);
67 }
68
69 for (; i < n; ++i) {
70 union { uint32_t u; float f; } v;
71 v.u = ((uint32_t)in[i]) << 16;
72 out[i] = v.f;
73 }
74}

77int ie_gemv_bf16_f32_avx2_impl(const uint16_t *W_bf16, const float *x, float *y,
78 size_t rows, size_t cols,
79 const uint16_t *bias_bf16) {
80 if (!W_bf16 || !x || !y || rows == 0u || cols == 0u) return -1;
81
82 const size_t blk_k = 256u;
83
84 for (size_t r = 0; r < rows; ++r) {
85 const uint16_t *row = W_bf16 + r * cols;
86
87 __m256 acc = _mm256_setzero_ps();
88 float tail_sum = 0.0f;
89
90 for (size_t k0 = 0; k0 < cols; k0 += blk_k) {
91 const size_t kend = (k0 + blk_k <= cols) ? (k0 + blk_k) : cols;
92 size_t c = k0;
93
94 for (; c + 16u <= kend; c += 16u) {
95 _mm_prefetch((const char *)(row + c + 64u), _MM_HINT_T0);
96 _mm_prefetch((const char *)(x + c + 64u), _MM_HINT_T0);
97
98 __m128i w16_0 = _mm_loadu_si128((const __m128i *)(const void *)(row + c));
99 __m128i w16_1 = _mm_loadu_si128((const __m128i *)(const void *)(row + c + 8u));
100 __m256 w0 = ie_bf16x8_to_f32(w16_0);
101 __m256 w1 = ie_bf16x8_to_f32(w16_1);
102 __m256 x0 = _mm256_loadu_ps(x + c);
103 __m256 x1 = _mm256_loadu_ps(x + c + 8u);
104 acc = _mm256_fmadd_ps(w0, x0, acc);
105 acc = _mm256_fmadd_ps(w1, x1, acc);
106 }
107
108 for (; c + 8u <= kend; c += 8u) {
109 _mm_prefetch((const char *)(row + c + 64u), _MM_HINT_T0);
110 _mm_prefetch((const char *)(x + c + 64u), _MM_HINT_T0);
111 __m128i w16 = _mm_loadu_si128((const __m128i *)(const void *)(row + c));
112 __m256 w = ie_bf16x8_to_f32(w16);
113 __m256 xv = _mm256_loadu_ps(x + c);
114 acc = _mm256_fmadd_ps(w, xv, acc);
115 }
116
117 for (; c < kend; ++c) {
118 union { uint32_t u; float f; } v;
119 v.u = ((uint32_t)row[c]) << 16;
120 tail_sum += v.f * x[c];
121 }
122 }
123
124 float sum = ie_hsum256_ps(acc) + tail_sum;
125
126 if (bias_bf16) {
127 union { uint32_t u; float f; } b;
128 b.u = ((uint32_t)bias_bf16[r]) << 16;
129 sum += b.f;
130 }
131
132 y[r] = sum;
133 }
134
135 return 0;
136}

139void ie_gemv_f32_avx2_impl(const float *W, const float *x, float *y,
140 size_t rows, size_t cols,
141 const float *bias, size_t bias_stride) {
142 for (size_t r = 0; r < rows; r++) {
143 const float *row = W + r * cols;
144
145 __m256 acc = _mm256_setzero_ps();
146
147 size_t c = 0;
148 for (; c + 8 <= cols; c += 8) {
149 __m256 wv = _mm256_loadu_ps(row + c);
150 __m256 xv = _mm256_loadu_ps(x + c);
151 acc = _mm256_fmadd_ps(wv, xv, acc);
152 }
153
154 float sum = ie_hsum256_ps(acc);
155
156 for (; c < cols; c++) {
157 sum += row[c] * x[c];
158 }
159
160 if (bias) {
161 sum += bias[r * bias_stride];
162 }
163
164 y[r] = sum;
165 }
