92__global__ void ie_gemv_rowwise_f32_kernel(const float *W,
93 const float *x,
94 float *y,
95 const float *bias,
96 size_t rows,
97 size_t cols) {
98 const size_t r = (size_t)blockIdx.x * (size_t)blockDim.x + (size_t)threadIdx.x;
99 if (r >= rows) return;
100
101 const float *Wr = W + r * cols;
102 float acc = 0.0f;
103
104 for (size_t c = 0; c < cols; ++c) {
105 acc += Wr[c] * x[c];
106 }
107
108 if (bias) acc += bias[r];
109 y[r] = acc;
110}
