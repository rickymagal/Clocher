52 __global__ void gemv_rowwise_f32_kernel(const float* __restrict__ W,
53  const float* __restrict__ x,
54  float* __restrict__ y,
55  int rows,
56  int cols,
57  int ldw,
58  size_t row_stride_e)
59 {
60  const int r = blockIdx.x; // row handled by this block
61  if (r >= rows) return;
62 
63  const float* row = W + r * row_stride_e;
64 
65  // Each thread accumulates partial over K with grid-stride loop on K.
66  float sum = 0.0f;
67  for (int k = threadIdx.x; k < cols; k += blockDim.x) {
68  sum += row[k] * x[k];
69  }
70 
71  // Block-wide reduction in shared memory
72  __shared__ float smem[256]; // assumes blockDim.x <= 256
73  smem[threadIdx.x] = sum;
74  __syncthreads();
75 
76  // Reduce to lane 0
77  for (int offset = blockDim.x / 2; offset > 0; offset >>= 1) {
78  if (threadIdx.x < offset) {
79  smem[threadIdx.x] += smem[threadIdx.x + offset];
80  }
81  __syncthreads();
82  }
83 
84  if (threadIdx.x == 0) {
85  y[r] = smem[0];
86  }
87 }
