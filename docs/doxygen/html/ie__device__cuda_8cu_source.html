93__global__ void ie_gemv_rowwise_f32_kernel(const float *W,
94 const float *x,
95 float *y,
96 const float *bias,
97 size_t rows,
98 size_t cols) {
99 const size_t r = (size_t)blockIdx.x * (size_t)blockDim.x + (size_t)threadIdx.x;
100 if (r >= rows) return;
101
102 const float *Wr = W + r * cols;
103 float acc = 0.0f;
104
105 for (size_t c = 0; c < cols; ++c) {
106 acc += Wr[c] * x[c];
107 }
108
109 if (bias) acc += bias[r];
110 y[r] = acc;
111}

142__global__ void ie_gemv_q4_0_f32_kernel(const uint8_t *W_blocks,
143 const uint8_t *W_scales,
144 size_t scale_bytes,
145 const float *x,
146 float *y,
147 size_t rows,
148 size_t cols,
149 const uint16_t *bias_bf16) {
150 const size_t r = (size_t)blockIdx.x * (size_t)blockDim.x + (size_t)threadIdx.x;
151 if (r >= rows) return;
152
153 if (scale_bytes != 1u && scale_bytes != 2u) {
154 return;
155 }
156
157 const size_t blocks_per_row = (cols + 31u) / 32u;
158 const size_t row_w_bytes = blocks_per_row * 16u;
159 const size_t row_s_bytes = blocks_per_row * (size_t)scale_bytes;
160
161 const uint8_t *wrow = W_blocks + r * row_w_bytes;
162 const uint8_t *srow = W_scales + r * row_s_bytes;
163
164 float acc = 0.0f;
165
166 for (size_t b = 0; b < blocks_per_row; ++b) {
167 float s = 0.0f;
168 if (scale_bytes == 2u) {
169 uint16_t s16 = 0;
170 const uint8_t *sp = srow + b * 2u;
171 s16 = (uint16_t)sp[0] | ((uint16_t)sp[1] << 8);
172 s = ie_bf16_to_f32_u16_dev(s16);
173 } else {
174 const uint8_t s8 = srow[b];
175 s = ie_log2_u8_q3_to_f32_dev(s8);
176 }
177
178 const uint8_t *blk = wrow + b * 16u;
179 const size_t base_c = b * 32u;
180 const size_t limit_c = (base_c + 32u <= cols) ? (base_c + 32u) : cols;
181
182 size_t c = base_c;
183 for (size_t i = 0; i < 16u && c < limit_c; ++i) {
184 const uint8_t byte = blk[i];
185
186 const int8_t w0 = ie_s4_from_u4_dev((uint8_t)(byte & 0x0F));
187 acc += (float)w0 * s * x[c];
188 ++c;
189 if (c >= limit_c) break;
190
191 const int8_t w1 = ie_s4_from_u4_dev((uint8_t)(byte >> 4));
192 acc += (float)w1 * s * x[c];
193 ++c;
194 }
195 }
196
197 if (bias_bf16) {
198 acc += ie_bf16_to_f32_u16_dev(bias_bf16[r]);
199 }
200 y[r] = acc;
201}

236extern "C" int ie_cuda_init(int device_ordinal,
237 char *out_name, size_t name_cap,
238 char *out_driver, size_t driver_cap) {
239 cudaError_t st = cudaSetDevice(device_ordinal);
240 if (st != cudaSuccess) {
241 ie_cuda_set_last_error_cuda("cudaSetDevice", st);
242 return -1;
243 }
244
245 st = cudaFree(0);
246 if (st != cudaSuccess) {
247 ie_cuda_set_last_error_cuda("cudaFree(0)", st);
248 return -2;
249 }
250
251 cudaDeviceProp prop;
252 std::memset(&prop, 0, sizeof(prop));
253 st = cudaGetDeviceProperties(&prop, device_ordinal);
254 if (st != cudaSuccess) {
255 ie_cuda_set_last_error_cuda("cudaGetDeviceProperties", st);
256 return -3;
257 }
258
259 if (out_name && name_cap) {
260 std::snprintf(out_name, name_cap, "%s", prop.name);
261 }
262
263 int drv = 0, rt = 0;
264 (void)cudaDriverGetVersion(&drv);
265 (void)cudaRuntimeGetVersion(&rt);
266
267 if (out_driver && driver_cap) {
268 std::snprintf(out_driver, driver_cap, "driver=%d runtime=%d", drv, rt);
269 }
270
271 ie_cuda_clear_last_error();
272 return 0;
273}

316extern "C" int ie_cuda_memcpy(void *dst, const void *src, size_t nbytes, ie_cuda_copy_kind_t kind) {
317 cudaMemcpyKind ck = cudaMemcpyDefault;
318 switch (kind) {
319 case IE_CUDA_COPY_H2D: ck = cudaMemcpyHostToDevice; break;
320 case IE_CUDA_COPY_D2H: ck = cudaMemcpyDeviceToHost; break;
321 case IE_CUDA_COPY_D2D: ck = cudaMemcpyDeviceToDevice; break;
322 case IE_CUDA_COPY_DEFAULT: default: ck = cudaMemcpyDefault; break;
323 }
324
325 const cudaError_t st = cudaMemcpy(dst, src, nbytes, ck);
326 if (st != cudaSuccess) {
327 ie_cuda_set_last_error_cuda("cudaMemcpy", st);

329 }
