19def read_safetensors_header(f):
20 hdr_len_bytes = f.read(8)
21 if len(hdr_len_bytes) != 8:
22 raise RuntimeError("Failed to read safetensors header length")
23 (hdr_len,) = struct.unpack("<Q", hdr_len_bytes)
24 hdr_json = f.read(hdr_len)
25 if len(hdr_json) != hdr_len:
26 raise RuntimeError("Failed to read safetensors header JSON")
27 header = json.loads(hdr_json.decode("utf-8"))
28 return header, 8 + hdr_len

31def bf16_to_f32_from_u16(u16: np.ndarray) -> np.ndarray:
32 u32 = (u16.astype(np.uint32) << 16)

36def f32_to_bf16_u16(x: np.ndarray) -> np.ndarray:

66 def get_meta(self, tensor_name: str):
67 shard_name = self.weight_map.get(tensor_name)
68 if shard_name is None:
69 raise KeyError(tensor_name)
70 header, data_base = self._load_shard_header(shard_name)
71 meta = header.get(tensor_name)
72 if meta is None:
73 raise SystemExit(f"ERROR: tensor '{tensor_name}' not found in header of {shard_name}")
74 dtype = meta.get("dtype")
75 shape = meta.get("shape")
76 offs = meta.get("data_offsets")
77 if dtype is None or shape is None or offs is None:
78 raise SystemExit(f"ERROR: malformed meta for '{tensor_name}' in {shard_name}")
79 start, end = int(offs[0]), int(offs[1])
80 nbytes = end - start
81 if nbytes <= 0:
82 raise SystemExit(f"ERROR: bad data_offsets for '{tensor_name}': {offs}")
83 return shard_name, data_base, dtype, [int(x) for x in shape], start, nbytes

85 def read_tensor_f32(self, tensor_name: str) -> np.ndarray:
86 shard_name, data_base, dtype, shape, start, nbytes = self.get_meta(tensor_name)
87 shard_path = self.hf_dir / shard_name
88 with open(shard_path, "rb") as f:
89 f.seek(data_base + start, os.SEEK_SET)
90 raw = f.read(nbytes)
91 if len(raw) != nbytes:
92 raise SystemExit(f"ERROR: unexpected EOF reading '{tensor_name}' from {shard_name}")
93
94 dt = str(dtype).upper()
95 if dt == "BF16":
96 u16 = np.frombuffer(raw, dtype=np.uint16)
97 f32 = bf16_to_f32_from_u16(u16)
98 elif dt in ("F16", "FLOAT16"):
99 f16 = np.frombuffer(raw, dtype=np.float16)
100 f32 = f16.astype(np.float32)
101 elif dt in ("F32", "FLOAT32"):
102 f32 = np.frombuffer(raw, dtype=np.float32)
103 else:
104 raise SystemExit(f"ERROR: unsupported dtype '{dtype}' for '{tensor_name}'")
105
106 if np.prod(shape, dtype=np.int64) * 4 != f32.nbytes:
107 raise SystemExit(f"ERROR: size mismatch for '{tensor_name}': shape={shape} dtype={dtype} bytes={nbytes}")
108
109 return f32.reshape(shape)

126def quantize_q4_0_blocks(mat_f32: np.ndarray, block_k: int = BLOCK_K):
127 mat = np.asarray(mat_f32, dtype=np.float32)
128 if mat.ndim != 2:
129 raise SystemExit(f"ERROR: quantize expects 2D, got {mat.ndim}D")
130 rows, cols = int(mat.shape[0]), int(mat.shape[1])
131 if cols % block_k != 0:
132 raise SystemExit(f"ERROR: cols={cols} not divisible by block_k={block_k}")
133 nb = cols // block_k
134
135 scales_u16 = np.empty((rows, nb), dtype=np.uint16)
136 blocks_u8 = np.empty((rows, nb, block_k // 2), dtype=np.uint8)
137
138 for r in range(rows):
139 row = mat[r]
140 for b in range(nb):
141 x = row[b * block_k : (b + 1) * block_k]
142 absmax = float(np.max(np.abs(x))) if x.size else 0.0
143 if absmax == 0.0:
144 scale = np.float32(1.0)
145 q = np.zeros((block_k,), dtype=np.int32)
146 else:
147 scale = np.float32(absmax / 7.0)
148 q = np.rint(x / scale).astype(np.int32)
149 q = np.clip(q, -7, 7)
150
151 scales_u16[r, b] = f32_to_bf16_u16(scale)[()]
152 q_u = (q + 8).astype(np.uint8)
153 lo = q_u[0::2]
154 hi = q_u[1::2] << 4
155 blocks_u8[r, b, :] = (lo | hi)
156
157 return blocks_u8, scales_u16

160def infer_layers(weight_map: dict) -> int:
161 max_layer = -1
162 prefix = "model.layers."
163 for k in weight_map.keys():
164 if not k.startswith(prefix):
165 continue
166 rest = k[len(prefix):]
167 parts = rest.split(".", 1)
168 if not parts:
169 continue
170 try:
171 lid = int(parts[0])
172 except ValueError:
173 continue
174 if lid > max_layer:
175 max_layer = lid
176 if max_layer < 0:
177 raise SystemExit("ERROR: could not infer layer count from safetensors index")
178 return max_layer + 1

231def append_tensor(bin_f, data: bytes, align: int) -> tuple[int, int]:
232 off = bin_f.tell()
233 aligned = align_up(off, align)
234 if aligned > off:
235 bin_f.write(b"\x00" * (aligned - off))
236 bin_f.write(data)
