66static uint8_t* read_file_all(const char* path, size_t* out_n) {
67 FILE* f = fopen(path, "rb");
68 if (!f) return NULL;
69 if (fseek(f, 0, SEEK_END) != 0) { fclose(f); return NULL; }
70 long sz = ftell(f);
71 if (sz < 0) { fclose(f); return NULL; }
72 if (fseek(f, 0, SEEK_SET) != 0) { fclose(f); return NULL; }
73
74 uint8_t* buf = (uint8_t*)malloc((size_t)sz + 1u);
75 if (!buf) { fclose(f); return NULL; }
76 if (sz > 0) {
77 if (fread(buf, 1, (size_t)sz, f) != (size_t)sz) { free(buf); fclose(f); return NULL; }
78 }
79 fclose(f);
80 buf[(size_t)sz] = 0;
81 if (out_n) *out_n = (size_t)sz;
82 return buf;
83}

157static char* json_find_string(const char* s, const char* key) {
158 char pat[128];
159 snprintf(pat, sizeof(pat), "\"%s\"", key);
160 const char* p = strstr(s, pat);
161 if (!p) return NULL;
162 p = strchr(p, ':');
163 if (!p) return NULL;
164 p++;
165 while (*p == ' ' || *p == '\t' || *p == '\n' || *p == '\r') p++;
166 if (*p != '"') return NULL;
167 p++;
168 const char* q = strchr(p, '"');
169 if (!q) return NULL;
170 size_t n = (size_t)(q - p);
171 char* out = (char*)malloc(n + 1u);
172 if (!out) return NULL;
173 memcpy(out, p, n);
174 out[n] = 0;
175 return out;
176}

227static int load_manifest(const char* path, blob_list_t* out) {
228 size_t n = 0;
229 uint8_t* txt = read_file_all(path, &n);
230 if (!txt) return -1;
231
232 const char* s = (const char*)txt;
233 const char* p = s;
234 while ((p = strchr(p, '{')) != NULL) {
235 const char* q = strchr(p, '}');
236 if (!q) break;
237
238 size_t obj_len = (size_t)(q - p + 1);
239 char* obj = (char*)malloc(obj_len + 1u);
240 if (!obj) { free(txt); return -1; }
241 memcpy(obj, p, obj_len);
242 obj[obj_len] = 0;
243
244 blob_ref_t b;
245 memset(&b, 0, sizeof(b));
246 b.name = json_find_string(obj, "name");
247 b.file = json_find_string(obj, "file");
248 if (!b.name || !b.file) { free(obj); continue; }
249 if (json_find_u64(obj, "offset", &b.offset) != 0) { free(obj); continue; }
250 if (json_find_u64(obj, "bytes", &b.bytes) != 0) { free(obj); continue; }
251
252 char* k = json_find_string(obj, "kind");
253 b.kind = kind_from_string(k);
254 free(k);
255
256 blobs_push(out, &b);
257 free(obj);
258 p = q + 1;
259 }
260
261 free(txt);
262 return (out->n > 0) ? 0 : -1;
263}

331static void normalize_name(const char* name, char* out_buf, size_t out_cap) {
332 /* Copy and apply simple rewrites. */
333 snprintf(out_buf, out_cap, "%s", name);
334
335 /* Remove obvious per-expert/per-rank suffixes if present. */
336 const char* patterns[] = {
337 ".expert_0", ".expert_1", ".expert_2", ".expert_3",
338 ".rank0", ".rank1", ".rank2", ".rank3",
339 NULL
340 };
341 for (int i = 0; patterns[i]; i++) {
342 char* p = strstr(out_buf, patterns[i]);
343 if (p) {
344 memmove(p, p + strlen(patterns[i]), strlen(p + strlen(patterns[i])) + 1u);
345 }
346 }
347
348 /* Collapse any accidental double dots. */
349 for (;;) {
350 char* dd = strstr(out_buf, "..");
351 if (!dd) break;
352 memmove(dd, dd + 1, strlen(dd));
353 }
354}

367static void build_groups_auto(const blob_list_t* blobs, group_list_t* out_groups) {
368 uint64_t next_gid = 1u;
369
370 char ni[1024];
371 char nj[1024];
372
373 int* assigned = (int*)calloc(blobs->n, sizeof(int));
374 if (!assigned) die("out of memory");
375
376 for (size_t i = 0; i < blobs->n; i++) {
377 if (assigned[i]) continue;
378
379 const blob_ref_t* bi = &blobs->v[i];
380 normalize_name(bi->name, ni, sizeof(ni));
381
382 group_t* g = groups_push(out_groups, next_gid++, i);
383 assigned[i] = 1;
384
385 for (size_t j = i + 1; j < blobs->n; j++) {
386 if (assigned[j]) continue;
387 const blob_ref_t* bj = &blobs->v[j];
388
389 if (bj->bytes != bi->bytes) continue;
390 if (bj->kind != bi->kind) continue;
391 if (strcmp(bj->file, bi->file) != 0) continue;
392
393 normalize_name(bj->name, nj, sizeof(nj));
394 if (strcmp(nj, ni) != 0) continue;
395
396 group_add_member(g, j);
397 assigned[j] = 1;
398 }
399 }
400
401 free(assigned);
402}

413static void mask_set(uint8_t* mask, uint64_t block_index, int value) {
414 const uint64_t byte_i = block_index >> 3;
415 const uint64_t bit_i = block_index & 7u;
416 const uint8_t bit = (uint8_t)(1u << (uint8_t)bit_i);
417 if (value) mask[byte_i] |= bit;
418 else mask[byte_i] &= (uint8_t)~bit;
419}

435static int extract_one(
436 FILE* f,
437 const blob_ref_t* def_ref,
438 const blob_ref_t* mem_ref,
439 uint64_t block_bytes,
440 FILE* exceptions_out,
441 FILE* masks_out,
442 uint64_t* exceptions_off_inout,
443 uint64_t* masks_off_inout,
444 dedup_spec_entry_t* entry_out
445) {
446 if (def_ref->bytes != mem_ref->bytes) return -1;
447 if (block_bytes == 0) return -1;
448 if ((def_ref->bytes % block_bytes) != 0) return -1;
449
450 const uint64_t block_count = def_ref->bytes / block_bytes;
451 const uint64_t mask_bytes = u64_ceil_div(block_count, 8u);
452
453 uint8_t* mask = (uint8_t*)calloc((size_t)mask_bytes, 1u);
454 if (!mask) return -1;
455
456 uint8_t* buf_def = (uint8_t*)malloc((size_t)block_bytes);
457 uint8_t* buf_mem = (uint8_t*)malloc((size_t)block_bytes);
458 if (!buf_def || !buf_mem) { free(mask); free(buf_def); free(buf_mem); return -1; }
459
460 uint64_t ex_blocks = 0;
461
462 /* Walk blocks; store only deltas. */
463 for (uint64_t i = 0; i < block_count; i++) {
464 const uint64_t off = i * block_bytes;
465
466 if (pread_exact(f, def_ref->offset + off, buf_def, (size_t)block_bytes) != 0) { free(mask); free(buf_def); free(buf_mem); return -1; }
467 if (pread_exact(f, mem_ref->offset + off, buf_mem, (size_t)block_bytes) != 0) { free(mask); free(buf_def); free(buf_mem); return -1; }
468
469 if (memcmp(buf_def, buf_mem, (size_t)block_bytes) != 0) {
470 mask_set(mask, i, 1);
471
472 /* Append this block to exceptions stream. */
473 if (fwrite_all(exceptions_out, buf_mem, (size_t)block_bytes) != 0) { free(mask); free(buf_def); free(buf_mem); return -1; }
474 ex_blocks++;
475 }
476 }
477
478 /* Append mask to masks stream. */
479 if (fwrite_all(masks_out, mask, (size_t)mask_bytes) != 0) { free(mask); free(buf_def); free(buf_mem); return -1; }
480
481 /* Fill entry. */
482 entry_out->payload_kind = (uint32_t)mem_ref->kind;
483 entry_out->original_bytes = mem_ref->bytes;
484 entry_out->block_bytes = block_bytes;
485 entry_out->block_count = block_count;
486
487 entry_out->defaults_off = 0; /* filled by caller (per-group) */
488 entry_out->exceptions_off = *exceptions_off_inout;
489 entry_out->exceptions_blocks = ex_blocks;
490
491 entry_out->mask_off = *masks_off_inout;
492 entry_out->mask_bytes = mask_bytes;
493
494 *exceptions_off_inout += ex_blocks * block_bytes;
495 *masks_off_inout += mask_bytes;
496
497 free(mask);
498 free(buf_def);
499 free(buf_mem);
500 return 0;
501}

512static args_t parse_args(int argc, char** argv) {

526 }

528 if (!a.manifest) die("missing --manifest");
529 if (!a.out_dir) die("missing --out-dir");
530 if (a.block_bytes == 0) die("--block-bytes must be > 0");

548int main(int argc, char** argv) {
549 args_t a = parse_args(argc, argv);
550
551 blob_list_t blobs;
552 memset(&blobs, 0, sizeof(blobs));
553 if (load_manifest(a.manifest, &blobs) != 0) die("failed to load manifest (expected array of objects with name/file/offset/bytes)");
554
555 group_list_t groups;
556 memset(&groups, 0, sizeof(groups));
557 build_groups_auto(&blobs, &groups);
558
559 /* Open output streams. */
560 char p_spec[4096], p_def[4096], p_ex[4096], p_mask[4096];
561 path_join(a.out_dir, "dedup.spec.bin", p_spec, sizeof(p_spec));
562 path_join(a.out_dir, "dedup.defaults.bin", p_def, sizeof(p_def));
563 path_join(a.out_dir, "dedup.exceptions.bin", p_ex, sizeof(p_ex));
564 path_join(a.out_dir, "dedup.masks.bin", p_mask, sizeof(p_mask));
565
566 FILE* f_def = fopen(p_def, "wb");
567 FILE* f_ex = fopen(p_ex, "wb");
568 FILE* f_msk = fopen(p_mask, "wb");
569 if (!f_def || !f_ex || !f_msk) die("failed to open output files (check --out-dir exists)");
570
571 uint64_t defaults_off = 0;
572 uint64_t exceptions_off = 0;
573 uint64_t masks_off = 0;
574
575 dedup_spec_builder_t* sb = dedup_spec_builder_create();
576 if (!sb) die("failed to allocate spec builder");
577
578 /* For simplicity in Step 1, we assume all blobs can be read via their own file handles. */
579 for (size_t gi = 0; gi < groups.n; gi++) {
580 group_t* g = &groups.v[gi];
581 const blob_ref_t* def_ref = &blobs.v[g->default_idx];
582
583 /* Write the default blob once per group into defaults.bin. */
584 FILE* f_in = open_rb(def_ref->file);
585 if (!f_in) die("failed to open input file for default blob");
586
587 uint8_t* tmp = (uint8_t*)malloc((size_t)def_ref->bytes);
588 if (!tmp) die("out of memory");
589 if (pread_exact(f_in, def_ref->offset, tmp, (size_t)def_ref->bytes) != 0) die("failed to read default blob bytes");
590 if (fwrite_all(f_def, tmp, (size_t)def_ref->bytes) != 0) die("failed to write defaults.bin");
591 free(tmp);
592
593 const uint64_t group_defaults_off = defaults_off;
594 defaults_off += def_ref->bytes;
595
596 /* Emit entries for each member. */
597 for (size_t mi = 0; mi < g->members_n; mi++) {
598 const size_t idx = g->members[mi];
599 const blob_ref_t* mem_ref = &blobs.v[idx];
600
601 /* All members must match the default's size to be in this group. */
602 if (mem_ref->bytes != def_ref->bytes) continue;
603
604 dedup_spec_entry_t e;
605 memset(&e, 0, sizeof(e));
606 e.group_id = g->group_id;
607
608 const uint64_t name_off = dedup_spec_builder_intern_string(sb, mem_ref->name);
609 if (!name_off) die("failed to intern name");
610 e.name_off = name_off;
611
612 e.defaults_off = group_defaults_off;
613
614 FILE* f_member = open_rb(mem_ref->file);
615 if (!f_member) die("failed to open input file for member blob");
616
617 /* Fast path: default itself => identical, store zero exceptions, zero mask. */
618 if (idx == g->default_idx) {
619 const uint64_t block_bytes = a.block_bytes;
620 if ((mem_ref->bytes % block_bytes) != 0) die("blob bytes must be divisible by --block-bytes");
621
622 const uint64_t block_count = mem_ref->bytes / block_bytes;
623 const uint64_t mask_bytes2 = u64_ceil_div(block_count, 8u);
624
625 uint8_t* zmask = (uint8_t*)calloc((size_t)mask_bytes2, 1u);
626 if (!zmask) die("out of memory");
627 if (fwrite_all(f_msk, zmask, (size_t)mask_bytes2) != 0) die("failed writing masks.bin");
628 free(zmask);
629
630 e.payload_kind = (uint32_t)mem_ref->kind;
631 e.original_bytes = mem_ref->bytes;
632 e.block_bytes = block_bytes;
633 e.block_count = block_count;
634
635 e.exceptions_off = exceptions_off;
636 e.exceptions_blocks = 0;
637
638 e.mask_off = masks_off;
639 e.mask_bytes = mask_bytes2;
640 masks_off += mask_bytes2;
641
642 if (dedup_spec_builder_add_entry(sb, &e) != 0) die("failed to add spec entry");
643
644 fclose(f_member);
645 continue;
646 }
647
648 if (extract_one(
649 f_member,
650 def_ref,
651 mem_ref,
652 a.block_bytes,
653 f_ex,
654 f_msk,
655 &exceptions_off,
656 &masks_off,
657 &e
658 ) != 0) {
659 die("extract_one failed (check manifest offsets/bytes and block sizing)");
660 }
661
662 if (dedup_spec_builder_add_entry(sb, &e) != 0) die("failed to add spec entry");
663 fclose(f_member);
664 }
665
666 fclose(f_in);
667 }
668
669 /* Finalize spec. */
670 if (dedup_spec_builder_write_file(sb, p_spec, defaults_off, exceptions_off, masks_off) != 0) {
671 die("failed to write dedup.spec.bin");
672 }
673
674 dedup_spec_builder_destroy(sb);
675 fclose(f_def);
676 fclose(f_ex);
677 fclose(f_msk);
678
679 groups_free(&groups);
680 blobs_free(&blobs);
681
682 fprintf(stderr, "[dedup_extract] wrote:\n");
683 fprintf(stderr, " %s\n %s\n %s\n %s\n", p_spec, p_def, p_ex, p_mask);
684 fprintf(stderr, "[dedup_extract] totals: defaults=%llu bytes, exceptions=%llu bytes, masks=%llu bytes\n",
685 (unsigned long long)defaults_off,
686 (unsigned long long)exceptions_off,
687 (unsigned long long)masks_off);
688
689 return 0;
690}
