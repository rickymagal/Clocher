16def read_safetensors_header(f):
17 hdr_len_bytes = f.read(8)
18 if len(hdr_len_bytes) != 8:
19 raise RuntimeError("Failed to read safetensors header length")
20 (hdr_len,) = struct.unpack("<Q", hdr_len_bytes)
21 hdr_json = f.read(hdr_len)
22 if len(hdr_json) != hdr_len:
23 raise RuntimeError("Failed to read safetensors header JSON")
24 header = json.loads(hdr_json.decode("utf-8"))
25 return header, 8 + hdr_len

60 def get_meta(self, tensor_name: str):
61 shard_name = self.weight_map.get(tensor_name)
62 if shard_name is None:
63 raise KeyError(tensor_name)
64 header, data_base = self._load_shard_header(shard_name)
65 meta = header.get(tensor_name)
66 if meta is None:
67 raise SystemExit(f"ERROR: tensor '{tensor_name}' not found in header of {shard_name}")
68 dtype = meta.get("dtype")
69 shape = meta.get("shape")
70 offs = meta.get("data_offsets")

79 def read_tensor_f32(self, tensor_name: str) -> np.ndarray:
80 shard_name, data_base, dtype, shape, start, nbytes = self.get_meta(tensor_name)
81 shard_path = self.hf_dir / shard_name
82 with open(shard_path, "rb") as f:
83 f.seek(data_base + start, os.SEEK_SET)
84 raw = f.read(nbytes)
85 if len(raw) != nbytes:
86 raise SystemExit(f"ERROR: unexpected EOF reading '{tensor_name}' from {shard_name}")
87
88 dt = str(dtype).upper()
89 if dt == "BF16":
90 u16 = np.frombuffer(raw, dtype=np.uint16)
91 f32 = bf16_to_f32_from_u16(u16)
92 elif dt in ("F16", "FLOAT16"):
93 f16 = np.frombuffer(raw, dtype=np.float16)
94 f32 = f16.astype(np.float32)
95 elif dt in ("F32", "FLOAT32"):
96 f32 = np.frombuffer(raw, dtype=np.float32)
97 else:
98 raise SystemExit(f"ERROR: unsupported dtype '{dtype}' for '{tensor_name}' (expected BF16/F16/F32)")
99
100 if np.prod(shape, dtype=np.int64) * 4 != f32.nbytes:
101 raise SystemExit(f"ERROR: size mismatch for '{tensor_name}': shape={shape} dtype={dtype} bytes={nbytes}")
102
103 return f32.reshape(shape)
