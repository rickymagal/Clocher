48 def make_virtual_slices(base: dict, expert_count: int) -> List[dict]:
49  # Returns a list of virtual tensor entries (index/name/dtype/shape/offset/nbytes) for each expert slice.
50  # We slice along dim0 (experts).
51  shape = list(base["shape"])
52  dtype = base["dtype"]
53  base_offset = int(base["offset"])
54  total_bytes = int(base["nbytes"])
55  esize = dtype_size(dtype)
56  # Elements per expert = total_elements / expert_count
57  total_elems = 1
58  for d in shape:
59  total_elems *= int(d)
60  if total_elems * esize != total_bytes:
61  raise SystemExit(f"IE tensor size mismatch for {base.get('name')}")
62 
63  if shape[0] != expert_count:
64  # Not an expert-fused tensor; no slices.
65  return []
66 
67  elems_per_expert = total_elems // expert_count
68  bytes_per_expert = elems_per_expert * esize
69 
70  out = []
71  # Slice shape drops the first dim
72  slice_shape = shape[1:]
73  for e in range(expert_count):
74  out.append({
75  "dtype": dtype,
76  "shape": slice_shape,
77  "offset": base_offset + e * bytes_per_expert,
78  "nbytes": bytes_per_expert,
79  # Name is filled by caller for context; index assigned later.
80  })
81  return out

119  nonlocal next_index
