1#!/usr/bin/env python3
2# =============================================================================
3# Grid-sweep benchmark harness + optional Prometheus Pushgateway publishing.
4#
5# This harness sweeps combinations of:
6# - act_quant : ["none","int8","fp8_e4m3","fp8_e5m2"]
7# - kv_quant : ["none","int8"]
8# - prefetch : ["off","on","auto"]
9# - nt_loads : [0,1]
10# - numa_policy : ["auto","compact","scatter"]
11#
12# It creates a per-combination subdirectory under --report-root, writes CSV and
13# summary.json snapshots, and (optionally) pushes a metrics snapshot per combo.
14#
15# NOTE: This is stdlib-only and does not invoke the engine binary by default.
16# You can point IE_ENGINE_BIN to an executable to integrate real runs.
17# For now, it emits synthetically shaped outputs for downstream tooling.
18# =============================================================================

45def _load_prompts_jsonl(path: Path) -> List[str]:
46 ps: List[str] = []
47 if not path.exists():
48 return ps
49 for line in path.read_text(encoding="utf-8").splitlines():
50 s = line.strip()
51 if not s or s.startswith("#"):
52 continue
53 try:
54 obj = json.loads(s)
55 t = obj.get("text")
56 if isinstance(t, str) and t:
57 ps.append(t)
58 except Exception:
59 # best-effort tolerant reader
60 pass
61 return ps

102 import urllib.parse

138def emit_metrics_text(summary: Dict[str, Any], labels: Dict[str, str]) -> str:
139 """Render a single snapshot suitable for Pushgateway."""
140 lines: List[str] = []
141
142 def put(name: str, val, l_extra: dict | None = None, mtype: str = "gauge"):
143 all_labels = dict(labels)
144 if l_extra:
145 all_labels.update(l_extra)
146 lines.append(f"# TYPE {name} {mtype}")
147 lines.append(f"{name}{_fmt_labels(all_labels)} {val}")
148
149 # Core
150 put("ie_tps", summary.get("tps_true", 0.0))
151 put("ie_latency_seconds", summary.get("wall_time_s", 0.0), {"stage": "e2e"})
152 put("ie_rss_bytes", int(float(summary.get("rss_peak_mb_max", 0.0)) * 1024 * 1024))
153 put("ie_tokens_total", int(summary.get("tokens_generated", 0)), mtype="counter")
154
155 # Spatial metrics
156 bpt = int(summary.get("ie_bytes_per_token") or 0)
157 bt = float(summary.get("bytes_touched") or 0.0)
158 bw = float(summary.get("effective_bandwidth_gbps") or 0.0)
159 put("ie_bytes_per_token", bpt)
160 put("ie_bytes_touched_total", bt)
161 put("ie_effective_bandwidth_gbps", bw)
162
163 # KV counts
164 put("ie_kv_hits_total", int(summary.get("kv_hits", 0)), mtype="counter")
165 put("ie_kv_misses_total", int(summary.get("kv_misses", 0)), mtype="counter")
166
167 # Config-as-labels snapshot
168 put("ie_config_info", 1, mtype="gauge")
169
170 return "\n".join(lines) + "\n"

175def _synthetic_evaluate(prompts: List[str]) -> Tuple[int, float, float, float]:
176 """Return (tokens_generated, wall_time_s, p50_ms, p95_ms)."""
177 # Keep simple and deterministic but non-zero:
178 n = max(1, len(prompts))
179 tokens = 64 * n
180 wall_s = max(0.050, 0.012 * n)
181 p50_ms = 6.0 + 0.25 * n
182 p95_ms = p50_ms * 2.0
183 return tokens, wall_s, p50_ms, p95_ms

186def _derive_bytes_touched(tokens: int, bpt: Optional[int], wall_s: float) -> Tuple[float, float]:
187 """Return (bytes_touched, effective_bandwidth_gbps)."""
188 if not bpt or tokens <= 0 or wall_s <= 0:
189 return 0.0, 0.0
190 bt = float(tokens) * float(bpt) # bytes
191 bw_gbps = (bt / 1_000_000_000.0) / float(wall_s) # GB/s
192 return bt, bw_gbps

259 )

408def _discover_git() -> Optional[str]:
409 try:
410 sha = subprocess.check_output(
411 ["git", "rev-parse", "--short", "HEAD"],
412 stderr=subprocess.DEVNULL,
413 ).decode().strip()
414 dirty = subprocess.call(
415 ["git", "diff", "--quiet", "--ignore-submodules"],
416 stdout=subprocess.DEVNULL,
417 stderr=subprocess.DEVNULL,
418 )
419 return f"{sha}{' DIRTY' if dirty != 0 else ''}"
420 except Exception:
421 return None
