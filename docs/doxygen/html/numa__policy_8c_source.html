51static int str_ieq(const char *a, const char *b) {
52 if (!a || !b) return 0;
53 while (*a && *b) {
54 if (tolower((unsigned char)*a) != tolower((unsigned char)*b)) return 0;
55 ++a;
56 ++b;
57 }
58 return (*a == '\0' && *b == '\0');

136int ie_numa_policy_init(ie_numa_policy_t *out, const ie_topology_t *topo) {

146 out->detected_nodes = ie_numa_detect_nodes();
147 if (out->detected_nodes <= 0) out->detected_nodes = 1;

162 (size_t)1099511627776ull);

165}
