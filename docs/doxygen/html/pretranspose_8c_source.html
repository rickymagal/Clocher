43static int load_cache(const char *path, size_t rows, size_t cols, float **data_out) {
44 if (!path) return -1;
45 FILE *f = fopen(path, "rb");
46 if (!f) return -1;
47 size_t count = rows * cols;
48 float *buf = (float*)malloc(count * sizeof(float));
49 if (!buf) { fclose(f); return -1; }
50 size_t got = fread(buf, sizeof(float), count, f);
51 fclose(f);
52 if (got != count) { free(buf); return -1; }
53 *data_out = buf;
54 return 0;
55}
