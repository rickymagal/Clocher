1/* File: engine/src/opt/pretranspose.c
2 * -----------------------------------------------------------------------------
3 * @file pretranspose.c
4 * @brief Build and cache blocked-K weight layout for GEMV.
5 *
6 * This module implements a light-weight re-packing of a row-major weight
7 * matrix W (rows x cols) into a per-row, contiguous "blocked-K" layout,
8 * suitable for cache-friendly inner loops. The packing is stable and
9 * reversible: each output row preserves the original order of columns, only
10 * grouped by contiguous blocks of size blk_k.
11 *
12 * The module also exposes a tiny on-disk cache keyed by a filename stem,
13 * a layout tag ("wxh", "woh", etc.), and the structural parameters
14 * (rows, cols, blk_k). The cache is optional and atomic on POSIX via
15 * rename(2).
16 *
17 * Thread-safety: all functions are re-entrant; no global state is modified.
18 * Ownership: the returned buffer in ::ie_layout_build_blockedK belongs to the
19 * caller and must be released with ::ie_layout_free (or free()).
20 */

76static int load_cache(const char *path, size_t rows, size_t cols, float **data_out) {
77 if (!path || !*path || !data_out) return -1;
78 FILE *f = fopen(path, "rb");
79 if (!f) return -1;
80
81 size_t count = rows * cols;
82 float *buf = (float *)malloc(count * sizeof(float));
83 if (!buf) {
84 fclose(f);
85 return -1;
86 }
87 size_t got = fread(buf, sizeof(float), count, f);
88 fclose(f);
89 if (got != count) {
90 free(buf);
91 return -1;
92 }
93 *data_out = buf;
94 return 0;
95}

109static int save_cache_atomic(const char *path, const float *data, size_t rows, size_t cols) {
110 if (!path || !*path || !data) return -1;
111
112 char tmp[4096];
113 int n = snprintf(tmp, sizeof(tmp), "%s.tmp", path);
114 if (n <= 0 || (size_t)n >= sizeof(tmp)) return -1;
115
116 FILE *f = fopen(tmp, "wb");
117 if (!f) return -1;
118
119 size_t count = rows * cols;
120 size_t put = fwrite(data, sizeof(float), count, f);
121 if (put != count) {
122 fclose(f);
123 (void)remove(tmp);
124 return -1;
125 }
126 if (fclose(f) != 0) {
127 (void)remove(tmp);
128 return -1;
129 }
130
131 if (rename(tmp, path) != 0) {
132 (void)remove(tmp);
133 return -1;
134 }
135 return 0;
136}

169int ie_layout_build_blockedK(const float *W,
170 size_t rows, size_t cols, size_t blk_k,
171 const char *cache_path,
172 ie_wblocked_desc_t *out) {
173 if (!W || !out || rows == 0 || cols == 0 || blk_k == 0) return -1;
174
175 float *buf = NULL;
176 if (cache_path && load_cache(cache_path, rows, cols, &buf) == 0) {
177 out->rows = rows;
178 out->cols = cols;
179 out->blk_k = blk_k;
180 out->data = buf;
181 return 0;
182 }
183
184 buf = (float *)malloc(rows * cols * sizeof(float));
185 if (!buf) return -1;
186
187 for (size_t r = 0; r < rows; ++r) {
188 const float *src_row = W + r * cols;
189 float *dst_row = buf + r * cols;
190 size_t written = 0;
191
192 for (size_t k0 = 0; k0 < cols; k0 += blk_k) {
193 const size_t klen = (k0 + blk_k <= cols) ? blk_k : (cols - k0);
194 memcpy(dst_row + written, src_row + k0, klen * sizeof(float));
195 written += klen;
196 }
197 }
198
199 if (cache_path) {
200 (void)save_cache_atomic(cache_path, buf, rows, cols);
201 }
202
203 out->rows = rows;
204 out->cols = cols;
205 out->blk_k = blk_k;
206 out->data = buf;
207 return 0;
208}
