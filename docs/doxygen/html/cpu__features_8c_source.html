111bool ie_cpu_features_detect(ie_cpu_features_t *out) {
112 if (!out) return false;
113
114 out->avx2 = false;
115 out->fma = false;
116
117 /* If your ie_cpu_features_t includes this field, we fill it. */
118#if defined(__GNUC__) || defined(_MSC_VER)
119 /* Some builds may not include avx512_bf16 in the struct; keep this guarded
120 * at the source level by only referencing the member name used in your ABI.
121 */
122 out->avx512_bf16 = false;
123#endif
124
125#if defined(__x86_64__) || defined(__i386__) || defined(_M_X64) || defined(_M_IX86)
126 unsigned a=0,b=0,c=0,d=0;
127
128 /* Leaf 0: max leaf availability */
129 ie_cpuid_ex(0, 0, &a, &b, &c, &d);
130 if (a == 0) {
131 return true; /* CPUID unsupported or unavailable */
132 }
133
134 /* Leaf 1: baseline features (AVX, OSXSAVE, FMA) */
135 ie_cpuid_ex(1, 0, &a, &b, &c, &d);
136 const int osxsave = (int)((c >> 27) & 1u);
137 const int avx = (int)((c >> 28) & 1u);
138 const int fma = (int)((c >> 12) & 1u);
139
140 /* Validate OS has enabled XMM/YMM via XGETBV */
141 unsigned xcr0 = osxsave ? ie_xgetbv_xcr0() : 0u;
142 const int ymm_ok = ((xcr0 & 0x6u) == 0x6u);
143
144 /* Leaf 7 subleaf 0: extended features (AVX2) */
145 ie_cpuid_ex(7, 0, &a, &b, &c, &d);
146 const int avx2 = (int)((b >> 5) & 1u);
147
148 out->avx2 = (avx && ymm_ok && avx2);
149 out->fma = (fma != 0);
150
151 /* Leaf 7 subleaf 1: AVX512_BF16 is in EAX bit 5 on supporting CPUs.
152 * We still store it even if the OS might not enable ZMM state; this flag is
153 * primarily used as a capability indicator for "do we even try BF16 kernels".
154 */
155 ie_cpuid_ex(7, 1, &a, &b, &c, &d);
156 {
157 const int avx512bf16 = (int)((a >> 5) & 1u);
158 out->avx512_bf16 = (avx512bf16 != 0);
159 }
160#endif
161
162 if (ie_env_flag("IE_DEBUG_KERNELS")) {
163 char buf[256];
164 (void)ie_cpu_features_to_string(out, buf, sizeof buf);
165 fprintf(stderr, "[cpu] %s\n", buf);
166 }
167
168 return true;
169}
