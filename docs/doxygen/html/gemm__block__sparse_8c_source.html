77void ie_gemv_block_sparse_f32(const ie_block_sparse_matrix_t *m,
78 const float *x,
79 float *y,
80 const float *bias) {
81 if (!m || !x || !y) {
82 return;
83 }
84
85 const uint32_t rows = m->rows;
86 const uint32_t cols = m->cols;
87 const uint32_t block_rows = m->block_rows;
88 const uint32_t block_cols = m->block_cols;
89 const uint32_t n_block_rows = m->n_block_rows;
90
91 if (rows == 0u || cols == 0u ||
92 block_rows == 0u || block_cols == 0u ||
93 n_block_rows == 0u) {
94 return;
95 }
96
97 /* For each block-row, process its constituent row stripes. */
98 for (uint32_t br = 0u; br < n_block_rows; ++br) {
99 const uint32_t row_start = br * block_rows;
100 const uint32_t row_end = row_start + block_rows;
101
102 /* Iterate within the block-row over actual rows (tail-safe). */
103 for (uint32_t local_r = 0u; local_r < block_rows; ++local_r) {
104 const uint32_t row = row_start + local_r;
105 if (row >= rows) {
106 break;
107 }
108
109 float acc = bias ? bias[row] : 0.0f;
110
111 const uint32_t blk_begin = m->row_ptr[br];
112 const uint32_t blk_end = m->row_ptr[br + 1u];
113
114 /* Accumulate contributions from all non-zero blocks in this block-row. */
115 for (uint32_t bi = blk_begin; bi < blk_end; ++bi) {
116 const uint32_t bc = m->col_idx[bi];
117 const uint32_t col_block_start = bc * block_cols;
118
119 const float *block = block_values_ptr(m, bi);
120 const float *block_row = block + (size_t)local_r * (size_t)block_cols;
121
122 /* Inner product of the local block row with the corresponding
123 * slice of x, with clipping for tail columns. */
124 for (uint32_t local_c = 0u; local_c < block_cols; ++local_c) {
125 const uint32_t col = col_block_start + local_c;
126 if (col >= cols) {
127 break;
128 }
129 acc += block_row[local_c] * x[col];
130 }
131 }
132
133 y[row] = acc;
134 }
135
136 (void)row_end; /* suppress unused-variable warning if not referenced */
137 }
