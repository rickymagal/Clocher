159static float *load_dense_matrix(const char *path,
160 uint32_t rows,
161 uint32_t cols) {
162 const size_t n = (size_t)rows * (size_t)cols;
163 const size_t bytes = n * sizeof(float);
164
165 FILE *f = fopen(path, "rb");
166 if (!f) {
167 fprintf(stderr, "failed to open input file '%s': %s\n",
168 path, strerror(errno));
169 return NULL;
170 }
171
172 float *buf = (float *)malloc(bytes);
173 if (!buf) {
174 fprintf(stderr, "out of memory allocating %zu bytes\n", bytes);
175 fclose(f);
176 return NULL;
177 }
178
179 size_t got = fread(buf, 1, bytes, f);
180 if (got != bytes) {
181 fprintf(stderr,
182 "short read: expected %zu bytes, got %zu bytes from '%s'\n",
183 bytes, got, path);
184 free(buf);
185 fclose(f);
186 return NULL;
187 }
188
189 fclose(f);
190 return buf;
191}

210static int block_is_nonzero(const float *W,
211 uint32_t rows,
212 uint32_t cols,
213 uint32_t br,
214 uint32_t bc,
215 uint32_t block_rows,
216 uint32_t block_cols,
217 float threshold) {
218 const uint32_t row_start = br * block_rows;
219 const uint32_t col_start = bc * block_cols;
220
221 for (uint32_t lr = 0; lr < block_rows; ++lr) {
222 uint32_t r = row_start + lr;
223 if (r >= rows) {
224 break;
225 }
226 const float *row = W + (size_t)r * (size_t)cols;
227 for (uint32_t lc = 0; lc < block_cols; ++lc) {
228 uint32_t c = col_start + lc;
229 if (c >= cols) {
230 break;
231 }
232 float v = row[c];
233 if (fabsf(v) >= threshold) {
234 return 1;
235 }
236 }
237 }
238 return 0;
239}

253static int dense_to_block_sparse(const float *W,
254 uint32_t rows,
255 uint32_t cols,
256 uint32_t block_rows,
257 uint32_t block_cols,
258 float threshold,
259 ie_block_sparse_matrix_t *out) {
260 if (!W || !out || rows == 0u || cols == 0u ||
261 block_rows == 0u || block_cols == 0u) {
262 return -1;
263 }
264
265 ie_block_sparse_init(out);
266
267 const uint32_t n_block_rows =
268 ie_block_sparse_n_block_rows(rows, block_rows);
269 const uint32_t n_block_cols =
270 ie_block_sparse_n_block_cols(cols, block_cols);
271
272 if (n_block_rows == 0u || n_block_cols == 0u) {
273 return -1;
274 }
275
276 /* First pass: detect which blocks are non-zero. */
277 const size_t total_blocks =
278 (size_t)n_block_rows * (size_t)n_block_cols;
279 uint8_t *nonzero = (uint8_t *)calloc(total_blocks, sizeof(uint8_t));
280 if (!nonzero) {
281 fprintf(stderr, "out of memory allocating nonzero flags\n");
282 return -1;
283 }
284
285 uint32_t nnzb = 0u;
286 for (uint32_t br = 0; br < n_block_rows; ++br) {
287 for (uint32_t bc = 0; bc < n_block_cols; ++bc) {
288 size_t idx = (size_t)br * (size_t)n_block_cols + (size_t)bc;
289 int nz = block_is_nonzero(W, rows, cols,
290 br, bc,
291 block_rows, block_cols,
292 threshold);
293 if (nz) {
294 nonzero[idx] = 1u;
295 ++nnzb;
296 }
297 }
298 }
299
300 if (nnzb == 0u) {
301 fprintf(stderr,
302 "warning: no non-zero blocks detected; result will be empty\n");
303 }
304
305 /* Allocate CSR-like arrays. */
306 uint32_t *row_ptr =
307 (uint32_t *)malloc((size_t)(n_block_rows + 1u) * sizeof(uint32_t));
308 uint32_t *col_idx =
309 (uint32_t *)malloc((size_t)nnzb * sizeof(uint32_t));
310 float *values = NULL;
311
312 if (!row_ptr || (!col_idx && nnzb > 0u)) {
313 fprintf(stderr, "out of memory allocating row_ptr/col_idx\n");
314 free(row_ptr);
315 free(col_idx);
316 free(nonzero);
317 return -1;
318 }
319
320 if (nnzb > 0u) {
321 const size_t block_elems =
322 (size_t)block_rows * (size_t)block_cols;
323 values = (float *)calloc((size_t)nnzb * block_elems, sizeof(float));
324 if (!values) {
325 fprintf(stderr, "out of memory allocating values\n");
326 free(row_ptr);
327 free(col_idx);
328 free(nonzero);
329 return -1;
330 }
331 }
332
333 /* Second pass: fill row_ptr, col_idx, values. */
334 uint32_t cursor = 0u;
335 for (uint32_t br = 0; br < n_block_rows; ++br) {
336 row_ptr[br] = cursor;
337 for (uint32_t bc = 0; bc < n_block_cols; ++bc) {
338 size_t idx = (size_t)br * (size_t)n_block_cols + (size_t)bc;
339 if (!nonzero[idx]) {
340 continue;
341 }
342 /* This block is non-zero: assign col_idx and copy data. */
343 const uint32_t blk_id = cursor;
344 col_idx[blk_id] = bc;
345
346 const uint32_t row_start = br * block_rows;
347 const uint32_t col_start = bc * block_cols;
348 float *blk = values + (size_t)blk_id *
349 (size_t)block_rows * (size_t)block_cols;
350
351 for (uint32_t lr = 0; lr < block_rows; ++lr) {
352 uint32_t r = row_start + lr;
353 if (r >= rows) {
354 break;
355 }
356 const float *row = W + (size_t)r * (size_t)cols;
357 for (uint32_t lc = 0; lc < block_cols; ++lc) {
358 uint32_t c = col_start + lc;
359 if (c >= cols) {
360 break;
361 }
362 blk[(size_t)lr * (size_t)block_cols + (size_t)lc] = row[c];
363 }
364 }
365
366 ++cursor;
367 }
368 }
369 row_ptr[n_block_rows] = cursor;
370
371 if (cursor != nnzb) {
372 fprintf(stderr,
373 "internal error: nnzb mismatch (expected %u, built %u)\n",
374 nnzb, cursor);
375 free(row_ptr);
376 free(col_idx);
377 free(values);
378 free(nonzero);
379 return -1;
380 }
381
382 free(nonzero);
383
384 /* Commit to descriptor. */
385 out->rows = rows;
386 out->cols = cols;
387 out->block_rows = block_rows;
388 out->block_cols = block_cols;
389 out->n_block_rows = n_block_rows;
390 out->n_block_cols = n_block_cols;
391 out->nnzb = nnzb;
392 out->row_ptr = row_ptr;
393 out->col_idx = col_idx;
394 out->values = values;
395
396 return 0;
397}

410static int write_block_sparse_file(const char *path,
411 const ie_block_sparse_matrix_t *m) {
412 if (!path || !m) {
413 return -1;
414 }
415
416 FILE *f = fopen(path, "wb");
417 if (!f) {
418 fprintf(stderr, "failed to open output file '%s': %s\n",
419 path, strerror(errno));
420 return -1;
421 }
422
423 ie_block_sparse_header_disk_t hdr;
424 memset(&hdr, 0, sizeof(hdr));
425 hdr.magic[0] = 'I';
426 hdr.magic[1] = 'E';
427 hdr.magic[2] = 'B';
428 hdr.magic[3] = 'S';
429 hdr.version = 1u;
430 hdr.rows = m->rows;
431 hdr.cols = m->cols;
432 hdr.block_rows = m->block_rows;
433 hdr.block_cols = m->block_cols;
434 hdr.nnzb = m->nnzb;
435 /* reserved[] left zeroed */
436
437 if (fwrite(&hdr, 1, sizeof(hdr), f) != sizeof(hdr)) {
438 fprintf(stderr, "failed to write header to '%s'\n", path);
439 fclose(f);
440 return -1;
441 }
442
443 const uint32_t n_block_rows = m->n_block_rows;
444 const uint32_t nnzb = m->nnzb;
445 const size_t block_elems =
446 (size_t)m->block_rows * (size_t)m->block_cols;
447 const size_t values_count = (size_t)nnzb * block_elems;
448
449 if (fwrite(m->row_ptr,
450 sizeof(uint32_t),
451 (size_t)(n_block_rows + 1u),
452 f) != (size_t)(n_block_rows + 1u)) {
453 fprintf(stderr, "failed to write row_ptr to '%s'\n", path);
454 fclose(f);
455 return -1;
456 }
457
458 if (nnzb > 0u) {
459 if (fwrite(m->col_idx,
460 sizeof(uint32_t),
461 (size_t)nnzb,
462 f) != (size_t)nnzb) {
463 fprintf(stderr, "failed to write col_idx to '%s'\n", path);
464 fclose(f);
465 return -1;
466 }
467
468 if (fwrite(m->values,
469 sizeof(float),
470 values_count,
471 f) != values_count) {
472 fprintf(stderr, "failed to write values to '%s'\n", path);
473 fclose(f);
474 return -1;
475 }
476 }
477
478 if (fclose(f) != 0) {
479 fprintf(stderr, "failed to close output file '%s': %s\n",
480 path, strerror(errno));
481 return -1;
482 }
483
484 return 0;
485}

491int main(int argc, char **argv) {
492 cli_opts_t opts;
493 if (parse_cli(argc, argv, &opts) != 0) {

498 fprintf(stderr,

500 " in = %s\n"
501 " out = %s\n"
502 " rows = %u\n"
503 " cols = %u\n"
504 " block_rows= %u\n"
505 " block_cols= %u\n"
506 " threshold = %g\n",
507 opts.in_path,
508 opts.out_path,
509 opts.rows,
510 opts.cols,
511 opts.block_rows,
512 opts.block_cols,
513 (double)opts.threshold);
514
515 float *dense = load_dense_matrix(opts.in_path,
516 opts.rows,
517 opts.cols);
518 if (!dense) {
519 return 1;
520 }
521
522 ie_block_sparse_matrix_t m;
523 if (dense_to_block_sparse(dense,
524 opts.rows,
525 opts.cols,
526 opts.block_rows,
527 opts.block_cols,
528 opts.threshold,
529 &m) != 0) {
530 fprintf(stderr, "failed to build block-sparse representation\n");
531 free(dense);
532 return 1;
533 }
534
535 free(dense);
536
537 const float density = ie_block_sparse_density(m.nnzb,
538 m.n_block_rows,
539 m.n_block_cols);
540 fprintf(stderr,
541 " nnzb = %u / %u blocks (density = %.6f)\n",
542 m.nnzb,
543 m.n_block_rows * m.n_block_cols,
544 (double)density);
545
546 if (write_block_sparse_file(opts.out_path, &m) != 0) {
547 fprintf(stderr, "failed to write output file\n");
548 ie_block_sparse_release(&m);
549 return 1;
550 }
551
552 ie_block_sparse_release(&m);
553 fprintf(stderr, "done.\n");
554 return 0;
