1#!/usr/bin/env python3
2"""
3dedup_extract_int4.py
4
5Lossless dedup extractor for byte-oriented tensors (including packed INT4 weights).
6
7This tool creates:
8 - model.defaults.bin
9 - model.masks.bin
10 - model.exceptions.bin
11 - model.dedup.json
12
13The core contract is byte-level and therefore dtype-agnostic:
14 - A tensor is reconstructed exactly as raw bytes.
15 - The mask has 1 bit per tensor byte.
16 - Exceptions store the exact original bytes at masked indices.
17
18Inputs:
19 1) A tensor map JSON describing where each tensor lives in the binary file.
20 Expected minimal schema:
21 {
22 "weights_bin": "model.ie.bin",
23 "tensors": [
24 { "index": 0, "name": "...", "offset": 123, "nbytes": 456 },
25 ...
26 ]
27 }
28
29 2) A groups JSON describing dedup groups:
30 [
31 {
32 "default_index": 17,
33 "targets": [18, 19, 20]
34 },
35 ...
36 ]
37
38Usage:
39 python3 tools/dedup_extract_int4.py \
40 --model-dir models/gpt-oss-20b \
41 --tensor-map models/gpt-oss-20b/model.tensor_map.json \
42 --groups tools/dedup_groups.json \
43 --out-prefix models/gpt-oss-20b/model
44
45Notes:
46 - This script intentionally does not parse safetensors or HF formats.
47 - It expects a simple (offset,nbytes) map to an existing binary file.
48"""

80def _ceil_div(a: int, b: int) -> int:
81 """Compute ceil(a/b) for non-negative integers."""

85def _mask_bytes(nbytes: int) -> int:

90def _bitset_set(mask: bytearray, bit_index: int) -> None:
91 """Set one bit in a little-endian bitset (bit 0 is LSB of byte 0)."""
92 byte_i = bit_index // 8
93 bit_i = bit_index % 8
94 mask[byte_i] |= (1 << bit_i)

97def _bitset_popcount(mask: bytes) -> int:
98 """Count set bits in a bitset."""
99 return sum(int(b).bit_count() for b in mask)

102def _read_slice(path: str, offset: int, nbytes: int) -> bytes:
103 """Read a byte slice from a file at a given offset."""
104 with open(path, "rb") as f:
105 f.seek(offset)

112def _diff_default_target(default_bytes: bytes, target_bytes: bytes) -> Tuple[bytes, bytes]:
113 """
114 Compute (mask, exceptions) between default_bytes and target_bytes.
115
116 The mask has 1 bit per byte; a 1 means target differs from default at that byte.
117 Exceptions are the raw target bytes at set-bit positions in ascending index order.
118 """
119 if len(default_bytes) != len(target_bytes):
120 raise ValueError("default_bytes and target_bytes must have same length")
121
122 n = len(target_bytes)
123 mask = bytearray(_mask_bytes(n))
124 exc = bytearray()
125
126 for i in range(n):
127 if target_bytes[i] != default_bytes[i]:
128 _bitset_set(mask, i)
129 exc.append(target_bytes[i])
130
131 return bytes(mask), bytes(exc)

134def _parse_tensor_map(tensor_map: object) -> Tuple[str, Dict[int, TensorRec]]:
135 """Parse the tensor map JSON object into a lookup table."""
136 if not isinstance(tensor_map, dict):
137 raise ValueError("tensor map JSON must be an object")
138
139 weights_bin = tensor_map.get("weights_bin")
140 tensors = tensor_map.get("tensors")
141 if not isinstance(weights_bin, str) or not isinstance(tensors, list):
142 raise ValueError("tensor map JSON must contain 'weights_bin' (str) and 'tensors' (list)")
143
144 out: Dict[int, TensorRec] = {}
145 for t in tensors:
146 if not isinstance(t, dict):
147 raise ValueError("tensor entry must be an object")
148 idx = int(t["index"])
149 name = str(t.get("name", f"tensor_{idx}"))
150 off = int(t["offset"])
151 nb = int(t["nbytes"])
152 out[idx] = TensorRec(index=idx, name=name, offset=off, nbytes=nb)
153
154 return weights_bin, out

198 drec = tensors[default_index]
