124static int utf8_next_cp(const char *s, size_t n, size_t *io_i, uint32_t *out_cp) {
125 if (!s || !io_i || !out_cp) return -1;
126 size_t i = *io_i;
127 if (i >= n) return 1;
128
129 unsigned char c0 = (unsigned char)s[i++];
130 if (c0 < 0x80) { *out_cp = (uint32_t)c0; *io_i = i; return 0; }
131
132 int need = 0;
133 uint32_t cp = 0;
134 if ((c0 & 0xE0) == 0xC0) { need = 1; cp = (uint32_t)(c0 & 0x1F); }
135 else if ((c0 & 0xF0) == 0xE0) { need = 2; cp = (uint32_t)(c0 & 0x0F); }
136 else if ((c0 & 0xF8) == 0xF0) { need = 3; cp = (uint32_t)(c0 & 0x07); }
137 else return -1;
138
139 if (i + (size_t)need > n) return -1;
140 for (int k = 0; k < need; ++k) {
141 unsigned char cx = (unsigned char)s[i++];
142 if ((cx & 0xC0) != 0x80) return -1;
143 cp = (cp << 6) | (uint32_t)(cx & 0x3F);
144 }
145
146 *out_cp = cp;
147 *io_i = i;
148 return 0;
149}

151static size_t utf8_put_cp(char *dst, size_t cap, uint32_t cp) {
152 if (!dst || cap == 0) return 0;
153 if (cp < 0x80) {

155 dst[0] = (char)cp;

637 for (size_t i = 0; i < 2048; ++i) m->cp_to_byte[i] = -1;
638 m->cp_to_byte_cap = 2048;
639
640 int used[256];
641 for (int i = 0; i < 256; ++i) used[i] = 0;
642
643 for (int b = 33; b <= 126; ++b) { m->byte_to_cp[b] = (uint32_t)b; used[b] = 1; }
644 for (int b = 161; b <= 172; ++b) { m->byte_to_cp[b] = (uint32_t)b; used[b] = 1; }
645 for (int b = 174; b <= 255; ++b) { m->byte_to_cp[b] = (uint32_t)b; used[b] = 1; }
646
647 int extra = 0;
648 for (int b = 0; b < 256; ++b) {
649 if (!used[b]) { m->byte_to_cp[b] = (uint32_t)(256 + extra); extra++; }
650 }
651
652 for (int b = 0; b < 256; ++b) {
653 uint32_t cp = m->byte_to_cp[b];
654 if (cp < m->cp_to_byte_cap) m->cp_to_byte[cp] = b;
655 }
656}
657
658static int byteunicode_cp_to_byte(const byteunicode_t *m, uint32_t cp, uint8_t *out_b) {
659 if (!m || !out_b) return -1;
660 if (cp < m->cp_to_byte_cap && m->cp_to_byte[cp] >= 0) { *out_b = (uint8_t)m->cp_to_byte[cp]; return 0; }
661 return -1;
662}

723typedef struct bytearena_s {
724 uint8_t **blocks;
725 size_t *sizes;
726 size_t nblocks;
727 size_t cap;
728 uint8_t *cur;
729 size_t cur_sz;
730 size_t cur_used;
731} bytearena_t;
732
733static void bytearena_free(bytearena_t *a) {
734 if (!a) return;
735 if (a->blocks) {
736 for (size_t i = 0; i < a->nblocks; ++i) free(a->blocks[i]);
737 }
738 free(a->blocks);
739 free(a->sizes);
740 memset(a, 0, sizeof(*a));
741}

743static int bytearena_new_block(bytearena_t *a, size_t need) {
744 size_t blk = (a->cur_sz ? a->cur_sz * 2 : (1u << 20));
745 if (blk < need) blk = need;
746 uint8_t *mem = (uint8_t *)malloc(blk);
747 if (!mem) return -1;
748
749 if (a->nblocks == a->cap) {
750 size_t nc = a->cap ? a->cap * 2 : 8;
751 uint8_t **nb = (uint8_t **)realloc(a->blocks, nc * sizeof(*nb));
752 size_t *ns = (size_t *)realloc(a->sizes, nc * sizeof(*ns));
753 if (!nb || !ns) { free(mem); free(nb); free(ns); return -1; }
754 a->blocks = nb;
755 a->sizes = ns;
756 a->cap = nc;
757 }
758
759 a->blocks[a->nblocks] = mem;
760 a->sizes[a->nblocks] = blk;
761 a->nblocks++;
762
763 a->cur = mem;
764 a->cur_sz = blk;
765 a->cur_used = 0;
766 return 0;
767}
768
769static uint8_t *bytearena_alloc(bytearena_t *a, size_t n) {
770 if (!a) return NULL;
771 if (n == 0) return (uint8_t *)"";
772 if (!a->cur || a->cur_used + n > a->cur_sz) {
773 if (bytearena_new_block(a, n) != 0) return NULL;
774 }
775 uint8_t *p = a->cur + a->cur_used;
776 a->cur_used += n;
777 return p;
778}

820static int parse_vocab_object(const char *json, ie_tok_gptoss_t *tok) {
821 const char *p = find_key_in_object(json, "vocab");
822 if (!p) return IE_TOK_GPTOSS_ERR_JSON;
823
824 p = skip_ws(p);
825 if (*p != '{') return IE_TOK_GPTOSS_ERR_JSON;
826 ++p;
827
828 uint32_t max_id = 0;
829 const char *q = p;
830 while (*q) {
831 q = skip_ws(q);
832 if (*q == '}') break;
833 char *kraw = NULL;
834 if (scan_json_string(&q, &kraw, NULL) != 0) return IE_TOK_GPTOSS_ERR_JSON;
835 if (json_unescape_inplace(kraw) != 0) { free(kraw); return IE_TOK_GPTOSS_ERR_JSON; }
836 q = skip_ws(q);
837 if (*q != ':') { free(kraw); return IE_TOK_GPTOSS_ERR_JSON; }
838 ++q;
839 int id = 0;
840 if (scan_json_int(&q, &id) != 0 || id < 0) { free(kraw); return IE_TOK_GPTOSS_ERR_JSON; }
841 if ((uint32_t)id > max_id) max_id = (uint32_t)id;
842 free(kraw);
843 q = skip_ws(q);
844 if (*q == ',') ++q;
845 }
846
847 tok->vocab_size = max_id + 1;
848 tok->id_to_tok = (char **)xcalloc(tok->vocab_size, sizeof(char *));
849 if (!tok->id_to_tok) return IE_TOK_GPTOSS_ERR_NOMEM;
850
851 if (strid_map_init(&tok->tok_to_id, &tok->tok_to_id_cap, tok->vocab_size) != 0) return IE_TOK_GPTOSS_ERR_NOMEM;
852
853 q = p;
854 while (*q) {
855 q = skip_ws(q);
856 if (*q == '}') break;
857 char *kraw = NULL;
858 if (scan_json_string(&q, &kraw, NULL) != 0) return IE_TOK_GPTOSS_ERR_JSON;
859 if (json_unescape_inplace(kraw) != 0) { free(kraw); return IE_TOK_GPTOSS_ERR_JSON; }
860 q = skip_ws(q);
861 if (*q != ':') { free(kraw); return IE_TOK_GPTOSS_ERR_JSON; }
862 ++q;
863 int id = 0;
864 if (scan_json_int(&q, &id) != 0 || id < 0 || (uint32_t)id >= tok->vocab_size) { free(kraw); return IE_TOK_GPTOSS_ERR_JSON; }
865
866 tok->id_to_tok[id] = kraw;
867 (void)strid_map_put(tok->tok_to_id, tok->tok_to_id_cap, tok->id_to_tok[id], (uint32_t)id);
868
869 q = skip_ws(q);
870 if (*q == ',') ++q;
871 }
872
873 return IE_TOK_GPTOSS_OK;
874}

876static int rebuild_tok_to_id(ie_tok_gptoss_t *tok) {
877 if (!tok) return -1;
878
879 free(tok->tok_to_id);
880 tok->tok_to_id = NULL;
881 tok->tok_to_id_cap = 0;
882
883 if (strid_map_init(&tok->tok_to_id, &tok->tok_to_id_cap, tok->vocab_size) != 0) return -1;
884
885 for (size_t i = 0; i < tok->vocab_size; i++) {
886 const char *s = tok->id_to_tok[i];
887 if (!s) continue;
888 if (strid_map_put(tok->tok_to_id, tok->tok_to_id_cap, s, (uint32_t)i) != 0) return -1;
889 }
890 return 0;

893static int parse_added_tokens_array(const char *json, ie_tok_gptoss_t *tok) {
894 const char *p = find_key_in_object(json, "added_tokens");
895 if (!p) return IE_TOK_GPTOSS_OK;
896
897 p = skip_ws(p);
898 if (!p || *p != '[') return IE_TOK_GPTOSS_ERR_JSON;
899 p++;
900
901 uint32_t max_id = (tok->vocab_size > 0) ? (uint32_t)(tok->vocab_size - 1) : 0;
902
903 const char *q = p;
904 for (;;) {
905 q = skip_ws(q);
906 if (!q) return IE_TOK_GPTOSS_ERR_JSON;
907 if (*q == ']') break;
908 if (*q != '{') return IE_TOK_GPTOSS_ERR_JSON;
909 q++;
910
911 int got_id = 0;
912 uint32_t id = 0;
913
914 for (;;) {
915 q = skip_ws(q);
916 if (!q) return IE_TOK_GPTOSS_ERR_JSON;
917 if (*q == '}') { q++; break; }
918
919 char *kraw = NULL;
920 if (scan_json_string(&q, &kraw, NULL) != 0) return IE_TOK_GPTOSS_ERR_JSON;
921 if (json_unescape_inplace(kraw) != 0) { free(kraw); return IE_TOK_GPTOSS_ERR_JSON; }
922
923 q = skip_ws(q);
924 if (!q || *q != ':') { free(kraw); return IE_TOK_GPTOSS_ERR_JSON; }
925 q++;
926
927 if (strcmp(kraw, "id") == 0) {
928 int v = 0;
929 if (scan_json_int(&q, &v) != 0) { free(kraw); return IE_TOK_GPTOSS_ERR_JSON; }
930 if (v < 0) { free(kraw); return IE_TOK_GPTOSS_ERR_JSON; }
931 id = (uint32_t)v;
932 got_id = 1;
933 } else {
934 if (skip_json_value(&q) != 0) { free(kraw); return IE_TOK_GPTOSS_ERR_JSON; }
935 }
936
937 free(kraw);
938
939 q = skip_ws(q);
940 if (!q) return IE_TOK_GPTOSS_ERR_JSON;
941 if (*q == ',') { q++; continue; }
942 if (*q == '}') { q++; break; }
943 }
944
945 if (got_id && id > max_id) max_id = id;
946
947 q = skip_ws(q);
948 if (!q) return IE_TOK_GPTOSS_ERR_JSON;
949 if (*q == ',') { q++; continue; }
950 if (*q == ']') break;
951 }
952
953 size_t need = (size_t)max_id + 1;
954 if (need > tok->vocab_size) {
955 size_t old = tok->vocab_size;
956 char **nt = (char **)realloc(tok->id_to_tok, need * sizeof(char *));
957 if (!nt) return IE_TOK_GPTOSS_ERR_NOMEM;
958 tok->id_to_tok = nt;
959 for (size_t i = old; i < need; i++) tok->id_to_tok[i] = NULL;
960 tok->vocab_size = (uint32_t)need;
961 if (rebuild_tok_to_id(tok) != 0) return IE_TOK_GPTOSS_ERR_NOMEM;
962 }
963
964 q = p;
965 for (;;) {
966 q = skip_ws(q);
967 if (!q) return IE_TOK_GPTOSS_ERR_JSON;
968 if (*q == ']') break;
969 if (*q != '{') return IE_TOK_GPTOSS_ERR_JSON;
970 q++;
971
972 int got_id = 0;
973 uint32_t id = 0;
974 char *content = NULL;
975
976 for (;;) {
977 q = skip_ws(q);
978 if (!q) { if (content) free(content); return IE_TOK_GPTOSS_ERR_JSON; }
979 if (*q == '}') { q++; break; }
980
981 char *kraw = NULL;
982 if (scan_json_string(&q, &kraw, NULL) != 0) { if (content) free(content); return IE_TOK_GPTOSS_ERR_JSON; }
983 if (json_unescape_inplace(kraw) != 0) { free(kraw); if (content) free(content); return IE_TOK_GPTOSS_ERR_JSON; }
984
985 q = skip_ws(q);
986 if (!q || *q != ':') { free(kraw); if (content) free(content); return IE_TOK_GPTOSS_ERR_JSON; }
987 q++;
988
989 if (strcmp(kraw, "id") == 0) {
990 int v = 0;
991 if (scan_json_int(&q, &v) != 0) { free(kraw); if (content) free(content); return IE_TOK_GPTOSS_ERR_JSON; }
992 if (v < 0) { free(kraw); if (content) free(content); return IE_TOK_GPTOSS_ERR_JSON; }
993 id = (uint32_t)v;
994 got_id = 1;
995 } else if (strcmp(kraw, "content") == 0) {
996 if (content) { free(content); content = NULL; }
997 if (scan_json_string(&q, &content, NULL) != 0) { free(kraw); return IE_TOK_GPTOSS_ERR_JSON; }
998 if (json_unescape_inplace(content) != 0) { free(kraw); free(content); return IE_TOK_GPTOSS_ERR_JSON; }
999 } else {
1000 if (skip_json_value(&q) != 0) { free(kraw); if (content) free(content); return IE_TOK_GPTOSS_ERR_JSON; }
1001 }
1002
1003 free(kraw);
1004
1005 q = skip_ws(q);
1006 if (!q) { if (content) free(content); return IE_TOK_GPTOSS_ERR_JSON; }
1007 if (*q == ',') { q++; continue; }
1008 if (*q == '}') { q++; break; }
1009 }
1010
1011 if (got_id && content) {
1012 if (id >= tok->vocab_size) { free(content); return IE_TOK_GPTOSS_ERR_JSON; }
1013 if (tok->id_to_tok[id] == NULL) tok->id_to_tok[id] = content;
1014 else free(content);
1015 if (strid_map_put(tok->tok_to_id, tok->tok_to_id_cap, tok->id_to_tok[id], id) != 0) return IE_TOK_GPTOSS_ERR_NOMEM;
1016 } else if (content) {
1017 free(content);
1018 }
1019
1020 q = skip_ws(q);
1021 if (!q) return IE_TOK_GPTOSS_ERR_JSON;
1022 if (*q == ',') { q++; continue; }
1023 if (*q == ']') break;
1024 }
1025
1026 return IE_TOK_GPTOSS_OK;
1027}

1029static int scan_merge_entry(const char **pp, char **a, char **b) {
1030 const char *p = skip_ws(*pp);
1031 if (!p || *p == '\0') return -1;
1032
1033 *a = NULL;
1034 *b = NULL;
1035
1036 if (*p == '"') {
1037 char *sraw = NULL;
1038 if (scan_json_string(&p, &sraw, NULL) != 0) return -1;
1039 if (json_unescape_inplace(sraw) != 0) { free(sraw); return -1; }
1040
1041 char *sp = strchr(sraw, ' ');
1042 if (!sp) { free(sraw); return -1; }
1043 *sp = '\0';
1044
1045 char *lhs = xstrdup_n(sraw, strlen(sraw));
1046 char *rhs = xstrdup_n(sp + 1, strlen(sp + 1));
1047 free(sraw);
1048
1049 if (!lhs || !rhs) { free(lhs); free(rhs); return -1; }
1050
1051 *a = lhs;
1052 *b = rhs;
1053 *pp = p;
1054 return 0;
1055 }
1056
1057 if (*p == '[') {
1058 p++;
1059 p = skip_ws(p);
1060
1061 char *lhs = NULL;
1062 char *rhs = NULL;
1063
1064 if (scan_json_string(&p, &lhs, NULL) != 0) return -1;
1065 if (json_unescape_inplace(lhs) != 0) { free(lhs); return -1; }
1066
1067 p = skip_ws(p);
1068 if (!p || *p != ',') { free(lhs); return -1; }
1069 p++;
1070 p = skip_ws(p);
1071
1072 if (scan_json_string(&p, &rhs, NULL) != 0) { free(lhs); return -1; }
1073 if (json_unescape_inplace(rhs) != 0) { free(lhs); free(rhs); return -1; }
1074
1075 p = skip_ws(p);
1076 if (!p || *p != ']') { free(lhs); free(rhs); return -1; }
1077 p++;
1078
1079 *a = lhs;
1080 *b = rhs;
1081 *pp = p;
1082 return 0;
1083 }
1084
1085 return -1;
1086}

1088static int parse_merges_array(const char *json, ie_tok_gptoss_t *tok) {
1089 const char *p = find_key_in_object(json, "merges");
1090 if (!p) return IE_TOK_GPTOSS_ERR_JSON;
1091
1092 p = skip_ws(p);
1093 if (!p || *p != '[') return IE_TOK_GPTOSS_ERR_JSON;
1094 ++p;
1095
1096 size_t merges = 0;
1097 const char *q = p;
1098 while (*q) {
1099 q = skip_ws(q);
1100 if (!q) return IE_TOK_GPTOSS_ERR_JSON;
1101 if (*q == ']') break;
1102
1103 char *lhs = NULL;
1104 char *rhs = NULL;
1105 if (scan_merge_entry(&q, &lhs, &rhs) != 0) return IE_TOK_GPTOSS_ERR_JSON;
1106 free(lhs);
1107 free(rhs);
1108 merges++;
1109
1110 q = skip_ws(q);
1111 if (!q) return IE_TOK_GPTOSS_ERR_JSON;
1112 if (*q == ',') ++q;
1113 }
1114
1115 if (intern_init(&tok->intern, merges * 4 + 1024, (size_t)(1u << 20)) != 0) return IE_TOK_GPTOSS_ERR_NOMEM;
1116 if (pairrank_map_init(&tok->pair_to_rank, &tok->pair_to_rank_cap, merges) != 0) return IE_TOK_GPTOSS_ERR_NOMEM;
1117
1118 q = p;
1119 uint32_t rank = 0;
1120 while (*q) {
1121 q = skip_ws(q);
1122 if (!q) return IE_TOK_GPTOSS_ERR_JSON;
1123 if (*q == ']') break;
1124
1125 char *lhs = NULL;
1126 char *rhs = NULL;
1127 if (scan_merge_entry(&q, &lhs, &rhs) != 0) return IE_TOK_GPTOSS_ERR_JSON;
1128
1129 uint32_t ida = 0, idb = 0;
1130 if (intern_get_or_add(&tok->intern, lhs, &ida) != 0 || intern_get_or_add(&tok->intern, rhs, &idb) != 0) {
1131 free(lhs); free(rhs);
1132 return IE_TOK_GPTOSS_ERR_NOMEM;
1133 }
1134
1135 uint64_t key = ((uint64_t)ida << 32) | (uint64_t)idb;
1136 (void)pairrank_map_put(tok->pair_to_rank, tok->pair_to_rank_cap, key, rank++);
1137
1138 free(lhs);
1139 free(rhs);
1140
1141 q = skip_ws(q);
1142 if (!q) return IE_TOK_GPTOSS_ERR_JSON;
1143 if (*q == ',') ++q;
1144 }
1145
1146 return IE_TOK_GPTOSS_OK;
1147}

1169static int rd_u32le(const unsigned char *buf, size_t n, size_t *io_off, uint32_t *out) {
1170 if (!buf || !io_off || !out) return -1;
1171 size_t off = *io_off;
1172 if (off + 4 > n) return -1;
1173 uint32_t v = 0;
1174 v |= (uint32_t)buf[off + 0];
1175 v |= (uint32_t)buf[off + 1] << 8;
1176 v |= (uint32_t)buf[off + 2] << 16;
1177 v |= (uint32_t)buf[off + 3] << 24;
1178 *io_off = off + 4;
1179 *out = v;
1180 return 0;
1181}

1285static int b64_val(int c) {
1286 if (c >= 'A' && c <= 'Z') return c - 'A';
1287 if (c >= 'a' && c <= 'z') return 26 + (c - 'a');
1288 if (c >= '0' && c <= '9') return 52 + (c - '0');
1289 if (c == '+') return 62;
1290 if (c == '/') return 63;
1291 if (c == '-') return 62;
1292 if (c == '_') return 63;
1293 return -1;
1294}

1296static int b64_decode(const char *s, size_t n, uint8_t **out, size_t *out_n) {
1297 if (!s || !out || !out_n) return -1;
1298 *out = NULL;
1299 *out_n = 0;
1300
1301 size_t max_out = (n / 4 + 1) * 3;
1302 uint8_t *buf = (uint8_t *)malloc(max_out ? max_out : 1);
1303 if (!buf) return -1;
1304
1305 size_t wi = 0;
1306 int quad[4];
1307 int qn = 0;
1308
1309 for (size_t i = 0; i < n; ++i) {
1310 unsigned char c = (unsigned char)s[i];
1311 if (c == '=') {
1312 quad[qn++] = -2;
1313 } else {
1314 int v = b64_val((int)c);
1315 if (v < 0) { free(buf); return -1; }
1316 quad[qn++] = v;
1317 }
1318
1319 if (qn == 4) {
1320 if (quad[0] < 0 || quad[1] < 0) { free(buf); return -1; }
1321
1322 uint32_t x = ((uint32_t)quad[0] << 18) | ((uint32_t)quad[1] << 12);
1323 uint32_t b2 = (quad[2] >= 0) ? (uint32_t)quad[2] : 0;
1324 uint32_t b3 = (quad[3] >= 0) ? (uint32_t)quad[3] : 0;
1325 x |= (b2 << 6) | b3;
1326
1327 buf[wi++] = (uint8_t)((x >> 16) & 0xFF);
1328 if (quad[2] != -2) buf[wi++] = (uint8_t)((x >> 8) & 0xFF);
1329 if (quad[3] != -2) buf[wi++] = (uint8_t)(x & 0xFF);
1330
1331 qn = 0;
1332 }
1333 }
1334
1335 if (qn != 0) { free(buf); return -1; }
1336
1337 *out = buf;
1338 *out_n = wi;
1339 return 0;
1340}

1478static int pretokenize_basic(const char *text, char ***out_parts, size_t *out_n) {
1479 if (!text || !out_parts || !out_n) return -1;

1481 *out_n = 0;
1482
1483 size_t cap = 0, n = 0;
1484 char **parts = NULL;
1485 const char *p = text;
1486
1487 while (*p) {
1488 if (isspace((unsigned char)*p)) {
1489 const char *s = p;
1490 while (*p && isspace((unsigned char)*p)) ++p;
1491 size_t len = (size_t)(p - s);
1492 char *seg = xstrdup_n(s, len);
1493 if (!seg) goto fail;
1494 if (n == cap) {
1495 size_t nc = cap ? cap * 2 : 64;
1496 char **np = (char **)realloc(parts, nc * sizeof(*np));
1497 if (!np) { free(seg); goto fail; }

1501 continue;
1502 }
1503

1506 size_t len = (size_t)(p - s);
1507
1508 int have_lead_space = 0;

1510 char *w = parts[n - 1];
1511 size_t wl = strlen(w);
1512 if (wl > 0 && w[wl - 1] == ' ') { w[wl - 1] = '\0'; have_lead_space = 1; }
1513 }
1514
1515 strbuf_t sb = {0};
1516 if (have_lead_space) {
1517 char sp = ' ';
1518 if (strbuf_append_bytes(&sb, &sp, 1) != 0) { strbuf_free(&sb); goto fail; }
1519 }
1520 if (strbuf_append_bytes(&sb, s, len) != 0) { strbuf_free(&sb); goto fail; }

1522 sb.v[sb.n] = '\0';
1523
1524 if (n == cap) {
1525 size_t nc = cap ? cap * 2 : 64;
1526 char **np = (char **)realloc(parts, nc * sizeof(*np));
1527 if (!np) { strbuf_free(&sb); goto fail; }
1528 parts = np; cap = nc;
1529 }
1530 parts[n++] = sb.v;
1531 }
1532
1533 size_t w = 0;
1534 for (size_t i = 0; i < n; ++i) {
1535 if (parts[i] && parts[i][0] == '\0') { free(parts[i]); parts[i] = NULL; continue; }
1536 parts[w++] = parts[i];
1537 }
1538 n = w;
1539
1540 *out_parts = parts;
1541 *out_n = n;
1542 return 0;
1543
1544fail:
1545 if (parts) {
1546 for (size_t i = 0; i < n; ++i) free(parts[i]);
1547 free(parts);
1548 }
1549 return -1;
1550}
