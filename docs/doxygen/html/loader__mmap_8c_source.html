140int ie_mmap_open(const char *path,
141 const void **out_addr, size_t *out_len,
142 int *out_is_fallback,
143 int flags) {
144 if (!path || !out_addr || !out_len || !out_is_fallback) return -22;
145
146#if defined(__linux__)
147 int fd = open(path, O_RDONLY | O_CLOEXEC);
148 if (fd < 0) return -errno;
149
150 struct stat st;
151 if (fstat(fd, &st) != 0) { int e = -errno; close(fd); return e; }
152 if (st.st_size <= 0) { close(fd); return -5; }
153
154 /* If POPULATE requested, prime the cache best-effort before mapping. */
155 if (flags & IE_MMAP_F_POPULATE) {
156 (void)ie_mmap_fadvise(fd, 0, (off_t)st.st_size, IE_PAGE_WILLNEED);
157 (void)ie_mmap_readahead(fd, 0, (size_t)st.st_size);
158 }
159
160 int mmap_flags = MAP_PRIVATE;
161 #ifdef MAP_POPULATE
162 if (flags & IE_MMAP_F_POPULATE) mmap_flags |= MAP_POPULATE;
163 #endif
164
165 void *p = mmap(NULL, (size_t)st.st_size, PROT_READ, mmap_flags, fd, 0);
166 int saved_errno = errno;
167 close(fd);
168
169 if (p == MAP_FAILED) {
170 /* Fallback to buffered read. */
171 FILE *f = fopen(path, "rb");
172 if (!f) return -saved_errno ? -saved_errno : -errno;
173
174 size_t n = (size_t)st.st_size;
175 void *buf = NULL;
176 /* 64-byte alignment tends to play well with prefetchers. */
177 int aerr = posix_memalign(&buf, 64, n ? n : 1u);
178 if (aerr != 0 || !buf) { fclose(f); return -12; }
179
180 size_t rd = fread(buf, 1, n, f);
181 fclose(f);
182 if (rd != n) { free(buf); return -5; }
183
184 *out_addr = buf;
185 *out_len = n;
186 *out_is_fallback = 1;
187 return 0;
188 }
189
190 /* Best-effort THP request if flagged. */
191 if (flags & IE_MMAP_F_HUGEPAGE) {
192 (void)ie_mmap_advise_thp(p, (size_t)st.st_size, /*enable=*/1);
193 }
194
195 *out_addr = p;
196 *out_len = (size_t)st.st_size;
197 *out_is_fallback = 0;
198 return 0;
199
200#else
201 /* Non-Linux: buffered read only. */
202 FILE *f = fopen(path, "rb");
203 if (!f) return -errno;
204
205 if (fseek(f, 0, SEEK_END) != 0) { int e = -errno; fclose(f); return e; }
206 long sz = ftell(f);
207 if (sz < 0) { int e = -errno; fclose(f); return e; }
208 if (fseek(f, 0, SEEK_SET) != 0) { int e = -errno; fclose(f); return e; }
209
210 size_t n = (size_t)sz;
211 void *buf = NULL;
212 int aerr = posix_memalign(&buf, 64, n ? n : 1u);
213 if (aerr != 0 || !buf) { fclose(f); return -12; }
214
215 size_t rd = fread(buf, 1, n, f);
216 fclose(f);
217 if (rd != n) { free(buf); return -5; }
218
219 (void)flags;
220 *out_addr = buf;
221 *out_len = n;
222 *out_is_fallback = 1;
223 return 0;
224#endif
225}

227int ie_mmap_apply_advise(const void *addr, size_t len, ie_mmap_advise_t advise) {
228 if (!addr || len == 0) return -22;
229
230 ie_page_policy pol = IE_PAGE_NORMAL;
231 switch (advise) {
232 case IE_MMAP_ADVISE_NONE: pol = IE_PAGE_NORMAL; break;
233 case IE_MMAP_ADVISE_WILLNEED: pol = IE_PAGE_WILLNEED; break;
234 case IE_MMAP_ADVISE_RANDOM: pol = IE_PAGE_RANDOM; break;
235 case IE_MMAP_ADVISE_SEQUENTIAL: pol = IE_PAGE_SEQUENTIAL; break;
236 default: return -22;
237 }
238 return ie_mmap_advise((void*)addr, len, pol);
239}
