103 static int parse_status_kb_line(const char *line, const char *key, uint64_t *out_kb) {
104  if (!line || !key || !out_kb) return 0;
105 
106  const size_t klen = strlen(key);
107  if (strncmp(line, key, klen) != 0) return 0;
108 
109  const char *p = line + klen;
110  while (*p == ' ' || *p == '\t') ++p;
111 
112  uint64_t val = 0;
113  int matched = 0;
114  while (*p >= '0' && *p <= '9') {
115  matched = 1;
116  val = val * 10 + (uint64_t)(*p - '0');
117  ++p;
118  }
119  if (!matched) return 0;
120 
121  *out_kb = val;
122  return 1;
123 }

201 uint32_t ie_metrics_sample_rss_peak(void) {
202  uint32_t best_mib = 0;
203 
204 #if defined(__linux__)
205  /* 1) Prefer VmHWM (KiB): process peak resident set size. */
206  {
207  uint64_t kb = 0;
208  const int rc = read_proc_kb_single("/proc/self/status", "VmHWM:", &kb);
209  if (rc == 0) {
210  const uint32_t mib = kib_to_mib_ceil(kb);
211  if (mib > best_mib) best_mib = mib;
212  dbg("[RSS] VmHWM: %llu kB -> %u MiB\n", (unsigned long long)kb, mib);
213  } else {
214  dbg("[RSS] VmHWM not found (rc=%d)\n", rc);
215  }
216  }
217 
218  /* 2) Fallback: VmRSS (KiB): current resident set size. */
219  if (best_mib == 0) {
220  uint64_t kb = 0;
221  const int rc = read_proc_kb_single("/proc/self/status", "VmRSS:", &kb);
222  if (rc == 0) {
223  const uint32_t mib = kib_to_mib_ceil(kb);
224  if (mib > best_mib) best_mib = mib;
225  dbg("[RSS] VmRSS: %llu kB -> %u MiB\n", (unsigned long long)kb, mib);
226  } else {
227  dbg("[RSS] VmRSS not found (rc=%d)\n", rc);
228  }
229  }
230 
231  /* 3) Try smaps_rollup Rss: (KiB): often useful with mmaps. */
232  if (best_mib == 0) {
233  uint64_t kb = 0;
234  const int rc = read_proc_kb_single("/proc/self/smaps_rollup", "Rss:", &kb);
235  if (rc == 0) {
236  const uint32_t mib = kib_to_mib_ceil(kb);
237  if (mib > best_mib) best_mib = mib;
238  dbg("[RSS] smaps_rollup Rss: %llu kB -> %u MiB\n", (unsigned long long)kb, mib);
239  } else {
240  dbg("[RSS] smaps_rollup Rss not found (rc=%d)\n", rc);
241  }
242  }
243 
244  /* 4) Final fallback: getrusage (Linux ru_maxrss is KiB). */
245  if (best_mib == 0) {
246  struct rusage ru;
247  if (getrusage(RUSAGE_SELF, &ru) == 0) {
248  const uint64_t kb = (uint64_t)ru.ru_maxrss;
249  const uint32_t mib = kib_to_mib_ceil(kb);
250  if (mib > best_mib) best_mib = mib;
251  dbg("[RSS] getrusage ru_maxrss: %llu kB -> %u MiB\n", (unsigned long long)kb, mib);
252  } else {
253  dbg("[RSS] getrusage failed: errno=%d\n", errno);
254  }
255  }
256 
257 #elif defined(__APPLE__)
258  /* macOS primary: mach_task_basic_info (bytes). */
259  {
260  mach_task_basic_info info;
261  mach_msg_type_number_t count = MACH_TASK_BASIC_INFO_COUNT;
262  const kern_return_t kr = task_info(mach_task_self(),
263  MACH_TASK_BASIC_INFO,
264  (task_info_t)&info,
265  &count);
266  if (kr == KERN_SUCCESS) {
267  uint64_t bytes = (uint64_t)info.resident_size_max;
268  if (bytes == 0) bytes = (uint64_t)info.resident_size;
269 
270  const uint32_t mib = (uint32_t)((bytes + (1024u * 1024u - 1)) / (1024u * 1024u));
271  if (mib > best_mib) best_mib = mib;
272  dbg("[RSS] mach_task_basic_info: %llu bytes -> %u MiB\n",
273  (unsigned long long)bytes, mib);
274  } else {
275  dbg("[RSS] mach task_info failed: kr=%d\n", (int)kr);
276  }
277  }
278 
279  /* Fallback: getrusage (macOS ru_maxrss is bytes). */
280  if (best_mib == 0) {
281  struct rusage ru;
282  if (getrusage(RUSAGE_SELF, &ru) == 0) {
283  const uint64_t bytes = (uint64_t)ru.ru_maxrss;
284  const uint32_t mib = (uint32_t)((bytes + (1024u * 1024u - 1)) / (1024u * 1024u));
285  if (mib > best_mib) best_mib = mib;
286  dbg("[RSS] getrusage ru_maxrss(bytes): %llu -> %u MiB\n",
287  (unsigned long long)bytes, mib);
288  }
289  }
290 
291 #else
292  /* Other OS: getrusage and conservatively treat ru_maxrss as KiB. */
293  {
294  struct rusage ru;
295  if (getrusage(RUSAGE_SELF, &ru) == 0) {
296  const uint64_t kb = (uint64_t)ru.ru_maxrss;
297  const uint32_t mib = kib_to_mib_ceil(kb);

303 #endif
304 
305  return best_mib;
