86static int parse_status_kb_line(const char* line, const char* key, uint64_t* out_kb) {
87 if (!line || !key || !out_kb) return 0;
88 size_t klen = strlen(key);
89 if (strncmp(line, key, klen) != 0) return 0;
90
91 /* Expected format: "Key:\t 12345 kB" (spaces/tabs vary) */
92 const char* p = line + klen;
93 while (*p == ' ' || *p == '\t') ++p;
94
95 /* Read number */
96 uint64_t val = 0;
97 int matched = 0;
98 while (*p >= '0' && *p <= '9') {
99 matched = 1;
100 val = val * 10 + (uint64_t)(*p - '0');
101 ++p;
102 }
103 if (!matched) return 0;
104
105 /* Skip spaces/tabs */
106 while (*p == ' ' || *p == '\t') ++p;
107
108 /* Optional unit "kB" */
109 if ((p[0] == 'k' || p[0] == 'K') && (p[1] == 'B' || p[1] == 'b')) {
110 /* ok */
111 }
112
113 *out_kb = val;
114 return 1;
115}
