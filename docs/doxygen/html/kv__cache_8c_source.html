59static int kv_validate_and_derive(ie_kv_cache* kv, const ie_kv_opts* opts) {
60 if (!kv || !opts) return -1;
61 if (opts->heads <= 0 || opts->head_dim <= 0 || opts->max_seq <= 0) return -1;
62 if (opts->storage != IE_KV_STORAGE_F32 &&
63 opts->storage != IE_KV_STORAGE_INT8 &&
64 opts->storage != IE_KV_STORAGE_FP8) return -1;
65
66 kv->heads = opts->heads;
67 kv->head_dim = opts->head_dim;
68 kv->max_seq = opts->max_seq;
69 kv->storage = opts->storage;
70 kv->group_size = (opts->group_size > 0 ? opts->group_size : 1);
71 kv->symmetric = opts->symmetric;
72 kv->fp8_format = opts->fp8_format;
73
74 kv->elem_count = (size_t)kv->heads * (size_t)kv->max_seq * (size_t)kv->head_dim;
75 kv->group_count = ceil_div_sz((size_t)kv->head_dim, kv->group_size);
76
77 kv->K = kv->V = NULL;
78 kv->scales_K = kv->scales_V = NULL;
79 kv->zeros_K = kv->zeros_V = NULL;
80
81 return 0;
82}

95int ie_kv_init(ie_kv_cache* kv, const ie_kv_opts* opts) {
96 if (kv_validate_and_derive(kv, opts) != 0) return -1;
97
98 const size_t elem_size = ie_kv_element_size(kv->storage);
99 if (elem_size == 0) return -1;
100
101 const size_t bytes = kv->elem_count * elem_size;
102
103 kv->K = calloc(1, bytes);
104 kv->V = calloc(1, bytes);
105 if (!kv->K || !kv->V) {
106 ie_kv_free(kv);
107 return -2;
108 }
109
110 if (kv->storage == IE_KV_STORAGE_INT8) {
111 const size_t params_len = (size_t)kv->max_seq * (size_t)kv->heads * kv->group_count;
112 kv->scales_K = (float*)calloc(params_len, sizeof(float));
113 kv->zeros_K = (int8_t*)calloc(params_len, sizeof(int8_t));
114 kv->scales_V = (float*)calloc(params_len, sizeof(float));
115 kv->zeros_V = (int8_t*)calloc(params_len, sizeof(int8_t));
116 if (!kv->scales_K || !kv->zeros_K || !kv->scales_V || !kv->zeros_V) {
117 ie_kv_free(kv);
118 return -2;
119 }
120 }
121
122 return 0;
123}

125void ie_kv_free(ie_kv_cache* kv) {
126 if (!kv) return;
127 if (kv->K) { free(kv->K); kv->K = NULL; }
128 if (kv->V) { free(kv->V); kv->V = NULL; }
129 if (kv->scales_K) { free(kv->scales_K); kv->scales_K = NULL; }
130 if (kv->zeros_K) { free(kv->zeros_K); kv->zeros_K = NULL; }
131 if (kv->scales_V) { free(kv->scales_V); kv->scales_V = NULL; }
132 if (kv->zeros_V) { free(kv->zeros_V); kv->zeros_V = NULL; }
133
134 /* Keep metadata so double-free is harmless; caller may reuse the struct. */
135 kv->elem_count = kv->group_count = 0;

138int ie_kv_store_token_f32(ie_kv_cache* kv, size_t t,
139 const float* K_f32, const float* V_f32) {
140 if (!kv || !K_f32 || !V_f32) return -1;
141 if (kv->storage != IE_KV_STORAGE_F32) return -2;
142 if (t >= (size_t)kv->max_seq) return -3;
143
144 const size_t H = (size_t)kv->heads;
145 const size_t D = (size_t)kv->head_dim;
146
147 float* Kdst = (float*)kv->K;
148 float* Vdst = (float*)kv->V;
149
150 /* Copy [heads, head_dim] block for token t. */
151 for (size_t h = 0; h < H; ++h) {
152 const size_t off = payload_index(t, h, 0, H, D);
153 memcpy(Kdst + off, K_f32 + h * D, D * sizeof(float));
154 memcpy(Vdst + off, V_f32 + h * D, D * sizeof(float));
155 }
156 return 0;
157}

159int ie_kv_store_token_int8_per_tensor(ie_kv_cache* kv, size_t t,
160 const float* K_f32, const float* V_f32) {
161 if (!kv || !K_f32 || !V_f32) return -1;
162 if (kv->storage != IE_KV_STORAGE_INT8) return -2;
163 if (t >= (size_t)kv->max_seq) return -3;
164
165 const size_t H = (size_t)kv->heads;
166 const size_t D = (size_t)kv->head_dim;
167 const size_t G = kv->group_count; /* per-tensor uses g=0 */
168
169 int8_t* Kdst = (int8_t*)kv->K;
170 int8_t* Vdst = (int8_t*)kv->V;
171
172 for (size_t h = 0; h < H; ++h) {
173 const float* Ksrc = K_f32 + h * D;
174 const float* Vsrc = V_f32 + h * D;
175
176 /* Compute per-head params from min/max. */
177 float Kmn = Ksrc[0], Kmx = Ksrc[0];
178 float Vmn = Vsrc[0], Vmx = Vsrc[0];
179 for (size_t d = 1; d < D; ++d) {
180 float kv = Ksrc[d];
181 float vv = Vsrc[d];
182 if (kv < Kmn) Kmn = kv; if (kv > Kmx) Kmx = kv;
183 if (vv < Vmn) Vmn = vv; if (vv > Vmx) Vmx = vv;
184 }
185
186 ie_act_i8_params pK, pV;
187 ie_act_i8_params_from_minmax(Kmn, Kmx, kv->symmetric, &pK.scale, &pK.zero_point);
188 ie_act_i8_params_from_minmax(Vmn, Vmx, kv->symmetric, &pV.scale, &pV.zero_point);
189
190 /* Quantize into destination payload. */
191 const size_t off = payload_index(t, h, 0, H, D);
192 ie_quantize_act_int8(Ksrc, Kdst + off, D, pK, kv->symmetric);
193 ie_quantize_act_int8(Vsrc, Vdst + off, D, pV, kv->symmetric);
194
195 /* Record parameters at g=0. */
196 const size_t pi = param_index(t, h, 0, H, G);
197 kv->scales_K[pi] = pK.scale;
198 kv->zeros_K[pi] = pK.zero_point;
199 kv->scales_V[pi] = pV.scale;
200 kv->zeros_V[pi] = pV.zero_point;
201 }
202 return 0;
203}

205int ie_kv_store_token_int8_per_group(ie_kv_cache* kv, size_t t,
206 const float* K_f32, const float* V_f32) {
207 if (!kv || !K_f32 || !V_f32) return -1;
208 if (kv->storage != IE_KV_STORAGE_INT8) return -2;
209 if (t >= (size_t)kv->max_seq) return -3;
210
211 const size_t H = (size_t)kv->heads;
212 const size_t D = (size_t)kv->head_dim;
213 const size_t G = kv->group_count;
214 const size_t Gs = kv->group_size;
215
216 int8_t* Kdst = (int8_t*)kv->K;
217 int8_t* Vdst = (int8_t*)kv->V;
218
219 /* Temporary parameter buffers per head. */
220 float* scales = (float*)malloc(G * sizeof(float));
221 int8_t* zeros = (int8_t*)malloc(G * sizeof(int8_t));
222 if (!scales || !zeros) { free(scales); free(zeros); return -4; }
223
224 for (size_t h = 0; h < H; ++h) {
225 const float* Ksrc = K_f32 + h * D;
226 const float* Vsrc = V_f32 + h * D;
227 const size_t off = payload_index(t, h, 0, H, D);
228
229 /* K */
230 ie_act_i8_group_params_from_data(Ksrc, D, Gs, kv->symmetric, scales, zeros);
231 ie_quantize_act_int8_per_group(Ksrc, Kdst + off, D, Gs, scales, zeros, kv->symmetric);
232 for (size_t g = 0; g < G; ++g) {
233 const size_t pi = param_index(t, h, g, H, G);
234 kv->scales_K[pi] = scales[g];
235 kv->zeros_K[pi] = zeros[g];
236 }
237
238 /* V */
239 ie_act_i8_group_params_from_data(Vsrc, D, Gs, kv->symmetric, scales, zeros);
240 ie_quantize_act_int8_per_group(Vsrc, Vdst + off, D, Gs, scales, zeros, kv->symmetric);
241 for (size_t g = 0; g < G; ++g) {
242 const size_t pi = param_index(t, h, g, H, G);
243 kv->scales_V[pi] = scales[g];
244 kv->zeros_V[pi] = zeros[g];
245 }
246 }
247
248 free(scales);
249 free(zeros);
250 return 0;
251}

253int ie_kv_store_token_fp8(ie_kv_cache* kv, size_t t,
254 const float* K_f32, const float* V_f32) {
255 if (!kv || !K_f32 || !V_f32) return -1;
256 if (kv->storage != IE_KV_STORAGE_FP8) return -2;
257 if (t >= (size_t)kv->max_seq) return -3;
258
259 const size_t H = (size_t)kv->heads;
260 const size_t D = (size_t)kv->head_dim;
261
262 uint8_t* Kdst = (uint8_t*)kv->K;
263 uint8_t* Vdst = (uint8_t*)kv->V;
264
265 /* Temporary FP8 buffers per head to avoid partial writes on error. */
266 uint8_t* tmp = (uint8_t*)malloc(D * sizeof(uint8_t));
267 if (!tmp) return -4;
268
269 for (size_t h = 0; h < H; ++h) {
270 const float* Ksrc = K_f32 + h * D;
271 const float* Vsrc = V_f32 + h * D;
272 const size_t off = payload_index(t, h, 0, H, D);
273
274 ie_quantize_act_fp8(Ksrc, tmp, D, kv->fp8_format);
275 memcpy(Kdst + off, tmp, D * sizeof(uint8_t));
276
277 ie_quantize_act_fp8(Vsrc, tmp, D, kv->fp8_format);
278 memcpy(Vdst + off, tmp, D * sizeof(uint8_t));
279 }
280
281 free(tmp);
282 return 0;
283}

285int ie_kv_load_token_f32(const ie_kv_cache* kv, size_t t,
286 float* K_out, float* V_out) {
287 if (!kv || !K_out || !V_out) return -1;
288 if (t >= (size_t)kv->max_seq) return -2;
289
290 const size_t H = (size_t)kv->heads;
291 const size_t D = (size_t)kv->head_dim;
292
293 if (kv->storage == IE_KV_STORAGE_F32) {
294 const float* Ksrc = (const float*)kv->K;
295 const float* Vsrc = (const float*)kv->V;
296 for (size_t h = 0; h < H; ++h) {
297 const size_t off = payload_index(t, h, 0, H, D);
298 memcpy(K_out + h * D, Ksrc + off, D * sizeof(float));
299 memcpy(V_out + h * D, Vsrc + off, D * sizeof(float));
300 }
301 return 0;
302 }
303
304 if (kv->storage == IE_KV_STORAGE_INT8) {
305 const int8_t* Kq = (const int8_t*)kv->K;
306 const int8_t* Vq = (const int8_t*)kv->V;
307 const size_t G = kv->group_count;
308 const size_t Gs = kv->group_size;
309
310 for (size_t h = 0; h < H; ++h) {
311 const size_t off = payload_index(t, h, 0, H, D);
312
313 /* Dequant K */
314 for (size_t d = 0; d < D; ++d) {
315 const size_t g = d / Gs;
316 const size_t pi = param_index(t, h, g, H, G);
317 const float s = kv->scales_K[pi];
318 const int z = (int)kv->zeros_K[pi];
319 K_out[h * D + d] = s * ((int)Kq[off + d] - z);
320 }
321
322 /* Dequant V */
323 for (size_t d = 0; d < D; ++d) {
324 const size_t g = d / Gs;
325 const size_t pi = param_index(t, h, g, H, G);
326 const float s = kv->scales_V[pi];
327 const int z = (int)kv->zeros_V[pi];
328 V_out[h * D + d] = s * ((int)Vq[off + d] - z);
329 }
330 }
331 return 0;
332 }
333
334 if (kv->storage == IE_KV_STORAGE_FP8) {
335 const uint8_t* K8 = (const uint8_t*)kv->K;
336 const uint8_t* V8 = (const uint8_t*)kv->V;
337
338 for (size_t h = 0; h < H; ++h) {
339 const size_t off = payload_index(t, h, 0, H, D);
340 ie_dequantize_act_fp8(K8 + off, K_out + h*D, D, kv->fp8_format);
341 ie_dequantize_act_fp8(V8 + off, V_out + h*D, D, kv->fp8_format);
342 }
343 return 0;
344 }
345
346 return -3; /* Unknown storage. */
347}
