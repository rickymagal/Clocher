118static int mul_overflow_sz(size_t a, size_t b, size_t *out) {
119 if (!out) return 1;
120 if (a == 0 || b == 0) {
121 *out = 0;
122 return 0;
123 }
124 if (a > (SIZE_MAX / b)) return 1;
125 *out = a * b;
126 return 0;
127}

138static int mul3_overflow_sz(size_t a, size_t b, size_t c, size_t *out) {
139 size_t ab = 0;
140 if (mul_overflow_sz(a, b, &ab)) return 1;
141 return mul_overflow_sz(ab, c, out);
142}

159static int kv_validate_and_derive(ie_kv_cache *kv, const ie_kv_opts *opts) {
160 if (!kv || !opts) return -1;
161 if (opts->heads <= 0 || opts->head_dim <= 0 || opts->max_seq <= 0) return -1;
162 if (opts->storage != IE_KV_STORAGE_F32 &&
163 opts->storage != IE_KV_STORAGE_INT8 &&
164 opts->storage != IE_KV_STORAGE_FP8) return -1;
165
166 kv->heads = opts->heads;
167 kv->head_dim = opts->head_dim;
168 kv->max_seq = opts->max_seq;
169 kv->storage = opts->storage;
170 kv->symmetric = opts->symmetric;
171 kv->fp8_format = opts->fp8_format;
172
173 if (kv->storage == IE_KV_STORAGE_INT8) {
174 kv->group_size = (opts->group_size > 0 ? opts->group_size : 1u);
175 kv->group_count = ceil_div_sz((size_t)kv->head_dim, kv->group_size);
176 } else {
177 kv->group_size = 1u;
178 kv->group_count = 1u;
179 }
180
181 if (mul3_overflow_sz((size_t)kv->heads, (size_t)kv->max_seq, (size_t)kv->head_dim,
182 &kv->elem_count)) {
183 return -1;
184 }
185
186 kv->K = kv->V = NULL;
187 kv->scales_K = kv->scales_V = NULL;
188 kv->zeros_K = kv->zeros_V = NULL;
189
190 kv->scratch_scales = NULL;
191 kv->scratch_zeros = NULL;
192 kv->scratch_u8 = NULL;
193
194 kv->backing = NULL;
195 kv->backing_bytes = 0;
196 kv->backing_owner = 0;
197
198 return 0;
199}

215static int kv_compute_slab_bytes(const ie_kv_cache *kv, size_t *out_bytes) {
216 if (!kv || !out_bytes) return -1;
217 *out_bytes = 0;
218
219 const size_t align = 64u;
220 const size_t elem_size = ie_kv_element_size(kv->storage);
221 if (elem_size == 0) return -1;
222
223 size_t payload_bytes = 0;
224 if (mul_overflow_sz(kv->elem_count, elem_size, &payload_bytes)) return -1;
225
226 size_t total = 0;
227
228 /* K and V payloads. */
229 total = align_up_sz(total, align);
230 total += payload_bytes;
231 total = align_up_sz(total, align);
232 total += payload_bytes;
233
234 /* INT8 params (two scales arrays + two zeros arrays). */
235 if (kv->storage == IE_KV_STORAGE_INT8) {
236 size_t params_len = 0;
237 if (mul3_overflow_sz((size_t)kv->max_seq, (size_t)kv->heads, kv->group_count, &params_len)) {
238 return -1;
239 }
240
241 const size_t scales_bytes = params_len * sizeof(float);
242 const size_t zeros_bytes = params_len * sizeof(int8_t);
243
244 total = align_up_sz(total, align);
245 total += scales_bytes; /* scales_K */
246 total = align_up_sz(total, align);
247 total += zeros_bytes; /* zeros_K */
248 total = align_up_sz(total, align);
249 total += scales_bytes; /* scales_V */
250 total = align_up_sz(total, align);
251 total += zeros_bytes; /* zeros_V */
252
253 /* Scratch for per-group store: [G] scales + [G] zeros. */
254 total = align_up_sz(total, align);
255 total += kv->group_count * sizeof(float);
256 total = align_up_sz(total, align);
257 total += kv->group_count * sizeof(int8_t);
258 }
259
260 /* FP8 scratch: [D] bytes. */
261 if (kv->storage == IE_KV_STORAGE_FP8) {
262 total = align_up_sz(total, align);
263 total += (size_t)kv->head_dim * sizeof(uint8_t);
264 }
265
266 *out_bytes = align_up_sz(total, align);
267 return 0;
268}

278static int kv_assign_from_slab(ie_kv_cache *kv, uint8_t *slab, size_t slab_bytes) {
279 if (!kv || !slab || slab_bytes == 0) return -1;
280
281 const size_t align = 64u;
282 const size_t elem_size = ie_kv_element_size(kv->storage);
283 if (elem_size == 0) return -1;
284
285 size_t payload_bytes = 0;
286 if (mul_overflow_sz(kv->elem_count, elem_size, &payload_bytes)) return -1;
287
288 size_t off = 0;
289
290 /* K */
291 off = align_up_sz(off, align);
292 if (off + payload_bytes > slab_bytes) return -1;
293 kv->K = slab + off;
294 off += payload_bytes;
295
296 /* V */
297 off = align_up_sz(off, align);
298 if (off + payload_bytes > slab_bytes) return -1;
299 kv->V = slab + off;
300 off += payload_bytes;
301
302 if (kv->storage == IE_KV_STORAGE_INT8) {
303 size_t params_len = 0;
304 if (mul3_overflow_sz((size_t)kv->max_seq, (size_t)kv->heads, kv->group_count, &params_len)) {
305 return -1;
306 }
307
308 const size_t scales_bytes = params_len * sizeof(float);
309 const size_t zeros_bytes = params_len * sizeof(int8_t);
310
311 off = align_up_sz(off, align);
312 if (off + scales_bytes > slab_bytes) return -1;
313 kv->scales_K = (float *)(void *)(slab + off);
314 off += scales_bytes;
315
316 off = align_up_sz(off, align);
317 if (off + zeros_bytes > slab_bytes) return -1;
318 kv->zeros_K = (int8_t *)(void *)(slab + off);
319 off += zeros_bytes;
320
321 off = align_up_sz(off, align);
322 if (off + scales_bytes > slab_bytes) return -1;
323 kv->scales_V = (float *)(void *)(slab + off);
324 off += scales_bytes;
325
326 off = align_up_sz(off, align);
327 if (off + zeros_bytes > slab_bytes) return -1;
328 kv->zeros_V = (int8_t *)(void *)(slab + off);
329 off += zeros_bytes;
330
331 /* Scratch */
332 off = align_up_sz(off, align);
333 if (off + kv->group_count * sizeof(float) > slab_bytes) return -1;
334 kv->scratch_scales = (float *)(void *)(slab + off);
335 off += kv->group_count * sizeof(float);
336
337 off = align_up_sz(off, align);
338 if (off + kv->group_count * sizeof(int8_t) > slab_bytes) return -1;
339 kv->scratch_zeros = (int8_t *)(void *)(slab + off);
340 off += kv->group_count * sizeof(int8_t);
341 }
342
343 if (kv->storage == IE_KV_STORAGE_FP8) {
344 off = align_up_sz(off, align);
345 if (off + (size_t)kv->head_dim * sizeof(uint8_t) > slab_bytes) return -1;
346 kv->scratch_u8 = (uint8_t *)(void *)(slab + off);
347 off += (size_t)kv->head_dim * sizeof(uint8_t);
348 }
349
350 return 0;
351}

377int ie_kv_init(ie_kv_cache *kv, const ie_kv_opts *opts) {
378 if (!kv) return -1;
379 if (kv_validate_and_derive(kv, opts) != 0) return -1;
380
381 size_t slab_bytes = 0;
382 if (kv_compute_slab_bytes(kv, &slab_bytes) != 0) return -2;
383
384 uint8_t *slab = (uint8_t *)calloc(1, slab_bytes);
385 if (!slab) return -3;
386
387 if (kv_assign_from_slab(kv, slab, slab_bytes) != 0) {
388 free(slab);
389 return -4;
390 }
391
392 kv->backing = slab;
393 kv->backing_bytes = slab_bytes;
394 kv->backing_owner = 1;
395 return 0;

411int ie_kv_init_layers(ie_kv_cache *kv_layers, int n_layers, const ie_kv_opts *opts) {
412 if (!kv_layers || n_layers <= 0 || !opts) return -1;
413
414 /* Derive one layer to compute per-layer slab bytes. */
415 ie_kv_cache tmp;
416 memset(&tmp, 0, sizeof(tmp));
417 if (kv_validate_and_derive(&tmp, opts) != 0) return -2;
418
419 size_t per_layer_bytes = 0;
420 if (kv_compute_slab_bytes(&tmp, &per_layer_bytes) != 0) return -3;
421
422 /* Make each slice start aligned. */
423 per_layer_bytes = align_up_sz(per_layer_bytes, 64u);
424
425 size_t total_bytes = 0;
426 if (mul_overflow_sz((size_t)n_layers, per_layer_bytes, &total_bytes)) return -4;
427
428 uint8_t *slab_all = (uint8_t *)calloc(1, total_bytes);
429 if (!slab_all) return -5;
430
431 for (int l = 0; l < n_layers; ++l) {
432 ie_kv_cache *kv = &kv_layers[l];
433 memset(kv, 0, sizeof(*kv));
434 if (kv_validate_and_derive(kv, opts) != 0) {
435 free(slab_all);
436 for (int j = 0; j < l; ++j) memset(&kv_layers[j], 0, sizeof(kv_layers[j]));
437 return -6;
438 }
439
440 uint8_t *slice = slab_all + (size_t)l * per_layer_bytes;
441 if (kv_assign_from_slab(kv, slice, per_layer_bytes) != 0) {
442 free(slab_all);
443 for (int j = 0; j < l; ++j) memset(&kv_layers[j], 0, sizeof(kv_layers[j]));
444 return -7;
445 }
446
447 kv->backing = slab_all;
448 kv->backing_bytes = total_bytes;
449 kv->backing_owner = (l == 0) ? 1 : 0;
450 }
451
452 return 0;
453}

533int ie_kv_store_token_int8_per_tensor(ie_kv_cache *kv, size_t t,
534 const float *K_f32, const float *V_f32) {
535 if (!kv || !K_f32 || !V_f32) return -1;
536 if (kv->storage != IE_KV_STORAGE_INT8) return -2;
537 if (t >= (size_t)kv->max_seq) return -3;
538
539 const size_t H = (size_t)kv->heads;
540 const size_t D = (size_t)kv->head_dim;
541 const size_t G = kv->group_count;
542
543 int8_t *Kdst = (int8_t *)kv->K;
544 int8_t *Vdst = (int8_t *)kv->V;
545
546 for (size_t h = 0; h < H; ++h) {
547 const float *Ksrc = K_f32 + h * D;
548 const float *Vsrc = V_f32 + h * D;
549
550 float Kmn = Ksrc[0], Kmx = Ksrc[0];
551 float Vmn = Vsrc[0], Vmx = Vsrc[0];
552 for (size_t d = 1; d < D; ++d) {
553 const float kvv = Ksrc[d];
554 const float vvv = Vsrc[d];
555 if (kvv < Kmn) Kmn = kvv;
556 if (kvv > Kmx) Kmx = kvv;
557 if (vvv < Vmn) Vmn = vvv;
558 if (vvv > Vmx) Vmx = vvv;
559 }
560
561 ie_act_i8_params pK, pV;
562 ie_act_i8_params_from_minmax(Kmn, Kmx, kv->symmetric, &pK.scale, &pK.zero_point);
563 ie_act_i8_params_from_minmax(Vmn, Vmx, kv->symmetric, &pV.scale, &pV.zero_point);
564
565 const size_t off = (t * H + h) * D;
566 ie_quantize_act_int8(Ksrc, Kdst + off, D, pK, kv->symmetric);
567 ie_quantize_act_int8(Vsrc, Vdst + off, D, pV, kv->symmetric);
568
569 for (size_t g = 0; g < G; ++g) {
570 const size_t pi = param_index(t, h, g, H, G);
571 kv->scales_K[pi] = pK.scale;
572 kv->zeros_K[pi] = pK.zero_point;
573 kv->scales_V[pi] = pV.scale;
574 kv->zeros_V[pi] = pV.zero_point;
575 }
576 }
577
578 return 0;
579}

694int ie_kv_load_token_f32(const ie_kv_cache *kv, size_t t,
695 float *K_out, float *V_out) {
696 if (!kv || !K_out || !V_out) return -1;
697
698 if (t >= (size_t)kv->max_seq) {
699 ie_kv_miss_add_local(1);
700 return -2;
701 }
702
703 const size_t H = (size_t)kv->heads;
704 const size_t D = (size_t)kv->head_dim;
705 const size_t slice = H * D;
706
707 if (kv->storage == IE_KV_STORAGE_F32) {
708 const float *Ksrc = (const float *)kv->K;
709 const float *Vsrc = (const float *)kv->V;
710 memcpy(K_out, Ksrc + t * slice, slice * sizeof(float));
711 memcpy(V_out, Vsrc + t * slice, slice * sizeof(float));
712 ie_kv_hit_add_local(1);
713 return 0;
714 }
715
716 if (kv->storage == IE_KV_STORAGE_INT8) {
717 const int8_t *Kq = (const int8_t *)kv->K;
718 const int8_t *Vq = (const int8_t *)kv->V;
719 const size_t G = kv->group_count;
720 const size_t Gs = kv->group_size;
721
722 for (size_t h = 0; h < H; ++h) {
723 const size_t off = (t * H + h) * D;
724
725 for (size_t d = 0; d < D; ++d) {
726 const size_t g = d / Gs;
727 const size_t pi = param_index(t, h, g, H, G);
728 const float s = kv->scales_K[pi];
729 const int z = (int)kv->zeros_K[pi];
730 K_out[h * D + d] = s * ((int)Kq[off + d] - z);
731 }
732
733 for (size_t d = 0; d < D; ++d) {
734 const size_t g = d / Gs;
735 const size_t pi = param_index(t, h, g, H, G);
736 const float s = kv->scales_V[pi];
737 const int z = (int)kv->zeros_V[pi];
738 V_out[h * D + d] = s * ((int)Vq[off + d] - z);
739 }
740 }
741
742 ie_kv_hit_add_local(1);
743 return 0;
744 }
745
746 if (kv->storage == IE_KV_STORAGE_FP8) {
747 const uint8_t *K8 = (const uint8_t *)kv->K;
748 const uint8_t *V8 = (const uint8_t *)kv->V;
749
750 for (size_t h = 0; h < H; ++h) {
751 const size_t off = (t * H + h) * D;
752 ie_dequantize_act_fp8(K8 + off, K_out + h * D, D, kv->fp8_format);
753 ie_dequantize_act_fp8(V8 + off, V_out + h * D, D, kv->fp8_format);
754 }
755
756 ie_kv_hit_add_local(1);
757 return 0;
758 }
759
760 ie_kv_miss_add_local(1);
761 return -3;
762}
