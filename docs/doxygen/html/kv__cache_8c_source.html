98static int kv_validate_and_derive(ie_kv_cache* kv, const ie_kv_opts* opts) {
99 if (!kv || !opts) return -1;
100 if (opts->heads <= 0 || opts->head_dim <= 0 || opts->max_seq <= 0) return -1;
101 if (opts->storage != IE_KV_STORAGE_F32 &&
102 opts->storage != IE_KV_STORAGE_INT8 &&
103 opts->storage != IE_KV_STORAGE_FP8) return -1;
104
105 kv->heads = opts->heads;
106 kv->head_dim = opts->head_dim;
107 kv->max_seq = opts->max_seq;
108 kv->storage = opts->storage;
109 kv->group_size = (opts->group_size > 0 ? opts->group_size : 1);
110 kv->symmetric = opts->symmetric;
111 kv->fp8_format = opts->fp8_format;
112
113 kv->elem_count = (size_t)kv->heads * (size_t)kv->max_seq * (size_t)kv->head_dim;
114 kv->group_count = ceil_div_sz((size_t)kv->head_dim, kv->group_size);
115
116 kv->K = kv->V = NULL;
117 kv->scales_K = kv->scales_V = NULL;
118 kv->zeros_K = kv->zeros_V = NULL;
119
120 return 0;
121}

134int ie_kv_init(ie_kv_cache* kv, const ie_kv_opts* opts) {
135 if (kv_validate_and_derive(kv, opts) != 0) return -1;
136
137 const size_t elem_size = ie_kv_element_size(kv->storage);
138 if (elem_size == 0) return -1;
139
140 const size_t bytes = kv->elem_count * elem_size;
141
142 kv->K = calloc(1, bytes);
143 kv->V = calloc(1, bytes);
144 if (!kv->K || !kv->V) {
145 ie_kv_free(kv);
146 return -2;
147 }
148
149 if (kv->storage == IE_KV_STORAGE_INT8) {
150 const size_t params_len = (size_t)kv->max_seq * (size_t)kv->heads * kv->group_count;
151 kv->scales_K = (float*)calloc(params_len, sizeof(float));
152 kv->zeros_K = (int8_t*)calloc(params_len, sizeof(int8_t));
153 kv->scales_V = (float*)calloc(params_len, sizeof(float));
154 kv->zeros_V = (int8_t*)calloc(params_len, sizeof(int8_t));
155 if (!kv->scales_K || !kv->zeros_K || !kv->scales_V || !kv->zeros_V) {
156 ie_kv_free(kv);
157 return -2;
158 }
159 }
160
161 return 0;
162}

165 if (!kv) return;
166 if (kv->K) { free(kv->K); kv->K = NULL; }
167 if (kv->V) { free(kv->V); kv->V = NULL; }
168 if (kv->scales_K) { free(kv->scales_K); kv->scales_K = NULL; }
169 if (kv->zeros_K) { free(kv->zeros_K); kv->zeros_K = NULL; }
170 if (kv->scales_V) { free(kv->scales_V); kv->scales_V = NULL; }
171 if (kv->zeros_V) { free(kv->zeros_V); kv->zeros_V = NULL; }
172
173 /* Keep metadata so double-free is harmless; caller may reuse the struct. */
174 kv->elem_count = 0;
175 kv->group_count = 0;

178int ie_kv_store_token_f32(ie_kv_cache* kv, size_t t,
179 const float* K_f32, const float* V_f32) {
180 if (!kv || !K_f32 || !V_f32) return -1;
181 if (kv->storage != IE_KV_STORAGE_F32) return -2;
182 if (t >= (size_t)kv->max_seq) return -3;
183
184 const size_t H = (size_t)kv->heads;
185 const size_t D = (size_t)kv->head_dim;
186
187 float* Kdst = (float*)kv->K;
188 float* Vdst = (float*)kv->V;
189
190 /* Copy [heads, head_dim] block for token t. */
191 for (size_t h = 0; h < H; ++h) {
192 const size_t off = payload_index(t, h, 0, H, D);
193 memcpy(Kdst + off, K_f32 + h * D, D * sizeof(float));
194 memcpy(Vdst + off, V_f32 + h * D, D * sizeof(float));
195 }
196 return 0;
197}

199int ie_kv_store_token_int8_per_tensor(ie_kv_cache* kv, size_t t,
200 const float* K_f32, const float* V_f32) {
201 if (!kv || !K_f32 || !V_f32) return -1;
202 if (kv->storage != IE_KV_STORAGE_INT8) return -2;
203 if (t >= (size_t)kv->max_seq) return -3;
204
205 const size_t H = (size_t)kv->heads;
206 const size_t D = (size_t)kv->head_dim;
207 const size_t G = kv->group_count; /* per-tensor uses g=0 */
208
209 int8_t* Kdst = (int8_t*)kv->K;
210 int8_t* Vdst = (int8_t*)kv->V;
211
212 for (size_t h = 0; h < H; ++h) {
213 const float* Ksrc = K_f32 + h * D;
214 const float* Vsrc = V_f32 + h * D;
215
216 /* Compute per-head params from min/max. */
217 float Kmn = Ksrc[0], Kmx = Ksrc[0];
218 float Vmn = Vsrc[0], Vmx = Vsrc[0];
219 for (size_t d = 1; d < D; ++d) {
220 float kvv = Ksrc[d];
221 float vvv = Vsrc[d];
222 if (kvv < Kmn) Kmn = kvv;
223 if (kvv > Kmx) Kmx = kvv;
224 if (vvv < Vmn) Vmn = vvv;
225 if (vvv > Vmx) Vmx = vvv;
226 }
227
228 ie_act_i8_params pK, pV;
229 ie_act_i8_params_from_minmax(Kmn, Kmx, kv->symmetric, &pK.scale, &pK.zero_point);
230 ie_act_i8_params_from_minmax(Vmn, Vmx, kv->symmetric, &pV.scale, &pV.zero_point);
231
232 /* Quantize into destination payload. */
233 const size_t off = payload_index(t, h, 0, H, D);
234 ie_quantize_act_int8(Ksrc, Kdst + off, D, pK, kv->symmetric);
235 ie_quantize_act_int8(Vsrc, Vdst + off, D, pV, kv->symmetric);
236
237 /* Record parameters at g=0. */
238 const size_t pi = param_index(t, h, 0, H, G);
239 kv->scales_K[pi] = pK.scale;
240 kv->zeros_K[pi] = pK.zero_point;
241 kv->scales_V[pi] = pV.scale;
242 kv->zeros_V[pi] = pV.zero_point;
243 }
244 return 0;
245}

296 const float* K_f32, const float* V_f32) {
297 if (!kv || !K_f32 || !V_f32) return -1;
298 if (kv->storage != IE_KV_STORAGE_FP8) return -2;
299 if (t >= (size_t)kv->max_seq) return -3;
300
301 const size_t H = (size_t)kv->heads;
302 const size_t D = (size_t)kv->head_dim;
303
304 uint8_t* Kdst = (uint8_t*)kv->K;
305 uint8_t* Vdst = (uint8_t*)kv->V;
306
307 /* Temporary FP8 buffers per head to avoid partial writes on error. */

309 if (!tmp) return -4;

315
316 ie_quantize_act_fp8(Ksrc, tmp, D, kv->fp8_format);
317 memcpy(Kdst + off, tmp, D * sizeof(uint8_t));
318
319 ie_quantize_act_fp8(Vsrc, tmp, D, kv->fp8_format);
320 memcpy(Vdst + off, tmp, D * sizeof(uint8_t));
321 }
322
323 free(tmp);
324 return 0;

327int ie_kv_load_token_f32(const ie_kv_cache* kv, size_t t,
328 float* K_out, float* V_out) {
329 if (!kv || !K_out || !V_out) return -1;
330
331 /* If request is out of the configured cache capacity, treat as miss. */
332 if (t >= (size_t)kv->max_seq) {
333 ie_kv_miss_add_local(1);
334 return -2;
335 }
336
337 const size_t H = (size_t)kv->heads;
338 const size_t D = (size_t)kv->head_dim;
339
340 if (kv->storage == IE_KV_STORAGE_F32) {
341 const float* Ksrc = (const float*)kv->K;
342 const float* Vsrc = (const float*)kv->V;
343 for (size_t h = 0; h < H; ++h) {
344 const size_t off = payload_index(t, h, 0, H, D);
345 memcpy(K_out + h * D, Ksrc + off, D * sizeof(float));
346 memcpy(V_out + h * D, Vsrc + off, D * sizeof(float));
347 }
348 ie_kv_hit_add_local(1);
349 return 0;
350 }
351
352 if (kv->storage == IE_KV_STORAGE_INT8) {
353 const int8_t* Kq = (const int8_t*)kv->K;
354 const int8_t* Vq = (const int8_t*)kv->V;
355 const size_t G = kv->group_count;
356 const size_t Gs = kv->group_size;
357
358 for (size_t h = 0; h < H; ++h) {
359 const size_t off = payload_index(t, h, 0, H, D);
360
361 /* Dequant K */
362 for (size_t d = 0; d < D; ++d) {
363 const size_t g = d / Gs;
364 const size_t pi = param_index(t, h, g, H, G);
365 const float s = kv->scales_K[pi];
366 const int z = (int)kv->zeros_K[pi];
367 K_out[h * D + d] = s * ((int)Kq[off + d] - z);
368 }
369
370 /* Dequant V */
371 for (size_t d = 0; d < D; ++d) {
372 const size_t g = d / Gs;
373 const size_t pi = param_index(t, h, g, H, G);
374 const float s = kv->scales_V[pi];
375 const int z = (int)kv->zeros_V[pi];
376 V_out[h * D + d] = s * ((int)Vq[off + d] - z);
377 }
378 }
379 ie_kv_hit_add_local(1);
380 return 0;
381 }
382
383 if (kv->storage == IE_KV_STORAGE_FP8) {
384 const uint8_t* K8 = (const uint8_t*)kv->K;
385 const uint8_t* V8 = (const uint8_t*)kv->V;
386
387 for (size_t h = 0; h < H; ++h) {
388 const size_t off = payload_index(t, h, 0, H, D);
389 ie_dequantize_act_fp8(K8 + off, K_out + h * D, D, kv->fp8_format);
390 ie_dequantize_act_fp8(V8 + off, V_out + h * D, D, kv->fp8_format);
391 }
392 ie_kv_hit_add_local(1);
393 return 0;
394 }
395
396 /* Unknown storage: treat as miss. */
397 ie_kv_miss_add_local(1);
398 return -3;
399}
