2"""
3Build a dedup/quant source bin+json from:
4 - a base IE manifest/bin (model.ie.json + model.ie.bin)
5 - a dedup pairing manifest (manifest.dedup.json)
6
7It will:
8 1) Copy all tensors referenced by kv_pair groups into a compact output bin.
9 2) Ensure fused expert tensors referenced by expert_pair groups exist in the output.
10 If missing in the base IE, it will synthesize them from per-expert weight tensors by:
11 - symmetric int4 quantization (group-wise, default group size 64)
12 - writing int4 blocks (packed nibbles) and per-group scales
13
14This script does not require torch.
15"""

62def _parse_ie(ie_path: str) -> Tuple[str, List[Dict[str, Any]]]:
63 ie = _load_json(ie_path)
64 weights_bin = ie.get("weights_bin") or ie.get("bin") or "model.ie.bin"
65 tensors = ie.get("tensors")
66 if tensors is None:
67 raise SystemExit("ERROR: IE json missing 'tensors'")
68
69 out: List[Dict[str, Any]] = []
70 if isinstance(tensors, list):
71 for t in tensors:
72 if isinstance(t, dict):
73 out.append(dict(t))
74 elif isinstance(tensors, dict):
75 for name, meta in tensors.items():
76 if isinstance(meta, dict):
77 d = dict(meta)
78 d.setdefault("name", name)
79 out.append(d)
80 else:
81 raise SystemExit(f"ERROR: unsupported tensors container type: {type(tensors)}")
82
83 norm: List[Dict[str, Any]] = []
84 for t in out:
85 name = t.get("name")
86 dtype = t.get("dtype")
87 shape = t.get("shape")
88 offset = t.get("offset")
89 nbytes = t.get("nbytes")
90 if not isinstance(name, str) or not name:
91 continue
92 if not isinstance(dtype, str) or not dtype:
93 raise SystemExit(f"ERROR: tensor '{name}' missing dtype")
94 if not isinstance(shape, list) or not all(isinstance(x, int) for x in shape):
95 raise SystemExit(f"ERROR: tensor '{name}' has invalid shape")
96 if not isinstance(offset, int):
97 raise SystemExit(f"ERROR: tensor '{name}' missing offset")
98 if not isinstance(nbytes, int):
99 npdt = _DTYPE_TO_NP.get(dtype)
100 if npdt is None:
101 raise SystemExit(f"ERROR: tensor '{name}' has unknown dtype '{dtype}' and missing nbytes")
102 esz = int(np.dtype(npdt).itemsize)
103 elems = 1
104 for d in shape:
105 elems *= int(d)
106 nbytes = elems * esz
107
108 norm.append(
109 {
110 "name": name,
111 "dtype": dtype,
112 "shape": shape,
113 "offset": int(offset),
114 "nbytes": int(nbytes),
115 }
116 )
117
118 norm.sort(key=lambda x: (x["offset"], x["name"]))
119 return weights_bin, norm

133def _read_weight_matrix_f32(bin_path: str, t: Dict[str, Any]) -> np.ndarray:
134 shape = t["shape"]
135 if len(shape) != 2:
136 raise SystemExit(f"ERROR: expected 2D weight for '{t['name']}', got shape={shape}")
137 raw = _read_tensor_bytes(bin_path, t)
138 dtype = t["dtype"]
139 npdt = _DTYPE_TO_NP.get(dtype)
140 if npdt is None:
141 raise SystemExit(f"ERROR: unsupported dtype '{dtype}' for '{t['name']}'")
142 if dtype in ("torch.bfloat16", "BF16"):
143 u16 = np.frombuffer(raw, dtype=np.uint16).reshape(shape)
144 return _bf16_to_f32(u16)
145 arr = np.frombuffer(raw, dtype=npdt).reshape(shape)
146 return arr.astype(np.float32, copy=False)

149def _pack_int4_nibbles(q: np.ndarray) -> np.ndarray:
150 if q.dtype != np.uint8:
151 q = q.astype(np.uint8, copy=False)
152 if q.shape[-1] % 2 == 1:
153 pad = np.zeros(q.shape[:-1] + (1,), dtype=np.uint8)
154 q = np.concatenate([q, pad], axis=-1)
155 lo = q[..., 0::2]
156 hi = q[..., 1::2]
157 return (lo | (hi << 4)).astype(np.uint8, copy=False)

160def _quant_int4_groupwise(w: np.ndarray, group_size: int) -> Tuple[np.ndarray, np.ndarray]:
161 if w.dtype != np.float32:
162 w = w.astype(np.float32, copy=False)
163 rows, cols = w.shape
164 groups = int(math.ceil(cols / group_size))
165 pad_cols = groups * group_size
166 if pad_cols != cols:
167 wpad = np.zeros((rows, pad_cols), dtype=np.float32)
168 wpad[:, :cols] = w
169 w = wpad
170 cols = pad_cols
171
172 w3 = w.reshape(rows, groups, group_size)
173 absmax = np.max(np.abs(w3), axis=2)
174 scales = absmax / 7.0
175 scales = np.where(scales == 0.0, 1.0, scales).astype(np.float32)
176
177 q = np.rint(w3 / scales[:, :, None]).astype(np.int32)
178 q = np.clip(q, -7, 7).astype(np.int16)
179 q_u8 = (q + 8).astype(np.uint8)
180
181 packed = _pack_int4_nibbles(q_u8)
182 packed = packed.reshape(rows, -1)
183 return packed, scales

190def _encode_scales_log2_u8_q3(scales_f32: np.ndarray, bias: int = 128, step: float = 0.125) -> bytes:
191 s = np.maximum(scales_f32, np.float32(2 ** -30))
192 l = np.log2(s) / np.float32(step)
193 code = np.rint(l).astype(np.int32) + int(bias)
194 code = np.clip(code, 0, 255).astype(np.uint8)
195 return code.tobytes(order="C")

207def _manifest_required_tensors(manifest: Dict[str, Any]) -> Tuple[List[str], List[str]]:
208 groups = manifest.get("groups", manifest)
209 if not isinstance(groups, list):
210 raise SystemExit("ERROR: manifest is neither a list nor an object with 'groups'")
211
212 kv: List[str] = []
213 fused: List[str] = []
214 for g in groups:
215 if not isinstance(g, dict):
216 continue
217 if g.get("kind") == "kv_pair":
218 members = g.get("members")
219 if isinstance(members, list):
220 for m in members:
221 if isinstance(m, str):
222 kv.append(m)
223 elif g.get("kind") == "expert_pair":
224 ft = g.get("fused_tensor")
225 if isinstance(ft, str):
226 fused.append(ft)
227 return sorted(set(kv)), sorted(set(fused))

230def _infer_expert_weight_names(fused_tensor: str, expert_count: int) -> Tuple[str, List[str]]:
231 base_prefix, last = fused_tensor.rsplit(".", 1)
232 if last.endswith("_scales"):
233 proj = last[: -len("_scales")]
234 elif last.endswith("_blocks"):
235 proj = last[: -len("_blocks")]
236 else:
237 raise SystemExit(f"ERROR: cannot infer per-expert weights from '{fused_tensor}'")
238
239 per = [f"{base_prefix}.{e}.{proj}.weight" for e in range(expert_count)]
240 return proj, per

243def _discover_expert_count(all_names: List[str], fused_tensor: str) -> int:
244 base_prefix, last = fused_tensor.rsplit(".", 1)
245 if last.endswith("_scales"):
246 proj = last[: -len("_scales")]
247 elif last.endswith("_blocks"):
248 proj = last[: -len("_blocks")]
249 else:
250 raise SystemExit(f"ERROR: cannot infer expert_count from '{fused_tensor}'")
251
252 pat = re.compile(re.escape(base_prefix) + r"\.(\d+)\." + re.escape(proj) + r"\.weight$")
253 idxs: List[int] = []
254 for name in all_names:
255 m = pat.match(name)
256 if m:
257 idxs.append(int(m.group(1)))
258 if not idxs:
259 raise SystemExit(f"ERROR: no per-expert weights found for '{fused_tensor}'")
260 max_idx = max(idxs)
261 missing = [i for i in range(max_idx + 1) if i not in set(idxs)]
262 if missing:
263 raise SystemExit(f"ERROR: missing expert indices for '{fused_tensor}': missing={missing[:10]} total_missing={len(missing)}")
264 return max_idx + 1

267def _build_dedup_source(
268 base_ie_json: str,
269 base_bin_path: str,
270 manifest_path: str,
271 out_bin_path: str,
272 out_ie_json_path: str,
273 group_size: int,
274 scale_encoding: str,
275 include_blocks: bool,
276) -> None:
277 weights_bin_rel, tensors = _parse_ie(base_ie_json)
278 base_dir = os.path.dirname(os.path.abspath(base_ie_json))
279 base_bin = base_bin_path or os.path.join(base_dir, weights_bin_rel)
280
281 by_name: Dict[str, Dict[str, Any]] = {t["name"]: t for t in tensors}
282 man = _load_json(manifest_path)
283 kv_names, fused_names = _manifest_required_tensors(man)
284
285 out_tensors: List[OutTensor] = []
286 cursor = 0
287
288 os.makedirs(os.path.dirname(os.path.abspath(out_bin_path)), exist_ok=True)
289 with open(out_bin_path, "wb") as out_f:
290 for name in kv_names:
291 t = by_name.get(name)
292 if t is None:
293 raise SystemExit(f"ERROR: kv tensor '{name}' not found in base IE json")
294 blob = _read_tensor_bytes(base_bin, t)
295 out_f.write(blob)
296 out_tensors.append(OutTensor(name=name, dtype=t["dtype"], shape=list(t["shape"]), offset=cursor, nbytes=len(blob)))
297 cursor += len(blob)
298
299 all_names = list(by_name.keys())
300 for fused in fused_names:
301 if fused in by_name:
302 t = by_name[fused]
303 blob = _read_tensor_bytes(base_bin, t)
304 out_f.write(blob)
305 out_tensors.append(OutTensor(name=fused, dtype=t["dtype"], shape=list(t["shape"]), offset=cursor, nbytes=len(blob)))
306 cursor += len(blob)
307 continue
308
309 expert_count = _discover_expert_count(all_names, fused)
310 proj, per_names = _infer_expert_weight_names(fused, expert_count)
311
312 blocks_list: List[np.ndarray] = []
313 scales_list: List[np.ndarray] = []
314 rows = None
315 cols = None
316
317 for pn in per_names:
318 wt = by_name.get(pn)
319 if wt is None:
320 raise SystemExit(f"ERROR: per-expert weight '{pn}' not found (needed for '{fused}')")
321 w = _read_weight_matrix_f32(base_bin, wt)
322 if rows is None:
323 rows, cols = w.shape
324 elif w.shape != (rows, cols):
325 raise SystemExit(f"ERROR: shape mismatch '{pn}': {w.shape} vs {(rows, cols)}")
326 b, s = _quant_int4_groupwise(w, group_size=group_size)
327 blocks_list.append(b)
328 scales_list.append(s)
329
330 blocks_fused = np.stack(blocks_list, axis=0)
331 scales_fused = np.stack(scales_list, axis=0)
332
333 base_prefix, last = fused.rsplit(".", 1)
334 if last.endswith("_scales"):
335 fused_scales_name = fused
336 fused_blocks_name = f"{base_prefix}.{proj}_blocks"
337 elif last.endswith("_blocks"):
338 fused_blocks_name = fused
339 fused_scales_name = f"{base_prefix}.{proj}_scales"
340 else:
341 raise SystemExit(f"ERROR: unsupported fused name '{fused}'")
342
343 if scale_encoding == "fp16":
344 scales_bytes = _encode_scales_fp16(scales_fused)
345 scales_dtype = "torch.float16"
346 elif scale_encoding == "log2_u8_q3":
347 scales_bytes = _encode_scales_log2_u8_q3(scales_fused)
348 scales_dtype = "torch.uint8"
349 else:
350 raise SystemExit(f"ERROR: unknown scale encoding '{scale_encoding}'")
351
352 out_f.write(scales_bytes)
353 out_tensors.append(
354 OutTensor(
355 name=fused_scales_name,
356 dtype=scales_dtype,
357 shape=[expert_count, int(rows), int(scales_fused.shape[2])],
358 offset=cursor,
359 nbytes=len(scales_bytes),
360 )
361 )
362 cursor += len(scales_bytes)
363
364 if include_blocks:
365 blocks_bytes = blocks_fused.astype(np.uint8).tobytes(order="C")
366 out_f.write(blocks_bytes)
367 out_tensors.append(
368 OutTensor(
369 name=fused_blocks_name,
370 dtype="torch.uint8",
371 shape=[expert_count, int(rows), int(blocks_fused.shape[2])],
372 offset=cursor,
373 nbytes=len(blocks_bytes),
374 )
375 )
376 cursor += len(blocks_bytes)
377
378 out_ie = {
379 "version": 1,
380 "dtype": "mixed",
381 "weights_bin": os.path.basename(out_bin_path),
382 "tensors": [
383 {"name": t.name, "dtype": t.dtype, "shape": t.shape, "offset": t.offset, "nbytes": t.nbytes}
384 for t in out_tensors
385 ],
386 "quant": {
387 "int4": {
388 "group_size": int(group_size),
389 "scale_encoding": scale_encoding,
390 "pack": "nibble_lohi",
391 "range": [-7, 7],
392 }
393 },
394 }
395 _save_json(out_ie_json_path, out_ie)
396
397 print(f"Wrote: {out_bin_path}")
398 print(f"Wrote: {out_ie_json_path}")
399 print(f"Tensors: {len(out_tensors)} Bytes: {sum(t.nbytes for t in out_tensors)}")
