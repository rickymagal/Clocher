23def write_f32_bin(path: Path, rows: int, cols: int):
24 """Write a simple 2D ramp into a float32 .bin file (row-major)."""
25 n = rows * cols
26 with open(path, "wb") as f:
27 for i in range(n):
28 # spread across a moderate range to avoid all-zero or saturations
29 val = (i - n / 2) / 123.0
30 f.write(struct.pack("<f", float(val)))
