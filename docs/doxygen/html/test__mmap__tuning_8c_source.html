1/* File: tests/c/test_mmap_tuning.c
2 * -----------------------------------------------------------------------------
3 * @file test_mmap_tuning.c
4 * @brief Sanity tests for mmap + paging advice + soft prefetch walks.
5 *
6 * This test is intentionally self-contained and relies only on POSIX APIs,
7 * so it can be built alongside the engine's mmap tuning modules without
8 * depending on their internal headers. It exercises the following:
9 *
10 * 1) Creates a temporary file and fills it with a deterministic byte pattern.
11 * 2) Maps the file read-only with mmap(2).
12 * 3) Issues paging hints (madvise/posix_madvise) for SEQUENTIAL and RANDOM.
13 * 4) Performs a soft prefetch walk (touch each page with a configurable stride).
14 * 5) Verifies a rolling checksum to ensure mapping correctness.
15 *
16 * The program exits with code 0 on success and non-zero on any failure.
17 */

81static int make_temp_file(size_t nbytes, int *out_fd, char *out_path, size_t out_cap) {
82 if (!out_fd) return -1;
83 char tmpl[] = "/tmp/mmap_tuning_test.XXXXXX";
84 int fd = mkstemp(tmpl);
85 if (fd < 0) {
86 perror("mkstemp");
87 return -1;
88 }
89
90 /* Unlink immediately; the file will be removed when closed. */
91 if (unlink(tmpl) != 0) {
92 perror("unlink");
93 close(fd);
94 return -1;
95 }
96
97 /* Pre-allocate and write pattern in 1 MiB chunks. */
98 const size_t chunk = 1u * 1024u * 1024u;
99 uint8_t *buf = (uint8_t *)malloc(chunk);
100 if (!buf) {
101 perror("malloc");
102 close(fd);
103 return -1;
104 }
105
106 size_t written = 0;
107 while (written < nbytes) {
108 size_t todo = (nbytes - written) < chunk ? (nbytes - written) : chunk;
109 for (size_t i = 0; i < todo; ++i) {
110 size_t idx = written + i;
111 buf[i] = (uint8_t)((idx * 131u) ^ 0xA5u);
112 }
113 ssize_t put = write(fd, buf, todo);
114 if (put < 0) {
115 perror("write");
116 free(buf);
117 close(fd);
118 return -1;
119 }
120 if ((size_t)put != todo) {
121 fprintf(stderr, "short write: %zd != %zu\n", put, todo);
122 free(buf);
123 close(fd);
124 return -1;
125 }
126 written += (size_t)put;
127 }
128 free(buf);
129
130 if (out_path && out_cap > 0) {
131 /* Not the real on-disk path (since unlinked), but return the template for logs. */
132 snprintf(out_path, out_cap, "%s", "(anonymous tmpfile)");
133 }
134
135 *out_fd = fd;
136 return 0;
137}

143static uint64_t checksum_bytes(const uint8_t *p, size_t n) {
144 uint64_t s0 = 1469598103934665603ull; /* FNV offset basis */
145 uint64_t s1 = 1099511628211ull; /* FNV prime (lower) */
146 for (size_t i = 0; i < n; ++i) {
147 s0 ^= p[i];
148 s0 *= s1;
149 }
150 return s0;
151}

161static int verify_pattern(const uint8_t *p, size_t n, uint64_t *outcs) {
162 uint64_t cs = checksum_bytes(p, n);
163 if (outcs) *outcs = cs;
164
165 /* Spot-check a few positions to ensure mapping content is correct. */
166 size_t probes[5] = {0, n / 7, n / 5, n / 3, n - 1};
167 for (int i = 0; i < 5; ++i) {
168 size_t idx = probes[i];
169 if (idx >= n) continue;
170 uint8_t want = (uint8_t)((idx * 131u) ^ 0xA5u);
171 if (p[idx] != want) {
172 fprintf(stderr, "mismatch at %zu: got 0x%02x, want 0x%02x\n",
173 idx, (unsigned)p[idx], (unsigned)want);
174 return -1;
175 }
176 }
177 return 0;
178}

195static int exercise_advice_and_walk(const uint8_t *base, size_t n, size_t stride) {
196 /* Try madvise first; fall back to posix_madvise if unavailable. */
197 int have_madvise = 1;
198#if !defined(MADV_RANDOM) || !defined(MADV_SEQUENTIAL)
199 have_madvise = 0;
200#endif
201
202 if (have_madvise) {
203 if (madvise((void *)base, n, MADV_SEQUENTIAL) != 0) {
204 /* Not fatal; some kernels refuse advice on tmpfs. */
205 fprintf(stderr, "[warn] madvise(SEQUENTIAL) failed: %s\n", strerror(errno));
206 }
207 } else {
208#if defined(POSIX_MADV_SEQUENTIAL)
209 if (posix_madvise((void *)base, n, POSIX_MADV_SEQUENTIAL) != 0) {
210 fprintf(stderr, "[warn] posix_madvise(SEQUENTIAL) failed\n");
211 }
212#endif
213 }
214
215 /* Soft prefetch walk: touch one byte per page/stride. */
216 volatile uint8_t sink = 0;
217 for (size_t i = 0; i < n; i += stride) {
218 sink ^= base[i];
219 }
220
221 if (have_madvise) {
222 if (madvise((void *)base, n, MADV_RANDOM) != 0) {
223 fprintf(stderr, "[warn] madvise(RANDOM) failed: %s\n", strerror(errno));
224 }
225 } else {
226#if defined(POSIX_MADV_RANDOM)
227 if (posix_madvise((void *)base, n, POSIX_MADV_RANDOM) != 0) {
228 fprintf(stderr, "[warn] posix_madvise(RANDOM) failed\n");
229 }
230#endif
231 }
232
233 /* Re-verify a checksum after the walk to ensure content is still valid. */
234 (void)sink;
235 uint64_t cs = 0;
236 if (verify_pattern(base, n, &cs) != 0) return -1;
237 fprintf(stderr, "[info] checksum after walk: 0x%016" PRIx64 "\n", cs);
238 return 0;
239}

254int main(void) {

256 int fd = -1;
257 void *addr = NULL;
258
259 const size_t total = cfg_bytes();
260 const long pagesz_l = sysconf(_SC_PAGESIZE);
261 const size_t pagesz = (pagesz_l > 0) ? (size_t)pagesz_l : 4096u;
262 const size_t stride = cfg_stride(pagesz);
263
264 fprintf(stderr, "[test] bytes=%zu, pagesz=%zu, stride=%zu\n",
265 total, pagesz, stride);
266
267 if (make_temp_file(total, &fd, NULL, 0) != 0) {
268 fprintf(stderr, "FAIL: could not create/write temp file\n");
269 goto done;
270 }
271
272 /* Map read-only, private. */
273 addr = mmap(NULL, total, PROT_READ, MAP_PRIVATE, fd, 0);
274 if (addr == MAP_FAILED) {
275 perror("mmap");
276 goto done;
277 }
278
279 /* Initial checksum + pattern verification. */
280 uint64_t cs0 = 0;
281 if (verify_pattern((const uint8_t *)addr, total, &cs0) != 0) {
282 fprintf(stderr, "FAIL: pattern mismatch after initial mmap\n");
283 goto done;
284 }
285 fprintf(stderr, "[info] checksum initial: 0x%016" PRIx64 "\n", cs0);
286
287 /* Exercise advice + soft prefetch walk. */
288 if (exercise_advice_and_walk((const uint8_t *)addr, total, stride) != 0) {
289 fprintf(stderr, "FAIL: exercise_advice_and_walk failed\n");
290 goto done;
291 }
292
293 fprintf(stderr, "[pass] mmap tuning test completed successfully\n");
294 rc = 0;
295
296done:
297 if (addr && addr != MAP_FAILED) munmap(addr, total);
298 if (fd >= 0) close(fd);
299 return rc;
300}
