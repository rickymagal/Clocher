3  * @file test_mmap_tuning.c
4  * @brief Sanity tests for mmap + paging advice + soft prefetch walks.
5  *
6  * This test is intentionally self-contained and relies only on POSIX APIs,
7  * so it can be built alongside the engine's mmap tuning modules without
8  * depending on their internal headers. It exercises the following:

10  * 1) Creates a temporary file and fills it with a deterministic byte pattern.

12  * 3) Issues paging hints (madvise/posix_madvise) for SEQUENTIAL and RANDOM.
13  * 4) Performs a soft prefetch walk (touch each page with a configurable stride).
14  * 5) Verifies a rolling checksum to ensure mapping correctness.
15  *
16  * The program exits with code 0 on success and non-zero on any failure.
17  */

97  /* Pre-allocate and write pattern in 1 MiB chunks. */
98  const size_t chunk = 1u * 1024u * 1024u;
99  uint8_t *buf = (uint8_t *)malloc(chunk);

104  }

109  for (size_t i = 0; i < todo; ++i) {

122  free(buf);

143 static uint64_t checksum_bytes(const uint8_t *p, size_t n) {
144  uint64_t s0 = 1469598103934665603ull; /* FNV offset basis */
145  uint64_t s1 = 1099511628211ull; /* FNV prime (lower) */
146  for (size_t i = 0; i < n; ++i) {
147  s0 ^= p[i];
148  s0 *= s1;

161 static int verify_pattern(const uint8_t *p, size_t n, uint64_t *outcs) {
162  uint64_t cs = checksum_bytes(p, n);
163  if (outcs) *outcs = cs;

166  size_t probes[5] = {0, n / 7, n / 5, n / 3, n - 1};
167  for (int i = 0; i < 5; ++i) {
168  size_t idx = probes[i];

195 static int exercise_advice_and_walk(const uint8_t *base, size_t n, size_t stride) {
196  /* Try madvise first; fall back to posix_madvise if unavailable. */
197  int have_madvise = 1;
198 #if !defined(MADV_RANDOM) || !defined(MADV_SEQUENTIAL)
199  have_madvise = 0;
200 #endif
201 
202  if (have_madvise) {
203  if (madvise((void *)base, n, MADV_SEQUENTIAL) != 0) {
204  /* Not fatal; some kernels refuse advice on tmpfs. */
205  fprintf(stderr, "[warn] madvise(SEQUENTIAL) failed: %s\n", strerror(errno));
206  }
207  } else {
208 #if defined(POSIX_MADV_SEQUENTIAL)
209  if (posix_madvise((void *)base, n, POSIX_MADV_SEQUENTIAL) != 0) {
210  fprintf(stderr, "[warn] posix_madvise(SEQUENTIAL) failed\n");
211  }
212 #endif
213  }
214 
215  /* Soft prefetch walk: touch one byte per page/stride. */
216  volatile uint8_t sink = 0;
217  for (size_t i = 0; i < n; i += stride) {
218  sink ^= base[i];
219  }
220 
221  if (have_madvise) {
222  if (madvise((void *)base, n, MADV_RANDOM) != 0) {
223  fprintf(stderr, "[warn] madvise(RANDOM) failed: %s\n", strerror(errno));
224  }
225  } else {
226 #if defined(POSIX_MADV_RANDOM)
227  if (posix_madvise((void *)base, n, POSIX_MADV_RANDOM) != 0) {
228  fprintf(stderr, "[warn] posix_madvise(RANDOM) failed\n");
229  }
230 #endif
231  }
232 
233  /* Re-verify a checksum after the walk to ensure content is still valid. */
234  (void)sink;
235  uint64_t cs = 0;
236  if (verify_pattern(base, n, &cs) != 0) return -1;
237  fprintf(stderr, "[info] checksum after walk: 0x%016" PRIx64 "\n", cs);
238  return 0;
239 }
