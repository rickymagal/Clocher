37def _pack_int4_from_int8(q_int8: np.ndarray) -> bytes:
38 rows, cols = q_int8.shape
39 if cols % 2 == 1:
40 q_int8 = np.pad(q_int8, ((0, 0), (0, 1)), mode="constant")
41 cols += 1
42 q_u = q_int8.astype(np.int16)
43 lo = (q_u[:, 0::2] & 0xF).astype(np.uint8)
44 hi = (q_u[:, 1::2] & 0xF).astype(np.uint8)
45 return ((hi << 4) | lo).tobytes()
