2# Streamed INT4 (weight-only, per-row) packer for huge HF checkpoints
3# - Writes packed INT4 bytes + FP16 scales incrementally (low RAM)
4# - Emits expanded manifest mapping each exact key to the INT4 spec

6# python3 scripts/gen_q4_bytes_stream.py \

10# --manifest quant/q4_manifest.expanded.json \

27# Filters (mesmos do teu pipeline)
28INC_W = re.compile(r".*\.weight$")
29EXC = [re.compile(x) for x in [

38def wanted(k: str) -> bool:
39 if not INC_W.match(k): return False
40 for rx in EXC:
41 if rx.match(k): return False
42 return True

44def pack_int4_rows(q_int8: np.ndarray) -> bytes:
45 # q_int8 shape: (rows, cols) in [-8..7]
46 rows, cols = q_int8.shape
47 if cols % 2:
48 # pad 1 coluna p/ emparelhar nibble
49 q_int8 = np.pad(q_int8, ((0,0),(0,1)), mode="constant")
50 cols += 1
51 # Usa 0xF para pegar nibble em complemento de dois
52 q_u = q_int8.astype(np.int16)
53 lo = (q_u[:, 0::2] & 0xF).astype(np.uint8)
54 hi = (q_u[:, 1::2] & 0xF).astype(np.uint8)
55 packed = (hi << 4) | lo # shape (rows, cols//2)
56 return packed.tobytes()

78 manifest = {}

83
84 for shard in shards:

86 for k, W in sd.items():
87 if not (wanted(k) and isinstance(W, torch.Tensor) and W.ndim == 2 and W.is_floating_point()):
88 continue

90 W = W.float() # garante FP32
91 rows, cols = W.shape
92 rows_done = 0
93
94 # Manifest entry para ESTA key
95 manifest[k] = {
96 "dtype": "int4",
97 "scheme": "weight_only",
98 "per": "row",
99 "bits": 4,
100 "pack": "nibble_lohi",
101 "scale_dtype": "fp16",
102 "zero_point": 0,
103 "symmetric": True,
104 "align": 256,
105 "rows": rows,
106 "cols": cols,
107 "packed_bin": str(q4_path),
108 "scale_bin": str(sc_path),
109 }
110
111 with torch.no_grad():
112 step = max(1, args.chunk_rows)
113 for start in range(0, rows, step):
114 end = min(start + step, rows)
115 chunk = W[start:end, :] # [r, cols]
116
117 # scales per-row: max(|w|)/7 -> range [-7..7]
118 maxabs = torch.amax(torch.abs(chunk), dim=1).clamp_min(1e-12) # [r]
119 scales = (maxabs / 7.0).cpu().numpy().astype(np.float16) # [r]
120 fs.write(scales.tobytes()); bytes_s += scales.nbytes
121
122 inv = (7.0 / maxabs).unsqueeze(1) # [r,1]
123 q = torch.round(chunk * inv).clamp_(-8, 7).to(torch.int8).cpu().numpy() # [r, cols]
124
125 fq.write(pack_int4_rows(q)); bytes_q += ( (q.shape[1] + (q.shape[1]&1)) // 2 ) * q.shape[0]
126 rows_done += (end - start)
127 total_rows += (end - start)
128
129 total_keys += 1
130 # solta memória cedo
131 del W
132 del sd
133 gc.collect()
134
135 fq.flush(); fs.flush()
136 fq.close(); fs.close()
137

139 json.dump(manifest, f, indent=2)

142 print(f"[ok] wrote manifest -> {man_path.resolve()}")
143 print(f"[ok] q4 bytes -> {q4_path}")
144 print(f"[ok] scales -> {sc_path}")
