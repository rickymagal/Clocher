106ie_device_kind_t ie_device_kind_from_str(const char *s) {
107 if (!s || !*s) return IE_DEV_CPU;
108 char buf[16] = {0};
109 size_t n = strlen(s);
110 if (n >= sizeof(buf)) n = sizeof(buf) - 1;
111 for (size_t i = 0; i < n; ++i) buf[i] = (char)tolower((unsigned char)s[i]);
112 if (strcmp(buf, "cpu") == 0) return IE_DEV_CPU;
113 if (strcmp(buf, "cuda") == 0) return IE_DEV_CUDA;
114 if (strcmp(buf, "ze") == 0) return IE_DEV_ZE;
115 return IE_DEV_CPU;
116}

377int ie_device_create(ie_device_kind_t kind, ie_device_t **out_dev) {
378 if (!out_dev) return -1;
379
380 ie_device_t *d = (ie_device_t*)calloc(1, sizeof(*d));
381 if (!d) return -2;
382
383 int ok = -1;
384 if (kind == IE_DEV_CUDA) {
385 ok = cuda_try_create(&d->impl, &d->vt);
386 } else if (kind == IE_DEV_ZE) {
387 ok = ze_try_create(&d->impl, &d->vt);
388 }
389
390 if (ok != 0) {
391 /* Fallback to CPU */
392 cpu_impl_t *ci = cpu_new();
393 if (!ci) { free(d); return -3; }
394 d->impl = ci;
395 d->vt.caps = cpu_caps;
396 d->vt.gemv_f32 = cpu_gemv_f32;
397 d->vt.memcpy = cpu_memcpy;
398 d->vt.destroy = cpu_destroy;
399 d->kind = IE_DEV_CPU;
400 } else {
401 d->kind = kind;
402 }
403
404 *out_dev = d;
405 return 0;
406}

444int ie_device_gemv_f32(ie_device_t *dev,
445 const float *W, const float *x, float *y,
446 size_t rows, size_t cols,
447 const float *bias, size_t blk_k) {
448 if (!dev || !dev->vt.gemv_f32) return -1;
449 int rc = dev->vt.gemv_f32(dev->impl, W, x, y, rows, cols, bias, blk_k);
450 if (rc == 0) return 0;
451
452 /* If GPU backend is stubbed, transparently fall back to CPU once. */
453 if (dev->kind != IE_DEV_CPU) {
454 ie_device_t *cpu = NULL;
455 if (ie_device_create(IE_DEV_CPU, &cpu) == 0) {
456 int rc2 = ie_device_gemv_f32(cpu, W, x, y, rows, cols, bias, blk_k);
457 ie_device_destroy(cpu);
458 return rc2;
459 }
460 }
461 return rc;
462}
