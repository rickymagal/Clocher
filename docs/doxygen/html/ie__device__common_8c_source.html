233static int cuda_gemv_f32_thunk(void *self,
234 const float *W,
235 const float *x,
236 float *y,
237 size_t rows,
238 size_t cols,
239 const float *bias,
240 size_t blk_k) {
241 (void)blk_k;
242
243 cuda_impl_t *ci = (cuda_impl_t*)self;
244 if (!ci || !W || !x || !y || rows == 0 || cols == 0) return -1;
245
246 const size_t W_bytes = rows * cols * sizeof(float);
247 const size_t x_bytes = cols * sizeof(float);
248 const size_t y_bytes = rows * sizeof(float);
249 const size_t b_bytes = bias ? (rows * sizeof(float)) : 0;
250
251 if (cuda_ensure_capacity(&ci->dW, &ci->dW_cap, W_bytes) != 0) return -2;
252 if (cuda_ensure_capacity(&ci->dx, &ci->dx_cap, x_bytes) != 0) return -3;
253 if (cuda_ensure_capacity(&ci->dy, &ci->dy_cap, y_bytes) != 0) return -4;
254 if (bias) {
255 if (cuda_ensure_capacity(&ci->dbias, &ci->dbias_cap, b_bytes) != 0) return -5;
256 }
257
258 if (ie_cuda_memcpy(ci->dW, W, W_bytes, IE_CUDA_COPY_H2D) != 0) return -6;
259 if (ie_cuda_memcpy(ci->dx, x, x_bytes, IE_CUDA_COPY_H2D) != 0) return -7;
260 if (bias) {
261 if (ie_cuda_memcpy(ci->dbias, bias, b_bytes, IE_CUDA_COPY_H2D) != 0) return -8;
262 }
263
264 if (ie_cuda_gemv_f32((const float*)ci->dW,
265 (const float*)ci->dx,
266 (float*)ci->dy,
267 rows,
268 cols,
269 bias ? (const float*)ci->dbias : NULL) != 0) {
270 return -9;
271 }
272
273 if (ie_cuda_memcpy(y, ci->dy, y_bytes, IE_CUDA_COPY_D2H) != 0) return -10;
274 return 0;

546ie_device_kind_t ie_device_kind_from_str(const char *s) {
547 if (!s) return IE_DEV_CPU;
548
549 char tmp[16] = {0};
550 size_t n = strlen(s);
551 if (n >= sizeof(tmp)) n = sizeof(tmp) - 1;
552
553 for (size_t i = 0; i < n; ++i) {
554 tmp[i] = (char)tolower((unsigned char)s[i]);
555 }
556
557 if (!strcmp(tmp, "cpu")) return IE_DEV_CPU;
558 if (!strcmp(tmp, "cuda")) return IE_DEV_CUDA;
559 if (!strcmp(tmp, "ze")) return IE_DEV_ZE;
560
561 return IE_DEV_CPU;
562}
