113 __device__ inline float ie_decode_fp8_e4m3_u8(uint8_t v) {
114  if (v == 0u) return 0.0f;
115  const uint8_t sign = (v >> 7) & 0x1;
116  const uint8_t exp = (v >> 3) & 0xF;
117  const uint8_t man = (v & 0x7);
118  if (exp == 0) return sign ? -0.0f : 0.0f;
119  const int bias = 7;
120  const int e = (int)exp - bias;
121  const float frac = (float)man / 8.0f;
122  const float val = (1.0f + frac) * __int2float_rn(1 << e);
123  return sign ? -val : val;
124 }

131 __device__ inline float ie_decode_fp8_e5m2_u8(uint8_t v) {
132  const uint8_t sign = (v >> 7) & 0x1;
133  const uint8_t exp = (v >> 2) & 0x1F;
134  const uint8_t man = (v & 0x3);
135  if (exp == 0) return sign ? -0.0f : 0.0f;
136  if (exp == 0x1F) {
