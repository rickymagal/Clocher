177__device__ __forceinline__ int8_t unpack_int4(const uint8_t byte, const int hi) {
178 const uint8_t v = hi ? (byte >> 4) : (byte & 0x0F);
179 const int8_t s = (v & 0x08) ? (int8_t)(v | 0xF0) : (int8_t)v;
180 return s;
181}

269__global__ void pack_w_blockedk_f32_kernel(
270 float *out,
271 const float *w,
272 int rows,
273 int cols,
274 int block_k
275) {
276 const int idx = (int)blockIdx.x * (int)blockDim.x + (int)threadIdx.x;
277 const int total = rows * cols;
278 if (idx >= total) return;
279
280 const int r = idx / cols;
281 const int c = idx % cols;
282
283 const int bk = (block_k > 0) ? block_k : 1;
284 const int cb = c / bk;
285 const int ci = c % bk;
286
287 const int cols_blk = (cols / bk) + ((cols % bk) ? 1 : 0);
288 const int out_idx = (r * cols_blk + cb) * bk + ci;
289 out[out_idx] = w[r * cols + c];
290}

338__global__ static void ie_cuda_touch_kernel(
339 const uint8_t *buf,
340 size_t nbytes,
341 size_t stride_bytes,
342 uint64_t tokens,
343 unsigned long long *out_acc
344) {
345 const size_t stride = (stride_bytes == 0) ? 1 : stride_bytes;
346 const size_t tid = (size_t)blockIdx.x * (size_t)blockDim.x + (size_t)threadIdx.x;
347 const size_t total_threads = (size_t)gridDim.x * (size_t)blockDim.x;
348
349 unsigned long long local = 0;
350
351 for (uint64_t t = 0; t < tokens; ++t) {
352 for (size_t i = tid * stride; i < nbytes; i += total_threads * stride) {
353 local += (unsigned long long)buf[i];
354 }
355 }
356
357 if (local) atomicAdd(out_acc, local);
358}

385 size_t stride_bytes,
386 uint64_t tokens,

388) {
389 if (bytes_per_token == 0 || tokens == 0) {
390 ie_cuda_clear_last_error();
391 return IE_CUDA_OK;
392 }
393
394 CUDA_GUARD(cudaFree(0));
395
396 if (!g_touch_buf || g_touch_buf_cap != bytes_per_token) {
397 if (g_touch_buf) {
398 (void)cudaFree(g_touch_buf);
399 g_touch_buf = NULL;
400 g_touch_buf_cap = 0;
401 }
402 CUDA_GUARD(cudaMalloc((void **)&g_touch_buf, bytes_per_token));
403 g_touch_buf_cap = bytes_per_token;
404 }
405
406 if (!g_touch_acc) {
407 CUDA_GUARD(cudaMalloc((void **)&g_touch_acc, sizeof(*g_touch_acc)));
408 }
409
410 if (seed && seed_len > 0) {
411 const size_t n_copy = (seed_len < g_touch_buf_cap) ? seed_len : g_touch_buf_cap;
412 CUDA_GUARD(cudaMemcpy(g_touch_buf, seed, n_copy, cudaMemcpyHostToDevice));
413 if (n_copy < g_touch_buf_cap) {
414 CUDA_GUARD(cudaMemset(g_touch_buf + n_copy, 1, g_touch_buf_cap - n_copy));
415 }
416 } else {
417 CUDA_GUARD(cudaMemset(g_touch_buf, 1, g_touch_buf_cap));
418 }
419
420 CUDA_GUARD(cudaMemset(g_touch_acc, 0, sizeof(*g_touch_acc)));
421
422 const int threads = 256;

427 ));
428 if (blocks_per_sm <= 0) blocks_per_sm = 1;
429

434 if (sm_count <= 0) sm_count = 1;
