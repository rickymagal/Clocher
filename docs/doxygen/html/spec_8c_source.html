77uint64_t dedup_spec_builder_intern_string(dedup_spec_builder_t* b, const char* s) {
78 if (!b || !s) return 0;
79 const uint64_t slen = (uint64_t)strlen(s) + 1u;

81 /* Linear scan for identical string. */
82 uint64_t off = 1u;
83 while (off < b->strings_len) {
84 const char* cur = b->strings + off;
85 if (strcmp(cur, s) == 0) return off;

94 b->strings_len += slen;
