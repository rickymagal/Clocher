29def pack_row_int4_sym(row: torch.Tensor):
30 a = row.abs().max().item()
31 scale = 0.0 if a == 0.0 else a / 7.0
32 if scale == 0.0:
33 q = np.zeros(row.numel(), dtype=np.int8)
34 else:
35 q = torch.clamp(torch.round(row / scale), -8, 7).to(torch.int8).cpu().numpy()
36 if q.size % 2 == 1:
37 q = np.concatenate([q, np.zeros(1, dtype=np.int8)], axis=0)
38 n = (q & 0x0F).astype(np.uint8)
39 lo = n[0::2]; hi = n[1::2]
40 packed = (lo | (hi << 4)).astype(np.uint8)
41 return packed.tobytes(), np.float16(scale).tobytes()
