58static int read_int_file(const char *path, int *out) {
59 FILE *f = fopen(path, "r");
60 if (!f) return -errno ? -errno : -1;
61 char buf[64];
62 if (!fgets(buf, sizeof(buf), f)) {
63 int e = ferror(f) ? -errno : -1;
64 fclose(f);
65 return e ? e : -1;
66 }
67 fclose(f);
68 char *end = NULL;
69 long v = strtol(buf, &end, 10);
70 if (end == buf) return -EINVAL;
71 *out = (int)v;
72 return 0;
73}

83static void apply_cpulist_to_node(const char *s, int node_id, int *cpu2node, int cap) {
84 const char *p = s ? s : "";
85 while (*p) {
86 /* skip separators and whitespace */
87 while (*p == ' ' || *p == '\t' || *p == ',' || *p == '\n' || *p == '\r') ++p;
88 if (!*p) break;
89 char *end = NULL;
90 long a = strtol(p, &end, 10);
91 if (end == p) { /* skip invalid token */
92 while (*p && *p != ',' && *p != '\n' && *p != '\r') ++p;
93 continue;
94 }
95 p = end;
96 long b = a;
97 if (*p == '-') {
98 ++p;
99 b = strtol(p, &end, 10);
100 if (end == p) b = a; /* treat lone '-' weirdness as single */
101 p = end;
102 }
103 if (a < 0) a = 0;
104 if (b < 0) b = 0;
105 if (a > b) { long t = a; a = b; b = t; }
106 for (long i = a; i <= b; ++i) {
107 if (i >= 0 && i < cap) cpu2node[(int)i] = node_id;
108 }
109 }
110}

122static int discover_sockets_sysfs(int n_cpus, int *cpu2socket) {
123 int max_socket_id = -1;
124 for (int cpu = 0; cpu < n_cpus; ++cpu) {
125 char path[256];
126 snprintf(path, sizeof(path),
127 "/sys/devices/system/cpu/cpu%d/topology/physical_package_id", cpu);
128 int sid = -1;
129 if (read_int_file(path, &sid) == 0 && sid >= 0) {
130 cpu2socket[cpu] = sid;
131 if (sid > max_socket_id) max_socket_id = sid;
132 }
133 }
134 if (max_socket_id < 0) return 0;
135 return max_socket_id + 1;
136}

150static int discover_nodes_sysfs(int n_cpus, int *cpu2node) {
151 const char *root = "/sys/devices/system/node";
152 DIR *d = opendir(root);
153 if (!d) return 0;
154
155 int max_node_id = -1;
156 struct dirent *de;
157 while ((de = readdir(d)) != NULL) {
158 if (strncmp(de->d_name, "node", 4) != 0) continue;
159 char *end = NULL;
160 long nid = strtol(de->d_name + 4, &end, 10);
161 if (end == de->d_name + 4 || nid < 0 || nid > INT_MAX) continue;
162
163 char path[512];
164 snprintf(path, sizeof(path), "%s/%s/cpulist", root, de->d_name);
165
166 FILE *f = fopen(path, "r");
167 if (!f) continue;
168
169 char buf[4096];
170 size_t n = fread(buf, 1, sizeof(buf) - 1, f);
171 fclose(f);
172 buf[n] = '\0';
173
174 apply_cpulist_to_node(buf, (int)nid, cpu2node, n_cpus);
175 if ((int)nid > max_node_id) max_node_id = (int)nid;
176 }
177 closedir(d);
178
179 if (max_node_id < 0) return 0;
180 /* Normalize: nodes are 0..max_node_id, but some may be sparse; still fine. */
181 return max_node_id + 1;
182}

192static void compute_first_cpu_per_socket(int n_cpus, int n_sockets,
193 const int *cpu2socket, int *out_first) {
194 for (int s = 0; s < n_sockets; ++s) out_first[s] = -1;
195 for (int cpu = 0; cpu < n_cpus; ++cpu) {
196 int s = cpu2socket[cpu];
197 if (s >= 0 && s < n_sockets) {
198 if (out_first[s] == -1 || cpu < out_first[s]) out_first[s] = cpu;
199 }
200 }
201}

205int ie_topology_init(ie_topology_t **out_topo) {
206 if (!out_topo) return -EINVAL;
207
208 int n_cpus = get_online_cpus();
209
210 ie_topology_t *t = (ie_topology_t *)calloc(1, sizeof(*t));
211 if (!t) return -ENOMEM;
212
213 t->n_cpus = n_cpus;
214
215 t->cpu_to_socket = (int *)malloc(sizeof(int) * n_cpus);
216 t->cpu_to_node = (int *)malloc(sizeof(int) * n_cpus);
217 if (!t->cpu_to_socket || !t->cpu_to_node) {
218 free(t->cpu_to_socket);
219 free(t->cpu_to_node);
220 free(t);
221 return -ENOMEM;
222 }
223 for (int i = 0; i < n_cpus; ++i) {
224 t->cpu_to_socket[i] = -1;
225 t->cpu_to_node[i] = -1;
226 }
227
228 /* Best-effort discovery */
229 int sockets = discover_sockets_sysfs(n_cpus, t->cpu_to_socket);
230 int nodes = discover_nodes_sysfs(n_cpus, t->cpu_to_node);
231
232 if (sockets <= 0) {
233 sockets = 1;
234 for (int i = 0; i < n_cpus; ++i) t->cpu_to_socket[i] = 0;
235 }
236 if (nodes <= 0) {
237 nodes = 1;
238 for (int i = 0; i < n_cpus; ++i) t->cpu_to_node[i] = 0;
239 }
240 t->n_sockets = sockets;
241 t->n_nodes = nodes;
242
243 t->first_cpu_socket = (int *)malloc(sizeof(int) * sockets);
244 if (!t->first_cpu_socket) {
245 free(t->cpu_to_socket);
246 free(t->cpu_to_node);
247 free(t);
248 return -ENOMEM;
249 }
250 compute_first_cpu_per_socket(n_cpus, sockets, t->cpu_to_socket, t->first_cpu_socket);
251
252 *out_topo = t;
253 return 0;
254}

338int ie_topology_bind_thread_to_socket(const ie_topology_t *topo, int socket) {
339 if (!topo) return -EINVAL;
340 if (socket < 0 || socket >= topo->n_sockets) return -EINVAL;
341
342 int cpu = ie_topology_first_cpu_on_socket(topo, socket);
343 if (cpu < 0) return -EINVAL;
344
345 cpu_set_t set;
346 CPU_ZERO(&set);
347 CPU_SET((unsigned)cpu, &set);
348
349 if (sched_setaffinity(0, sizeof(set), &set) != 0) {
350 return -errno ? -errno : -1;
351 }
352 return 0;
353}
