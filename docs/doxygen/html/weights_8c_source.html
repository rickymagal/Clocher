96static void join_path(char *out, size_t outsz, const char *dir, const char *file) {
97 if (!out || outsz == 0) return;
98 out[0] = '\0';
99 if (!file || !*file) return;
100
101 if (!dir || !*dir) {
102 cpyz(out, outsz, file);
103 return;
104 }
105
106 const size_t ld = strlen(dir);
107 const size_t lf = strlen(file);
108 const int need_slash = (ld > 0 && dir[ld - 1] != '/');
109
110 size_t pos = 0;
111
112 size_t copyd = ld;
113 if (copyd >= outsz) copyd = outsz - 1;
114 if (copyd > 0) {
115 memcpy(out + pos, dir, copyd);
116 pos += copyd;
117 }
118
119 if (pos < outsz - 1 && need_slash) out[pos++] = '/';
120
121 if (pos < outsz - 1) {
122 size_t room = (outsz - 1) - pos;
123 size_t copyf = (lf > room) ? room : lf;
124 if (copyf > 0) {
125 memcpy(out + pos, file, copyf);
126 pos += copyf;
127 }
128 }
129
130 out[pos] = '\0';
131}

144static int read_all_text(const char *path, char **buf, size_t *len_out) {
145 if (!path || !buf) return -1;
146 *buf = NULL;
147 if (len_out) *len_out = 0;
148
149 FILE *f = fopen(path, "rb");
150 if (!f) return -2;
151
152 if (fseek(f, 0, SEEK_END) != 0) {
153 fclose(f);
154 return -3;
155 }
156 long n = ftell(f);
157 if (n < 0) {
158 fclose(f);
159 return -3;
160 }
161 if (fseek(f, 0, SEEK_SET) != 0) {
162 fclose(f);
163 return -3;
164 }
165
166 size_t size = (size_t)n;
167 char *mem = (char *)malloc(size + 1);
168 if (!mem) {
169 fclose(f);
170 return -4;
171 }
172
173 size_t rd = fread(mem, 1, size, f);
174 fclose(f);
175 if (rd != size) {
176 free(mem);
177 return -5;
178 }
179
180 if (size >= 3 && (unsigned char)mem[0] == 0xEF && (unsigned char)mem[1] == 0xBB &&
181 (unsigned char)mem[2] == 0xBF) {
182 size -= 3;
183 memmove(mem, mem + 3, size);
184 }
185
186 mem[size] = '\0';
187 *buf = mem;
188 if (len_out) *len_out = size;
189 return 0;
190}

200static int scan_json_key_int(const char *json, const char *key, int *out_val) {
201 if (!json || !key || !out_val) return -1;
202 const char *p = json;
203 const size_t klen = strlen(key);
204 while ((p = strstr(p, "\""))) {
205 ++p;
206 if (strncmp(p, key, klen) == 0 && p[klen] == '"') {
207 const char *c = p + klen + 1;
208 while (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n') ++c;
209 if (*c != ':') continue;
210 ++c;
211 while (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n') ++c;
212 char *end = NULL;
213 long v = strtol(c, &end, 10);

268static int scan_json_object_range(const char *json, const char *key, const char **out_begin,
269 const char **out_end) {
270 if (!json || !key || !out_begin || !out_end) return -1;
271 *out_begin = *out_end = NULL;
272
273 const size_t klen = strlen(key);
274 const char *p = json;
275 while ((p = strstr(p, "\""))) {
276 ++p;
277 if (strncmp(p, key, klen) == 0 && p[klen] == '"') {
278 const char *c = p + klen + 1;
279 while (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n') ++c;
280 if (*c != ':') continue;
281 ++c;
282 while (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n') ++c;
283 if (*c != '{') return 0;
284 int depth = 0;
285 const char *start = c + 1;
286 const char *q = c;
287 do {
288 if (*q == '{')
289 ++depth;
290 else if (*q == '}')
291 --depth;
292 ++q;
293 } while (*q && depth > 0);
294 if (depth == 0) {
295 *out_begin = start;
296 *out_end = q - 1;
297 return 1;
298 }
299 return 0;
300 }
301 }
302 return 0;

315static int scan_json_key_string_in_range(const char *begin, const char *end, const char *key,
316 char *dst, size_t dstsz) {
317 if (!begin || !end || !key || !dst || dstsz == 0) return -1;
318 dst[0] = '\0';
319 const size_t klen = strlen(key);
320 const char *p = begin;
321 while (p < end) {
322 const char *q = memchr(p, '"', (size_t)(end - p));
323 if (!q) break;
324 ++q;
325 if ((size_t)(end - q) >= klen && memcmp(q, key, klen) == 0 && q[klen] == '"') {
326 const char *c = q + klen + 1;
327 while (c < end && (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n')) ++c;
328 if (c >= end || *c != ':') {
329 p = q + 1;
330 continue;
331 }
332 ++c;
333 while (c < end && (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n')) ++c;
334 if (c >= end || *c != '"') {
335 p = q + 1;
336 continue;
337 }
338 ++c;
339 const char *start = c;
340 while (c < end && *c != '"') ++c;
341 size_t n = (size_t)(c - start);
342 if (n >= dstsz) n = dstsz - 1;
343 memcpy(dst, start, n);
344 dst[n] = '\0';
345 return 1;
346 }
347 p = q + 1;
348 }
349 return 0;

361static ssize_t ie_pread(int fd, void *buf, size_t count, off_t offset) {
362#if defined(_XOPEN_SOURCE) || defined(_POSIX_C_SOURCE)
363 return pread(fd, buf, count, offset);
364#else
365 off_t cur = lseek(fd, 0, SEEK_CUR);
366 if (cur == (off_t)-1) return -1;
367 if (lseek(fd, offset, SEEK_SET) == (off_t)-1) return -1;
368 ssize_t r = read(fd, buf, count);
369 (void)lseek(fd, cur, SEEK_SET);
370 return r;
371#endif
372}

381static int ascii_ieq(const char *a, const char *b) {
382 if (!a || !b) return 0;
383 while (*a && *b) {
384 if ((unsigned char)tolower((unsigned char)*a) != (unsigned char)tolower((unsigned char)*b)) {
385 return 0;
386 }
387 ++a;
388 ++b;
389 }

511int ie_weights_open(const char *json_path, const char *bin_path, ie_weights_t *out) {
512 if (!out) return IE_IO_ERR_ARGS;
513 memset(out, 0, sizeof(*out));
514
515 if (!json_path || !*json_path) return IE_IO_ERR_ARGS;
516 cpyz(out->json_path, sizeof(out->json_path), json_path);
517
518 char *jbuf = NULL;
519 size_t jlen = 0;
520 if (read_all_text(json_path, &jbuf, &jlen) != 0 || !jbuf || jlen == 0) {
521 return IE_IO_ERR_JSON;
522 }
523
524 int ver = 0;
525 (void)scan_json_key_int(jbuf, "version", &ver);
526 if (ver <= 0) ver = 1;
527 out->version = ver;
528
529 char dtype[16];
530 dtype[0] = '\0';
531 (void)scan_json_key_string(jbuf, "dtype", dtype, sizeof(dtype));
532 if (dtype[0] == '\0') cpyz(dtype, sizeof(dtype), "float32");
533 cpyz(out->dtype, sizeof(out->dtype), dtype);
534
535 char resolved_bin[512];
536 resolved_bin[0] = '\0';
537 if (bin_path && *bin_path) {
538 cpyz(resolved_bin, sizeof(resolved_bin), bin_path);
539 } else {
540 char bin_key[256];
541 bin_key[0] = '\0';
542 (void)scan_json_key_string(jbuf, "bin", bin_key, sizeof(bin_key));
543 if (bin_key[0] == '\0') {
544 free(jbuf);
545 return IE_IO_ERR_BIN_UNSPEC;
546 }
547
548 const char *slash = last_slash(json_path);
549 if (slash) {
550 char dir[512];

552 if (dn >= sizeof(dir)) dn = sizeof(dir) - 1;
553 memcpy(dir, json_path, dn);

555 join_path(resolved_bin, sizeof(resolved_bin), dir, bin_key);
556 } else {
557 cpyz(resolved_bin, sizeof(resolved_bin), bin_key);
558 }
559 }
560 cpyz(out->weights_path, sizeof(out->weights_path), resolved_bin);
561
562 struct stat st;
563 if (stat(out->weights_path, &st) != 0 || st.st_size < 0) {
564 free(jbuf);
565 return IE_IO_ERR_STAT;
566 }
567 out->bin_size_bytes = (size_t)st.st_size;
568 out->loaded = 1;
569
570#if (IE_WEIGHTS_HAS_DEDUP_FIELDS == 1)

573 const int dedup_strict = env_flag_get("IE_DEDUP_STRICT", 0);
574
575 if (dedup_enabled) {
576 const char *slash = last_slash(json_path);
577 char dir[512];
578 dir[0] = '\0';
579 if (slash) {
580 size_t dn = (size_t)(slash - json_path);
581 if (dn >= sizeof(dir)) dn = sizeof(dir) - 1;
582 memcpy(dir, json_path, dn);
583 dir[dn] = '\0';
584 } else {
585 cpyz(dir, sizeof(dir), ".");
586 }
587
588 char dedup_json[512];
589 dedup_json[0] = '\0';
590 join_path(dedup_json, sizeof(dedup_json), dir, "model.dedup.json");
591

598

601 if (dst == IE_WDEDUP_OK && dh) {
602 out->is_dedup = 1;
603 out->dedup_handle = (void *)dh;
604 } else {
605 if (dedup_strict) {
606 free(jbuf);
607 return IE_IO_ERR_JSON;
608 }
609 out->is_dedup = 0;

611 }
612 } else {
613 if (dedup_strict) {
614 free(jbuf);

616 }
617 }
618 }
619 }

831static float fp16_to_fp32(uint16_t h) {
832 uint32_t sign = (uint32_t)(h & 0x8000u) << 16;
833 uint32_t exp = (h & 0x7C00u) >> 10;
834 uint32_t mant = (h & 0x03FFu);
835 uint32_t f;
836 if (exp == 0) {
837 if (mant == 0) {
838 f = sign;
839 } else {
840 int e = -1;
841 do {
842 e++;
843 mant <<= 1;
844 } while ((mant & 0x0400u) == 0);
845 mant &= 0x03FFu;
846 f = sign | ((uint32_t)(127 - 15 - e) << 23) | (mant << 13);
847 }
848 } else if (exp == 0x1F) {
849 f = sign | 0x7F800000u | (mant << 13);
850 } else {
851 f = sign | ((exp + (127 - 15)) << 23) | (mant << 13);
852 }
853 float out;

869int ie_weights_decode_int4(const char *packed_path, const char *scales_path, size_t rows, size_t cols,
870 int per_row, float *dst) {
871 if (!packed_path || !scales_path || !dst || rows == 0 || cols == 0) return IE_IO_ERR_ARGS;
872
873 int fdw = open(packed_path, O_RDONLY);
874 if (fdw < 0) return IE_IO_ERR_OPEN;
875
876 struct stat stw;
877 if (fstat(fdw, &stw) != 0) {
878 close(fdw);
879 return IE_IO_ERR_STAT;
880 }
881
882 const size_t need_packed = rows * ie_int4_rowbytes(cols);
883 if ((size_t)stw.st_size < need_packed) {
884 close(fdw);
885 return IE_IO_ERR_READ;
886 }
887
888 uint8_t *buf_packed = (uint8_t *)malloc(need_packed);
889 if (!buf_packed) {
890 close(fdw);
891 return IE_IO_ERR_ALLOC;
892 }
893
894 ssize_t rd = ie_pread(fdw, buf_packed, need_packed, 0);
895 close(fdw);
896 if (rd < 0 || (size_t)rd != need_packed) {
897 free(buf_packed);
898 return IE_IO_ERR_READ;
899 }
900
901 int fds = open(scales_path, O_RDONLY);
902 if (fds < 0) {
903 free(buf_packed);
904 return IE_IO_ERR_OPEN;
905 }
906
907 struct stat sts;
908 if (fstat(fds, &sts) != 0) {
909 close(fds);
910 free(buf_packed);
911 return IE_IO_ERR_STAT;
912 }
913
914 size_t nsc = per_row ? rows : 1;
915 size_t need16 = nsc * sizeof(uint16_t);
916 size_t need32 = nsc * sizeof(float);
917
918 int is16 = ((size_t)sts.st_size == need16);
919 int is32 = ((size_t)sts.st_size == need32);
920 if (!(is16 || is32)) {
921 close(fds);
922 free(buf_packed);
923 return IE_IO_ERR_READ;
924 }
925
926 float *buf_scales = (float *)malloc(need32);
927 if (!buf_scales) {
928 close(fds);
929 free(buf_packed);
930 return IE_IO_ERR_ALLOC;
931 }
932
933 if (is32) {
934 ssize_t rds = ie_pread(fds, buf_scales, need32, 0);
935 close(fds);
936 if (rds < 0 || (size_t)rds != need32) {
937 free(buf_scales);
938 free(buf_packed);
939 return IE_IO_ERR_READ;
940 }
941 } else {
942 uint16_t *tmp = (uint16_t *)malloc(need16);
943 if (!tmp) {
944 close(fds);
945 free(buf_scales);
946 free(buf_packed);
947 return IE_IO_ERR_ALLOC;
948 }
949 ssize_t rds = ie_pread(fds, tmp, need16, 0);
950 close(fds);
951 if (rds < 0 || (size_t)rds != need16) {
952 free(tmp);
953 free(buf_scales);
954 free(buf_packed);
955 return IE_IO_ERR_READ;
956 }
957 for (size_t i = 0; i < nsc; ++i) buf_scales[i] = fp16_to_fp32(tmp[i]);
958 free(tmp);
959 }
960
961 int qst = 0;
962 if (per_row) {
963 qst = ie_int4_dequantize_per_row(buf_packed, rows, cols, buf_scales, dst);
964 } else {
965 qst = ie_int4_dequantize_per_tensor(buf_packed, rows, cols, buf_scales[0], dst);
966 }
967
968 free(buf_scales);
969 free(buf_packed);
970
971 if (qst != IE_INT4_STATUS_OK) return IE_IO_ERR_DECODE;
972 return IE_IO_OK;
973}
