522 const char *c = p + klen + 1;

946 }

949 ssize_t r1 = ie_pread(fd, buf, sizeof(buf), 0);
950 ssize_t r2 = 0;

959 return 0;
960 }
961
962 WLOGD("touch_file_small: ok path='%s' size=%zu", path, (size_t)st.st_size);

1506static float fp16_to_fp32(uint16_t h) {
1507 uint32_t sign = (uint32_t)(h & 0x8000u) << 16;
1508 uint32_t exp = (h & 0x7C00u) >> 10;
1509 uint32_t mant = (h & 0x03FFu);
1510 uint32_t f;
1511 if (exp == 0) {
1512 if (mant == 0) {
1513 f = sign;
1514 } else {
1515 int e = -1;
1516 do {
1517 e++;
1518 mant <<= 1;
1519 } while ((mant & 0x0400u) == 0);
1520 mant &= 0x03FFu;
1521 f = sign | ((uint32_t)(127 - 15 - e) << 23) | (mant << 13);
1522 }
1523 } else if (exp == 0x1F) {
1524 f = sign | 0x7F800000u | (mant << 13);
1525 } else {
1526 f = sign | ((exp + (127 - 15)) << 23) | (mant << 13);
1527 }
1528 float out;
1529 memcpy(&out, &f, sizeof(out));
1530 return out;
1531}
