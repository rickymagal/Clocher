57static void cpyz(char *dst, size_t dstsz, const char *src) {

60 size_t n = strlen(src);

63 dst[n] = '\0';

81static void join_path(char *out, size_t outsz, const char *dir, const char *file) {
82 if (!out || outsz == 0) return;
83 out[0] = '\0';
84 if (!file || !*file) return;
85
86 if (!dir || !*dir) { cpyz(out, outsz, file); return; }
87
88 const size_t ld = strlen(dir);
89 const size_t lf = strlen(file);
90 const int need_slash = (ld > 0 && dir[ld - 1] != '/');
91
92 size_t pos = 0;
93
94 size_t copyd = ld;
95 if (copyd >= outsz) copyd = outsz - 1;
96 if (copyd > 0) { memcpy(out + pos, dir, copyd); pos += copyd; }
97
98 if (pos < outsz - 1 && need_slash) { out[pos++] = '/'; }
99
100 if (pos < outsz - 1) {
101 size_t room = (outsz - 1) - pos;
102 size_t copyf = (lf > room) ? room : lf;
103 if (copyf > 0) { memcpy(out + pos, file, copyf); pos += copyf; }
104 }
105
106 out[pos] = '\0';
107}

114static int read_all_text(const char *path, char **buf, size_t *len_out) {
115 if (!path || !buf) return -1;
116 *buf = NULL; if (len_out) *len_out = 0;
117
118 FILE *f = fopen(path, "rb");
119 if (!f) return -2;
120
121 if (fseek(f, 0, SEEK_END) != 0) { fclose(f); return -3; }
122 long n = ftell(f);
123 if (n < 0) { fclose(f); return -3; }
124 if (fseek(f, 0, SEEK_SET) != 0) { fclose(f); return -3; }
125
126 size_t size = (size_t)n;
127 char *mem = (char*)malloc(size + 1);
128 if (!mem) { fclose(f); return -4; }
129
130 size_t rd = fread(mem, 1, size, f);
131 fclose(f);
132 if (rd != size) { free(mem); return -5; }
133
134 /* Strip UTF-8 BOM if present */
135 if (size >= 3 &&
136 (unsigned char)mem[0] == 0xEF &&
137 (unsigned char)mem[1] == 0xBB &&
138 (unsigned char)mem[2] == 0xBF) {
139 size -= 3;
140 memmove(mem, mem + 3, size);
141 }
142
143 mem[size] = '\0';
144 *buf = mem;
145 if (len_out) *len_out = size;
146 return 0;
147}

155 if (!json || !key || !out_val) return -1;

157 const size_t klen = strlen(key);

212static int scan_json_object_range(const char *json,
213 const char *key,
214 const char **out_begin,
215 const char **out_end) {
216 if (!json || !key || !out_begin || !out_end) return -1;
217 *out_begin = *out_end = NULL;
218
219 const size_t klen = strlen(key);
220 const char *p = json;
221 while ((p = strstr(p, "\""))) {
222 ++p;
223 if (strncmp(p, key, klen) == 0 && p[klen] == '"') {
224 const char *c = p + klen + 1;
225 while (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n') ++c;
226 if (*c != ':') continue;
227 ++c;
228 while (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n') ++c;
229 if (*c != '{') return 0;
230 /* scan balanced braces */
231 int depth = 0;
232 const char *start = c + 1;
233 const char *q = c;
234 do {
235 if (*q == '{') ++depth;
236 else if (*q == '}') --depth;
237 ++q;
238 } while (*q && depth > 0);
239 if (depth == 0) {
240 *out_begin = start;
241 *out_end = q - 1; /* position of '}' */
242 return 1;
243 }
244 return 0;
245 }
246 }
247 return 0;
248}

255static int scan_json_key_string_in_range(const char *begin,
256 const char *end,
257 const char *key,
258 char *dst,
259 size_t dstsz) {
260 if (!begin || !end || !key || !dst || dstsz == 0) return -1;
261 dst[0] = '\0';
262 const size_t klen = strlen(key);
263 const char *p = begin;
264 while (p < end) {
265 const char *q = memchr(p, '"', (size_t)(end - p));
266 if (!q) break;
267 ++q;
268 if ((size_t)(end - q) >= klen && memcmp(q, key, klen) == 0 && q[klen] == '"') {
269 const char *c = q + klen + 1;
270 while (c < end && (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n')) ++c;
271 if (c >= end || *c != ':') { p = q + 1; continue; }
272 ++c;
273 while (c < end && (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n')) ++c;
274 if (c >= end || *c != '"') { p = q + 1; continue; }
275 ++c;
276 const char *start = c;
277 while (c < end && *c != '"') ++c;
278 size_t n = (size_t)(c - start);
279 if (n >= dstsz) n = dstsz - 1;
280 memcpy(dst, start, n);
281 dst[n] = '\0';
282 return 1;
283 }
284 p = q + 1;
285 }
286 return 0;

293static ssize_t ie_pread(int fd, void *buf, size_t count, off_t offset) {
294#if defined(_XOPEN_SOURCE) || defined(_POSIX_C_SOURCE)
295 return pread(fd, buf, count, offset);
296#else
297 off_t cur = lseek(fd, 0, SEEK_CUR);
298 if (cur == (off_t)-1) return -1;
299 if (lseek(fd, offset, SEEK_SET) == (off_t)-1) return -1;
300 ssize_t r = read(fd, buf, count);
301 (void)lseek(fd, cur, SEEK_SET);
302 return r;
303#endif
304}

310static int ascii_ieq(const char *a, const char *b) {
311 if (!a || !b) return 0;
312 while (*a && *b) {
313 if ((unsigned char)tolower(*a) != (unsigned char)tolower(*b)) return 0;
314 ++a; ++b;
315 }
316 return *a == '\0' && *b == '\0';

326int ie_weights_open(const char *json_path, const char *bin_path, ie_weights_t *out) {
327 if (!out) return IE_IO_ERR_ARGS;
328 memset(out, 0, sizeof(*out));
329
330 if (!json_path || !*json_path) return IE_IO_ERR_ARGS;
331 cpyz(out->json_path, sizeof(out->json_path), json_path);
332
333 /* Read JSON text */
334 char *jbuf = NULL; size_t jlen = 0;
335 if (read_all_text(json_path, &jbuf, &jlen) != 0 || !jbuf || jlen == 0) {
336 return IE_IO_ERR_JSON;
337 }
338
339 /* Parse relaxed header keys */
340 int ver = 0; (void)scan_json_key_int(jbuf, "version", &ver);
341 if (ver <= 0) ver = 1;
342 out->version = ver;
343
344 char dtype[16]; dtype[0] = '\0';
345 (void)scan_json_key_string(jbuf, "dtype", dtype, sizeof(dtype));
346 if (dtype[0] == '\0') strcpy(dtype, "float32");
347 cpyz(out->dtype, sizeof(out->dtype), dtype);
348
349 /* Determine weights path (absolute or relative to JSON dir) */
350 char resolved_bin[512]; resolved_bin[0] = '\0';
351 if (bin_path && *bin_path) {
352 cpyz(resolved_bin, sizeof(resolved_bin), bin_path);
353 } else {
354 char bin_key[256]; bin_key[0] = '\0';
355 (void)scan_json_key_string(jbuf, "bin", bin_key, sizeof(bin_key));
356 if (bin_key[0] == '\0') { free(jbuf); return IE_IO_ERR_BIN_UNSPEC; }
357
358 const char *slash = last_slash(json_path);
359 if (slash) {
360 char dir[512]; size_t dn = (size_t)(slash - json_path);
361 if (dn >= sizeof(dir)) dn = sizeof(dir) - 1;
362 memcpy(dir, json_path, dn); dir[dn] = '\0';
363 join_path(resolved_bin, sizeof(resolved_bin), dir, bin_key);
364 } else {
365 cpyz(resolved_bin, sizeof(resolved_bin), bin_key);
366 }
367 }
368 cpyz(out->weights_path, sizeof(out->weights_path), resolved_bin);
369
370 /* Stat the weights file to get its size */
371 struct stat st;
372 if (stat(out->weights_path, &st) != 0 || st.st_size < 0) {
373 free(jbuf);
374 return IE_IO_ERR_STAT;
375 }
376 out->bin_size_bytes = (size_t)st.st_size;
377 out->loaded = 1;
378
379 free(jbuf);
380 return IE_IO_OK;
381}

439int ie_weights_read_int4_meta(const char *json_path, struct ie_int4_quant_meta *out_meta) {
440 if (!json_path || !out_meta) return IE_IO_ERR_ARGS;
441 memset(out_meta, 0, sizeof(*out_meta));
442
443 /* Read JSON */
444 char *jbuf = NULL; size_t jlen = 0;
445 if (read_all_text(json_path, &jbuf, &jlen) != 0 || !jbuf || jlen == 0) {
446 return IE_IO_ERR_JSON;
447 }
448
449 /* dtype must indicate INT4/Q4 (root-level) to proceed */
450 char dtype[16]; dtype[0] = '\0';
451 (void)scan_json_key_string(jbuf, "dtype", dtype, sizeof(dtype));
452 if (!(ascii_ieq(dtype, "int4") || ascii_ieq(dtype, "q4") || ascii_ieq(dtype, "mixed"))) {
453 free(jbuf);
454 out_meta->present = 0;
455 return IE_IO_OK;
456 }
457
458 /* Find "quant": { ... } section */
459 const char *qb = NULL, *qe = NULL;
460 if (scan_json_object_range(jbuf, "quant", &qb, &qe) != 1 || !qb || !qe || qb >= qe) {
461 free(jbuf);
462 out_meta->present = 0;
463 return IE_IO_OK; /* treat as missing metadata */
464 }
465
466 /* Extract fields from the quant object */
467 char per[32]; per[0] = '\0';
468 (void)scan_json_key_string_in_range(qb, qe, "per", per, sizeof(per));
469 out_meta->per_row = ascii_ieq(per, "row") ? 1 : 0;
470
471 char scale_bin_rel[512]; scale_bin_rel[0] = '\0';
472 (void)scan_json_key_string_in_range(qb, qe, "scale_bin", scale_bin_rel, sizeof(scale_bin_rel));
473
474 char pack[32]; pack[0] = '\0';
475 (void)scan_json_key_string_in_range(qb, qe, "pack", pack, sizeof(pack));
476 cpyz(out_meta->pack, sizeof(out_meta->pack), pack);
477
478 char zp_str[32]; zp_str[0] = '\0';
479 (void)scan_json_key_string_in_range(qb, qe, "zp", zp_str, sizeof(zp_str));
480 if (zp_str[0]) out_meta->zero_point = atoi(zp_str);
481
482 char sym_str[32]; sym_str[0] = '\0';
483 (void)scan_json_key_string_in_range(qb, qe, "symmetric", sym_str, sizeof(sym_str));
484 if (sym_str[0]) out_meta->symmetric = (ascii_ieq(sym_str, "true") || strcmp(sym_str, "1") == 0) ? 1 : 0;
485
486 /* Resolve scale_bin relative to JSON directory if needed */
487 if (scale_bin_rel[0]) {
488 const char *slash = last_slash(json_path);
489 if (slash) {
490 char dir[512]; size_t dn = (size_t)(slash - json_path);
491 if (dn >= sizeof(dir)) dn = sizeof(dir) - 1;
492 memcpy(dir, json_path, dn); dir[dn] = '\0';
493 join_path(out_meta->scale_bin, sizeof(out_meta->scale_bin), dir, scale_bin_rel);
494 } else {
495 cpyz(out_meta->scale_bin, sizeof(out_meta->scale_bin), scale_bin_rel);
496 }
497 } else {
498 out_meta->scale_bin[0] = '\0';
499 }
500
501 out_meta->present = 1;
502 free(jbuf);
503 return IE_IO_OK;
504}

522int ie_weights_decode_int4(const char *packed_path,
523 const char *scales_path,
524 size_t rows,
525 size_t cols,
526 int per_row,
527 float *dst) {
528 if (!packed_path || !scales_path || !dst || rows == 0 || cols == 0) return IE_IO_ERR_ARGS;
529
530 /* Read packed blob */
531 int fdw = open(packed_path, O_RDONLY);
532 if (fdw < 0) return IE_IO_ERR_OPEN;
533
534 struct stat stw;
535 if (fstat(fdw, &stw) != 0) { close(fdw); return IE_IO_ERR_STAT; }
536
537 const size_t need_packed = rows * ie_int4_rowbytes(cols);
538 if ((size_t)stw.st_size < need_packed) { close(fdw); return IE_IO_ERR_READ; }
539
540 uint8_t *buf_packed = (uint8_t*)malloc(need_packed);
541 if (!buf_packed) { close(fdw); return IE_IO_ERR_ALLOC; }
542
543 ssize_t rd = ie_pread(fdw, buf_packed, need_packed, 0);
544 close(fdw);
545 if (rd < 0 || (size_t)rd != need_packed) { free(buf_packed); return IE_IO_ERR_READ; }
546
547 /* Read scales */
548 int fds = open(scales_path, O_RDONLY);
549 if (fds < 0) { free(buf_packed); return IE_IO_ERR_OPEN; }
550
551 struct stat sts;
552 if (fstat(fds, &sts) != 0) { close(fds); free(buf_packed); return IE_IO_ERR_STAT; }
553
554 size_t need_scales = per_row ? (rows * sizeof(float)) : sizeof(float);
555 if ((size_t)sts.st_size < need_scales) { close(fds); free(buf_packed); return IE_IO_ERR_READ; }
556
557 float *buf_scales = (float*)malloc(need_scales);
558 if (!buf_scales) { close(fds); free(buf_packed); return IE_IO_ERR_ALLOC; }
559
560 rd = ie_pread(fds, buf_scales, need_scales, 0);
561 close(fds);
562 if (rd < 0 || (size_t)rd != need_scales) { free(buf_scales); free(buf_packed); return IE_IO_ERR_READ; }
563
564 /* Dequantize */
565 int qst = 0;
566 if (per_row) {
567 qst = ie_int4_dequantize_per_row(buf_packed, rows, cols, buf_scales, dst);
568 } else {
569 qst = ie_int4_dequantize_per_tensor(buf_packed, rows, cols, buf_scales[0], dst);
570 }
571
572 free(buf_scales);
573 free(buf_packed);
574
575 if (qst != IE_INT4_STATUS_OK) return IE_IO_ERR_DECODE;
576 return IE_IO_OK;
577}
