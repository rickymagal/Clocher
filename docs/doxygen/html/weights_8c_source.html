744 static float fp16_to_fp32(uint16_t h) {
745  uint32_t sign = (uint32_t)(h & 0x8000u) << 16;
746  uint32_t exp = (h & 0x7C00u) >> 10;
747  uint32_t mant = (h & 0x03FFu);
748  uint32_t f;
749  if (exp == 0) {
750  if (mant == 0) {
751  f = sign;
752  } else {
753  int e = -1;
754  do {
755  e++;
756  mant <<= 1;
757  } while ((mant & 0x0400u) == 0);
758  mant &= 0x03FFu;
759  f = sign | ((uint32_t)(127 - 15 - e) << 23) | (mant << 13);
760  }
761  } else if (exp == 0x1F) {
762  f = sign | 0x7F800000u | (mant << 13);
763  } else {
764  f = sign | ((exp + (127 - 15)) << 23) | (mant << 13);
765  }
766  float out;
767  memcpy(&out, &f, sizeof(out));
768  return out;
769 }
