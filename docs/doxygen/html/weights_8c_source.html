95static void join_path(char *out, size_t outsz, const char *dir, const char *file) {
96 if (!out || outsz == 0) return;
97 out[0] = '\0';
98 if (!file || !*file) return;
99
100 if (!dir || !*dir) { cpyz(out, outsz, file); return; }
101
102 const size_t ld = strlen(dir);
103 const size_t lf = strlen(file);
104 const int need_slash = (ld > 0 && dir[ld - 1] != '/');
105
106 size_t pos = 0;
107
108 size_t copyd = ld;
109 if (copyd >= outsz) copyd = outsz - 1;
110 if (copyd > 0) { memcpy(out + pos, dir, copyd); pos += copyd; }
111
112 if (pos < outsz - 1 && need_slash) { out[pos++] = '/'; }
113
114 if (pos < outsz - 1) {
115 size_t room = (outsz - 1) - pos;
116 size_t copyf = (lf > room) ? room : lf;
117 if (copyf > 0) { memcpy(out + pos, file, copyf); pos += copyf; }
118 }
119
120 out[pos] = '\0';
121}

132static int read_all_text(const char *path, char **buf, size_t *len_out) {
133 if (!path || !buf) return -1;
134 *buf = NULL; if (len_out) *len_out = 0;
135
136 FILE *f = fopen(path, "rb");
137 if (!f) return -2;
138
139 if (fseek(f, 0, SEEK_END) != 0) { fclose(f); return -3; }
140 long n = ftell(f);
141 if (n < 0) { fclose(f); return -3; }
142 if (fseek(f, 0, SEEK_SET) != 0) { fclose(f); return -3; }
143
144 size_t size = (size_t)n;
145 char *mem = (char*)malloc(size + 1);
146 if (!mem) { fclose(f); return -4; }
147
148 size_t rd = fread(mem, 1, size, f);
149 fclose(f);
150 if (rd != size) { free(mem); return -5; }
151
152 /* Strip UTF-8 BOM if present */
153 if (size >= 3 &&
154 (unsigned char)mem[0] == 0xEF &&
155 (unsigned char)mem[1] == 0xBB &&
156 (unsigned char)mem[2] == 0xBF) {
157 size -= 3;
158 memmove(mem, mem + 3, size);
159 }
160
161 mem[size] = '\0';
162 *buf = mem;
163 if (len_out) *len_out = size;
164 return 0;
165}

177 if (!json || !key || !out_val) return -1;

244static int scan_json_object_range(const char *json,
245 const char *key,
246 const char **out_begin,
247 const char **out_end) {
248 if (!json || !key || !out_begin || !out_end) return -1;
249 *out_begin = *out_end = NULL;
250
251 const size_t klen = strlen(key);
252 const char *p = json;
253 while ((p = strstr(p, "\""))) {
254 ++p;
255 if (strncmp(p, key, klen) == 0 && p[klen] == '"') {
256 const char *c = p + klen + 1;
257 while (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n') ++c;
258 if (*c != ':') continue;
259 ++c;
260 while (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n') ++c;
261 if (*c != '{') return 0;
262 /* scan balanced braces */
263 int depth = 0;
264 const char *start = c + 1;
265 const char *q = c;
266 do {
267 if (*q == '{') ++depth;
268 else if (*q == '}') --depth;
269 ++q;
270 } while (*q && depth > 0);
271 if (depth == 0) {
272 *out_begin = start;
273 *out_end = q - 1; /* position of '}' */
274 return 1;
275 }
276 return 0;
277 }
278 }
279 return 0;

293static int scan_json_key_string_in_range(const char *begin,
294 const char *end,
295 const char *key,
296 char *dst,
297 size_t dstsz) {
298 if (!begin || !end || !key || !dst || dstsz == 0) return -1;
299 dst[0] = '\0';
300 const size_t klen = strlen(key);
301 const char *p = begin;
302 while (p < end) {
303 const char *q = memchr(p, '"', (size_t)(end - p));
304 if (!q) break;
305 ++q;
306 if ((size_t)(end - q) >= klen && memcmp(q, key, klen) == 0 && q[klen] == '"') {
307 const char *c = q + klen + 1;
308 while (c < end && (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n')) ++c;
309 if (c >= end || *c != ':') { p = q + 1; continue; }
310 ++c;
311 while (c < end && (*c == ' ' || *c == '\t' || *c == '\r' || *c == '\n')) ++c;
312 if (c >= end || *c != '"') { p = q + 1; continue; }
313 ++c;
314 const char *start = c;
315 while (c < end && *c != '"') ++c;
316 size_t n = (size_t)(c - start);
317 if (n >= dstsz) n = dstsz - 1;
318 memcpy(dst, start, n);
319 dst[n] = '\0';
320 return 1;
321 }
322 p = q + 1;
323 }
324 return 0;

337static ssize_t ie_pread(int fd, void *buf, size_t count, off_t offset) {
338#if defined(_XOPEN_SOURCE) || defined(_POSIX_C_SOURCE)
339 return pread(fd, buf, count, offset);

342 if (cur == (off_t)-1) return -1;
343 if (lseek(fd, offset, SEEK_SET) == (off_t)-1) return -1;

358static int ascii_ieq(const char *a, const char *b) {
359 if (!a || !b) return 0;
360 while (*a && *b) {
361 if ((unsigned char)tolower(*a) != (unsigned char)tolower(*b)) return 0;

374int ie_weights_open(const char *json_path, const char *bin_path, ie_weights_t *out) {
375 if (!out) return IE_IO_ERR_ARGS;
376 memset(out, 0, sizeof(*out));
377
378 if (!json_path || !*json_path) return IE_IO_ERR_ARGS;
379 cpyz(out->json_path, sizeof(out->json_path), json_path);
380
381 /* Read JSON text */
382 char *jbuf = NULL; size_t jlen = 0;
383 if (read_all_text(json_path, &jbuf, &jlen) != 0 || !jbuf || jlen == 0) {
384 return IE_IO_ERR_JSON;
385 }
386
387 /* Parse relaxed header keys */
388 int ver = 0; (void)scan_json_key_int(jbuf, "version", &ver);
389 if (ver <= 0) ver = 1;
390 out->version = ver;
391
392 char dtype[16]; dtype[0] = '\0';
393 (void)scan_json_key_string(jbuf, "dtype", dtype, sizeof(dtype));
394 if (dtype[0] == '\0') strcpy(dtype, "float32");
395 cpyz(out->dtype, sizeof(out->dtype), dtype);
396
397 /* Determine weights path (absolute or relative to JSON dir) */
398 char resolved_bin[512]; resolved_bin[0] = '\0';
399 if (bin_path && *bin_path) {
400 cpyz(resolved_bin, sizeof(resolved_bin), bin_path);
401 } else {
402 char bin_key[256]; bin_key[0] = '\0';
403 (void)scan_json_key_string(jbuf, "bin", bin_key, sizeof(bin_key));
404 if (bin_key[0] == '\0') { free(jbuf); return IE_IO_ERR_BIN_UNSPEC; }
405
406 const char *slash = last_slash(json_path);
407 if (slash) {
408 char dir[512]; size_t dn = (size_t)(slash - json_path);
409 if (dn >= sizeof(dir)) dn = sizeof(dir) - 1;
410 memcpy(dir, json_path, dn); dir[dn] = '\0';
411 join_path(resolved_bin, sizeof(resolved_bin), dir, bin_key);
412 } else {
413 cpyz(resolved_bin, sizeof(resolved_bin), bin_key);
414 }
415 }
416 cpyz(out->weights_path, sizeof(out->weights_path), resolved_bin);
417
418 /* Stat the weights file to get its size */
419 struct stat st;
420 if (stat(out->weights_path, &st) != 0 || st.st_size < 0) {
421 free(jbuf);
422 return IE_IO_ERR_STAT;
423 }
424 out->bin_size_bytes = (size_t)st.st_size;
425 out->loaded = 1;
426
427 free(jbuf);
428 return IE_IO_OK;
429}

492int ie_weights_read_int4_meta(const char *json_path, struct ie_int4_quant_meta *out_meta) {
493 if (!json_path || !out_meta) return IE_IO_ERR_ARGS;
494 memset(out_meta, 0, sizeof(*out_meta));
495
496 /* Read JSON */
497 char *jbuf = NULL; size_t jlen = 0;
498 if (read_all_text(json_path, &jbuf, &jlen) != 0 || !jbuf || jlen == 0) {
499 return IE_IO_ERR_JSON;
500 }
501
502 /* dtype must indicate INT4/Q4 (root-level) to proceed */
503 char dtype[16]; dtype[0] = '\0';
504 (void)scan_json_key_string(jbuf, "dtype", dtype, sizeof(dtype));
505 if (!(ascii_ieq(dtype, "int4") || ascii_ieq(dtype, "q4") || ascii_ieq(dtype, "mixed"))) {
506 free(jbuf);
507 out_meta->present = 0;
508 return IE_IO_OK;
509 }
510
511 /* Find "quant": { ... } section */
512 const char *qb = NULL, *qe = NULL;
513 if (scan_json_object_range(jbuf, "quant", &qb, &qe) != 1 || !qb || !qe || qb >= qe) {
514 free(jbuf);
515 out_meta->present = 0;
516 return IE_IO_OK; /* treat as missing metadata */
517 }
518
519 /* Extract fields from the quant object */
520 char per[32]; per[0] = '\0';
521 (void)scan_json_key_string_in_range(qb, qe, "per", per, sizeof(per));
522 out_meta->per_row = ascii_ieq(per, "row") ? 1 : 0;
523
524 char scale_bin_rel[512]; scale_bin_rel[0] = '\0';
525 (void)scan_json_key_string_in_range(qb, qe, "scale_bin", scale_bin_rel, sizeof(scale_bin_rel));
526
527 char pack[32]; pack[0] = '\0';
528 (void)scan_json_key_string_in_range(qb, qe, "pack", pack, sizeof(pack));
529 cpyz(out_meta->pack, sizeof(out_meta->pack), pack);
530
531 char zp_str[32]; zp_str[0] = '\0';
532 (void)scan_json_key_string_in_range(qb, qe, "zp", zp_str, sizeof(zp_str));
533 if (zp_str[0]) out_meta->zero_point = atoi(zp_str);
534
535 char sym_str[32]; sym_str[0] = '\0';
536 (void)scan_json_key_string_in_range(qb, qe, "symmetric", sym_str, sizeof(sym_str));
537 if (sym_str[0]) out_meta->symmetric = (ascii_ieq(sym_str, "true") || strcmp(sym_str, "1") == 0) ? 1 : 0;
538
539 /* Resolve scale_bin relative to JSON directory if needed */
540 if (scale_bin_rel[0]) {
541 const char *slash = last_slash(json_path);
542 if (slash) {
543 char dir[512]; size_t dn = (size_t)(slash - json_path);
544 if (dn >= sizeof(dir)) dn = sizeof(dir) - 1;
545 memcpy(dir, json_path, dn); dir[dn] = '\0';
546 join_path(out_meta->scale_bin, sizeof(out_meta->scale_bin), dir, scale_bin_rel);
547 } else {
548 cpyz(out_meta->scale_bin, sizeof(out_meta->scale_bin), scale_bin_rel);
549 }
550 } else {
551 out_meta->scale_bin[0] = '\0';
552 }
553
554 out_meta->present = 1;
555 free(jbuf);
556 return IE_IO_OK;
557}

568static float fp16_to_fp32(uint16_t h) {
569 uint32_t sign = (uint32_t)(h & 0x8000u) << 16;
570 uint32_t exp = (h & 0x7C00u) >> 10;
571 uint32_t mant = (h & 0x03FFu);
572 uint32_t f;
573 if (exp == 0) {
574 if (mant == 0) {
575 f = sign; /* zero */
576 } else {
577 /* subnormal: normalize mantissa */
578 int e = -1;
579 do { e++; mant <<= 1; } while ((mant & 0x0400u) == 0);
580 mant &= 0x03FFu;
581 f = sign | ((uint32_t)(127 - 15 - e) << 23) | (mant << 13);
582 }
583 } else if (exp == 0x1F) {
584 f = sign | 0x7F800000u | (mant << 13); /* Inf/NaN */
585 } else {
586 f = sign | ((exp + (127 - 15)) << 23) | (mant << 13);
587 }
588 float out;
589 memcpy(&out, &f, sizeof(out));
590 return out;
591}

612int ie_weights_decode_int4(const char *packed_path,
613 const char *scales_path,
614 size_t rows,
615 size_t cols,
616 int per_row,
617 float *dst) {
618 if (!packed_path || !scales_path || !dst || rows == 0 || cols == 0) return IE_IO_ERR_ARGS;
619
620 /* Read packed blob */
621 int fdw = open(packed_path, O_RDONLY);
622 if (fdw < 0) return IE_IO_ERR_OPEN;
623
624 struct stat stw;
625 if (fstat(fdw, &stw) != 0) { close(fdw); return IE_IO_ERR_STAT; }
626
627 const size_t need_packed = rows * ie_int4_rowbytes(cols);
628 if ((size_t)stw.st_size < need_packed) { close(fdw); return IE_IO_ERR_READ; }
629
630 uint8_t *buf_packed = (uint8_t*)malloc(need_packed);
631 if (!buf_packed) { close(fdw); return IE_IO_ERR_ALLOC; }
632
633 ssize_t rd = ie_pread(fdw, buf_packed, need_packed, 0);
634 close(fdw);
635 if (rd < 0 || (size_t)rd != need_packed) { free(buf_packed); return IE_IO_ERR_READ; }
636
637 /* Read scales (auto-detect fp16 or fp32) */
638 int fds = open(scales_path, O_RDONLY);
639 if (fds < 0) { free(buf_packed); return IE_IO_ERR_OPEN; }
640
641 struct stat sts;
642 if (fstat(fds, &sts) != 0) { close(fds); free(buf_packed); return IE_IO_ERR_STAT; }
643
644 size_t nsc = per_row ? rows : 1;
645 size_t need16 = nsc * sizeof(uint16_t);
646 size_t need32 = nsc * sizeof(float);
647
648 int is16 = ((size_t)sts.st_size == need16);
649 int is32 = ((size_t)sts.st_size == need32);
650 if (!(is16 || is32)) { close(fds); free(buf_packed); return IE_IO_ERR_READ; }
651
652 float *buf_scales = (float*)malloc(need32);
653 if (!buf_scales) { close(fds); free(buf_packed); return IE_IO_ERR_ALLOC; }
654
655 if (is32) {
656 ssize_t rds = ie_pread(fds, buf_scales, need32, 0);
657 close(fds);
658 if (rds < 0 || (size_t)rds != need32) { free(buf_scales); free(buf_packed); return IE_IO_ERR_READ; }
659 } else {
660 uint16_t *tmp = (uint16_t*)malloc(need16);
661 if (!tmp) { close(fds); free(buf_scales); free(buf_packed); return IE_IO_ERR_ALLOC; }
662 ssize_t rds = ie_pread(fds, tmp, need16, 0);
663 close(fds);
664 if (rds < 0 || (size_t)rds != need16) { free(tmp); free(buf_scales); free(buf_packed); return IE_IO_ERR_READ; }
665 for (size_t i = 0; i < nsc; ++i) buf_scales[i] = fp16_to_fp32(tmp[i]);
666 free(tmp);
667 }
668
669 /* Dequantize */
670 int qst = 0;
671 if (per_row) {
672 qst = ie_int4_dequantize_per_row(buf_packed, rows, cols, buf_scales, dst);
673 } else {
674 qst = ie_int4_dequantize_per_tensor(buf_packed, rows, cols, buf_scales[0], dst);
675 }
676
677 free(buf_scales);
678 free(buf_packed);
679
680 if (qst != IE_INT4_STATUS_OK) return IE_IO_ERR_DECODE;
681 return IE_IO_OK;
682}
