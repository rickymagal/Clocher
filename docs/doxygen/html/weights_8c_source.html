655 static float fp16_to_fp32(uint16_t h) {
656  uint32_t sign = (uint32_t)(h & 0x8000u) << 16;
657  uint32_t exp = (h & 0x7C00u) >> 10;
658  uint32_t mant = (h & 0x03FFu);
659  uint32_t f;
660  if (exp == 0) {
661  if (mant == 0) {
662  f = sign; /* zero */
663  } else {
664  /* subnormal: normalize mantissa */
665  int e = -1;
666  do { e++; mant <<= 1; } while ((mant & 0x0400u) == 0);
667  mant &= 0x03FFu;
668  f = sign | ((uint32_t)(127 - 15 - e) << 23) | (mant << 13);
669  }
670  } else if (exp == 0x1F) {
671  f = sign | 0x7F800000u | (mant << 13); /* Inf/NaN */
672  } else {
673  f = sign | ((exp + (127 - 15)) << 23) | (mant << 13);
674  }
675  float out;
676  memcpy(&out, &f, sizeof(out));
677  return out;
678 }

699 int ie_weights_decode_int4(const char *packed_path,
700  const char *scales_path,
701  size_t rows,
702  size_t cols,
703  int per_row,
704  float *dst) {
705  if (!packed_path || !scales_path || !dst || rows == 0 || cols == 0) return IE_IO_ERR_ARGS;
706 
707  /* Read packed blob */
708  int fdw = open(packed_path, O_RDONLY);
709  if (fdw < 0) return IE_IO_ERR_OPEN;
710 
711  struct stat stw;
712  if (fstat(fdw, &stw) != 0) { close(fdw); return IE_IO_ERR_STAT; }
713 
714  const size_t need_packed = rows * ie_int4_rowbytes(cols);
715  if ((size_t)stw.st_size < need_packed) { close(fdw); return IE_IO_ERR_READ; }
716 
717  uint8_t *buf_packed = (uint8_t*)malloc(need_packed);
718  if (!buf_packed) { close(fdw); return IE_IO_ERR_ALLOC; }
719 
720  ssize_t rd = ie_pread(fdw, buf_packed, need_packed, 0);
721  close(fdw);
722  if (rd < 0 || (size_t)rd != need_packed) { free(buf_packed); return IE_IO_ERR_READ; }
723 
724  /* Read scales (auto-detect fp16 or fp32) */
725  int fds = open(scales_path, O_RDONLY);
726  if (fds < 0) { free(buf_packed); return IE_IO_ERR_OPEN; }
727 
728  struct stat sts;
729  if (fstat(fds, &sts) != 0) { close(fds); free(buf_packed); return IE_IO_ERR_STAT; }
730 
731  size_t nsc = per_row ? rows : 1;
732  size_t need16 = nsc * sizeof(uint16_t);
733  size_t need32 = nsc * sizeof(float);
734 
735  int is16 = ((size_t)sts.st_size == need16);
736  int is32 = ((size_t)sts.st_size == need32);
737  if (!(is16 || is32)) { close(fds); free(buf_packed); return IE_IO_ERR_READ; }
738 
739  float *buf_scales = (float*)malloc(need32);
740  if (!buf_scales) { close(fds); free(buf_packed); return IE_IO_ERR_ALLOC; }
741 
742  if (is32) {
743  ssize_t rds = ie_pread(fds, buf_scales, need32, 0);
744  close(fds);
745  if (rds < 0 || (size_t)rds != need32) { free(buf_scales); free(buf_packed); return IE_IO_ERR_READ; }
746  } else {
747  uint16_t *tmp = (uint16_t*)malloc(need16);
748  if (!tmp) { close(fds); free(buf_scales); free(buf_packed); return IE_IO_ERR_ALLOC; }
749  ssize_t rds = ie_pread(fds, tmp, need16, 0);
750  close(fds);
751  if (rds < 0 || (size_t)rds != need16) { free(tmp); free(buf_scales); free(buf_packed); return IE_IO_ERR_READ; }
752  for (size_t i = 0; i < nsc; ++i) buf_scales[i] = fp16_to_fp32(tmp[i]);
753  free(tmp);
754  }
755 
756  /* Dequantize */
757  int qst = 0;
758  if (per_row) {
759  qst = ie_int4_dequantize_per_row(buf_packed, rows, cols, buf_scales, dst);
760  } else {
761  qst = ie_int4_dequantize_per_tensor(buf_packed, rows, cols, buf_scales[0], dst);
762  }
763 
764  free(buf_scales);
765  free(buf_packed);
766 
767  if (qst != IE_INT4_STATUS_OK) return IE_IO_ERR_DECODE;
768  return IE_IO_OK;
769 }
