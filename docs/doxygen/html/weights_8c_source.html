753 static float fp16_to_fp32(uint16_t h) {
754  uint32_t sign = (uint32_t)(h & 0x8000u) << 16;
755  uint32_t exp = (h & 0x7C00u) >> 10;
756  uint32_t mant = (h & 0x03FFu);
757  uint32_t f;
758  if (exp == 0) {
759  if (mant == 0) {
760  f = sign;
761  } else {
762  int e = -1;
763  do {
764  e++;
765  mant <<= 1;
766  } while ((mant & 0x0400u) == 0);
767  mant &= 0x03FFu;
768  f = sign | ((uint32_t)(127 - 15 - e) << 23) | (mant << 13);
769  }
770  } else if (exp == 0x1F) {
771  f = sign | 0x7F800000u | (mant << 13);
772  } else {
773  f = sign | ((exp + (127 - 15)) << 23) | (mant << 13);
774  }
775  float out;
776  memcpy(&out, &f, sizeof(out));
777  return out;
778 }
