75static void join_path(char *out, size_t outsz, const char *dir, const char *file) {
76 if (!out || outsz == 0) return;
77 out[0] = '\0';
78 if (!file || !*file) return;
79
80 if (!dir || !*dir) { cpyz(out, outsz, file); return; }
81
82 const size_t ld = strlen(dir);
83 const size_t lf = strlen(file);
84 const int need_slash = (ld > 0 && dir[ld - 1] != '/');
85
86 size_t pos = 0;
87
88 /* copy dir */
89 size_t copyd = ld;
90 if (copyd >= outsz) copyd = outsz - 1;
91 if (copyd > 0) { memcpy(out + pos, dir, copyd); pos += copyd; }
92
93 if (pos < outsz - 1 && need_slash) { out[pos++] = '/'; }
94
95 /* copy file */
96 if (pos < outsz - 1) {
97 size_t room = (outsz - 1) - pos;
98 size_t copyf = (lf > room) ? room : lf;
99 if (copyf > 0) { memcpy(out + pos, file, copyf); pos += copyf; }
100 }
101
102 out[pos] = '\0';
103}

115static int read_all_text(const char *path, char **buf, size_t *len_out) {
116 if (!path || !buf) return -1;
117 *buf = NULL; if (len_out) *len_out = 0;
118
119 FILE *f = fopen(path, "rb");
120 if (!f) return -2;
121
122 if (fseek(f, 0, SEEK_END) != 0) { fclose(f); return -3; }
123 long n = ftell(f);
124 if (n < 0) { fclose(f); return -3; }
125 if (fseek(f, 0, SEEK_SET) != 0) { fclose(f); return -3; }
126
127 size_t size = (size_t)n;
128 char *mem = (char*)malloc(size + 1);
129 if (!mem) { fclose(f); return -4; }
130
131 size_t rd = fread(mem, 1, size, f);
132 fclose(f);
133 if (rd != size) { free(mem); return -5; }
134
135 /* Strip UTF-8 BOM if present */
136 if (size >= 3 &&
137 (unsigned char)mem[0] == 0xEF &&
138 (unsigned char)mem[1] == 0xBB &&
139 (unsigned char)mem[2] == 0xBF) {
140 size -= 3;
141 memmove(mem, mem + 3, size);
142 }
143
144 mem[size] = '\0';
145 *buf = mem;
146 if (len_out) *len_out = size;
147 return 0;
148}

228static ssize_t ie_pread(int fd, void *buf, size_t count, off_t offset) {
229 /* If pread is declared, use it directly. */
230 #if defined(_XOPEN_SOURCE) || defined(_POSIX_C_SOURCE)
231 /* Some libcs still require the symbol even if declared, so try and use it. */
232 #ifdef __GLIBC__
233 return pread(fd, buf, count, offset);
234 #else
235 /* Attempt direct call; if not linked, fallback below at compile-time. */
236 return pread(fd, buf, count, offset);
237 #endif
238 #else
239 /* Fallback: save/restore file position around a plain read. */
240 off_t cur = lseek(fd, 0, SEEK_CUR);
241 if (cur == (off_t)-1) return -1;
242 if (lseek(fd, offset, SEEK_SET) == (off_t)-1) return -1;
243 ssize_t r = read(fd, buf, count);
244 /* Best-effort restore; ignore restore error to preserve original read errno. */
245 (void)lseek(fd, cur, SEEK_SET);
246 return r;
247 #endif
248}

254int ie_weights_open(const char *json_path, const char *bin_path, ie_weights_t *out) {
255 if (!out) return IE_IO_ERR_ARGS;
256 memset(out, 0, sizeof(*out));
257
258 if (!json_path || !*json_path) return IE_IO_ERR_ARGS;
259 cpyz(out->json_path, sizeof(out->json_path), json_path);
260
261 /* Read JSON text */
262 char *jbuf = NULL; size_t jlen = 0;
263 if (read_all_text(json_path, &jbuf, &jlen) != 0 || !jbuf || jlen == 0) {
264 return IE_IO_ERR_JSON;
265 }
266
267 /* Parse relaxed header keys */
268 int ver = 0; (void)scan_json_key_int(jbuf, "version", &ver);
269 if (ver <= 0) ver = 1;
270 out->version = ver;
271
272 char dtype[16]; dtype[0] = '\0';
273 (void)scan_json_key_string(jbuf, "dtype", dtype, sizeof(dtype));
274 if (dtype[0] == '\0') cpyz(dtype, sizeof(dtype), "float32");
275 cpyz(out->dtype, sizeof(out->dtype), dtype);
276
277 /* Determine weights path */
278 char resolved_bin[512]; resolved_bin[0] = '\0';
279 if (bin_path && *bin_path) {
280 cpyz(resolved_bin, sizeof(resolved_bin), bin_path);
281 } else {
282 char bin_key[256]; bin_key[0] = '\0';
283 (void)scan_json_key_string(jbuf, "bin", bin_key, sizeof(bin_key));
284 if (bin_key[0] == '\0') { free(jbuf); return IE_IO_ERR_BIN_UNSPEC; }
285
286 const char *slash = last_slash(json_path);
287 if (slash) {
288 char dir[512]; size_t dn = (size_t)(slash - json_path);
289 if (dn >= sizeof(dir)) dn = sizeof(dir) - 1;
290 memcpy(dir, json_path, dn); dir[dn] = '\0';
291 join_path(resolved_bin, sizeof(resolved_bin), dir, bin_key);
292 } else {
293 cpyz(resolved_bin, sizeof(resolved_bin), bin_key);
294 }
295 }
296 cpyz(out->weights_path, sizeof(out->weights_path), resolved_bin);
297
298 /* Stat the weights file to get its size */
299 struct stat st;
300 if (stat(out->weights_path, &st) != 0 || st.st_size < 0) {
301 free(jbuf);
302 return IE_IO_ERR_STAT;
303 }
304 out->bin_size_bytes = (size_t)st.st_size;
305 out->loaded = 1;
306
307 free(jbuf);
308 return IE_IO_OK;
309}
