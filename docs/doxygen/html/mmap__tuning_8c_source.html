136int ie_mmap_readahead(int fd, off_t offset, size_t length) {
137 if (fd < 0) { errno = EBADF; return -1; }
138#if defined(__linux__)
139 /* Try the Linux readahead(2) syscall first. */
140 #ifdef SYS_readahead
141 /* NB: readahead returns 0 on success, or -1 with errno on failure. */
142 ssize_t rc = (ssize_t)syscall(SYS_readahead, fd, (off64_t)offset, (size_t)length);
143 if (rc == 0) return 0;
144 if (errno == ENOSYS || errno == EINVAL || errno == EPERM) {
145 /* Fall back to posix_fadvise WILLNEED. */
146 (void)ie_mmap_fadvise(fd, offset, (off_t)length, IE_PAGE_WILLNEED);

162size_t ie_mmap_kernel_hugepagesize(void) {
163#if defined(__linux__)
164 FILE *f = fopen("/proc/meminfo", "r");
165 if (!f) return 0;
166 char line[256];
167 size_t kbytes = 0;
168 while (fgets(line, (int)sizeof(line), f)) {
169 /* Example: "Hugepagesize: 2048 kB" */
170 if (strncmp(line, "Hugepagesize:", 13) == 0) {
171 unsigned long v = 0;
172 if (sscanf(line + 13, "%lu kB", &v) == 1) {
173 kbytes = (size_t)v;
174 break;
175 }
176 }
177 }
178 fclose(f);
179 return kbytes ? (kbytes * 1024u) : 0u;
180#else
181 return 0u;
182#endif
183}

185int ie_mmap_prefault_touch_ro(const void *addr, size_t length, size_t stride,
186 uint64_t *checksum) {
187 if (!addr || length == 0) { errno = EINVAL; return -1; }
188 const size_t step = (stride == 0 ? 64u : stride);
189 const volatile unsigned char *p = (const volatile unsigned char *)addr;
190 uint64_t acc = 0u;
191 /* Bound the loop to [0, length). Use volatile loads to keep it intact. */
192 for (size_t i = 0; i < length; i += step) {
193 acc += p[i];
194 }
195 /* Touch the last byte to ensure the tail page is demand-paged too. */
196 acc += p[length - 1u];
197 if (checksum) *checksum = acc;
198 return 0;
199}
