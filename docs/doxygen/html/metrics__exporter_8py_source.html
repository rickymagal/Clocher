3 # Minimal Prometheus text exporter for engine benchmark results.
4 #
5 # Two modes:
6 # 1) --summary FILE
7 # Serve metrics based on a single JSON summary object produced by
8 # scripts/true_tps_strict.sh (the final block it prints).
9 #
10 # 2) --dir PATH
11 # Serve metrics computed from the most recent *.json file in PATH where
12 # each file is a single-run JSON line (also printed by true_tps_strict.sh).
13 # If both --dir and --summary are specified, --summary wins.
14 #
15 # This version also surfaces memory-related metrics and configuration knobs:
16 # - bytes touched, effective bandwidth, bytes-per-token (BPT)
17 # - activation quantization (act_quant), kv quantization (kv_quant)
18 # - prefetch controls, non-temporal loads, NUMA policy

194  try:
195  if self.path != "/metrics":

200 
201  obj: Dict[str, Any] = {}
202  if self.summary_path is not None:

204  elif self.dir_path is not None:
