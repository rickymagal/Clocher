4#!/usr/bin/env python3
5"""
6Minimal Prometheus text exporter for engine benchmark results.
7
8Two modes:
9 1) --summary FILE
10 Serve metrics based on a single JSON summary object produced by
11 scripts/true_tps_strict.sh (the final block it prints).
12
13 2) --dir PATH
14 Serve metrics computed from the most recent *.json file in PATH where each
15 file is a single-run JSON line (also printed by true_tps_strict.sh).
16 If both --dir and --summary are specified, --summary wins.
17
18The server exposes /metrics on the given port and refreshes metrics on demand.
19
20Examples:
21 python3 scripts/metrics_exporter.py --summary /tmp/final_summary.json
22 python3 scripts/metrics_exporter.py --dir /tmp/runs --port 9109
23"""

36def _find_latest_json(path: Path) -> Optional[Path]:
37 """Return the newest *.json file under path, or None."""
38 files = sorted(path.glob("*.json"), key=lambda p: p.stat().st_mtime, reverse=True)
39 return files[0] if files else None

42def _load_one_json_file(path: Path) -> Dict[str, Any]:
43 """Load a JSON object from file, return {} on error."""
44 try:
45 with open(path, "r", encoding="utf-8") as f:
46 txt = f.read().strip()
47 # handle either JSON object or a one-line object with trailing newline
48 return json.loads(txt)
49 except Exception:
50 return {}

53def _gauge_line(name: str, value: Any, labels: Dict[str, str] | None = None) -> str:
54 """Render a Prometheus gauge sample line."""
55 if value is None:
56 return ""
57 if labels:
58 lab = ",".join([f'{k}="{v}"' for (k, v) in labels.items()])
59 return f"{name}{{{lab}}} {value}\n"
60 return f"{name} {value}\n"

124class Handler(BaseHTTPRequestHandler):
125 """HTTP handler serving /metrics from either a summary file or dir."""
126
127 # These are set by the server factory below.
128 summary_path: Optional[Path] = None
129 dir_path: Optional[Path] = None

131 def do_GET(self) -> None: # noqa: N802
132 try:
133 if self.path != "/metrics":
134 self.send_response(404)
135 self.end_headers()
136 self.wfile.write(b"not found\n")
137 return
138
139 obj: Dict[str, Any] = {}
140 if self.summary_path is not None:
141 obj = _load_one_json_file(self.summary_path)
142 elif self.dir_path is not None:
143 latest = _find_latest_json(self.dir_path)
144 if latest is not None:
145 obj = _load_one_json_file(latest)
146
147 txt = collect_metrics_from_summary(obj if obj else {})
148 payload = txt.encode("utf-8")
149
150 self.send_response(200)
151 self.send_header("Content-Type", "text/plain; version=0.0.4")
152 self.send_header("Content-Length", str(len(payload)))
153 self.end_headers()
154 self.wfile.write(payload)
155 except Exception as e:
156 err = f"# exporter_error{{msg=\"{type(e).__name__}\"}} 1\n"
157 payload = err.encode("utf-8")
158 self.send_response(200)
159 self.send_header("Content-Type", "text/plain; version=0.0.4")
160 self.send_header("Content-Length", str(len(payload)))
161 self.end_headers()
162 self.wfile.write(payload)

165def make_server(addr: str, port: int, summary: Optional[Path], directory: Optional[Path]) -> HTTPServer:
166 """Create an HTTPServer serving metrics from the given source."""
167 class _Factory(Handler):
168 summary_path = summary
169 dir_path = directory
170 return HTTPServer((addr, port), _Factory)

173def main() -> None:
174 ap = argparse.ArgumentParser()
175 ap.add_argument("--addr", default="0.0.0.0", help="Bind address (default: 0.0.0.0)")
176 ap.add_argument("--port", type=int, default=9109, help="Port (default: 9109)")
177 ap.add_argument("--summary", type=str, help="Path to final summary JSON file")
178 ap.add_argument("--dir", type=str, help="Directory of per-run JSON files")
179 args = ap.parse_args()
180
181 summary = Path(args.summary) if args.summary else None
182 directory = Path(args.dir) if args.dir else None
183 if not summary and not directory:
184 raise SystemExit("ERROR: specify --summary FILE or --dir DIR")
185
186 if summary and not summary.is_file():
187 raise SystemExit(f"ERROR: summary file not found: {summary}")
188 if directory and not directory.is_dir():
189 raise SystemExit(f"ERROR: directory not found: {directory}")
190
191 srv = make_server(args.addr, args.port, summary, directory)
192 print(f"[metrics_exporter] serving on http://{args.addr}:{args.port}/metrics", flush=True)
193 try:
194 srv.serve_forever()
195 except KeyboardInterrupt:
