1#!/usr/bin/env python3
2# =============================================================================
3# Minimal Prometheus text exporter for engine benchmark results.
4#
5# Two modes:
6# 1) --summary FILE
7# Serve metrics based on a single JSON summary object produced by
8# scripts/true_tps_strict.sh (the final block it prints).
9#
10# 2) --dir PATH
11# Serve metrics computed from the most recent *.json file in PATH where
12# each file is a single-run JSON line (also printed by true_tps_strict.sh).
13# If both --dir and --summary are specified, --summary wins.
14#
15# This version also surfaces memory-related metrics and configuration knobs:
16# - bytes touched, effective bandwidth, bytes-per-token (BPT)
17# - activation quantization (act_quant), kv quantization (kv_quant)
18# - prefetch controls, non-temporal loads, NUMA policy
19# =============================================================================

74def collect_metrics_from_summary(obj: Dict[str, Any]) -> str:
75 """Convert a summary JSON object to Prometheus text."""
76 host = socket.gethostname()
77
78 # Core labels (stable)
79 base_labels = {
80 "host": host,
81 "device": _label_str(obj.get("device")), # "cpu" | "gpu" | etc.
82 "model_dir": _label_str(obj.get("model_dir")),
83 "precision": _label_str(obj.get("precision")),
84 "affinity": _label_str(obj.get("affinity")),
85 "pretranspose": _label_str(obj.get("pretranspose")),
86 "batch": _label_str(obj.get("batch")),
87 "threads": _label_str(obj.get("threads")),
88 }
89
90 # Config/knobs as labels-only info gauges
91 cfg_labels = {
92 "host": host,
93 "act_quant": _label_str(obj.get("act_quant")), # "none"|"int8"|"fp8_e4m3"|...
94 "kv_quant": _label_str(obj.get("kv_quant")), # "none"|"int8"|...
95 "prefetch": _label_str(obj.get("prefetch")), # "off"|"on"|"auto"|N
96 "nt_loads": _label_str(obj.get("nt_loads")), # 0|1
97 "numa_policy": _label_str(obj.get("numa_policy")), # "auto"|"compact"|"scatter"|...
98 "ie_bytes_per_token": _label_str(obj.get("ie_bytes_per_token")),
99 "ie_stride_bytes": _label_str(obj.get("ie_stride_bytes")),
100 "ie_require_model": _label_str(obj.get("ie_require_model")),
101 "ie_verify_touch": _label_str(obj.get("ie_verify_touch")),
102 "prompts": _label_str(obj.get("prompts")),
103 "engine_bin": _label_str(obj.get("engine_bin")),
104 "git_commit": _label_str(obj.get("git_commit")),
105 }
106
107 # Numeric metrics
108 tokens_generated = obj.get("tokens_generated") or obj.get("tokens_sum") or 0
109 wall_time_s = obj.get("wall_time_s") or obj.get("wall_sum") or 0.0
110 tps_true = obj.get("tps_true") or 0.0
111 p50_ms = obj.get("latency_p50_ms") or obj.get("p50_mean") or 0.0
112 p95_ms = obj.get("latency_p95_ms") or obj.get("p95_mean") or 0.0
113 rss_mean_mb = obj.get("rss_peak_mb") or obj.get("rss_mean") or 0.0
114 rss_max_mb = obj.get("rss_peak_mb_max") or obj.get("rss_max") or 0.0
115 kv_hits = obj.get("kv_hits") or 0
116 kv_misses = obj.get("kv_misses") or 0
117
118 # Memory/IO related (optional in summary)
119 bytes_per_token = obj.get("ie_bytes_per_token")
120 bytes_touched = obj.get("bytes_touched") # may be bytes or GB depending on source
121 if isinstance(bytes_touched, str):
122 try:
123 bytes_touched = float(bytes_touched)
124 except Exception:
125 bytes_touched = None
126 eff_bw_gbps = obj.get("effective_bandwidth_gbps")
127
128 # If not provided, attempt a naive derivation
129 if bytes_touched is None and bytes_per_token and tokens_generated:
130 try:
131 bytes_touched = float(bytes_per_token) * float(tokens_generated)
132 except Exception:
133 bytes_touched = None
134
135 out = []
136
137 # Core totals
138 out.append(_help_type("ie_tokens_generated_total", "Total tokens generated across runs"))
139 out.append(_gauge_line("ie_tokens_generated_total", tokens_generated, base_labels))
140
141 out.append(_help_type("ie_wall_time_seconds", "Total wall time across runs (s)"))
142 out.append(_gauge_line("ie_wall_time_seconds", wall_time_s, base_labels))
143
144 out.append(_help_type("ie_tps_true", "True tokens-per-second across runs"))
145 out.append(_gauge_line("ie_tps_true", tps_true, base_labels))
146
147 # Latency
148 out.append(_help_type("ie_latency_p50_ms", "Mean p50 latency (ms)"))
149 out.append(_gauge_line("ie_latency_p50_ms", p50_ms, base_labels))
150
151 out.append(_help_type("ie_latency_p95_ms", "Mean p95 latency (ms)"))
152 out.append(_gauge_line("ie_latency_p95_ms", p95_ms, base_labels))
153
154 # Memory
155 out.append(_help_type("ie_rss_peak_mb_mean", "Mean peak RSS (MB) across runs"))
156 out.append(_gauge_line("ie_rss_peak_mb_mean", rss_mean_mb, base_labels))
157
158 out.append(_help_type("ie_rss_peak_mb_max", "Max peak RSS (MB) across runs"))
159 out.append(_gauge_line("ie_rss_peak_mb_max", rss_max_mb, base_labels))
160
161 # KV cache
162 out.append(_help_type("ie_kv_hits_total", "Total KV cache hits during runs"))
163 out.append(_gauge_line("ie_kv_hits_total", kv_hits, base_labels))
164
165 out.append(_help_type("ie_kv_misses_total", "Total KV cache misses during runs"))
166 out.append(_gauge_line("ie_kv_misses_total", kv_misses, base_labels))
167
168 # Spatial metrics
169 out.append(_help_type("ie_bytes_per_token", "Model bytes processed per generated token (bytes/token)"))
170 out.append(_gauge_line("ie_bytes_per_token", bytes_per_token or 0, base_labels))
171
172 out.append(_help_type("ie_bytes_touched_total", "Total bytes touched across runs"))
173 out.append(_gauge_line("ie_bytes_touched_total", bytes_touched or 0, base_labels))
174
175 out.append(_help_type("ie_effective_bandwidth_gbps", "Effective aggregate bandwidth (GB/s)"))
176 out.append(_gauge_line("ie_effective_bandwidth_gbps", eff_bw_gbps or 0, base_labels))
177
178 # Build/config info as labels
179 out.append("# HELP ie_config_info Build/runtime knobs as labels\n# TYPE ie_config_info gauge\n")
180 out.append(_gauge_line("ie_config_info", 1, cfg_labels))
181
182 return "".join(out)

187class Handler(BaseHTTPRequestHandler):
188 """HTTP handler serving /metrics from either a summary file or dir."""
189
190 summary_path: Optional[Path] = None

193 def do_GET(self) -> None: # noqa: N802
194 try:
195 if self.path != "/metrics":
196 self.send_response(404)
197 self.end_headers()
198 self.wfile.write(b"not found\n")
199 return
200
201 obj: Dict[str, Any] = {}
202 if self.summary_path is not None:
203 obj = _load_one_json_file(self.summary_path)
204 elif self.dir_path is not None:
205 latest = _find_latest_json(self.dir_path)
206 if latest is not None:
207 obj = _load_one_json_file(latest)
208
209 txt = collect_metrics_from_summary(obj if obj else {})
210 payload = txt.encode("utf-8")
211
212 self.send_response(200)
213 self.send_header("Content-Type", "text/plain; version=0.0.4")
214 self.send_header("Content-Length", str(len(payload)))
215 self.end_headers()
216 self.wfile.write(payload)
217 except Exception as e:
218 err = f"# exporter_error{{msg=\"{type(e).__name__}\"}} 1\n"
219 payload = err.encode("utf-8")
220 self.send_response(200)
221 self.send_header("Content-Type", "text/plain; version=0.0.4")
222 self.send_header("Content-Length", str(len(payload)))
223 self.end_headers()
224 self.wfile.write(payload)
