44 def export_torch(checkpoint: str, key: str, out_bin: str, do_transpose: bool):
45  """
46  Export a 2D tensor from a PyTorch checkpoint to row-major float32 .bin.
47 
48  Returns
49  -------
50  tuple(int, int)
51  The tensor shape as (rows, cols).
52  """
53  try:
54  import torch
55  except Exception:
56  print("ERROR: torch is required. pip install torch", file=sys.stderr)
57  sys.exit(2)
58 
59  ckpt = torch.load(checkpoint, map_location="cpu")
60  if hasattr(ckpt, "state_dict"):
61  sd = ckpt.state_dict()
62  elif isinstance(ckpt, dict) and "state_dict" in ckpt:
63  sd = ckpt["state_dict"]
64  elif isinstance(ckpt, dict):
65  sd = ckpt
66  else:
67  print("ERROR: unsupported checkpoint format", file=sys.stderr)
68  sys.exit(2)
69 
70  if key not in sd:
71  print(f"ERROR: key '{key}' not found in state_dict", file=sys.stderr)
72  sys.exit(3)
73 
74  t = sd[key]
75  if not hasattr(t, "shape"):
76  print(f"ERROR: key '{key}' is not tensor-like", file=sys.stderr)
77  sys.exit(3)
78 
79  ten = t.detach().cpu().float()
80  if ten.ndim != 2:
81  print(f"ERROR: selected tensor is {ten.ndim}D; expected 2D [rows, cols]", file=sys.stderr)
82  sys.exit(3)
83 
84  if do_transpose:
85  ten = ten.t().contiguous()
86 
87  rows, cols = int(ten.shape[0]), int(ten.shape[1])
88  os.makedirs(os.path.dirname(out_bin) or ".", exist_ok=True)
89  with open(out_bin, "wb") as f:
90  f.write(ten.numpy().astype("float32").tobytes())
91  return rows, cols

94 def export_onnx(onnx_path: str, init_name: str, out_bin: str, do_transpose: bool):
95  """
96  Export a 2D initializer from an ONNX model to row-major float32 .bin.
97 
98  Returns
99  -------
100  tuple(int, int)
101  The tensor shape as (rows, cols).
102  """
103  try:
104  import onnx
105  from onnx import numpy_helper
106  except Exception:
107  print("ERROR: onnx is required. pip install onnx", file=sys.stderr)
108  sys.exit(2)
109 
110  model = onnx.load(onnx_path)
111  inits = {init.name: init for init in model.graph.initializer}
112  if init_name not in inits:
113  print(f"ERROR: initializer '{init_name}' not found", file=sys.stderr)
114  sys.exit(3)
115 
116  arr = numpy_helper.to_array(inits[init_name]).astype("float32", copy=False)
117  if arr.ndim != 2:
118  print(f"ERROR: selected initializer is {arr.ndim}D; expected 2D [rows, cols]", file=sys.stderr)
119  sys.exit(3)
120 
121  if do_transpose:
122  arr = arr.T.copy()
123 
124  rows, cols = int(arr.shape[0]), int(arr.shape[1])
125  os.makedirs(os.path.dirname(out_bin) or ".", exist_ok=True)
126  with open(out_bin, "wb") as f:
127  f.write(arr.astype("float32", copy=False).tobytes())
128  return rows, cols
