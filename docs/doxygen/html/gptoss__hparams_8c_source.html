57static int gptoss_join_path(char *out, size_t outsz, const char *dir, const char *file) {
58 if (!out || outsz == 0 || !file || !*file) return IE_IO_ERR_ARGS;
59 out[0] = '\0';
60
61 if (!dir || !*dir) {
62 const int n = snprintf(out, outsz, "%s", file);
63 if (n < 0 || (size_t)n >= outsz) return IE_IO_ERR_DECODE;
64 return IE_IO_OK;
65 }
66
67 const size_t ld = strlen(dir);
68 const int need_slash = (ld > 0 && dir[ld - 1] != '/');
69
70 const int n = snprintf(out, outsz, need_slash ? "%s/%s" : "%s%s", dir, file);
71 if (n < 0 || (size_t)n >= outsz) return IE_IO_ERR_DECODE;
72 return IE_IO_OK;
73}

82static int gptoss_read_all_text(const char *path, char **out_buf, size_t *out_len) {
83 if (!path || !out_buf) return IE_IO_ERR_ARGS;
84 *out_buf = NULL;
85 if (out_len) *out_len = 0;
86
87 FILE *f = fopen(path, "rb");
88 if (!f) return IE_IO_ERR_OPEN;
89
90 if (fseek(f, 0, SEEK_END) != 0) {
91 fclose(f);
92 return IE_IO_ERR_READ;
93 }
94 long n = ftell(f);
95 if (n < 0) {
96 fclose(f);
97 return IE_IO_ERR_READ;
98 }
99 if (fseek(f, 0, SEEK_SET) != 0) {
100 fclose(f);
101 return IE_IO_ERR_READ;
102 }
103
104 size_t size = (size_t)n;
105 char *buf = (char *)malloc(size + 1u);
106 if (!buf) {
107 fclose(f);
108 return IE_IO_ERR_ALLOC;
109 }
110
111 const size_t rd = fread(buf, 1, size, f);
112 fclose(f);
113 if (rd != size) {
114 free(buf);
115 return IE_IO_ERR_READ;
116 }
117
118 /* Strip UTF-8 BOM if present. */
119 if (size >= 3u &&
120 (unsigned char)buf[0] == 0xEFu &&
121 (unsigned char)buf[1] == 0xBBu &&
122 (unsigned char)buf[2] == 0xBFu) {
123 memmove(buf, buf + 3, size - 3u);
124 size -= 3u;
125 }
126
127 buf[size] = '\0';
128 *out_buf = buf;
129 if (out_len) *out_len = size;
130 return IE_IO_OK;
131}

137static const char *gptoss_skip_ws(const char *p) {
138 if (!p) return NULL;
139 while (*p == ' ' || *p == '\t' || *p == '\r' || *p == '\n') ++p;

143static const char *gptoss_find_value_start(const char *json, const char *key) {
144 if (!json || !key || !*key) return NULL;
145
146 const size_t klen = strlen(key);
147 const char *p = json;
148
149 for (;;) {
150 const char *q = strstr(p, "\"");
151 if (!q) break;
152 q++; /* past quote */
153 if (strncmp(q, key, klen) == 0 && q[klen] == '"') {
154 const char *c = q + klen + 1;
155 c = gptoss_skip_ws(c);
156 if (!c || *c != ':') {
157 p = q;
158 continue;
159 }
160 c++;
161 c = gptoss_skip_ws(c);
162 return c;
163 }
164 p = q;
165 }
166 return NULL;
167}

214static int gptoss_scan_string(const char *json, const char *key, char *out, size_t outsz) {
215 if (!json || !key || !out || outsz == 0) return -1;
216 out[0] = '\0';
217
218 const char *v = gptoss_find_value_start(json, key);
219 if (!v) return 0;
220
221 v = gptoss_skip_ws(v);
222 if (!v || *v != '"') return 0;
223 v++;
224
225 size_t di = 0;
226 for (const char *p = v; *p; ++p) {
227 if (*p == '"') break;
228 if (*p == '\\') {
229 const char n = *(p + 1);
230 if (n == '"' || n == '\\' || n == '/') {
231 if (di + 1 < outsz) out[di++] = n;
232 p++;
233 continue;
234 }
235 if (n != '\0') {
236 if (di + 1 < outsz) out[di++] = n;
237 p++;
238 continue;
239 }
240 }
241 if (di + 1 < outsz) out[di++] = *p;
242 }
243
244 out[di] = '\0';
245 return 1;
246}

248static int gptoss_scan_object_region(const char *json,
249 const char *key,
250 const char **out_obj,
251 size_t *out_len) {
252 if (!json || !key || !out_obj || !out_len) return -1;
253 *out_obj = NULL;
254 *out_len = 0;
255
256 const char *v = gptoss_find_value_start(json, key);
257 if (!v) return 0;
258
259 v = gptoss_skip_ws(v);
260 if (!v || *v != '{') return 0;
261
262 const char *start = v;
263 int depth = 0;
264 int in_str = 0;
265 int esc = 0;
266
267 for (const char *p = v; *p; ++p) {
268 const char ch = *p;
269
270 if (in_str) {
271 if (esc) {
272 esc = 0;
273 continue;
274 }
275 if (ch == '\\') {
276 esc = 1;
277 continue;
278 }
279 if (ch == '"') {
280 in_str = 0;
281 continue;
282 }
283 continue;
284 }
285
286 if (ch == '"') {
287 in_str = 1;
288 continue;
289 }
290 if (ch == '{') {
291 depth++;
292 continue;
293 }
294 if (ch == '}') {
295 depth--;
296 if (depth == 0) {
297 *out_obj = start;
298 *out_len = (size_t)((p + 1) - start);
299 return 1;
300 }
301 continue;
302 }
303 }
304
305 return 0;
306}

312static const char *iejson_find_tensor_region(const char *json, const char *tensor_name) {
313 if (!json || !tensor_name || !*tensor_name) return NULL;
314
315 const size_t nlen = strlen(tensor_name);
316
317 size_t need1 = strlen("\"name\":\"\"") + nlen;
318 char *needle1 = (char *)malloc(need1 + 1u);
319 if (!needle1) return NULL;
320 snprintf(needle1, need1 + 1u, "\"name\":\"%s\"", tensor_name);
321
322 const char *p = strstr(json, needle1);
323 free(needle1);
324 if (p) return p;
325
326 size_t need2 = strlen("\"name\": \"\"") + nlen;
327 char *needle2 = (char *)malloc(need2 + 1u);
328 if (!needle2) return NULL;
329 snprintf(needle2, need2 + 1u, "\"name\": \"%s\"", tensor_name);
330
331 p = strstr(json, needle2);

336static int iejson_parse_shape_dims(const char *p, uint32_t *dims, size_t max_dims, size_t *out_ndims) {
337 if (!p || !dims || max_dims == 0 || !out_ndims) return 0;
338 *out_ndims = 0;
339
340 const char *s = strstr(p, "\"shape\"");
341 if (!s) return 0;
342 s = strchr(s, '[');
343 if (!s) return 0;
344 s++;
345
346 size_t nd = 0;
347 for (;;) {
348 s = gptoss_skip_ws(s);
349 if (!s || !*s) return 0;
350 if (*s == ']') break;
351 if (!isdigit((unsigned char)*s)) return 0;
352
353 errno = 0;
354 char *end = NULL;
355 unsigned long v = strtoul(s, &end, 10);
356 if (end == s || errno != 0) return 0;
357 if (v > 0xFFFFFFFFul) return 0;
358
359 if (nd < max_dims) dims[nd] = (uint32_t)v;
360 nd++;
361 s = end;
362 s = gptoss_skip_ws(s);
363 if (*s == ',') {
364 s++;
365 continue;
366 }
367 if (*s == ']') break;
368 }
369
370 *out_ndims = nd;
371 return 1;
372}

436static int gptoss_parse_config_json(const char *json, gptoss_hparams_ex_t *out_ex) {
437 if (!json || !out_ex) return IE_IO_ERR_ARGS;
438
439 uint32_t n_layers = 0, n_heads = 0, d_model_cfg = 0, d_ff_cfg = 0, vocab_cfg = 0, max_seq_cfg = 0;
440
441 if (gptoss_scan_u32(json, "num_hidden_layers", &n_layers) != 1) return IE_IO_ERR_JSON;
442 if (gptoss_scan_u32(json, "num_attention_heads", &n_heads) != 1) return IE_IO_ERR_JSON;
443 if (gptoss_scan_u32(json, "hidden_size", &d_model_cfg) != 1) return IE_IO_ERR_JSON;
444 if (gptoss_scan_u32(json, "intermediate_size", &d_ff_cfg) != 1) return IE_IO_ERR_JSON;
445 if (gptoss_scan_u32(json, "vocab_size", &vocab_cfg) != 1) return IE_IO_ERR_JSON;
446
447 if (gptoss_scan_u32(json, "max_position_embeddings", &max_seq_cfg) != 1) {
448 if (gptoss_scan_u32(json, "max_seq_len", &max_seq_cfg) != 1) {
449 if (gptoss_scan_u32(json, "seq_length", &max_seq_cfg) != 1) {
450 return IE_IO_ERR_JSON;
451 }
452 }
453 }
454
455 uint32_t n_kv_heads = 0;
456 if (gptoss_scan_u32(json, "num_key_value_heads", &n_kv_heads) != 1) {
457 n_kv_heads = n_heads;
458 }
459
460 if (n_layers == 0 || n_heads == 0 || n_kv_heads == 0 || d_model_cfg == 0 || d_ff_cfg == 0 ||
461 vocab_cfg == 0 || max_seq_cfg == 0) {
462 return IE_IO_ERR_JSON;
463 }
464
465 out_ex->core.n_layers = n_layers;
466 out_ex->core.n_heads = n_heads;
467 out_ex->core.n_kv_heads = n_kv_heads;
468 out_ex->core.d_model = d_model_cfg;
469 out_ex->core.d_ff = d_ff_cfg;
470 out_ex->core.vocab_size = vocab_cfg;
471 out_ex->core.max_seq = max_seq_cfg;
472
473 (void)gptoss_scan_bool(json, "tie_word_embeddings", &out_ex->tie_word_embeddings);
474
475 {
476 float eps = 0.0f;
477 if (gptoss_scan_f32(json, "rms_norm_eps", &eps) == 1) out_ex->rms_norm_eps = eps;
478 else if (gptoss_scan_f32(json, "rms_epsilon", &eps) == 1) out_ex->rms_norm_eps = eps;
479 else if (gptoss_scan_f32(json, "layer_norm_eps", &eps) == 1) out_ex->rms_norm_eps = eps;
480 }
481
482 {
483 float theta = 0.0f;
484 if (gptoss_scan_f32(json, "rope_theta", &theta) == 1) out_ex->rope_theta = theta;
485 }
486
487 {
488 const char *obj = NULL;
489 size_t obj_len = 0;
490 if (gptoss_scan_object_region(json, "rope_scaling", &obj, &obj_len) == 1 && obj && obj_len > 2) {
491 int got_type = gptoss_scan_string(obj, "type", out_ex->rope_scaling_type,
492 sizeof(out_ex->rope_scaling_type));
493 if (got_type != 1) {
494 (void)gptoss_scan_string(obj, "rope_type", out_ex->rope_scaling_type,
495 sizeof(out_ex->rope_scaling_type));
496 }
497
498 {
499 float factor = 0.0f;
500 int got_factor = gptoss_scan_f32(obj, "factor", &factor);
501 if (got_factor != 1) {
502 got_factor = gptoss_scan_f32(obj, "scaling_factor", &factor);
503 }
504 if (got_factor == 1 && factor > 0.0f) out_ex->rope_scaling_factor = factor;
505 }
506
507 {
508 uint32_t orig = 0;
509 if (gptoss_scan_u32(obj, "original_max_position_embeddings", &orig) == 1) {
510 out_ex->rope_scaling_original_max_position_embeddings = orig;
511 }
512 }
513 }
514 }
515
516 return IE_IO_OK;
517}

558static int gptoss_apply_iejson_inference(const char *model_dir, gptoss_hparams_ex_t *io_ex) {
559 if (!model_dir || !io_ex) return IE_IO_ERR_ARGS;
560
561 char iep[1024];
562 const int r = gptoss_resolve_iejson_path(model_dir, iep, sizeof(iep));
563 if (r != IE_IO_OK) return r;
564
565 char *buf = NULL;
566 size_t len = 0;
567 const int rc = gptoss_read_all_text(iep, &buf, &len);
568 if (rc != IE_IO_OK) return rc;
569 if (!buf || len == 0) {
570 free(buf);
571 return IE_IO_ERR_READ;
572 }
573
574 /* Infer q_dim, kv_dim, and d_model from q_proj/k_proj shapes. */
575 uint32_t q_dims[4] = {0, 0, 0, 0};
576 uint32_t k_dims[4] = {0, 0, 0, 0};
577 uint32_t v_dims[4] = {0, 0, 0, 0};
578 size_t qn = 0, kn = 0, vn = 0;
579
580 const int has_q = gptoss_iejson_get_shape(buf, "model.layers.0.self_attn.q_proj.weight", q_dims, 4, &qn);
581 const int has_k = gptoss_iejson_get_shape(buf, "model.layers.0.self_attn.k_proj.weight", k_dims, 4, &kn);
582 const int has_v = gptoss_iejson_get_shape(buf, "model.layers.0.self_attn.v_proj.weight", v_dims, 4, &vn);
583
584 if (has_q && has_k && has_v && qn >= 2 && kn >= 2 && vn >= 2) {
585 const uint32_t q_dim = q_dims[0];
586 const uint32_t d_model_ie = q_dims[1];
587 const uint32_t kv_dim_k = k_dims[0];
588 const uint32_t kv_dim_v = v_dims[0];
589
590 if (q_dim > 0 && d_model_ie > 0 && kv_dim_k > 0 && kv_dim_k == kv_dim_v) {
591 if (io_ex->core.d_model != d_model_ie) {
592 fprintf(stderr,
593 "[gptoss_hparams] Note: config hidden_size=%u but model.ie.json q_proj in_dim=%u; using %u.\n",
594 (unsigned)io_ex->core.d_model,
595 (unsigned)d_model_ie,
596 (unsigned)d_model_ie);
597 io_ex->core.d_model = d_model_ie;
598 }
599
600 if (io_ex->core.n_heads == 0) {
601 free(buf);
602 return IE_IO_ERR_JSON;
603 }
604 if ((q_dim % io_ex->core.n_heads) != 0) {
605 fprintf(stderr,
606 "[gptoss_hparams] Error: q_dim=%u not divisible by n_heads=%u.\n",
607 (unsigned)q_dim, (unsigned)io_ex->core.n_heads);
608 free(buf);
609 return IE_IO_ERR_DECODE;
610 }
611 const uint32_t d_head = q_dim / io_ex->core.n_heads;
612 if (d_head == 0) {
613 free(buf);
614 return IE_IO_ERR_DECODE;
615 }
616
617 uint32_t n_kv = io_ex->core.n_kv_heads;
618 if (n_kv == 0 || (kv_dim_k != (n_kv * d_head))) {
619 if ((kv_dim_k % d_head) != 0) {
620 fprintf(stderr,
621 "[gptoss_hparams] Error: kv_dim=%u not divisible by d_head=%u.\n",
622 (unsigned)kv_dim_k, (unsigned)d_head);
623 free(buf);
624 return IE_IO_ERR_DECODE;
625 }
626 n_kv = kv_dim_k / d_head;
627 fprintf(stderr,
628 "[gptoss_hparams] Note: overriding n_kv_heads to %u based on kv_dim=%u and d_head=%u.\n",
629 (unsigned)n_kv, (unsigned)kv_dim_k, (unsigned)d_head);
630 }
631
632 if (n_kv == 0 || n_kv > io_ex->core.n_heads) {
633 fprintf(stderr,
634 "[gptoss_hparams] Error: inferred n_kv_heads=%u invalid for n_heads=%u.\n",
635 (unsigned)n_kv, (unsigned)io_ex->core.n_heads);
636 free(buf);
637 return IE_IO_ERR_DECODE;
638 }
639
640 io_ex->core.d_head = d_head;
641 io_ex->core.n_kv_heads = n_kv;
642 }
643 }
644
645 /* Infer d_ff from MoE down_proj blocks: shape [n_experts, d_ff, ...] */
646 {
647 uint32_t dp_dims[8] = {0};
648 size_t dpn = 0;
649 const int has_dp = gptoss_iejson_get_shape(buf, "model.layers.0.mlp.experts.down_proj_blocks", dp_dims, 8, &dpn);
650 if (has_dp && dpn >= 2 && dp_dims[1] > 0) {
651 const uint32_t dff = dp_dims[1];
652 if (io_ex->core.d_ff != dff) {
653 fprintf(stderr,
654 "[gptoss_hparams] Note: overriding d_ff from %u to %u based on down_proj_blocks.\n",
655 (unsigned)io_ex->core.d_ff, (unsigned)dff);
656 io_ex->core.d_ff = dff;
657 }
658 }
659 }
660
661 free(buf);
662 return IE_IO_OK;
663}

665static int gptoss_finalize_and_validate(const char *model_dir, gptoss_hparams_ex_t *io_ex) {
666 if (!io_ex) return IE_IO_ERR_ARGS;
667
668 /* Clamp max_seq to avoid accidental enormous KV allocations.
669 Override with IE_MAX_SEQ (must be > 0). */
670 {
671 const uint32_t max_cfg = io_ex->core.max_seq;
672 const uint32_t max_env = gptoss_env_u32("IE_MAX_SEQ", 0u);
673 uint32_t max_use = max_cfg;
674
675 if (max_env > 0u) {
676 max_use = (max_env < max_cfg) ? max_env : max_cfg;
677 } else {
678 const uint32_t clamp_default = 4096u;
679 if (max_use > clamp_default) {
680 fprintf(stderr,
681 "[gptoss_hparams] Note: clamping max_seq from %u to %u (set IE_MAX_SEQ to override).\n",
682 (unsigned)max_use, (unsigned)clamp_default);
683 max_use = clamp_default;
684 }
685 }
686
687 if (max_use == 0u) return IE_IO_ERR_JSON;
688 io_ex->core.max_seq = max_use;
689 }
690
691 if (io_ex->core.n_layers == 0 || io_ex->core.n_heads == 0 || io_ex->core.n_kv_heads == 0 ||
692 io_ex->core.d_model == 0 || io_ex->core.d_ff == 0 ||
693 io_ex->core.vocab_size == 0 || io_ex->core.max_seq == 0) {
694 fprintf(stderr, "[gptoss_hparams] Error: missing required hyperparameters after load.\n");
695 return IE_IO_ERR_JSON;
696 }
697
698 /* Apply IEJSON inference (d_head, d_ff fixes, possible kv-head override). */
699 if (model_dir) {
700 const int rc = gptoss_apply_iejson_inference(model_dir, io_ex);
701 if (rc != IE_IO_OK) return rc;
702 }
703
704 /* Export linear RoPE scaling to the RoPE helper, if present and not overridden. */
705 gptoss_apply_rope_scaling_env_if_absent(io_ex);
706
707 /* Validate after IEJSON overrides. */
708 if (io_ex->core.d_head == 0 || io_ex->core.d_model == 0) return IE_IO_ERR_DECODE;
709 if (io_ex->core.n_heads == 0 || io_ex->core.n_kv_heads == 0) return IE_IO_ERR_DECODE;
710 if (io_ex->core.vocab_size == 0) return IE_IO_ERR_DECODE;
711
712 if (io_ex->core.n_kv_heads > io_ex->core.n_heads) {
713 fprintf(stderr,
714 "[gptoss_hparams] Error: n_kv_heads=%u > n_heads=%u.\n",
715 (unsigned)io_ex->core.n_kv_heads, (unsigned)io_ex->core.n_heads);
716 return IE_IO_ERR_DECODE;
717 }
718
719 if ((io_ex->core.n_heads % io_ex->core.n_kv_heads) != 0) {
720 fprintf(stderr,
721 "[gptoss_hparams] Error: n_heads=%u not divisible by n_kv_heads=%u.\n",
722 (unsigned)io_ex->core.n_heads, (unsigned)io_ex->core.n_kv_heads);
723 return IE_IO_ERR_DECODE;
724 }
725
726 fprintf(stderr,
727 "[gptoss_hparams] Loaded: layers=%u heads=%u kv_heads=%u d_model=%u d_head=%u d_ff=%u vocab=%u max_seq=%u\n",
728 (unsigned)io_ex->core.n_layers,
729 (unsigned)io_ex->core.n_heads,
730 (unsigned)io_ex->core.n_kv_heads,
731 (unsigned)io_ex->core.d_model,
732 (unsigned)io_ex->core.d_head,
733 (unsigned)io_ex->core.d_ff,
734 (unsigned)io_ex->core.vocab_size,
735 (unsigned)io_ex->core.max_seq);
736
737 if (io_ex->rope_scaling_type[0] != '\0') {
738 fprintf(stderr,
739 "[gptoss_hparams] RoPE: theta=%.9g scaling_type='%s' scaling_factor=%.9g original_max_pos=%u\n",
740 (double)io_ex->rope_theta,
741 io_ex->rope_scaling_type,
742 (double)io_ex->rope_scaling_factor,
743 (unsigned)io_ex->rope_scaling_original_max_position_embeddings);
744 } else {
745 fprintf(stderr,
746 "[gptoss_hparams] RoPE: theta=%.9g scaling=none\n",
747 (double)io_ex->rope_theta);
748 }
749
750 return IE_IO_OK;
751}
