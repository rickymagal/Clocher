57  return;
58  }

62  /* Map amax -> 127, avoid divide-by-zero. */
63  const float denom = (amax > 0.0f) ? amax : 1.0f;
64  *out_scale = denom / 127.0f;
65  *out_zero = 0;
66  } else {
