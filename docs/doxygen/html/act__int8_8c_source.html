51void ie_act_i8_params_from_minmax(float minv, float maxv, int symmetric,
52 float* out_scale, int8_t* out_zero) {
53 /* Ensure valid range; if degenerate, pick unit scale. */
54 if (!(maxv > minv)) {
55 *out_scale = 1.0f;
56 *out_zero = 0;
57 return;
58 }
59
60 if (symmetric) {
61 const float amax = ie_maxf(ie_fabsf(minv), ie_fabsf(maxv));
62 /* Map amax -> 127, avoid divide-by-zero. */
63 const float denom = (amax > 0.0f) ? amax : 1.0f;
64 *out_scale = denom / 127.0f;
65 *out_zero = 0;
66 } else {
67 /* Map minv -> -128, maxv -> 127 */
68 const float qrange = 127.0f - (-128.0f); /* 255 */
69 const float scale = (maxv - minv) / qrange;
70 const float zp_f = -(-128.0f) - (minv / (scale > 0.0f ? scale : 1.0f));
71 /* Round and clamp zero point into INT8 range. */
72 int zp_i = (int)ie_rne(zp_f);
73 if (zp_i < -128) zp_i = -128;
74 if (zp_i > 127) zp_i = 127;
75
76 *out_scale = (scale > 0.0f ? scale : 1.0f);
77 *out_zero = (int8_t)zp_i;
78 }
79}

81void ie_quantize_act_int8(const float* src, int8_t* dst, size_t n,
82 ie_act_i8_params params, int symmetric) {
83 const float inv_scale = (params.scale > 0.0f) ? (1.0f / params.scale) : 1.0f;
84 if (symmetric) {
