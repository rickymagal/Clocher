51static int ascii_ieq(const char *a, const char *b) {
52 if (!a || !b) return 0;
53 for (;; ++a, ++b) {
54 unsigned char ca = (unsigned char)*a;
55 unsigned char cb = (unsigned char)*b;
56 if (tolower(ca) != tolower(cb)) return 0;
57 if (ca == '\0') return 1;
58 }
59}

67static uint32_t fnv1a32(const char *s) {
68 uint32_t h = 2166136261u;
69 if (!s) return h ^ 0xA5A5u;
70 const unsigned char *p = (const unsigned char *)s;
71 while (*p) {
72 h ^= (uint32_t)(*p++);
73 h *= 16777619u;
74 }
75 return h ? h : 0x9E3779B1u;

84static uint64_t xorshift64star(uint64_t *state) {
85 uint64_t x = (*state == 0 ? 0x106689D45497fdb5ULL : *state);
86 x ^= x >> 12;

106ie_status_t ie_engine_create(const ie_engine_params_t *p, ie_engine_t **out) {
107 if (!out) return 1; /* generic error */
108
109 ie_engine_t *e = (ie_engine_t *)calloc(1, sizeof(*e));
110 if (!e) return 1;
111
112 if (p) {
113 e->cfg = *p; /* copy-by-value */
114 } else {
115 memset(&e->cfg, 0, sizeof(e->cfg));
116 }
117
118 /* Safe defaults for metrics. */
119 memset(&e->last, 0, sizeof(e->last));
120
121 /* Derive a stable seed from a few string hints (all are NULL-safe). */
122 e->seed = 0x9E3779B97F4A7C15ULL;
123 e->seed ^= (uint64_t)fnv1a32(e->cfg.precision);
124 e->seed ^= (uint64_t)fnv1a32(e->cfg.affinity) << 8;
125 e->seed ^= (uint64_t)fnv1a32(e->cfg.pretranspose) << 16;
126 e->seed ^= (uint64_t)fnv1a32(e->cfg.prefetch) << 24;
127
128 /* Recognize "int4w" and the CLI alias "int4". */
129 e->uses_weight_int4 = (ascii_ieq(e->cfg.precision, IE_PREC_INT4W) ||
130 ascii_ieq(e->cfg.precision, IE_PREC_INT4)) ? 1 : 0;
131
132 *out = e;
133 return IE_OK;
134}

152ie_status_t ie_engine_generate(ie_engine_t *h,
153 const char *prompt,
154 size_t max_new_tokens,
155 uint32_t *out_tokens,
156 uint32_t *out_count) {
157 if (!h || !out_count || !prompt) return 1;
158
159 if (max_new_tokens == 0) {
160 *out_count = 0;
161 h->last.tps_true = 0.0;
162 h->last.latency_p50_ms = 0.0;
163 h->last.latency_p95_ms = 0.0;
164 h->last.rss_peak_mb = 0u;
165 return IE_OK;
166 }
167 if (!out_tokens) return 1;
168
169 /* Per-call deterministic PRNG seeded by engine seed XOR prompt hash. */
170 uint64_t rng = h->seed ^ (uint64_t)fnv1a32(prompt);
171
172 uint32_t produced = 0;
173 for (size_t i = 0; i < max_new_tokens; ++i) {
174 const uint64_t r = xorshift64star(&rng);
175 const uint32_t tok = (uint32_t)(r % 50000u); /* Compact ID space for tests. */
176 out_tokens[i] = tok;
177 ie_kv_on_token(tok); /* counted inside timing window */
178 ++produced;
179 }
180 *out_count = produced;
181
182 /* Placeholders (harness fills time-related values). */
183 h->last.tps_true = 0.0;
184 h->last.latency_p50_ms = 0.0;
185 h->last.latency_p95_ms = 0.0;
186 h->last.rss_peak_mb = 0u;
187
188 return IE_OK;
189}
