133static void join_path_(char *dst, size_t cap, const char *a, const char *b) {
134 if (!dst || cap == 0) return;
135
136 size_t i = 0;
137 if (a) {
138 for (; a[i] && i + 1 < cap; ++i) dst[i] = a[i];
139 }
140
141 const int need_slash = (i > 0 && dst[i - 1] != '/');
142 if (need_slash && i + 1 < cap) dst[i++] = '/';
143
144 if (b) {
145 size_t j = 0;
146 for (; b[j] && i + 1 < cap; ++j, ++i) dst[i] = b[j];
147 }
148
149 dst[i < cap ? i : (cap - 1)] = '\0';
150}

230static int streq_icase_(const char *a, const char *b)
231{
232 /* ASCII case-insensitive string equality. */
233 if (a == NULL || b == NULL) {
234 return 0;
235 }
236 for (;;) {
237 unsigned char ca = (unsigned char)*a++;
238 unsigned char cb = (unsigned char)*b++;
239 ca = (unsigned char)tolower((int)ca);
240 cb = (unsigned char)tolower((int)cb);
241 if (ca != cb) {
242 return 0;
243 }
244 if (ca == 0) {
245 return 1;
246 }
247 }

275static void maybe_correct_head_dim_(const char *model_dir, ie_gptoss_hparams_t *hp) {
276 if (!model_dir || !hp) return;
277 if (hp->n_heads == 0) return;
278
279 char path[4096];
280 join_path_(path, sizeof(path), model_dir, "tensor_map.json");
281
282 tensor_map_t map;
283 if (tensor_map_load(path, &map) != 0) return;
284
285 const tensor_desc_t *q = tensor_map_find(&map, "model.layers.0.self_attn.q_proj.weight");
286 if (q && q->ndim >= 2 && q->shape) {
287 const uint32_t n_heads = hp->n_heads;
288 const uint32_t q0 = q->shape[0];
289 const uint32_t q1 = q->shape[1];
290
291 uint32_t new_d_head = hp->d_head;
292
293 if (q0 != 0 && (q0 % n_heads) == 0 && q0 != hp->d_model) {
294 new_d_head = q0 / n_heads;
295 } else if (q1 != 0 && (q1 % n_heads) == 0 && q1 != hp->d_model) {
296 new_d_head = q1 / n_heads;
297 }
298
299 if (new_d_head != 0 && new_d_head != hp->d_head) {
300 ie_log_warn("correcting head_dim from %" PRIu32 " to %" PRIu32 " using %s (shape=[%" PRIu32 ",%" PRIu32 "])",
301 hp->d_head,
302 new_d_head,
303 "model.layers.0.self_attn.q_proj.weight",
304 q0,
305 q1);
306 hp->d_head = new_d_head;
307 }
308 }
309
310 tensor_map_free(&map);
311}

380static void json_write_escaped_(FILE *fp, const char *s) {
381 fputc('"', fp);
382 if (s) {
383 for (const unsigned char *p = (const unsigned char *)s; *p; ++p) {
384 const unsigned char ch = *p;
385 switch (ch) {
386 case '"': fputs("\\\"", fp); break;
387 case '\\': fputs("\\\\", fp); break;
388 case '\b': fputs("\\b", fp); break;
389 case '\f': fputs("\\f", fp); break;
390 case '\n': fputs("\\n", fp); break;
391 case '\r': fputs("\\r", fp); break;
392 case '\t': fputs("\\t", fp); break;
393 default:
394 if (ch < 0x20) {
395 fprintf(fp, "\\u%04x", (unsigned)ch);
396 } else {
397 fputc((int)ch, fp);
398 }
399 break;
400 }
401 }
402 }
403 fputc('"', fp);
404}

453static void decode_piece_preview_(const ie_tok_gptoss_t *tok, uint32_t token_id, char *dst, size_t cap) {
454 if (!dst || cap == 0) return;
455 dst[0] = '\0';
456 if (!tok) return;
457
458 size_t need = 0;
459 const int rc0 = ie_tok_gptoss_decode(tok, &token_id, 1u, NULL, &need);
460 if (rc0 != 0 || need == 0 || need > (1u << 16)) {
461 (void)snprintf(dst, cap, "<decode_err>");
462 return;
463 }
464
465 char *tmp = (char *)malloc(need);
466 if (!tmp) {
467 (void)snprintf(dst, cap, "<oom>");
468 return;
469 }
470
471 size_t inout = need;
472 const int rc1 = ie_tok_gptoss_decode(tok, &token_id, 1u, tmp, &inout);
473 if (rc1 != 0) {
474 free(tmp);
475 (void)snprintf(dst, cap, "<decode_err>");
476 return;
477 }
478
479 tmp[need - 1] = '\0';
480
481 size_t j = 0;
482 for (size_t i = 0; tmp[i] && j + 1 < cap; ++i) {
483 unsigned char c = (unsigned char)tmp[i];
484 if (c < 0x20 || c == 0x7f) c = ' ';
485 dst[j++] = (char)c;
486 }
487 dst[j] = '\0';
488
489 free(tmp);
490}

492static void debug_dump_topk_(const ie_tok_gptoss_t *tok,
493 const float *logits,
494 size_t logits_len,
495 int k,
496 const char *tag) {
497 if (!tok || !logits || logits_len == 0 || k <= 0 || !tag) return;
498 if (k > 64) k = 64;
499 if ((size_t)k > logits_len) k = (int)logits_len;
500
501 uint32_t top_id[64];
502 float top_val[64];
503 for (int i = 0; i < k; ++i) {
504 top_id[i] = 0;
505 top_val[i] = -INFINITY;
506 }
507
508 for (size_t i = 0; i < logits_len; ++i) {
509 const float v = logits[i];
510 if (v <= top_val[k - 1]) continue;
511
512 int pos = k - 1;
513 while (pos > 0 && v > top_val[pos - 1]) {
514 top_val[pos] = top_val[pos - 1];
515 top_id[pos] = top_id[pos - 1];
516 --pos;
517 }
518 top_val[pos] = v;
519 top_id[pos] = (uint32_t)i;
520 }
521
522 ie_log_info("debug_topk(%s): k=%d", tag, k);
523 for (int i = 0; i < k; ++i) {
524 char piece[160];
525 decode_piece_preview_(tok, top_id[i], piece, sizeof(piece));
526 ie_log_info("debug_topk(%s): rank=%d id=%" PRIu32 " logit=%.9g piece=\"%s\"",
527 tag, i, top_id[i], (double)top_val[i], piece);
528 }
529}

535ie_status_t ie_engine_create(const ie_engine_params_t *p,
536 const char *device,
537 const char *model_dir,
538 ie_engine_t **out) {
539 if (!model_dir || !out) {
540 ie_log_error("ie_engine_create: bad args (model_dir=%p out=%p)", (const void *)model_dir, (void *)out);
541 return IE_ERR_BADARG;
542 }
543 *out = NULL;
544
545 const char *dev_str = device ? device : "cpu";
546 ie_log_info("ie_engine_create: begin (device=%s model_dir=%s)", dev_str, model_dir);
547
548 if (p) {
549 ie_log_info("ie_engine_create: params precision=%s sparsity=%s affinity=%s pretranspose=%s prefetch=%s threads=%d",
550 (p->precision ? p->precision : "(null)"),
551 (p->sparsity ? p->sparsity : "(null)"),
552 (p->affinity ? p->affinity : "(null)"),
553 (p->pretranspose ? p->pretranspose : "(null)"),
554 (p->prefetch ? p->prefetch : "(null)"),
555 p->threads);
556 if (is_precision_int4_(p)) {
557 ie_log_info("ie_engine_create: q4 kernel=%s (runtime-dispatched)", ie_q4_kernel_selected_name());
558 }
559 }
560
561 ie_engine_t *e = (ie_engine_t *)calloc(1, sizeof(*e));
562 if (!e) {
563 ie_log_error("ie_engine_create: OOM allocating engine struct");
564 return IE_ERR_OOM;
565 }
566
567 ie_log_info("ie_engine_create: device_create (device=%s)", dev_str);
568 const ie_device_kind_t kind = ie_device_kind_from_str(dev_str);
569 const int dev_rc = ie_device_create(kind, &e->dev);
570 if (dev_rc != 0 || !e->dev) {
571 ie_log_error("ie_engine_create: device_create failed (rc=%d device=%s)", dev_rc, dev_str);
572 free(e);
573 return IE_ERR_UNSUPPORTED;
574 }
575
576 const char *tok_path = NULL;
577 char tok_path_buf[4096];
578 if (p && p->tokenizer_path && p->tokenizer_path[0] != '\0') {
579 tok_path = p->tokenizer_path;
580 } else {
581 join_path_(tok_path_buf, sizeof(tok_path_buf), model_dir, "tokenizer.json");
582 tok_path = tok_path_buf;
583 }
584 ie_log_info("ie_engine_create: tokenizer_open (path=%s)", tok_path);
585
586 const int tok_rc = ie_tok_gptoss_open(tok_path, &e->tok);
587 if (tok_rc != 0 || !e->tok) {
588 ie_log_error("ie_engine_create: tokenizer_open failed (rc=%d path=%s)", tok_rc, tok_path);
589 ie_device_destroy(e->dev);
590 free(e);
591 return IE_ERR_MODEL;
592 }
593
594 const uint32_t tok_vocab = ie_tok_gptoss_vocab_size(e->tok);
595 ie_log_info("ie_engine_create: tokenizer_open ok (vocab=%" PRIu32 ")", tok_vocab);
596
597 const char *weights_json = NULL;
598 const char *bin_override = "";
599
600 char weights_json_buf[PATH_MAX];
601 char compat_json_buf[PATH_MAX];
602 char q4_bin_buf[PATH_MAX];
603
604 const int want_int4 = (p && is_precision_int4_(p));
605
606 if (p && p->weights_json_path && p->weights_json_path[0] != '\0') {
607 weights_json = p->weights_json_path;
608 } else if (want_int4) {
609 join_path_(compat_json_buf, sizeof(compat_json_buf), model_dir, "model.ie.compat.json");
610 if (path_exists_(compat_json_buf)) {
611 weights_json = compat_json_buf;
612 }
613 }
614
615 if (weights_json == NULL) {
616 join_path_(weights_json_buf, sizeof(weights_json_buf), model_dir, "model.ie.json");
617 weights_json = weights_json_buf;
618 }
619
620 if (p && p->weights_bin_path && p->weights_bin_path[0] != '\0') {
621 bin_override = p->weights_bin_path;
622 } else if (want_int4) {
623 join_path_(q4_bin_buf, sizeof(q4_bin_buf), model_dir, "model.q4.bin");
624 if (path_exists_(q4_bin_buf)) {
625 bin_override = q4_bin_buf;
626 }
627 }
628
629 ie_log_info("ie_engine_create: weights_open (json=%s bin_override=%s)", weights_json,
630 (bin_override[0] != '\0' ? bin_override : "(null)"));
631 const int w_rc = ie_weights_open(weights_json, bin_override, &e->w);
632 if (w_rc != 0) {
633 ie_log_error("ie_engine_create: weights_open failed (rc=%d json=%s)", w_rc, weights_json);
634 ie_tok_gptoss_close(e->tok);
635 ie_device_destroy(e->dev);
636 free(e);
637 return IE_ERR_MODEL;
638 }
639
640 ie_log_info("ie_engine_create: weights_open ok (dtype=%s bin=%s bin_size_bytes=%zu dedup=%d)",
641 e->w.dtype,
642 e->w.weights_path,
643 e->w.bin_size_bytes,
644 e->w.is_dedup ? 1 : 0);
645
646 /* Load base hparams first (shapes/sizes). */
647 ie_log_info("ie_engine_create: hparams_load (dir=%s)", model_dir);
648 const int hp_rc = gptoss_hparams_load(model_dir, &e->hp);
649 if (hp_rc != 0) {
650 ie_log_error("ie_engine_create: hparams_load failed (rc=%d dir=%s)", hp_rc, model_dir);
651 ie_weights_close(&e->w);
652 ie_tok_gptoss_close(e->tok);
653 ie_device_destroy(e->dev);
654 free(e);
655 return IE_ERR_MODEL;
656 }
657
658 /* Load extended config (eps/theta/rope scaling). */
659 ie_log_info("ie_engine_create: hparams_load_ex (dir=%s)", model_dir);
660 const int ex_rc = gptoss_hparams_load_ex(model_dir, &e->hpx);
661 if (ex_rc != 0) {
662 ie_log_warn("ie_engine_create: hparams_load_ex failed (rc=%d dir=%s). Continuing with base hparams only.", ex_rc, model_dir);
663 memset(&e->hpx, 0, sizeof(e->hpx));
664 }
665
666 log_hparams_(&e->hp);
667 if (ex_rc == 0) log_hparams_ex_(&e->hpx);
668
669 if (tok_vocab != 0 && e->hp.vocab_size != 0 && tok_vocab != e->hp.vocab_size) {
670 ie_log_warn("ie_engine_create: tokenizer vocab (%" PRIu32 ") != hparams vocab (%" PRIu32 ")",
671 tok_vocab,
672 e->hp.vocab_size);
673 }
674

833 ie_log_error("ie_engine_generate_ex: bad args (e=%p prompt=%p out_n_tokens=%p)",
834 (const void *)e,
835 (const void *)prompt,
836 (void *)out_n_tokens);
837 return IE_ERR_BADARG;
838 }

840 ie_log_error("ie_engine_generate_ex: bad args (max_new=%zu out_tokens=NULL)", max_new);
841 return IE_ERR_BADARG;

844 *out_n_tokens = 0;
845
846 if (out_stats) {
847 memset(out_stats, 0, sizeof(*out_stats));
848 }
849
850 if (max_new == 0) {
851 ie_log_info("ie_engine_generate_ex: max_new=0; nothing to do");
852 return IE_OK;
853 }
854
855 const double t_total0 = now_s_();
856
857 ie_log_info("ie_engine_generate_ex: begin (engine=%p max_new=%zu)", (const void *)e, max_new);
858 log_prompt_preview_(prompt);
859
860 const int log_tokens = env_flag_("IE_API_LOG_TOKENS", 0);
861 const int log_every = env_int_("IE_API_LOG_EVERY", 0, 0, 1000000);
862 const int dbg_decode = env_flag_("IE_API_DEBUG_DECODE", 0);
863 const int dbg_decode_every = env_int_("IE_API_DEBUG_DECODE_EVERY", 0, 0, 1000000);
864

870 if (rc != 0) {

872 return IE_ERR_MODEL;

874
875 ie_log_info("ie_engine_generate_ex: tokenizer size-query ok (need_tokens=%" PRIu32 ")", need);
876
877 uint32_t n_prompt = (need == 0) ? 1u : need;
878 uint32_t *prompt_ids = (uint32_t *)malloc((size_t)n_prompt * sizeof(uint32_t));
879 if (!prompt_ids) {
880 ie_log_error("ie_engine_generate_ex: OOM allocating prompt_ids (n_prompt=%" PRIu32 ")", n_prompt);
881 return IE_ERR_OOM;
882 }
883
884 if (need == 0) {
885 prompt_ids[0] = 0;
886 n_prompt = 1;
887 ie_log_warn("ie_engine_generate_ex: prompt encoded to empty; forcing token_id=0");
888 } else {
889 uint32_t inout = n_prompt;
890 rc = ie_tok_gptoss_encode(e->tok, prompt, prompt_ids, &inout);
891 if (rc != 0) {
892 ie_log_error("ie_engine_generate_ex: tokenizer encode failed (rc=%d capacity=%" PRIu32 ")", rc, n_prompt);
893 free(prompt_ids);
894 return IE_ERR_MODEL;
895 }
896 n_prompt = inout;
897 if (n_prompt == 0) {
898 prompt_ids[0] = 0;
899 n_prompt = 1;
900 ie_log_warn("ie_engine_generate_ex: encode returned 0 tokens; forcing token_id=0");
901 }
902 }
903
904 ie_log_info("ie_engine_generate_ex: encoded prompt tokens=%" PRIu32, n_prompt);
905
906 size_t trace_prompt_index = 0;
907 const int trace_prompt = trace_ids_prompt_index_(&trace_prompt_index);
908 if (trace_prompt) {

919 pos += (size_t)snprintf(line + pos, sizeof(line) - pos, " %u", (unsigned)prompt_ids[i]);
920 }

922 (void)snprintf(line + pos, sizeof(line) - pos, " ...");
923 }
924 ie_log_info("%s", line);
925 }
926
927 if (e->hp.n_layers == 0) {
928 ie_log_error("ie_engine_generate_ex: invalid hparams (n_layers=0)");

935 if (!kv_layers) {
936 ie_log_error("ie_engine_generate_ex: OOM allocating kv_layers (n_layers=%zu)", n_layers);
937 free(prompt_ids);

948 if (want_seq < 16) want_seq = 16;
949
950 int32_t hp_seq = (int32_t)e->hp.max_seq;
951 if (hp_seq <= 0) hp_seq = want_seq;
952
953 kv_opts.max_seq = (want_seq < hp_seq) ? want_seq : hp_seq;
954
955 ie_log_info("ie_engine_generate_ex: kv_opts heads=%" PRIi32 " head_dim=%" PRIi32 " want_seq=%" PRIi32 " hp_seq=%" PRIi32 " max_seq=%" PRIi32,
956 kv_opts.heads,
957 kv_opts.head_dim,
958 want_seq,
959 hp_seq,
960 kv_opts.max_seq);
961
962 if (kv_opts.heads <= 0 || kv_opts.head_dim <= 0 || kv_opts.max_seq <= 0) {
963 ie_log_error("ie_engine_generate_ex: invalid kv_opts (heads=%" PRIi32 " head_dim=%" PRIi32 " max_seq=%" PRIi32 ")",
964 kv_opts.heads,
965 kv_opts.head_dim,
966 kv_opts.max_seq);
967 free(kv_layers);
968 free(prompt_ids);
969 return IE_ERR_MODEL;
970 }
971
972 if (ie_kv_init_layers(kv_layers, (int)e->hp.n_layers, &kv_opts) != 0) {
973 ie_log_error("ie_engine_generate_ex: kv_init_layers failed (n_layers=%" PRIu32 ")", e->hp.n_layers);
974 free(kv_layers);
975 free(prompt_ids);

985 const ie_status_t st = status_from_rc_(rc);
986 ie_log_error("ie_engine_generate_ex: prefill failed (rc=%d -> %s)", rc, status_str_(st));
987 ie_kv_free_layers(kv_layers, (int)e->hp.n_layers);
988 free(kv_layers);
989 free(prompt_ids);
990 return st;
991 }

1003

1018 }
1019

1022 if (trace_prompt) {

1024 }
1025

1028 }

1031 if (rc != 0) {

1034 produced,
1035 next,
1036 rc,
1037 status_str_(st));
1038 ie_kv_free_layers(kv_layers, (int)e->hp.n_layers);
1039 free(kv_layers);
1040 free(prompt_ids);
1041 return st;
1042 }
1043
1044
1045 if (produced == 0) {
1046 const double t_now = now_s_();
1047 ttft_s = (t_now > t_total0) ? (t_now - t_total0) : 0.0;
1048 }
1049
1050 if (dbg_topk > 0 && dbg_topk_every > 0 && (produced % (size_t)dbg_topk_every) == 0) {
1051 char tag[64];
1052 (void)snprintf(tag, sizeof(tag), "step_%zu", produced);
1053 debug_dump_topk_(e->tok, e->logits, e->logits_len, dbg_topk, tag);
1054 }
1055
1056 if (dbg_decode && dbg_decode_every > 0 && (produced % (size_t)dbg_decode_every) == 0) {
1057 size_t need_bytes = 0;
1058 const uint32_t cnt = (uint32_t)(produced + 1);
1059 const int drc = ie_tok_gptoss_decode(e->tok, (const uint32_t *)out_tokens, cnt, NULL, &need_bytes);
1060 if (drc == 0 && need_bytes > 0 && need_bytes < (size_t)(1u << 22)) {
1061 char *txt = (char *)malloc(need_bytes);
1062 if (txt) {
1063 size_t inout = need_bytes;
1064 const int drc2 = ie_tok_gptoss_decode(e->tok, (const uint32_t *)out_tokens, cnt, txt, &inout);
1065 if (drc2 == 0) {
1066 txt[need_bytes - 1] = '\0';
1067 ie_log_info("ie_engine_generate_ex: decoded[%u] bytes=%zu preview=\"%s%s\"",
1068 cnt,
1069 inout,
1070 txt,
1071 (inout > 120 ? "..." : ""));
1072 } else {
1073 ie_log_warn("ie_engine_generate_ex: decode failed (rc=%d) at step=%zu", drc2, produced);
1074 }
1075 free(txt);
1076 }
1077 } else {
1078 ie_log_warn("ie_engine_generate_ex: decode size-query failed (rc=%d need_bytes=%zu) at step=%zu",
1079 drc,
1080 need_bytes,
1081 produced);
1082 }
1083 }
1084 }
1085
1086 const double t_decode1 = now_s_();
1087
1088 *out_n_tokens = produced;
1089
1090 ie_log_info("ie_engine_generate_ex: done (produced=%zu requested=%zu)", produced, max_new);
1091
1092 ie_kv_free_layers(kv_layers, (int)e->hp.n_layers);
1093 free(kv_layers);
1094 free(prompt_ids);
1095
1096 const double t_total1 = now_s_();
1097
1098 if (out_stats) {
1099 out_stats->wall_time_s = (t_total1 > t_total0) ? (t_total1 - t_total0) : 0.0;
1100 out_stats->prefill_time_s = (t_prefill1 > t_prefill0) ? (t_prefill1 - t_prefill0) : 0.0;
1101 out_stats->decode_time_s = (t_decode1 > t_decode0) ? (t_decode1 - t_decode0) : 0.0;
1102 out_stats->ttft_s = ttft_s;
1103
1104 out_stats->tps_decode = (out_stats->decode_time_s > 0.0) ? ((double)produced / out_stats->decode_time_s) : 0.0;
1105 out_stats->tps_total = (out_stats->wall_time_s > 0.0) ? ((double)produced / out_stats->wall_time_s) : 0.0;
1106 }
1107
1108 return IE_OK;
