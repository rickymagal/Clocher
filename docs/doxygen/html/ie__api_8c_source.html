66 const uint64_t off = 1469598103934665603ULL;
67 const uint64_t prime = 1099511628211ULL;
68 uint64_t h = off;
69 if (!s) return h;

124 const uint64_t v = seed + 0x9e3779b97f4a7c15ULL * (uint64_t)(i + 1) + (ctr << 1);
