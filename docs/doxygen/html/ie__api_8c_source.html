53static int ascii_ieq(const char *a, const char *b) {
54 if (!a || !b) return 0;
55 for (;; ++a, ++b) {
56 unsigned char ca = (unsigned char)*a;
57 unsigned char cb = (unsigned char)*b;
58 if (tolower(ca) != tolower(cb)) return 0;
59 if (ca == '\0') return 1;
60 }
61}

69static uint32_t fnv1a32(const char *s) {
70 uint32_t h = 2166136261u;
71 if (!s) return h ^ 0xA5A5u;
72 const unsigned char *p = (const unsigned char *)s;
73 while (*p) {
74 h ^= (uint32_t)(*p++);
75 h *= 16777619u;
76 }

115ie_status_t ie_engine_create(const ie_engine_params_t *p, ie_engine_t **out) {
116 if (!out) return 1; /* generic error */
117
118 ie_engine_t *e = (ie_engine_t *)calloc(1, sizeof(*e));
119 if (!e) return 1;
120
121 if (p) {
122 e->cfg = *p; /* copy-by-value */
123 } else {
124 memset(&e->cfg, 0, sizeof(e->cfg));
125 }
126
127 /* Safe defaults for metrics. */
128 memset(&e->last, 0, sizeof(e->last));
129
130 /* Derive a stable seed from a few string hints (all are NULL-safe). */
131 e->seed = 0x9E3779B97F4A7C15ULL;
132 e->seed ^= (uint64_t)fnv1a32(e->cfg.precision);
133 e->seed ^= (uint64_t)fnv1a32(e->cfg.affinity) << 8;
134 e->seed ^= (uint64_t)fnv1a32(e->cfg.pretranspose) << 16;
135 e->seed ^= (uint64_t)fnv1a32(e->cfg.prefetch) << 24;
136 e->seed ^= (uint64_t)fnv1a32(e->cfg.sparsity) << 32;
137
138 /* Recognize "int4w" and the CLI alias "int4". */
139 e->uses_weight_int4 = (ascii_ieq(e->cfg.precision, IE_PREC_INT4W) ||
140 ascii_ieq(e->cfg.precision, IE_PREC_INT4)) ? 1 : 0;
141
142 /* Recognize block-sparse hint. */
143 e->uses_block_sparse = (ascii_ieq(e->cfg.sparsity, "block") ||
144 ascii_ieq(e->cfg.sparsity, "blocksparse")) ? 1 : 0;
145
146 *out = e;
147 return IE_OK;
148}

167 const char *prompt,
168 size_t max_new_tokens,
169 uint32_t *out_tokens,
170 uint32_t *out_count) {
171 if (!h || !out_count || !prompt) return 1;
172
173 if (max_new_tokens == 0) {
174 *out_count = 0;
175 h->last.tps_true = 0.0;
176 h->last.latency_p50_ms = 0.0;
177 h->last.latency_p95_ms = 0.0;
178 h->last.rss_peak_mb = 0u;
179 return IE_OK;
180 }
181 if (!out_tokens) return 1;
182
183 /* Per-call deterministic PRNG seeded by engine seed XOR prompt hash. */
184 uint64_t rng = h->seed ^ (uint64_t)fnv1a32(prompt);
185
186 uint32_t produced = 0;
187 for (size_t i = 0; i < max_new_tokens; ++i) {
188 const uint64_t r = xorshift64star(&rng);
189 const uint32_t tok = (uint32_t)(r % 50000u); /* Compact ID space for tests. */
190 out_tokens[i] = tok;
191 ie_kv_on_token(tok); /* counted inside timing window */
192 ++produced;
193 }
194 *out_count = produced;
195
196 /* Placeholders (harness fills time-related values). */
197 h->last.tps_true = 0.0;
198 h->last.latency_p50_ms = 0.0;
199 h->last.latency_p95_ms = 0.0;
200 h->last.rss_peak_mb = 0u;
201
202 return IE_OK;
203}
