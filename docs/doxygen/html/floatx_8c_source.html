62void ie_fp32_to_fp16(const float *in, uint16_t *out, size_t n) {
63 if (!in || !out || n == 0) return;
64
65 for (size_t i = 0; i < n; ++i) {
66 union { float f; uint32_t u; } v = { .f = in[i] };
67 uint32_t x = v.u;
68
69 uint32_t sign = (x >> 31) & 0x1u;
70 int32_t exp = (int32_t)((x >> 23) & 0xFFu) - 127; /* unbiased exp */
71 uint32_t frac = x & 0x7FFFFFu; /* 23 bits */
72
73 uint16_t h;
74 if (exp == 128) { /* Inf / NaN */
75 h = (uint16_t)((sign << 15) | 0x7C00u | (frac ? 0x200u : 0));
76 } else if (exp > 15) { /* Overflow -> Inf */
77 h = (uint16_t)((sign << 15) | 0x7C00u);
78 } else if (exp >= -14) { /* Normalized half */
79 /* Put implicit leading 1 back and round to 10 bits. */
80 uint32_t mant = (frac | 0x800000u); /* 24 bits */
81 int shift = 13 - exp; /* align to 10-bit mantissa */
82 uint32_t rounding = (shift > 0) ? ((1u << (shift - 1)) - 1u) : 0;
83 uint32_t half_mant = (shift > 0) ? ((mant + rounding + ((mant >> shift) & 1u)) >> shift)
84 : (mant << (-shift));
85 if (half_mant & 0x400u) { /* mantissa overflow after rounding */
86 half_mant >>= 1;
87 exp += 1;
88 }
89 if (exp > 15) { /* overflow to Inf */
90 h = (uint16_t)((sign << 15) | 0x7C00u);
91 } else {
92 uint16_t he = (uint16_t)((exp + 15) & 0x1Fu);
93 h = (uint16_t)((sign << 15) | (he << 10) | (half_mant & 0x3FFu));
94 }
95 } else if (exp >= -24) { /* Subnormal half */
96 uint32_t mant = (frac | 0x800000u);
97 int rshift = (-14 - exp) + 1; /* +1 to position for subnormal */
98 uint32_t sub = mant >> (rshift + 13); /* to 10 bits */
99 /* round-to-nearest-even */
100 uint32_t rem = mant & ((1u << (rshift + 13)) - 1u);
101 uint32_t halfway = 1u << (rshift + 12);
102 if (rem > halfway || (rem == halfway && (sub & 1u))) sub++;
103 h = (uint16_t)((sign << 15) | (sub & 0x3FFu));
104 } else {
105 /* Underflow -> signed zero */
106 h = (uint16_t)(sign << 15);
107 }
108
109 out[i] = h;
110 }

144 }
