73 static void gemv_generic_impl(const float *W, const float *x, float *y,
74  size_t rows, size_t cols, const float *bias,
75  size_t blk_k) {
76  const size_t BK = (blk_k > 0 ? blk_k : cols);
77 
78  for (size_t r = 0; r < rows; ++r) {
79  const float *wrow = W + r * cols;
80  float acc = 0.0f;
81 
82  size_t kofs = 0;
83  for (size_t k0 = 0; k0 < cols; k0 += BK) {
84  const size_t klen = (k0 + BK <= cols) ? BK : (cols - k0);
85  const float *wblk = wrow + kofs;
86  const float *xblk = x + k0;
87 
88  for (size_t k = 0; k < klen; ++k) acc += wblk[k] * xblk[k];
89  kofs += klen;
90  }
91 
92  if (bias) acc += bias[r];
93  y[r] = acc;
94  }
95 }

201 static inline float ie_decode_fp8_e4m3(uint8_t v) {
202  if (v == 0u) return 0.0f;
203  const uint8_t sign = (v >> 7) & 0x1;
204  const uint8_t exp = (v >> 3) & 0xF;
205  const uint8_t man = (v & 0x7);
206  if (exp == 0) return sign ? -0.0f : 0.0f;
207  const int bias = 7;
208  const int e = (int)exp - bias;
209  const float frac = (float)man / 8.0f;
210  const float val = (1.0f + frac) * (float)(1u << e);
211  return sign ? -val : val;
212 }

219 static inline float ie_decode_fp8_e5m2(uint8_t v) {
220  const uint8_t sign = (v >> 7) & 0x1;
221  const uint8_t exp = (v >> 2) & 0x1F;
222  const uint8_t man = (v & 0x3);
223  if (exp == 0) return sign ? -0.0f : 0.0f;
224  if (exp == 0x1F) {
225  if (man == 0) return sign ? -(__builtin_inff()) : (__builtin_inff());
226  volatile float qnan = __builtin_nanf("0x1");
227  return qnan;
228  }
229  const int bias = 15;
230  const int e = (int)exp - bias;
231  const float frac = (float)man / 4.0f;
232  const float val = (1.0f + frac) * (float)(1u << e);
233  return sign ? -val : val;
234 }
