95static void gemv_generic_impl(const float *W, const float *x, float *y,
96 size_t rows, size_t cols, const float *bias,
97 size_t blk_k) {
98 const size_t BK = (blk_k > 0 ? blk_k : cols);
99
100 for (size_t r = 0; r < rows; ++r) {
101 const float *wrow = W + r * cols;
102 float acc = 0.0f;
103
104 size_t kofs = 0;
105 for (size_t k0 = 0; k0 < cols; k0 += BK) {
106 const size_t klen = (k0 + BK <= cols) ? BK : (cols - k0);
107 const float *wblk = wrow + kofs;
108 const float *xblk = x + k0;
109
110 for (size_t k = 0; k < klen; ++k) acc += wblk[k] * xblk[k];
111 kofs += klen;
112 }
113
114 if (bias) acc += bias[r];
115 y[r] = acc;
116 }
117}

230static inline float ie_decode_fp8_e4m3(uint8_t v) {
231 if (v == 0u) return 0.0f;
232 const uint8_t sign = (v >> 7) & 0x1;
233 const uint8_t exp = (v >> 3) & 0xF;
234 const uint8_t man = (v & 0x7);
235
236 if (exp == 0) return sign ? -0.0f : 0.0f;
237
238 const int bias = 7;
239 const int e = (int)exp - bias;
240 const float frac = (float)man / 8.0f;
241 const float val = ldexpf(1.0f + frac, e);
242 return sign ? -val : val;
243}

256static inline float ie_decode_fp8_e5m2(uint8_t v) {
257 const uint8_t sign = (v >> 7) & 0x1;
258 const uint8_t exp = (v >> 2) & 0x1F;
259 const uint8_t man = (v & 0x3);
260
261 if (exp == 0) return sign ? -0.0f : 0.0f;
262
263 if (exp == 0x1F) {
264 if (man == 0) return sign ? -(__builtin_inff()) : (__builtin_inff());
265 volatile float qnan = __builtin_nanf("0x1");
266 return qnan;
267 }
268
269 const int bias = 15;
270 const int e = (int)exp - bias;
271 const float frac = (float)man / 4.0f;
272 const float val = ldexpf(1.0f + frac, e);
273 return sign ? -val : val;
274}

399
400 for (size_t r = 0; r < rows; ++r) {

402 float acc = 0.0f;

406 const size_t klen = (k0 + BK <= cols) ? BK : (cols - k0);
407 const float *wblk = wrow + kofs;
408 const uint8_t *xblk = x_fp8 + k0;

411 const uint8_t b = xblk[k];

414 }
