57static void ie_gemv_f32_generic_impl(const float *W, const float *x, float *y,
58 size_t rows, size_t cols,
59 const float *bias, size_t bias_stride) {
60 for (size_t r = 0; r < rows; r++) {
61 const float *row = W + r * cols;
62 float sum = 0.0f;
63
64 for (size_t c = 0; c < cols; c++) {
65 sum += row[c] * x[c];
66 }
67
68 if (bias) {
69 sum += bias[r * bias_stride];
70 }
71
72 y[r] = sum;
73 }
74}

76static inline float ie_bf16_bits_to_f32(uint16_t v) {

92static int ie_gemv_bf16_generic_impl(const uint16_t *W_bf16, const float *x, float *y,
93 size_t rows, size_t cols,
94 const uint16_t *bias_bf16) {
95 if (!W_bf16 || !x || !y || rows == 0u || cols == 0u) return -1;
96
97 for (size_t r = 0; r < rows; ++r) {
98 float acc = 0.0f;
99 const size_t base = r * cols;
100 for (size_t c = 0; c < cols; ++c) {
101 const float w = ie_bf16_bits_to_f32(W_bf16[base + c]);
102 acc += w * x[c];
103 }
104 if (bias_bf16) acc += ie_bf16_bits_to_f32(bias_bf16[r]);
105 y[r] = acc;
106 }
107
108 return 0;
109}

140static void ie_gemv_bf16_job_run(void *ctx, unsigned start, unsigned end) {
141 ie_gemv_bf16_job_t *job = (ie_gemv_bf16_job_t *)ctx;
142 if (!job || start >= end) return;
143 const size_t s = (size_t)start;
144 const size_t n = (size_t)(end - start);
145 const uint16_t *w = job->W_bf16 + s * job->cols;
146 float *y = job->y + s;
147 const uint16_t *b = job->bias_bf16 ? (job->bias_bf16 + s) : NULL;
148 (void)job->impl(w, job->x, y, n, job->cols, b);
149}

225int ie_gemv_bf16_f32(const uint16_t *W_bf16, const float *x, float *y,
226 size_t rows, size_t cols,
227 const uint16_t *bias_bf16) {
228 ie_kernels_lazy_init();
229 ie_bf16_tp_init();
230 if (g_bf16_tp && g_bf16_threads > 1u && rows >= g_bf16_threads * 2u) {
231 ie_gemv_bf16_job_t job = {
232 .W_bf16 = W_bf16,
233 .x = x,
234 .y = y,
235 .rows = rows,
236 .cols = cols,
237 .bias_bf16 = bias_bf16,
238 .impl = g_gemv_bf16,
239 };
240 unsigned grainsize = (unsigned)((rows + g_bf16_threads - 1u) / g_bf16_threads);
241 ie_tp_parallel_for(g_bf16_tp, (unsigned)rows, grainsize, ie_gemv_bf16_job_run, &job);
242 return 0;
243 }
244 return g_gemv_bf16(W_bf16, x, y, rows, cols, bias_bf16);
245}
