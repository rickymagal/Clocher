1/* File: engine/src/opt/stream.c
2 * -----------------------------------------------------------------------------
3 * @file stream.c
4 * @brief Prefetch and non-temporal (streaming) heuristics and helpers.
5 *
6 * @details
7 * This module implements:
8 * - Best-effort cache size discovery on Linux via sysfs:
9 * "/sys/devices/system/cpu/cpu0/cache/index<N>/size"
10 * with size strings like "32K" or "12M".
11 * - Policy initialization and environment overrides (documented in ie_stream.h).
12 * - Prefetch helpers (T0 and NTA) with graceful no-op on non-x86.
13 * - Optional non-temporal copy/memset helpers for float arrays, using AVX-512F
14 * or AVX2 when available, with safe scalar fallback.
15 *
16 * Environment variables (interpreted here):
17 * - IE_STREAM_PREFETCH Override prefetch lead distance (bytes).
18 * - IE_STREAM_NT_THRESHOLD Override non-temporal threshold (bytes).
19 * - IE_STREAM_FORCE_NT "1" => always NT, "0" => never NT, else heuristic.
20 */

54static size_t ie_read_cache_size_file(const char *path) {
55 FILE *f = fopen(path, "r");
56 if (!f) return 0;
57
58 char buf[64] = {0};
59 if (!fgets(buf, sizeof buf, f)) {
60 fclose(f);
61 return 0;
62 }
63 fclose(f);
64
65 char *p = buf;
66 while (*p && (*p == ' ' || *p == '\t')) p++;
67
68 long long v = 0;
69 char unit = 0;
70 if (sscanf(p, "%lld%c", &v, &unit) < 1) return 0;
71 if (v <= 0) return 0;
72
73 if (unit == 'M' || unit == 'm') return (size_t)v * 1024ull * 1024ull;
74 if (unit == 'K' || unit == 'k') return (size_t)v * 1024ull;
75 return (size_t)v;
76}

93static void ie_detect_caches_linux(size_t *l1, size_t *l2, size_t *l3) {
94 if (l1) *l1 = 0;
95 if (l2) *l2 = 0;
96 if (l3) *l3 = 0;
97
98 const char *paths[] = {
99 "/sys/devices/system/cpu/cpu0/cache/index0/size",
100 "/sys/devices/system/cpu/cpu0/cache/index1/size",
101 "/sys/devices/system/cpu/cpu0/cache/index2/size",
102 "/sys/devices/system/cpu/cpu0/cache/index3/size",
103 "/sys/devices/system/cpu/cpu0/cache/index4/size",
104 };
105
106 size_t found[5] = {0};
107 for (int i = 0; i < 5; ++i) found[i] = ie_read_cache_size_file(paths[i]);
108
109 /* Sort descending (N=5 selection sort). */
110 for (int i = 0; i < 5; ++i) {
111 for (int j = i + 1; j < 5; ++j) {
112 if (found[j] > found[i]) {
113 size_t t = found[i];
114 found[i] = found[j];
115 found[j] = t;
116 }
117 }
118 }
119
120 if (l3) *l3 = found[0];
121 if (l2) *l2 = found[1];
122 if (l1) *l1 = found[2];
123}

161int ie_stream_policy_init(struct ie_stream_policy *p) {
162 if (!p) return -EINVAL;
163 memset(p, 0, sizeof *p);
164
165 size_t l1 = 0, l2 = 0, l3 = 0;
166 ie_detect_caches_linux(&l1, &l2, &l3);
167
168 p->l1_bytes = l1;
169 p->l2_bytes = l2;
170 p->l3_bytes = l3;
171
172 if (p->l1_bytes == 0) p->l1_bytes = 32u * 1024u;
173 if (p->l2_bytes == 0) p->l2_bytes = 1u * 1024u * 1024u;
174 if (p->l3_bytes == 0) p->l3_bytes = 8u * 1024u * 1024u;
175
176 p->prefetch_distance = 6u * IE_CACHELINE_BYTES;
177 p->nt_threshold_bytes = 2u * p->l3_bytes;
178
179 p->prefetch_hint = 0;
180 p->enable_prefetch = true;
181 p->force_nt_on = false;
182 p->force_nt_off = false;
183
184 const char *s = NULL;
185
186 s = getenv("IE_STREAM_PREFETCH");
187 if (s && *s) {
188 size_t v = (size_t)strtoull(s, NULL, 10);
189 if (v > 0) p->prefetch_distance = v;
190 }
191
192 s = getenv("IE_STREAM_NT_THRESHOLD");
193 if (s && *s) {
194 size_t v = (size_t)strtoull(s, NULL, 10);
195 if (v > 0) p->nt_threshold_bytes = v;
196 }
197
198 /* IE_STREAM_FORCE_NT:
199 * - "1" => always use non-temporal stores
200 * - "0" => never use non-temporal stores
201 * - otherwise => ignore (keep heuristic)
202 */
203 s = getenv("IE_STREAM_FORCE_NT");
204 if (s && *s) {
205 int v = atoi(s);
206 if (v == 1) {
207 p->force_nt_on = true;
208 p->force_nt_off = false;
209 } else if (v == 0) {
210 p->force_nt_on = false;
211 p->force_nt_off = true;
212 }
213 }
214
215 return 0;
216}

251size_t ie_stream_recommend_prefetch_distance(const struct ie_stream_policy *p,
252 size_t stride_bytes) {
253 if (!p) return 4u * IE_CACHELINE_BYTES;
254
255 size_t lead = p->prefetch_distance ? p->prefetch_distance
256 : (6u * IE_CACHELINE_BYTES);
257
258 if (lead < IE_CACHELINE_BYTES) lead = IE_CACHELINE_BYTES;
259 if (lead > p->l1_bytes / 2u) lead = p->l1_bytes / 2u;
260
261 if (stride_bytes > 0 && lead < stride_bytes) lead = stride_bytes;
262
263 return lead;
264}

381static void ie_nt_copy_f32_avx512(float *dst, const float *src, size_t n) {
382#if IE_X86 && defined(__AVX512F__)
383 size_t i = 0;
384
385 /* Prologue until destination aligns to 64B (best for 512-bit streams). */
386 uintptr_t addr = (uintptr_t)dst;
387 size_t mis = (size_t)(addr & (64u - 1u));
388 if (mis) {
389 size_t pro = (64u - mis) / sizeof(float);
390 if (pro > n) pro = n;
391 ie_scalar_copy_f32(dst, src, pro);
392 dst += pro;
393 src += pro;
394 n -= pro;
395 }
396
397 for (; i + 16 <= n; i += 16) {
398 __m512 v = _mm512_loadu_ps(src + i);
399 _mm512_stream_ps(dst + i, v);
400 }
401 ie_scalar_copy_f32(dst + i, src + i, n - i);
402#else
403 (void)dst;
404 (void)src;
405 (void)n;
406#endif
407}
