39 static size_t ie_read_cache_size_file(const char *path) {
40  FILE *f = fopen(path, "r");
41  if (!f) return 0;
42  char buf[64] = {0};
43  if (!fgets(buf, sizeof buf, f)) { fclose(f); return 0; }
44  fclose(f);
45 
46  char *end = buf;
47  while (*end && (*end == ' ' || *end == '\t')) end++;
48  long long v = 0;
49  char unit = 0;
50  if (sscanf(end, "%lld%c", &v, &unit) < 1) return 0;
51 
52  size_t bytes = 0;
53  if (unit == 'M' || unit == 'm') bytes = (size_t)v * 1024ull * 1024ull;
54  else if (unit == 'K' || unit == 'k') bytes = (size_t)v * 1024ull;
55  else bytes = (size_t)v;

68 static void ie_detect_caches_linux(size_t *l1, size_t *l2, size_t *l3) {
69  /* Common indices: 0=L1D, 2=L2, 3=L3 on many systems, but vendor layouts vary. */
70  const char *paths[] = {
71  "/sys/devices/system/cpu/cpu0/cache/index0/size",
72  "/sys/devices/system/cpu/cpu0/cache/index1/size",
73  "/sys/devices/system/cpu/cpu0/cache/index2/size",
74  "/sys/devices/system/cpu/cpu0/cache/index3/size",
75  "/sys/devices/system/cpu/cpu0/cache/index4/size",
76  };
77  size_t found[5] = {0};
78  for (int i = 0; i < 5; ++i) found[i] = ie_read_cache_size_file(paths[i]);
79 
80  /* Pick largest as L3, next as L2, next as L1D (very rough but robust enough). */
81  size_t a[5]; memcpy(a, found, sizeof a);
82  /* Simple selection sort for 5 elements. */
83  for (int i = 0; i < 5; ++i) {
84  for (int j = i + 1; j < 5; ++j) if (a[j] > a[i]) { size_t t = a[i]; a[i] = a[j]; a[j] = t; }
85  }
86  *l3 = a[0];
87  *l2 = a[1];
88  *l1 = a[2];
89 }

158 size_t ie_stream_recommend_prefetch_distance(const struct ie_stream_policy *p,
159  size_t stride_bytes) {
160  if (!p) return 4u * IE_CACHELINE_BYTES;
161  /* Do not lead more than ~ half L1 by default; also not less than one line. */
162  size_t lead = p->prefetch_distance ? p->prefetch_distance : (6u * IE_CACHELINE_BYTES);
163  if (lead < IE_CACHELINE_BYTES) lead = IE_CACHELINE_BYTES;
164  if (lead > p->l1_bytes / 2u) lead = p->l1_bytes / 2u;
165  /* If strides are large, keep at least one stride ahead. */
166  if (stride_bytes > 0 && lead < stride_bytes) lead = stride_bytes;
167  return lead;
168 }

226  uintptr_t addr = (uintptr_t)dst;
