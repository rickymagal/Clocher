70 static void ie_detect_caches_linux(size_t *l1, size_t *l2, size_t *l3) {
71  if (l1) *l1 = 0;
72  if (l2) *l2 = 0;
73  if (l3) *l3 = 0;
74 
75  /* Common indices: vendor layouts vary; read a small set and rank by size. */
76  const char *paths[] = {
77  "/sys/devices/system/cpu/cpu0/cache/index0/size",
78  "/sys/devices/system/cpu/cpu0/cache/index1/size",
79  "/sys/devices/system/cpu/cpu0/cache/index2/size",
80  "/sys/devices/system/cpu/cpu0/cache/index3/size",
81  "/sys/devices/system/cpu/cpu0/cache/index4/size",
82  };
83 
84  size_t found[5] = {0};
85  for (int i = 0; i < 5; ++i) found[i] = ie_read_cache_size_file(paths[i]);
86 
87  /* Sort descending (tiny N=5 selection sort). */
88  for (int i = 0; i < 5; ++i) {
89  for (int j = i + 1; j < 5; ++j) {
90  if (found[j] > found[i]) {
91  size_t t = found[i];
92  found[i] = found[j];
93  found[j] = t;
94  }
95  }
96  }
97 
98  /* Pick largest as L3, next as L2, next as L1D (rough but robust enough). */
99  if (l3) *l3 = found[0];
100  if (l2) *l2 = found[1];
101  if (l1) *l1 = found[2];
102 }

219 size_t ie_stream_recommend_prefetch_distance(const struct ie_stream_policy *p,
220  size_t stride_bytes) {
221  if (!p) return 4u * IE_CACHELINE_BYTES;
222 
223  size_t lead = p->prefetch_distance ? p->prefetch_distance
224  : (6u * IE_CACHELINE_BYTES);
225 
226  if (lead < IE_CACHELINE_BYTES) lead = IE_CACHELINE_BYTES;
227  if (lead > p->l1_bytes / 2u) lead = p->l1_bytes / 2u;
228 
229  if (stride_bytes > 0 && lead < stride_bytes) lead = stride_bytes;
230 
231  return lead;
232 }

344

346  uintptr_t addr = (uintptr_t)dst;
347  size_t mis = (size_t)(addr & (64u - 1u));
348  if (mis) {
349  size_t pro = (64u - mis) / sizeof(float);
350  if (pro > n) pro = n;

355  }
