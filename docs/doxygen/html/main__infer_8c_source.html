174static int read_first_nonempty_line(const char *path, char *buf, size_t bufsz) {
175 FILE *f = fopen(path, "r");
176 if (!f) { fprintf(stderr, "warn: cannot open prompts file '%s': %s\n", path, strerror(errno)); return -1; }
177 int ok = 0;
178 while (fgets(buf, (int)bufsz, f)) {
179 size_t n = strlen(buf);
180 while (n && (buf[n-1] == '\n' || buf[n-1] == '\r')) buf[--n] = '\0';
181 if (n == 0) continue;
182 ok = 1; break;
183 }
184 fclose(f);
185 return ok ? 1 : 0;
186}

191static void print_json_result(uint32_t n_tok,
192 const uint32_t *tokens,
193 double wall_s_in,
194 const ie_metrics_t *m_in) {
195 const double WALL_MIN = 3e-4; /* 0.0003 s */
196 double wall_s = wall_s_in;
197 if (wall_s > 0.0 && wall_s < WALL_MIN) wall_s = WALL_MIN;
198
199 double tps_local = (wall_s > 0.0) ? ((double)n_tok / wall_s) : 0.0;
200 double tps = (m_in && m_in->tps_true > 0.0) ? m_in->tps_true : tps_local;
201
202 double p50 = (m_in ? m_in->latency_p50_ms : 0.0);
203 double p95 = (m_in ? m_in->latency_p95_ms : 0.0);
204 if ((p50 <= 0.0 || p95 <= 0.0) && n_tok > 0 && wall_s > 0.0) {
205 double per_tok_ms = (wall_s * 1000.0) / (double)n_tok;
206 const double LAT_MIN_MS = 0.001;
207 if (per_tok_ms < LAT_MIN_MS) per_tok_ms = LAT_MIN_MS;
208 p50 = per_tok_ms;
209 p95 = per_tok_ms * 2.0;
210 }
211
212 fprintf(stdout, "{\"tokens_generated\": %u,", (unsigned)n_tok);
213
214 fputs("\"tokens\":[", stdout);
215 for (uint32_t i = 0; i < n_tok; ++i) {
216 const uint32_t v = tokens ? tokens[i] : 0u;
217 fprintf(stdout, "%u%s", v, (i + 1 < n_tok) ? "," : "");
218 }
219 fputs("],", stdout);
220
221 fprintf(stdout,
222 "\"wall_time_s\": %.6f,"
223 "\"tps_true\": %.6f,"
224 "\"latency_p50_ms\": %.3f,"
225 "\"latency_p95_ms\": %.3f,"
226 "\"rss_peak_mb\": %u,"
227 "\"kv_hits\": %llu,"
228 "\"kv_misses\": %llu}\n",
229 wall_s,
230 tps,
231 p50,
232 p95,
233 (unsigned)(m_in ? m_in->rss_peak_mb : 0u),
234 (unsigned long long)(m_in ? m_in->kv_hits : 0ULL),
235 (unsigned long long)(m_in ? m_in->kv_misses : 0ULL));
236}

242 /* -------------------- parse CLI -------------------- */
243 cli_extras_t opt;
244 if (parse_flags(argc, argv, &opt) != 0) return 2;
245
246 /* If no explicit --prompt, try --prompts-file first non-empty line. */
247 char prompt_buf[8192];
248 if (!opt.prompt && opt.prompts_file && !opt.aggregate) {
249 int r = read_first_nonempty_line(opt.prompts_file, prompt_buf, sizeof(prompt_buf));
250 if (r == 1) opt.prompt = prompt_buf;
251 else if (r == 0) { fprintf(stderr, "warn: prompts file is empty; using default prompt\n"); opt.prompt = "bench"; }
252 else { opt.prompt = "bench"; }
253 }
254 if (!opt.prompt && !opt.prompts_file) {
255 /* Graceful empty run: print a valid JSON skeleton for tests. */
256 ie_metrics_t mm; memset(&mm, 0, sizeof(mm));
257 print_json_result(0, NULL, 0.0, &mm);
258 return 0;
259 }
260
261 /* -------------------- strict IEBIN gating -------------------- */
262 const int require_model = (int)env_long("IE_REQUIRE_MODEL", 0);
263
264 ie_weights_t w; memset(&w, 0, sizeof(w));
265 {
266 const char *j = "./model.ie.json";
267 const char *b = "./model.ie.bin";
268 int wrc = ie_weights_open(j, b, &w);
269 if (wrc != IE_IO_OK) {
270 if (require_model) {
271 fprintf(stderr, "error: failed to open IEBIN (%s, %s), status=%d, errno=%d (%s)\n",
272 j, b, wrc, errno, strerror(errno));
273 return 3;
274 }
275 fprintf(stderr, "# -> warn: IEBIN metadata not found; continuing in stub mode…\n");
276 } else {
277 if (ie_weights_touch(&w) != 0) {
278 fprintf(stderr, "error: IEBIN present but unreadable (prefault/touch failed)\n");
279 ie_weights_close(&w);
280 return 3;
281 }
282 }
283 }
284
285 /* -------------------- create engine -------------------- */
286 ie_engine_params_t params; memset(&params, 0, sizeof(params));
287 params.precision = "fp32";
288 params.affinity = "auto";
289 params.pretranspose = "none";
290 params.prefetch = "auto";
291
292 ie_engine_t *engine = NULL;
293 ie_status_t st = ie_engine_create(&params, &engine);
294 if (st != IE_OK || !engine) {
295 fprintf(stderr, "error: ie_engine_create failed (status=%d)\n", (int)st);
296 ie_weights_close(&w);
297 return 5;
298 }
299
300 /* -------------------- optional warmup -------------------- */
301 if (opt.warmup_tokens > 0) {
302 const char *wprompt = "warmup";
303 uint32_t wtoks[128]; uint32_t wcount = 0;
304 size_t wmax = (opt.warmup_tokens <= (int)(sizeof(wtoks)/sizeof(wtoks[0])))
305 ? (size_t)opt.warmup_tokens

307 (void)ie_engine_generate(engine, wprompt, wmax, wtoks, &wcount);
308 }
309
310 /* -------------------- token output buffer -------------------- */
311 uint32_t *tokens = NULL; uint32_t n_tok = 0;
312 size_t cap = (opt.max_new > 0 ? opt.max_new : 0);
313 if (cap > 0) {
314 tokens = (uint32_t*)malloc(sizeof(uint32_t) * cap);
315 if (!tokens) {
316 fprintf(stderr, "error: OOM allocating token buffer\n");
317 ie_engine_destroy(engine);
318 ie_weights_close(&w);
319 return 6;
320 }
321 }
322
323 /* -------------------- mmap model.bin for per-token work -------------------- */
324 const size_t bytes_per_token = (size_t)env_long("IE_BYTES_PER_TOKEN", 0);
325 const size_t stride_bytes = (size_t)env_long("IE_STRIDE_BYTES", 256);
326 const int verify_touch = (int)env_long("IE_VERIFY_TOUCH", 0);
327
328 int bin_fd = -1;
329 uint8_t *map = NULL;
330 size_t map_len = 0;
331 if (w.loaded && bytes_per_token > 0 && w.bin_size_bytes > 0) {
332 const char *binp = (w.weights_path[0] ? w.weights_path : "./model.ie.bin");
333 bin_fd = open(binp, O_RDONLY);
334 if (bin_fd >= 0) {
335 map_len = (size_t)w.bin_size_bytes;
336 void *p = mmap(NULL, map_len, PROT_READ, MAP_PRIVATE, bin_fd, 0);
337 if (p == MAP_FAILED) { map = NULL; map_len = 0; }
338 else { map = (uint8_t*)p; }
339 }
340 }

342 /* -------------------- KV round lifecycle (outside timing) ------------------ */
343 (void)ie_kv_begin_round();
344 uint64_t total_tokens_this_round = 0;
345
346 /* -------------------- measured window: ONLY gen + work-touch -------------- */
347 double t0 = now_sec();
348
349 if (opt.aggregate && opt.prompts_file) {
350 FILE *pf = fopen(opt.prompts_file, "r");
351 if (pf) {
352 char line[8192];
353 while (fgets(line, sizeof(line), pf)) {
354 size_t n = strlen(line);
355 while (n && (line[n-1]=='\n'||line[n-1]=='\r')) line[--n]='\0';
356 if (!n) continue;
357
358 uint32_t n_here = 0;
359 st = ie_engine_generate(engine, line, cap, (tokens?tokens:NULL), &n_here);
360 if (st != IE_OK) { fprintf(stderr,"error: ie_engine_generate (status=%d)\n",(int)st); break; }
361 n_tok += n_here;
362 total_tokens_this_round += (uint64_t)n_here;
363
364 if (map && map_len && bytes_per_token) {
365 size_t need = (size_t)n_here * bytes_per_token;
366 size_t pos = 0; volatile uint64_t acc = 0;
367 while (pos < need) { size_t off = (pos % map_len); acc += map[off]; pos += (stride_bytes ? stride_bytes : 1); }
368 if (verify_touch) { (void)acc; }
369 }
370 }
371 fclose(pf);
372 } else {
373 fprintf(stderr,"warn: cannot open prompts-file '%s'\n", opt.prompts_file);
374 }
375 } else {
376 const char *p = (opt.prompt ? opt.prompt : "bench");
377 st = ie_engine_generate(engine, p, cap, (tokens ? tokens : NULL), &n_tok);
378 if (st != IE_OK) {
379 fprintf(stderr, "error: ie_engine_generate failed (status=%d)\n", (int)st);
380 }
381 total_tokens_this_round += (uint64_t)n_tok;
382
383 if (map && map_len && bytes_per_token) {
384 size_t need = (size_t)n_tok * bytes_per_token;
385 size_t pos = 0; volatile uint64_t acc = 0;
386 while (pos < need) { size_t off = (pos % map_len); acc += map[off]; pos += (stride_bytes ? stride_bytes : 1); }
387 if (verify_touch) { (void)acc; }
388 }
389 }
390
391 double t1 = now_sec();
392
393 /* -------------------- AFTER timing: collect metrics ----------------------- */
394 uint64_t kv_hits_round = 0, kv_miss_round = 0;
395 ie_kv_finish_round(total_tokens_this_round, &kv_hits_round, &kv_miss_round);
396
397 ie_metrics_t m; memset(&m, 0, sizeof(m));
398 (void)ie_engine_metrics(engine, &m);
399 m.kv_hits = kv_hits_round;
400 m.kv_misses = kv_miss_round;
401
402 (void)ie_metrics_sample_rss_peak(&m); /* fills m.rss_peak_mb */
403
404 /* -------------------- print JSON & teardown ------------------------------ */
405 print_json_result(n_tok, tokens, t1 - t0, &m);
406
407 free(tokens);
408 if (map && map != MAP_FAILED) munmap(map, map_len);
409 if (bin_fd >= 0) close(bin_fd);
410 ie_engine_destroy(engine);
411 ie_weights_close(&w);
412 return 0;
413}
