193 static int ascii_ieq(const char *a, const char *b) {
194  if (!a || !b) return 0;
195  while (*a && *b) {
196  unsigned char ca = (unsigned char)*a;
197  unsigned char cb = (unsigned char)*b;
198  unsigned char la = (ca >= 'A' && ca <= 'Z') ? (unsigned char)(ca + 32) : ca;
199  unsigned char lb = (cb >= 'A' && cb <= 'Z') ? (unsigned char)(cb + 32) : cb;
200  if (la != lb) return 0;
201  ++a;
202  ++b;
203  }
204  return *a == '\0' && *b == '\0';
205 }
