165 static int ascii_ieq(const char *a, const char *b) {
166  if (!a || !b) return 0;
167  while (*a && *b) {
168  unsigned char ca = (unsigned char)*a;
169  unsigned char cb = (unsigned char)*b;
170  unsigned char la = (ca >= 'A' && ca <= 'Z') ? (unsigned char)(ca + 32) : ca;
171  unsigned char lb = (cb >= 'A' && cb <= 'Z') ? (unsigned char)(cb + 32) : cb;
172  if (la != lb) return 0;
173  ++a; ++b;
174  }
175  return *a == '\0' && *b == '\0';
176 }
