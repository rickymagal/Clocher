127static int ascii_ieq(const char *a, const char *b) {
128 if (!a || !b) return 0;
129 while (*a && *b) {
130 unsigned char ca = (unsigned char)*a;
131 unsigned char cb = (unsigned char)*b;
132 unsigned char la = (ca >= 'A' && ca <= 'Z') ? (unsigned char)(ca + 32) : ca;
133 unsigned char lb = (cb >= 'A' && cb <= 'Z') ? (unsigned char)(cb + 32) : cb;
134 if (la != lb) return 0;
135 ++a;
136 ++b;
137 }
138 return *a == '\0' && *b == '\0';
139}

164static void join_path(char *out, size_t outsz, const char *dir, const char *leaf) {
165 if (!out || outsz == 0) return;
166 out[0] = '\0';
167 if (!leaf) return;

169 if (path_is_abs(leaf)) {

175 safe_strcpy(out, outsz, leaf);

181 (void)snprintf(out, outsz, "%s%s", dir, leaf);
182 } else {
183 (void)snprintf(out, outsz, "%s/%s", dir, leaf);

284typedef struct cudart_api {
285 void *handle;
286 int ok;
287
288 cudaFree_fn_t cudaFree;
289 cudaMalloc_fn_t cudaMalloc;
290 cudaMemcpy_fn_t cudaMemcpy;
291 cudaMemset_fn_t cudaMemset;
292 cudaDeviceSynchronize_fn_t cudaDeviceSynchronize;
293 cudaGetErrorString_fn_t cudaGetErrorString;
294} cudart_api_t;

296static cudart_api_t *cudart_get_api(void) {
297 static cudart_api_t api;
298 static int inited = 0;
299
300 if (inited) return api.ok ? &api : NULL;
301 inited = 1;
302
303 memset(&api, 0, sizeof(api));
304
305 api.handle = dlopen("libcudart.so", RTLD_NOW | RTLD_LOCAL);
306 if (!api.handle) api.handle = dlopen("libcudart.so.13.0", RTLD_NOW | RTLD_LOCAL);
307 if (!api.handle) api.handle = dlopen("libcudart.so.12", RTLD_NOW | RTLD_LOCAL);
308
309 if (!api.handle) {
310 api.ok = 0;
311 return NULL;
312 }
313
314 {
315 union { void *p; cudaFree_fn_t f; } u;
316 u.p = dlsym(api.handle, "cudaFree");
317 api.cudaFree = u.f;
318 }
319 {
320 union { void *p; cudaMalloc_fn_t f; } u;
321 u.p = dlsym(api.handle, "cudaMalloc");
322 api.cudaMalloc = u.f;
323 }
324 {
325 union { void *p; cudaMemcpy_fn_t f; } u;
326 u.p = dlsym(api.handle, "cudaMemcpy");
327 api.cudaMemcpy = u.f;
328 }
329 {
330 union { void *p; cudaMemset_fn_t f; } u;
331 u.p = dlsym(api.handle, "cudaMemset");
332 api.cudaMemset = u.f;
333 }
334 {
335 union { void *p; cudaDeviceSynchronize_fn_t f; } u;
336 u.p = dlsym(api.handle, "cudaDeviceSynchronize");
337 api.cudaDeviceSynchronize = u.f;
338 }
339 {
340 union { void *p; cudaGetErrorString_fn_t f; } u;
341 u.p = dlsym(api.handle, "cudaGetErrorString");
342 api.cudaGetErrorString = u.f;
343 }
344
345 if (!api.cudaFree || !api.cudaMalloc || !api.cudaMemcpy || !api.cudaMemset || !api.cudaDeviceSynchronize) {
346 (void)dlclose(api.handle);
347 api.handle = NULL;
348 api.ok = 0;
349 return NULL;
350 }
351
352 api.ok = 1;
353 return &api;
354}

363/* Strict "touch" that guarantees real CUDA work without any custom kernel. */
364static int cudart_touch_bytes(size_t bytes_per_token,
365 uint64_t tokens,
366 size_t stride_bytes,
367 int verify_touch) {
368 UNUSED(stride_bytes);
369
370 cudart_api_t *api = cudart_get_api();
371 if (!api) return -1;
372
373 if (bytes_per_token == 0 || tokens == 0) return 0;
374
375 void *d = NULL;
376 cudart_err_t e = api->cudaMalloc(&d, bytes_per_token);
377 if (e != 0 || !d) return 1;
378
379 for (uint64_t t = 0; t < tokens; ++t) {
380 int pattern = (int)(t & 0xFFu);
381 e = api->cudaMemset(d, pattern, bytes_per_token);
382 if (e != 0) {
383 (void)api->cudaFree(d);
384 return 2;
385 }
386 }
387
388 e = api->cudaDeviceSynchronize();
389 if (e != 0) {
390 (void)api->cudaFree(d);
391 return 3;
392 }
393
394 if (verify_touch) {
395 uint64_t tmp = 0;
396 size_t n = min_size(sizeof(tmp), bytes_per_token);
397 e = api->cudaMemcpy(&tmp, d, n, CUDA_MEMCPY_DEVICE_TO_HOST);
398 if (e != 0) {
399 (void)api->cudaFree(d);
400 return 4;
401 }
402 e = api->cudaDeviceSynchronize();
403 if (e != 0) {
404 (void)api->cudaFree(d);
405 return 5;
406 }
407 (void)tmp;
408 }
409
410 (void)api->cudaFree(d);
411 (void)api->cudaDeviceSynchronize();
412 return 0;
413}

692 if (v < 0) { fprintf(stderr, "error: --pf-x >= 0\n"); return -1; }
693 out->pf_x_bytes = (size_t)v;
694
695 } else if (!strcmp(a, "--force-ntw")) {
696 if (++i >= argc) { usage(); return -1; }
697 long v = safe_atoi(argv[i]);
698 if (v != 0 && v != 1) { fprintf(stderr, "error: --force-ntw 0|1\n"); return -1; }
699 out->force_ntw = (int)v;
700
701 } else if (!strcmp(a, "--dedup")) {
702 if (++i >= argc) { usage(); return -1; }
703 long v = safe_atoi(argv[i]);
704 if (v != 0 && v != 1) { fprintf(stderr, "error: --dedup 0|1\n"); return -1; }
705 out->dedup_enable = (int)v;
706
707 } else if (!strcmp(a, "--dedup-policy")) {
708 if (++i >= argc) { usage(); return -1; }
709 out->dedup_policy = argv[i];
710
711 } else if (!strcmp(a, "--dedup-cache-mb")) {
712 if (++i >= argc) { usage(); return -1; }
713 long v = safe_atoi(argv[i]);
714 if (v < 0) v = 0;
715 out->dedup_cache_mb = v;
716
717 } else if (!strcmp(a, "--dedup-hot-bytes")) {
718 if (++i >= argc) { usage(); return -1; }
719 long v = safe_atoi(argv[i]);
720 if (v < 0) v = 0;
721 out->dedup_hot_bytes = v;
722
723 } else if (!strcmp(a, "--dedup-hot-list")) {
724 if (++i >= argc) { usage(); return -1; }
725 out->dedup_hot_list = argv[i];
726
727 } else if (a[0] == '-') {
728 fprintf(stderr, "error: unknown flag '%s'\n", a);
729 usage();
730 return -1;
731
732 } else {
733 if (!out->prompt) out->prompt = a;
734 }
735 }
736
737 return 0;
738}

742/* -------------------------------------------------------------------------- */

744static int read_first_nonempty_line(const char *path, char *buf, size_t bufsz) {
745 FILE *f = fopen(path, "r");
746 if (!f) {
747 fprintf(stderr, "warn: cannot open prompts file '%s': %s\n", path, strerror(errno));
748 return -1;
749 }
750
751 int ok = 0;
752 while (fgets(buf, (int)bufsz, f)) {
753 size_t n = strlen(buf);
754 while (n && (buf[n - 1] == '\n' || buf[n - 1] == '\r')) buf[--n] = '\0';
755 if (n == 0) continue;
756 ok = 1;
757 break;
758 }
759
760 (void)fclose(f);
761 return ok ? 1 : 0;
762}

764/* -------------------------------------------------------------------------- */
765/* JSON emitter */
766/* -------------------------------------------------------------------------- */
767
768static void print_json_result(uint32_t n_tok,
769 const uint32_t *tokens,
770 double wall_s_in,
771 uint64_t kv_hits,
772 uint64_t kv_misses,
773 uint32_t rss_peak_mb) {
774 const double WALL_MIN = 3e-4;
775 double wall_s = wall_s_in;
776 if (wall_s > 0.0 && wall_s < WALL_MIN) wall_s = WALL_MIN;
777
778 const double tps_true = (wall_s > 0.0) ? ((double)n_tok / wall_s) : 0.0;
779
780 double p50 = 0.0, p95 = 0.0;
781 if (n_tok > 0 && wall_s > 0.0) {
782 double per_tok_ms = (wall_s * 1000.0) / (double)n_tok;
783 if (per_tok_ms < 0.001) per_tok_ms = 0.001;
784 p50 = per_tok_ms;
785 p95 = per_tok_ms * 2.0;
786 }
787
788 fprintf(stdout, "{\"tokens_generated\": %u,", (unsigned)n_tok);
789
790 fputs("\"tokens\":[", stdout);
791 if (tokens && n_tok > 0) {
792 for (uint32_t i = 0; i < n_tok; ++i) {
793 fprintf(stdout, "%u%s", tokens[i], (i + 1 < n_tok) ? "," : "");
794 }
795 }
796 fputs("],", stdout);
797
798 fprintf(stdout,
799 "\"wall_time_s\": %.6f,"
800 "\"tps_true\": %.6f,"
801 "\"latency_p50_ms\": %.3f,"
802 "\"latency_p95_ms\": %.3f,"
803 "\"rss_peak_mb\": %u,"
804 "\"kv_hits\": %llu,"
805 "\"kv_misses\": %llu}\n",
806 wall_s,
807 tps_true,
808 p50,
809 p95,
810 (unsigned)rss_peak_mb,
811 (unsigned long long)kv_hits,
812 (unsigned long long)kv_misses);
813}

1098 tokens_to_print,
1099 (t1 - t0),
1100 m.kv_hits,
1101 m.kv_misses,
1102 rss_mib);
1103
1104 free(tokens);
1105 ie_engine_destroy(engine);
1106 ie_weights_close(&w);
1107 return 0;
