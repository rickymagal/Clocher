125static int ascii_ieq(const char *a, const char *b) {
126 if (!a || !b) return 0;
127 while (*a && *b) {
128 unsigned char ca = (unsigned char)*a;
129 unsigned char cb = (unsigned char)*b;
130 unsigned char la = (ca >= 'A' && ca <= 'Z') ? (unsigned char)(ca + 32) : ca;
131 unsigned char lb = (cb >= 'A' && cb <= 'Z') ? (unsigned char)(cb + 32) : cb;
132 if (la != lb) return 0;
133 ++a; ++b;
134 }
135 return *a == '\0' && *b == '\0';
136}

275static int parse_flags(int argc, char **argv, cli_extras_t *out) {
276 cli_extras_defaults(out);
277 for (int i = 1; i < argc; ++i) {
278 const char *a = argv[i];
279 if (!strcmp(a,"--help") || !strcmp(a,"-h")) { usage(); return -1; }
280 else if (!strcmp(a,"--prompt")) { if (++i>=argc) { usage(); return -1; } out->prompt = argv[i]; }
281 else if (!strcmp(a,"--max-new")) { if (++i>=argc) { usage(); return -1; } long v=safe_atoi(argv[i]); if(v<0){fprintf(stderr,"error: --max-new >= 0\n");return -1;} out->max_new=(size_t)v; }
282 else if (!strcmp(a,"--threads")) { if (++i>=argc) { usage(); return -1; } long v=safe_atoi(argv[i]); out->threads=(int)v; }
283 else if (!strcmp(a,"--precision")) {
284 if (++i>=argc) { usage(); return -1; }
285 const char *m = argv[i];
286 if (ascii_ieq(m, IE_PREC_FP32)) out->prec=CLI_PREC_FP32, out->precision_label=IE_PREC_FP32, out->precision_from_flag=1;
287 else if (ascii_ieq(m, IE_PREC_BF16)) out->prec=CLI_PREC_BF16, out->precision_label=IE_PREC_BF16, out->precision_from_flag=1;
288 else if (ascii_ieq(m, IE_PREC_FP16)) out->prec=CLI_PREC_FP16, out->precision_label=IE_PREC_FP16, out->precision_from_flag=1;
289 else if (ascii_ieq(m, IE_PREC_INT8W)) { out->precision_label = IE_PREC_INT8W; out->precision_from_flag=1; }
290 else if (ascii_ieq(m, IE_PREC_INT4W) || ascii_ieq(m, IE_PREC_INT4)) { out->precision_label = IE_PREC_INT4W; out->precision_from_flag=1; }
291 else { fprintf(stderr,"error: unknown precision '%s'\n", m); return -1; }
292 }
293 else if (!strcmp(a,"--affinity")) { if (++i>=argc) { usage(); return -1; } out->affinity = argv[i]; }
294 else if (!strcmp(a,"--pretranspose")){ if (++i>=argc) { usage(); return -1; } const char*p=argv[i];
295 if(!strcmp(p,"none")) out->pretx=CLI_PRETX_NONE;
296 else if(!strcmp(p,"woh")) out->pretx=CLI_PRETX_WOH;
297 else if(!strcmp(p,"wxh")) out->pretx=CLI_PRETX_WXH;
298 else if(!strcmp(p,"all")) out->pretx=CLI_PRETX_ALL;
299 else { fprintf(stderr,"error: unknown pretranspose '%s'\n", p); return -1; }
300 }
301 else if (!strcmp(a,"--device")) { if (++i>=argc) { usage(); return -1; } out->device = argv[i]; }
302 else if (!strcmp(a,"--model-dir")) { if (++i>=argc) { usage(); return -1; } out->model_dir = argv[i]; }
303 else if (!strcmp(a,"--model-json")) { if (++i>=argc) { usage(); return -1; } out->model_json = argv[i]; }
304 else if (!strcmp(a,"--model-bin")) { if (++i>=argc) { usage(); return -1; } out->model_bin = argv[i]; }
305 else if (!strcmp(a,"--prompts-file")){ if (++i>=argc) { usage(); return -1; } out->prompts_file = argv[i]; }
306 else if (!strcmp(a,"--batch")) { if (++i>=argc) { usage(); return -1; } out->batch = (int)safe_atoi(argv[i]); }
307 else if (!strcmp(a,"--prefetch")) { if (++i>=argc) { usage(); return -1; } out->prefetch = argv[i]; }
308 else if (!strcmp(a,"--warmup") || !strcmp(a,"--warmup-tokens")) {
309 if (++i>=argc) { usage(); return -1; }
310 long v = safe_atoi(argv[i]); if (v < 0) v = 0; out->warmup_tokens = (int)v;
311 }
312 else if (!strcmp(a,"--rounds")) {
313 if (++i>=argc) { usage(); return -1; }
314 long v = safe_atoi(argv[i]); if (v < 1) v = 1; out->rounds = (int)v;
315 }
316 else if (!strcmp(a,"--aggregate")) { out->aggregate = 1; }
317 else if (!strcmp(a,"--pf-w")) { if (++i>=argc) { usage(); return -1; } long v=safe_atoi(argv[i]); if(v<0){fprintf(stderr,"error: --pf-w >= 0\n");return -1;} out->pf_w_bytes=(size_t)v; }
318 else if (!strcmp(a,"--pf-x")) { if (++i>=argc) { usage(); return -1; } long v=safe_atoi(argv[i]); if(v<0){fprintf(stderr,"error: --pf-x >= 0\n");return -1;} out->pf_x_bytes=(size_t)v; }
319 else if (!strcmp(a,"--force-ntw")) { if (++i>=argc) { usage(); return -1; } long v=safe_atoi(argv[i]); if(v!=0 && v!=1){fprintf(stderr,"error: --force-ntw 0|1\n");return -1;} out->force_ntw=(int)v; }
320 else if (a[0] == '-') { fprintf(stderr,"error: unknown flag '%s'\n", a); usage(); return -1; }
321 else { out->prompt = a; } /* positional */
322 }
323 return 0;
324}

337static int read_first_nonempty_line(const char *path, char *buf, size_t bufsz) {
338 FILE *f = fopen(path, "r");
339 if (!f) { fprintf(stderr, "warn: cannot open prompts file '%s': %s\n", path, strerror(errno)); return -1; }
340 int ok = 0;
341 while (fgets(buf, (int)bufsz, f)) {
342 size_t n = strlen(buf);
343 while (n && (buf[n-1] == '\n' || buf[n-1] == '\r')) buf[--n] = '\0';
344 if (n == 0) continue;
345 ok = 1; break;
346 }
347 fclose(f);
348 return ok ? 1 : 0;
349}

364static void print_json_result(uint32_t n_tok,
365 const uint32_t *tokens,
366 double wall_s_in,
367 uint64_t kv_hits,
368 uint64_t kv_misses,
369 uint32_t rss_peak_mb) {
370 const double WALL_MIN = 3e-4;
371 double wall_s = wall_s_in;
372 if (wall_s > 0.0 && wall_s < WALL_MIN) wall_s = WALL_MIN;
373
374 const double tps_true = (wall_s > 0.0) ? ((double)n_tok / wall_s) : 0.0;
375
376 double p50 = 0.0, p95 = 0.0;
377 if (n_tok > 0 && wall_s > 0.0) {
378 double per_tok_ms = (wall_s * 1000.0) / (double)n_tok;
379 if (per_tok_ms < 0.001) per_tok_ms = 0.001;
380 p50 = per_tok_ms;
381 p95 = per_tok_ms * 2.0;
382 }
383
384 fprintf(stdout, "{\"tokens_generated\": %u,", (unsigned)n_tok);
385
386 fputs("\"tokens\":[", stdout);
387 if (tokens && n_tok > 0) {
388 for (uint32_t i = 0; i < n_tok; ++i) {
389 fprintf(stdout, "%u%s", tokens[i], (i + 1 < n_tok) ? "," : "");
390 }
391 }
392 fputs("],", stdout);
393
394 fprintf(stdout,
395 "\"wall_time_s\": %.6f,"
396 "\"tps_true\": %.6f,"
397 "\"latency_p50_ms\": %.3f,"
398 "\"latency_p95_ms\": %.3f,"
399 "\"rss_peak_mb\": %u,"
400 "\"kv_hits\": %llu,"
401 "\"kv_misses\": %llu}\n",
402 wall_s,
403 tps_true,
404 p50,
405 p95,
406 (unsigned)rss_peak_mb,
407 (unsigned long long)kv_hits,
408 (unsigned long long)kv_misses);
409}

435int main(int argc, char **argv) {
436 /* -------------------- parse CLI -------------------- */
437 cli_extras_t opt;
438 if (parse_flags(argc, argv, &opt) != 0) return 2;
439
440 /* Export per-tensor prefetch/stream overrides for kernels (if provided). */
441 if (opt.pf_w_bytes > 0) {
442 char buf[32]; snprintf(buf, sizeof buf, "%zu", opt.pf_w_bytes);
443 setenv("IE_GEMV_PFDIST_W", buf, 1);
444 }
445 if (opt.pf_x_bytes > 0) {
446 char buf[32]; snprintf(buf, sizeof buf, "%zu", opt.pf_x_bytes);
447 setenv("IE_GEMV_PFDIST_X", buf, 1);
448 }
449 if (opt.force_ntw == 0 || opt.force_ntw == 1) {
450 setenv("IE_GEMV_FORCE_NTW", opt.force_ntw ? "1" : "0", 1);
451 }
452
453 /* Optional: change working dir early if requested. */
454 if (opt.model_dir && *opt.model_dir) {
455 if (chdir(opt.model_dir) != 0) {
456 fprintf(stderr, "error: --model-dir '%s' is not accessible: %s\n",
457 opt.model_dir, strerror(errno));
458 return 3;
459 }
460 }
461
462 /* Precision from env if not provided on the command line. */
463 if (!opt.precision_from_flag) {
464 const char *envp = env_str("IE_PRECISION", env_str("PRECISION", IE_PREC_FP32));
465 if (ascii_ieq(envp, IE_PREC_INT4W) || ascii_ieq(envp, IE_PREC_INT4))
466 opt.precision_label = IE_PREC_INT4W;
467 else if (ascii_ieq(envp, IE_PREC_INT8W))
468 opt.precision_label = IE_PREC_INT8W;
469 else if (ascii_ieq(envp, IE_PREC_BF16))
470 opt.precision_label = IE_PREC_BF16;
471 else if (ascii_ieq(envp, IE_PREC_FP16))
472 opt.precision_label = IE_PREC_FP16;
473 else
474 opt.precision_label = IE_PREC_FP32;
475 }
476
477 /* If no explicit --prompt, try --prompts-file first non-empty line. */
478 char prompt_buf[8192];
479 if (!opt.prompt && opt.prompts_file && !opt.aggregate) {
480 int r = read_first_nonempty_line(opt.prompts_file, prompt_buf, sizeof(prompt_buf));
481 if (r == 1) opt.prompt = prompt_buf;
482 else if (r == 0) { fprintf(stderr, "warn: prompts file is empty; using default prompt\n"); opt.prompt = "bench"; }
483 else { opt.prompt = "bench"; }
484 }
485 if (!opt.prompt && !opt.prompts_file) {
486 ie_metrics_t mm; memset(&mm, 0, sizeof(mm));
487 print_json_result(0, NULL, 0.0, 0, 0, 0);
488 return 0;
489 }
490
491 /* -------------------- strict IEBIN gating -------------------- */
492 const int require_model = (int)env_long("IE_REQUIRE_MODEL", 0);
493
494 ie_weights_t w; memset(&w, 0, sizeof(w));
495 {
496 const char *j = (opt.model_json && *opt.model_json) ? opt.model_json : "./model.ie.json";
497 const char *b = (opt.model_bin && *opt.model_bin ) ? opt.model_bin : "./model.ie.bin";
498 int wrc = ie_weights_open(j, b, &w);
499 if (wrc != IE_IO_OK) {
500 if (require_model) {
501 fprintf(stderr, "error: failed to open IEBIN (%s, %s), status=%d, errno=%d (%s)\n",
502 j, b, wrc, errno, strerror(errno));
503 return 3;
504 }
505 fprintf(stderr, "# -> warn: IEBIN metadata not found; continuing in stub mode\n");
506 } else {
507 if (ie_weights_touch(&w) != 0) {
508 fprintf(stderr, "error: IEBIN present but unreadable (prefault/touch failed)\n");
509 ie_weights_close(&w);
510 return 3;
511 }
512 }
513 }
514
515 /* -------------------- create engine -------------------- */
516 ie_engine_params_t params; memset(&params, 0, sizeof(params));
517 params.precision = opt.precision_label;
518 params.affinity = opt.affinity;
519 params.pretranspose = (opt.pretx==CLI_PRETX_NONE ? "none" :
520 (opt.pretx==CLI_PRETX_WOH ? "woh" :
521 (opt.pretx==CLI_PRETX_WXH ? "wxh" : "all")));
522 params.prefetch = opt.prefetch;
523 params.threads = opt.threads;
524 UNUSED(opt.device);
525
526 ie_engine_t *engine = NULL;
527 ie_status_t st = ie_engine_create(&params, &engine);
528 if (st != IE_OK || !engine) {
529 fprintf(stderr, "error: ie_engine_create failed (status=%d)\n", (int)st);
530 ie_weights_close(&w);
531 return 5;
532 }
533
534 /* -------------------- optional warmup (outside timing) ------------------- */
535 if (opt.warmup_tokens > 0) {
536 const char *wprompt = "warmup";
537 uint32_t wtoks[128]; uint32_t wcount = 0;
538 size_t wmax = (opt.warmup_tokens <= (int)(sizeof(wtoks)/sizeof(wtoks[0])))
539 ? (size_t)opt.warmup_tokens
540 : (sizeof(wtoks)/sizeof(wtoks[0]));
541 (void)ie_engine_generate(engine, wprompt, wmax, wtoks, &wcount);
542 }
543
544 /* -------------------- token output buffer -------------------- */
545 uint32_t *tokens = NULL;
546 size_t cap = (opt.max_new > 0 ? opt.max_new : 0);
547 size_t cap_alloc = (cap > 0 ? cap : 1);
548 tokens = (uint32_t*)malloc(sizeof(uint32_t) * cap_alloc);
549 if (!tokens) {
550 fprintf(stderr, "error: OOM allocating token buffer\n");
551 ie_engine_destroy(engine);
552 ie_weights_close(&w);
553 return 6;
554 }
555
556 /* -------------------- mmap model.bin for per-token work ------------------ */
557 const size_t bytes_per_token = (size_t)env_long("IE_BYTES_PER_TOKEN", 0);
558 const size_t stride_bytes = (size_t)env_long("IE_STRIDE_BYTES", 256);
559 const int verify_touch = (int)env_long("IE_VERIFY_TOUCH", 0);
560
561 int bin_fd = -1;
562 uint8_t *map = NULL;
563 size_t map_len = 0;
564 if (w.loaded && bytes_per_token > 0 && w.bin_size_bytes > 0) {
565 const char *binp = (w.weights_path[0] ? w.weights_path : "./model.ie.bin");
566 bin_fd = open(binp, O_RDONLY);
567 if (bin_fd >= 0) {
568 map_len = (size_t)w.bin_size_bytes;
569 void *p = mmap(NULL, map_len, PROT_READ, MAP_PRIVATE, bin_fd, 0);
570 if (p == MAP_FAILED) { map = NULL; map_len = 0; }
571 else { map = (uint8_t*)p; }
572 }
573 }
574
575 /* -------------------- KV round lifecycle (outside timing) ---------------- */
576 (void)ie_kv_begin_round();
577 uint64_t total_tokens_this_round = 0;
578
579 /* -------------------- measured window: ONLY gen + work-touch ------------- */
580 double t0 = now_sec();
581
582 uint32_t tokens_generated_total = 0;
583
584 for (int r = 0; r < (opt.rounds > 0 ? opt.rounds : 1); ++r) {
585 if (opt.aggregate && opt.prompts_file) {
586 FILE *pf = fopen(opt.prompts_file, "r");
587 if (pf) {
588 char line[8192];
589 while (fgets(line, sizeof(line), pf)) {
590 size_t n = strlen(line);
591 while (n && (line[n-1]=='\n'||line[n-1]=='\r')) line[--n]='\0';
592 if (!n) continue;
593
594 uint32_t n_here = 0;
595 st = ie_engine_generate(engine, line, cap, tokens, &n_here);
596 if (st != IE_OK) { fprintf(stderr,"error: ie_engine_generate (status=%d)\n",(int)st); break; }
597 tokens_generated_total += n_here;
598 total_tokens_this_round += (uint64_t)n_here;
599
600 if (map && map_len && bytes_per_token) {
601 size_t need = (size_t)n_here * bytes_per_token;
602 size_t pos = 0; volatile uint64_t acc = 0;
603 while (pos < need) { size_t off = (pos % map_len); acc += map[off]; pos += (stride_bytes ? stride_bytes : 1); }
604 if (verify_touch) { (void)acc; }
605 }
606 }
607 fclose(pf);
608 } else {
609 fprintf(stderr,"warn: cannot open prompts-file '%s'\n", opt.prompts_file);
610 }
611 } else {
612 const char *p = (opt.prompt ? opt.prompt : "bench");
613 uint32_t n_here = 0;
614 st = ie_engine_generate(engine, p, cap, tokens, &n_here);
615 if (st != IE_OK) {
616 fprintf(stderr, "error: ie_engine_generate failed (status=%d)\n", (int)st);
617 }
618 tokens_generated_total += n_here;
619 total_tokens_this_round += (uint64_t)n_here;
620
621 if (map && map_len && bytes_per_token) {
622 size_t need = (size_t)n_here * bytes_per_token;
623 size_t pos = 0; volatile uint64_t acc = 0;
624 while (pos < need) { size_t off = (pos % map_len); acc += map[off]; pos += (stride_bytes ? stride_bytes : 1); }
625 if (verify_touch) { (void)acc; }
626 }
627 }
628 }
629
630 double t1 = now_sec();
631
632 /* -------------------- AFTER timing: collect metrics ---------------------- */
633 uint64_t kv_hits_round = 0, kv_miss_round = 0;
634 ie_kv_finish_round(total_tokens_this_round, &kv_hits_round, &kv_miss_round);
635
636 ie_metrics_t m; memset(&m, 0, sizeof(m));
637 (void)ie_engine_metrics(engine, &m);
638 if (kv_hits_round || kv_miss_round) {
639 m.kv_hits = kv_hits_round;
640 m.kv_misses = kv_miss_round;
641 }
642
643 uint32_t rss_mib = ie_metrics_sample_rss_peak();
644 m.rss_peak_mb = rss_mib;
645
646 /* -------------------- print JSON & teardown ------------------------------ */
647 const uint32_t *tokens_to_print = (opt.aggregate || opt.rounds > 1) ? NULL : tokens;
648
649 print_json_result(tokens_generated_total,
650 tokens_to_print,
