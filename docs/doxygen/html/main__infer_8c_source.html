128 static int ascii_ieq(const char *a, const char *b) {
129  if (!a || !b) return 0;
130  while (*a && *b) {
131  unsigned char ca = (unsigned char)*a;
132  unsigned char cb = (unsigned char)*b;
133  unsigned char la = (ca >= 'A' && ca <= 'Z') ? (unsigned char)(ca + 32) : ca;
134  unsigned char lb = (cb >= 'A' && cb <= 'Z') ? (unsigned char)(cb + 32) : cb;
135  if (la != lb) return 0;

137  }
