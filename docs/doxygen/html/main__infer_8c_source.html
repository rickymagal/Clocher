118static int ascii_ieq(const char *a, const char *b) {
119 if (!a || !b) return 0;
120 while (*a && *b) {
121 unsigned char ca = (unsigned char)*a++;
122 unsigned char cb = (unsigned char)*b++;
123 if (ca >= 'A' && ca <= 'Z') ca = (unsigned char)(ca + 32);
124 if (cb >= 'A' && cb <= 'Z') cb = (unsigned char)(cb + 32);
125 if (ca != cb) return 0;
126 }
127 return (*a == '\0' && *b == '\0');
128}

353static int model_touch_open(model_mmap_touch_t *ctx, const char *path) {
354 if (!ctx) return 1;
355 memset(ctx, 0, sizeof(*ctx));
356 ctx->fd = -1;
357
358 if (!path || !*path) return 2;
359
360 int fd = open(path, O_RDONLY);
361 if (fd < 0) return 3;
362
363 struct stat st;
364 if (fstat(fd, &st) != 0) {
365 (void)close(fd);
366 return 4;
367 }
368 if (st.st_size <= 0) {
369 (void)close(fd);
370 return 5;
371 }
372
373 size_t sz = (size_t)st.st_size;
374 void *base = mmap(NULL, sz, PROT_READ, MAP_PRIVATE, fd, 0);
375 if (base == MAP_FAILED) {
376 (void)close(fd);
377 return 6;
378 }
379
380 ctx->fd = fd;
381 ctx->base = base;
382 ctx->size = sz;
383 ctx->cursor = 0;
384 return 0;
385}

391static void model_touch_close(model_mmap_touch_t *ctx) {
392 if (!ctx) return;
393 if (ctx->base && ctx->base != MAP_FAILED && ctx->size) (void)munmap(ctx->base, ctx->size);
394 ctx->base = NULL;
395 ctx->size = 0;
396 if (ctx->fd >= 0) (void)close(ctx->fd);
397 ctx->fd = -1;
398 ctx->cursor = 0;
399}

413static int model_touch_bytes(model_mmap_touch_t *ctx,
414 size_t bytes_to_touch,
415 size_t stride_bytes,
416 int verify_touch) {
417 if (!ctx || !ctx->base || ctx->size == 0) {
418 return 1;
419 }
420
421 if (bytes_to_touch == 0) {
422 return 0;
423 }
424
425 if (stride_bytes == 0) {
426 stride_bytes = 1;
427 }
428
429 const size_t size = ctx->size;
430
431 size_t n = bytes_to_touch;
432 if (n > size) {
433 n = size;
434 }
435
436 volatile const unsigned char *p = (volatile const unsigned char *)ctx->base;
437 volatile uint64_t acc = 0;
438
439 size_t start = ctx->cursor % size;
440
441 if (start + n <= size) {
442 size_t end = start + n;
443
444 for (size_t off = start; off < end; off += stride_bytes) {
445 acc ^= (uint64_t)p[off];
446 }
447
448 ctx->cursor = end % size;
449 } else {
450 size_t first = size - start;
451
452 for (size_t off = start; off < size; off += stride_bytes) {
453 acc ^= (uint64_t)p[off];
454 }
455
456 size_t rem = n - first;
457
458 for (size_t off = 0; off < rem; off += stride_bytes) {
459 acc ^= (uint64_t)p[off];
460 }
461
462 ctx->cursor = rem % size;
463 }
464
465 if (verify_touch) {
466 if (acc == 0x9e3779b97f4a7c15ULL) {
467 fprintf(stderr, "touch verify: improbable accumulator value (ignore)\n");
468 }
469 }
470
471 return 0;
472}

492typedef struct cudart_api {
493 void *handle;
494 int ok;
495 cudaFree_fn_t cudaFree;
496 cudaMalloc_fn_t cudaMalloc;
497 cudaMemcpy_fn_t cudaMemcpy;
498 cudaMemset_fn_t cudaMemset;
499 cudaDeviceSynchronize_fn_t cudaDeviceSynchronize;
500} cudart_api_t;

506static cudart_api_t *cudart_get_api(void) {
507 static cudart_api_t api;
508 static int inited = 0;
509 if (inited) return api.ok ? &api : NULL;
510 inited = 1;
511
512 memset(&api, 0, sizeof(api));
513
514 api.handle = dlopen("libcudart.so", RTLD_NOW | RTLD_LOCAL);
515 if (!api.handle) api.handle = dlopen("libcudart.so.12", RTLD_NOW | RTLD_LOCAL);
516 if (!api.handle) api.handle = dlopen("libcudart.so.11.0", RTLD_NOW | RTLD_LOCAL);
517
518 if (!api.handle) {
519 api.ok = 0;
520 return NULL;
521 }
522
523 { union { void *p; cudaFree_fn_t f; } u; u.p = dlsym(api.handle, "cudaFree"); api.cudaFree = u.f; }
524 { union { void *p; cudaMalloc_fn_t f; } u; u.p = dlsym(api.handle, "cudaMalloc"); api.cudaMalloc = u.f; }
525 { union { void *p; cudaMemcpy_fn_t f; } u; u.p = dlsym(api.handle, "cudaMemcpy"); api.cudaMemcpy = u.f; }
526 { union { void *p; cudaMemset_fn_t f; } u; u.p = dlsym(api.handle, "cudaMemset"); api.cudaMemset = u.f; }
527 { union { void *p; cudaDeviceSynchronize_fn_t f; } u; u.p = dlsym(api.handle, "cudaDeviceSynchronize"); api.cudaDeviceSynchronize = u.f; }
528
529 if (!api.cudaFree || !api.cudaMalloc || !api.cudaMemcpy || !api.cudaMemset || !api.cudaDeviceSynchronize) {
530 (void)dlclose(api.handle);
531 api.handle = NULL;
532 api.ok = 0;
533 return NULL;
534 }
535
536 api.ok = 1;
537 return &api;
538}

558static int cudart_touch_bytes(size_t bytes_per_token,
559 uint64_t tokens,
560 size_t stride_bytes,
561 int verify_touch) {
562 UNUSED(stride_bytes);
563
564 cudart_api_t *api = cudart_get_api();
565 if (!api) return -1;
566 if (bytes_per_token == 0 || tokens == 0) return 0;
567
568 void *d = NULL;
569 cudart_err_t e = api->cudaMalloc(&d, bytes_per_token);
570 if (e != 0 || !d) return 1;
571
572 for (uint64_t t = 0; t < tokens; ++t) {
573 int pattern = (int)(t & 0xFFu);
574 e = api->cudaMemset(d, pattern, bytes_per_token);
575 if (e != 0) {
576 (void)api->cudaFree(d);
577 return 2;
578 }
579 }
580
581 e = api->cudaDeviceSynchronize();
582 if (e != 0) {
583 (void)api->cudaFree(d);
584 return 3;
585 }
586
587 if (verify_touch) {
588 uint64_t tmp = 0;
589 size_t n = min_size(sizeof(tmp), bytes_per_token);
590 e = api->cudaMemcpy(&tmp, d, n, CUDA_MEMCPY_DEVICE_TO_HOST);
591 if (e != 0) {
592 (void)api->cudaFree(d);
593 return 4;
594 }
595 e = api->cudaDeviceSynchronize();
596 if (e != 0) {
597 (void)api->cudaFree(d);
598 return 5;
599 }
600 (void)tmp;
601 }
602
603 (void)api->cudaFree(d);
604 (void)api->cudaDeviceSynchronize();
605 return 0;
606}

1156
1157 print_json_result(tokens_generated_total,
1158 tokens_to_print,
1159 (t1 - t0),
1160 kv_hits_round,
1161 kv_miss_round,
1162 rss_mib);
1163
1164 if (mt_ok) model_touch_close(&mt);
1165
1166 free(tokens);
1167 ie_engine_destroy(engine);
1168 ie_weights_close(&w);
1169 return 0;
