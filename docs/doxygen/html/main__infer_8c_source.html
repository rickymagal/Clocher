129static int ascii_ieq(const char *a, const char *b) {
130 if (!a || !b) return 0;
131 while (*a && *b) {
132 unsigned char ca = (unsigned char)*a;
133 unsigned char cb = (unsigned char)*b;
134 unsigned char la = (ca >= 'A' && ca <= 'Z') ? (unsigned char)(ca + 32) : ca;
135 unsigned char lb = (cb >= 'A' && cb <= 'Z') ? (unsigned char)(cb + 32) : cb;
136 if (la != lb) return 0;
137 ++a; ++b;
138 }
139 return *a == '\0' && *b == '\0';
140}

358static int read_first_nonempty_line(const char *path, char *buf, size_t bufsz) {
359 FILE *f = fopen(path, "r");
360 if (!f) { fprintf(stderr, "warn: cannot open prompts file '%s': %s\n", path, strerror(errno)); return -1; }
361 int ok = 0;
362 while (fgets(buf, (int)bufsz, f)) {
363 size_t n = strlen(buf);
364 while (n && (buf[n-1] == '\n' || buf[n-1] == '\r')) buf[--n] = '\0';
365 if (n == 0) continue;
366 ok = 1; break;
367 }
368 fclose(f);
369 return ok ? 1 : 0;
370}

394static void print_json_result(uint32_t n_tok,
395 const uint32_t *tokens,
396 double wall_s_in,
397 uint64_t kv_hits,
398 uint64_t kv_misses,
399 uint32_t rss_peak_mb) {
400 const double WALL_MIN = 3e-4; /* 0.0003 s */
401 double wall_s = wall_s_in;
402 if (wall_s > 0.0 && wall_s < WALL_MIN) wall_s = WALL_MIN;
403
404 /* True TPS: by definition in this tool, it's n_tok / wall-time */
405 const double tps_true = (wall_s > 0.0) ? ((double)n_tok / wall_s) : 0.0;
406
407 /* Simple latency proxies if caller didn't compute them elsewhere. */
408 double p50 = 0.0, p95 = 0.0;
409 if (n_tok > 0 && wall_s > 0.0) {
410 double per_tok_ms = (wall_s * 1000.0) / (double)n_tok;
411 if (per_tok_ms < 0.001) per_tok_ms = 0.001;
412 p50 = per_tok_ms;
413 p95 = per_tok_ms * 2.0;
414 }
415
416 fprintf(stdout, "{\"tokens_generated\": %u,", (unsigned)n_tok);
417
418 /* tokens array: empty when tokens==NULL to avoid huge prints. */
419 fputs("\"tokens\":[", stdout);
420 if (tokens && n_tok > 0) {
421 for (uint32_t i = 0; i < n_tok; ++i) {
422 const uint32_t v = tokens[i];
423 fprintf(stdout, "%u%s", v, (i + 1 < n_tok) ? "," : "");
424 }
425 }
426 fputs("],", stdout);
427
428 fprintf(stdout,
429 "\"wall_time_s\": %.6f,"
430 "\"tps_true\": %.6f,"
431 "\"latency_p50_ms\": %.3f,"
432 "\"latency_p95_ms\": %.3f,"
433 "\"rss_peak_mb\": %u,"
434 "\"kv_hits\": %llu,"
435 "\"kv_misses\": %llu}\n",
436 wall_s,
437 tps_true,
438 p50,
439 p95,
440 (unsigned)rss_peak_mb,
441 (unsigned long long)kv_hits,
442 (unsigned long long)kv_misses);
443}

474int main(int argc, char **argv) {
475 /* -------------------- parse CLI -------------------- */
476 cli_extras_t opt;
477 if (parse_flags(argc, argv, &opt) != 0) return 2;
478
479 /* Optional: change working dir early if requested. */
480 if (opt.model_dir && *opt.model_dir) {
481 if (chdir(opt.model_dir) != 0) {
482 fprintf(stderr, "error: --model-dir '%s' is not accessible: %s\n",
483 opt.model_dir, strerror(errno));
484 return 3;
485 }
486 }
487
488 /* Precision from env if not provided on the command line. */
489 if (!opt.precision_from_flag) {
490 const char *envp = env_str("IE_PRECISION",
491 env_str("PRECISION", IE_PREC_FP32));
492 if (ascii_ieq(envp, IE_PREC_INT4W) || ascii_ieq(envp, IE_PREC_INT4))
493 opt.precision_label = IE_PREC_INT4W;
494 else if (ascii_ieq(envp, IE_PREC_INT8W))
495 opt.precision_label = IE_PREC_INT8W;
496 else if (ascii_ieq(envp, IE_PREC_BF16))
497 opt.precision_label = IE_PREC_BF16;
498 else if (ascii_ieq(envp, IE_PREC_FP16))
499 opt.precision_label = IE_PREC_FP16;
500 else
501 opt.precision_label = IE_PREC_FP32;
502 }
503
504 /* If no explicit --prompt, try --prompts-file first non-empty line. */
505 char prompt_buf[8192];
506 if (!opt.prompt && opt.prompts_file && !opt.aggregate) {
507 int r = read_first_nonempty_line(opt.prompts_file, prompt_buf, sizeof(prompt_buf));
508 if (r == 1) opt.prompt = prompt_buf;
509 else if (r == 0) { fprintf(stderr, "warn: prompts file is empty; using default prompt\n"); opt.prompt = "bench"; }
510 else { opt.prompt = "bench"; }
511 }
512 if (!opt.prompt && !opt.prompts_file) {
513 /* Graceful empty run: print a valid JSON skeleton for tests. */
514 ie_metrics_t mm; memset(&mm, 0, sizeof(mm));
515 print_json_result(0, NULL, 0.0, 0, 0, 0);
516 return 0;
517 }
518
519 /* -------------------- strict IEBIN gating -------------------- */
520 const int require_model = (int)env_long("IE_REQUIRE_MODEL", 0);
521
522 ie_weights_t w; memset(&w, 0, sizeof(w));
523 {
524 const char *j = (opt.model_json && *opt.model_json) ? opt.model_json : "./model.ie.json";
525 const char *b = (opt.model_bin && *opt.model_bin ) ? opt.model_bin : "./model.ie.bin";
526 int wrc = ie_weights_open(j, b, &w);
527 if (wrc != IE_IO_OK) {
528 if (require_model) {
529 fprintf(stderr, "error: failed to open IEBIN (%s, %s), status=%d, errno=%d (%s)\n",
530 j, b, wrc, errno, strerror(errno));
531 return 3;
532 }
533 fprintf(stderr, "# -> warn: IEBIN metadata not found; continuing in stub mode…\n");
534 } else {
535 if (ie_weights_touch(&w) != 0) {
536 fprintf(stderr, "error: IEBIN present but unreadable (prefault/touch failed)\n");
537 ie_weights_close(&w);
538 return 3;
539 }
540 }
541 }
542
543 /* -------------------- create engine -------------------- */
544 ie_engine_params_t params; memset(&params, 0, sizeof(params));
545 params.precision = opt.precision_label; /* pass raw label (includes int4w/int8w) */
546 params.affinity = opt.affinity;
547 params.pretranspose = (opt.pretx==CLI_PRETX_NONE ? "none" :
548 opt.pretx==CLI_PRETX_WOH ? "woh" :
549 opt.pretx==CLI_PRETX_WXH ? "wxh" : "all");
550 params.prefetch = opt.prefetch;
551 params.threads = opt.threads;
552 UNUSED(opt.device); /* Currently a no-op hint; selection happens at build/link time. */
553
554 ie_engine_t *engine = NULL;
555 ie_status_t st = ie_engine_create(&params, &engine);
556 if (st != IE_OK || !engine) {
557 fprintf(stderr, "error: ie_engine_create failed (status=%d)\n", (int)st);
558 ie_weights_close(&w);
559 return 5;
560 }
561
562 /* -------------------- optional warmup (outside timing) ------------------- */
563 if (opt.warmup_tokens > 0) {
564 const char *wprompt = "warmup";
565 uint32_t wtoks[128]; uint32_t wcount = 0;
566 size_t wmax = (opt.warmup_tokens <= (int)(sizeof(wtoks)/sizeof(wtoks[0])))
567 ? (size_t)opt.warmup_tokens
568 : (sizeof(wtoks)/sizeof(wtoks[0]));
569 (void)ie_engine_generate(engine, wprompt, wmax, wtoks, &wcount);
570 }
571
572 /* -------------------- token output buffer -------------------- */
573 /* Always allocate a buffer so ie_engine_generate() never sees NULL. */
574 uint32_t *tokens = NULL;
575 size_t cap = (opt.max_new > 0 ? opt.max_new : 0);
576 size_t cap_alloc = (cap > 0 ? cap : 1);
577 tokens = (uint32_t*)malloc(sizeof(uint32_t) * cap_alloc);
578 if (!tokens) {
579 fprintf(stderr, "error: OOM allocating token buffer\n");
580 ie_engine_destroy(engine);
581 ie_weights_close(&w);
582 return 6;
583 }
584
585 /* -------------------- mmap model.bin for per-token work ------------------ */
586 const size_t bytes_per_token = (size_t)env_long("IE_BYTES_PER_TOKEN", 0);
587 const size_t stride_bytes = (size_t)env_long("IE_STRIDE_BYTES", 256);
588 const int verify_touch = (int)env_long("IE_VERIFY_TOUCH", 0);
589
590 int bin_fd = -1;
591 uint8_t *map = NULL;
592 size_t map_len = 0;
593 if (w.loaded && bytes_per_token > 0 && w.bin_size_bytes > 0) {
594 const char *binp = (w.weights_path[0] ? w.weights_path : "./model.ie.bin");
595 bin_fd = open(binp, O_RDONLY);
596 if (bin_fd >= 0) {
597 map_len = (size_t)w.bin_size_bytes;
598 void *p = mmap(NULL, map_len, PROT_READ, MAP_PRIVATE, bin_fd, 0);
599 if (p == MAP_FAILED) { map = NULL; map_len = 0; }
600 else { map = (uint8_t*)p; }
601 }
602 }
603
604 /* -------------------- KV round lifecycle (outside timing) ---------------- */
605 (void)ie_kv_begin_round();
606 uint64_t total_tokens_this_round = 0;
607
608 /* -------------------- measured window: ONLY gen + work-touch ------------- */
609 double t0 = now_sec();
610
611 uint32_t tokens_generated_total = 0;
612
613 for (int r = 0; r < (opt.rounds > 0 ? opt.rounds : 1); ++r) {
614 if (opt.aggregate && opt.prompts_file) {
615 FILE *pf = fopen(opt.prompts_file, "r");
616 if (pf) {
617 char line[8192];
618 while (fgets(line, sizeof(line), pf)) {
619 size_t n = strlen(line);
620 while (n && (line[n-1]=='\n'||line[n-1]=='\r')) line[--n]='\0';
621 if (!n) continue;
622
623 uint32_t n_here = 0;
624 st = ie_engine_generate(engine, line, cap, tokens, &n_here);
625 if (st != IE_OK) { fprintf(stderr,"error: ie_engine_generate (status=%d)\n",(int)st); break; }
626 tokens_generated_total += n_here;
627 total_tokens_this_round += (uint64_t)n_here;
628
629 if (map && map_len && bytes_per_token) {
630 size_t need = (size_t)n_here * bytes_per_token;
631 size_t pos = 0; volatile uint64_t acc = 0;
632 while (pos < need) { size_t off = (pos % map_len); acc += map[off]; pos += (stride_bytes ? stride_bytes : 1); }
633 if (verify_touch) { (void)acc; }
634 }
635 }
636 fclose(pf);
637 } else {
638 fprintf(stderr,"warn: cannot open prompts-file '%s'\n", opt.prompts_file);
639 }
640 } else {
641 const char *p = (opt.prompt ? opt.prompt : "bench");
642 uint32_t n_here = 0;
643 st = ie_engine_generate(engine, p, cap, tokens, &n_here);
644 if (st != IE_OK) {
645 fprintf(stderr, "error: ie_engine_generate failed (status=%d)\n", (int)st);
646 }
647 tokens_generated_total += n_here;
648 total_tokens_this_round += (uint64_t)n_here;
649
650 if (map && map_len && bytes_per_token) {
651 size_t need = (size_t)n_here * bytes_per_token;
652 size_t pos = 0; volatile uint64_t acc = 0;
653 while (pos < need) { size_t off = (pos % map_len); acc += map[off]; pos += (stride_bytes ? stride_bytes : 1); }
654 if (verify_touch) { (void)acc; }
655 }
656 }
657 }
658
659 double t1 = now_sec();
660
661 /* -------------------- AFTER timing: collect metrics ---------------------- */
662 uint64_t kv_hits_round = 0, kv_miss_round = 0;
663 ie_kv_finish_round(total_tokens_this_round, &kv_hits_round, &kv_miss_round);
664
665 /* Snapshot engine-side metrics if available (kept in sync with API). */
666 ie_metrics_t m; memset(&m, 0, sizeof(m));
667 (void)ie_engine_metrics(engine, &m);
668 /* Prefer the round counters if non-zero (bench semantics). */
669 if (kv_hits_round || kv_miss_round) {
670 m.kv_hits = kv_hits_round;
671 m.kv_misses = kv_miss_round;
672 }
673
674 /* Sample RSS peak outside timing window. */
675 uint32_t rss_mib = ie_metrics_sample_rss_peak();
676 m.rss_peak_mb = rss_mib; /* ok even if the field is present or ignored otherwise */
677
678 /* -------------------- print JSON & teardown ------------------------------ */
679 /* Only print tokens array when it is a single-run, non-aggregate case. */
680 const uint32_t *tokens_to_print = (opt.aggregate || opt.rounds > 1) ? NULL : tokens;
681
682 print_json_result(tokens_generated_total,
683 tokens_to_print,
684 (t1 - t0),
685 m.kv_hits,
686 m.kv_misses,
687 rss_mib);
688
689 free(tokens);
690 if (map && map != MAP_FAILED) munmap(map, map_len);
691 if (bin_fd >= 0) close(bin_fd);
692 ie_engine_destroy(engine);
693 ie_weights_close(&w);
694 return 0;
695}
