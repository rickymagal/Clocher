141static float ie_q4_0_row_dot_f32(const uint8_t *blocks,
142 const uint8_t *scales,
143 size_t scale_bytes,
144 const float *x,
145 size_t cols) {
146 const size_t n_blocks = cols / 32;
147 float acc = 0.0f;
148 const int dbg_nan = ie_q4_debug_nan_enabled_();
149
150 for (size_t b = 0; b < n_blocks; ++b) {
151 const uint8_t *q = blocks + b * 16;
152 const float d = ie_q4_load_scale_f32(scales + b * scale_bytes, scale_bytes);
153
154 for (size_t j = 0; j < 32; ++j) {
155 const uint8_t byte = q[j >> 1];
156 const uint8_t nibble = (j & 1) ? (uint8_t)(byte >> 4) : (uint8_t)(byte & 0x0F);
157 const int32_t w = (nibble >= 8u) ? ((int32_t)nibble - 16) : (int32_t)nibble;
158 acc += (d * (float)w) * x[b * 32 + j];
159 if (dbg_nan && !isfinite(acc)) {
160 fprintf(stderr,
161 "[q4_nan] block=%zu j=%zu scale=%g w=%d x=%g acc=%g\n",
162 b, j, (double)d, (int)w, (double)x[b * 32 + j], (double)acc);
163 return acc;
164 }
165 }
166 }
167
168 return acc;
169}

184static int ie_gemv_q4_0_f32_generic_impl(const uint8_t *W_blocks,
185 const uint8_t *W_scales,
186 size_t scale_bytes,
187 const float *x,
188 float *y,
189 size_t rows,
190 size_t cols,
191 const uint16_t *bias_bf16) {
192 if (!W_blocks || !W_scales || !x || !y) {
193 return 1;
194 }
195 if (cols == 0 || (cols % 32) != 0) {
196 return 2;
197 }
198 if (!(scale_bytes == 1 || scale_bytes == 2)) {
199 return 3;
200 }
201
202 const size_t n_blocks = cols / 32;
203 const size_t row_block_bytes = n_blocks * 16;
204 const size_t row_scale_bytes = n_blocks * scale_bytes;
205
206 for (size_t r = 0; r < rows; ++r) {
207 const uint8_t *row_blocks = W_blocks + r * row_block_bytes;
208 const uint8_t *row_scales = W_scales + r * row_scale_bytes;
209
210 float v = ie_q4_0_row_dot_f32(row_blocks, row_scales, scale_bytes, x, cols);
211 if (bias_bf16) {
212 v += ie_bf16_to_f32_scalar(bias_bf16[r]);
213 }
214 y[r] = v;
215 }
216
217 return 0;
218}
