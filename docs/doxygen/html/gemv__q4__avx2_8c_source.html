133static IE_TARGET_AVX2_FMA float ie_q4_0_block_dot_f32_avx2(const uint8_t *q,
134 const float *x,
135 float scale) {
136 const __m128i v = _mm_loadu_si128((const __m128i *)q);
137
138 const __m128i mask = _mm_set1_epi8(0x0F);
139 const __m128i lo = _mm_and_si128(v, mask);
140 const __m128i hi = _mm_and_si128(_mm_srli_epi16(v, 4), mask);
141
142 const __m128i lut =
143 _mm_setr_epi8((char)0, (char)1, (char)2, (char)3, (char)4, (char)5, (char)6, (char)7,
144 (char)-8, (char)-7, (char)-6, (char)-5, (char)-4, (char)-3, (char)-2, (char)-1);
145
146 const __m128i w_even = _mm_shuffle_epi8(lut, lo);
147 const __m128i w_odd = _mm_shuffle_epi8(lut, hi);
148
149 const __m128i w0_15 = _mm_unpacklo_epi8(w_even, w_odd);
150 const __m128i w16_31 = _mm_unpackhi_epi8(w_even, w_odd);
151
152 const __m256 scale_v = _mm256_set1_ps(scale);
153
154 __m256 acc = _mm256_setzero_ps();
155
156 /* 0..7 */
157 {
158 const __m128i w8 = _mm_cvtepi8_epi16(w0_15);
159 const __m256i wi32 = _mm256_cvtepi16_epi32(w8);
160 const __m256 wf = _mm256_cvtepi32_ps(wi32);
161 const __m256 xs = _mm256_loadu_ps(x + 0);
162 acc = _mm256_fmadd_ps(_mm256_mul_ps(wf, scale_v), xs, acc);
163 }
164
165 /* 8..15 */

167 const __m128i w8 = _mm_cvtepi8_epi16(_mm_srli_si128(w0_15, 8));

169 const __m256 wf = _mm256_cvtepi32_ps(wi32);
170 const __m256 xs = _mm256_loadu_ps(x + 8);

172 }

208IE_TARGET_AVX2_FMA int ie_gemv_q4_0_f32_avx2_impl(const uint8_t *W_blocks,
209 const uint8_t *W_scales,
210 size_t scale_bytes,
211 const float *x,
212 float *y,
213 size_t rows,
214 size_t cols,
215 const uint16_t *bias_bf16) {
216 if (!W_blocks || !W_scales || !x || !y) {
217 return 1;
218 }
219 if (cols == 0 || (cols % 32) != 0) {
220 return 2;
221 }
222 if (!(scale_bytes == 1 || scale_bytes == 2)) {
223 return 3;
224 }
225
226 const size_t n_blocks = cols / 32;
227 const size_t row_block_bytes = n_blocks * 16;
228 const size_t row_scale_bytes = n_blocks * scale_bytes;
229
230 for (size_t r = 0; r < rows; ++r) {
231 const uint8_t *row_blocks = W_blocks + r * row_block_bytes;
232 const uint8_t *row_scales = W_scales + r * row_scale_bytes;
233
234 float acc = 0.0f;
235 for (size_t b = 0; b < n_blocks; ++b) {
236 const float d = ie_q4_load_scale_f32(row_scales + b * scale_bytes, scale_bytes);
237 acc += ie_q4_0_block_dot_f32_avx2(row_blocks + b * 16, x + b * 32, d);
238 }
239
240 if (bias_bf16) {
241 acc += ie_bf16_to_f32_scalar(bias_bf16[r]);
242 }
243 y[r] = acc;
244 }
245
246 return 0;
247}
