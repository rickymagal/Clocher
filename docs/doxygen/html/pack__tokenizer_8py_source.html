5"""
6Packs a HuggingFace-style tokenizer.json into the engine's packed tokenizer
7format (IETOK1) expected by engine/src/io/tokenizer_gptoss.c.
8
9This packer writes:
10 magic: 6 bytes "IETOK1"
11 version: u16le 1
12 vocab_size: u32le
13 merges_cnt: u32le
14 off_vocab: u32le
15 off_merges: u32le
16 off_special:u32le
17 reserved: u32le
18
19Then at off_vocab:
20 repeated vocab_size times:
21 u32le len
22 len bytes (utf-8 token text)
23
24Then at off_merges:
25 repeated merges_cnt times:
26 u32le len_a
27 len_a bytes
28 u32le len_b
29 len_b bytes
30
31No special-tokens section is currently written; off_special == end of file.
32"""

82def _normalize_merges(merges: List) -> List[Tuple[str, str]]:
83 out: List[Tuple[str, str]] = []
84 for m in merges:
85 if isinstance(m, list) and len(m) == 2 and all(isinstance(x, str) for x in m):
86 out.append((m[0], m[1]))
87 continue
88 if isinstance(m, str):
89 sp = m.split()
90 if len(sp) == 2:
91 out.append((sp[0], sp[1]))
92 continue
93 return out

96def _build_id_to_tok(vocab: Dict[str, int], added: List[Tuple[int, str]]) -> Tuple[List[str], int, int]:
97 id_to_tok: Dict[int, str] = {}
98 max_id = -1
99
100 for tok, tid in vocab.items():
101 if not isinstance(tok, str):
102 continue
103 if not isinstance(tid, int) or tid < 0:
104 continue
105 id_to_tok[tid] = tok
106 if tid > max_id:
107 max_id = tid
108
109 added_count = 0
110 for tid, content in added:
111 if tid < 0:
112 continue
113 id_to_tok[tid] = content
114 if tid > max_id:
115 max_id = tid
116 added_count += 1
117
118 if max_id < 0:
119 _die("error: could not find any token ids in tokenizer.json")
120
121 vocab_size = max_id + 1
122 table: List[str] = ["" for _ in range(vocab_size)]
123 missing: List[int] = []
124 for i in range(vocab_size):
125 s = id_to_tok.get(i)
126 if s is None:
127 missing.append(i)
128 else:
129 table[i] = s
130
131 if missing:
132 preview = ", ".join(str(x) for x in missing[:20])
133 _die(f"error: tokenizer has gaps in ids; missing {len(missing)} ids (first: {preview})")
134
135 base_vocab_count = len(vocab)
136 return table, base_vocab_count, added_count

139def main() -> None:
140 ap = argparse.ArgumentParser()
141 ap.add_argument("--model-dir", required=True, help="Model directory containing tokenizer.json")
142 args = ap.parse_args()
143 md = args.model_dir
144
145 tok_json = os.path.join(md, "tokenizer.json")
146 if not os.path.exists(tok_json):
147 _die(f"error: tokenizer.json not found at {tok_json}")
148
149 j = _load_json(tok_json)
150 vocab, merges_raw = _extract_vocab_and_merges(j)
151 added = _extract_added_tokens(j)
152
153 id_table, base_vocab_count, added_count = _build_id_to_tok(vocab, added)
154 merges = _normalize_merges(merges_raw)
155
156 vocab_size = len(id_table)
157 merges_count = len(merges)
158
159 header_len = 6 + 2 + (6 * 4)
160
161 blob = bytearray()
162 blob += MAGIC
163 blob += struct.pack("<H", VERSION_U16)
164
165 # Placeholder u32 fields
166 blob += struct.pack("<IIIIII", 0, 0, 0, 0, 0, 0)
167
168 if len(blob) != header_len:
169 _die("error: internal header size mismatch")
170
171 off_vocab = header_len
172
173 # Vocab section
174 for s in id_table:
175 b = s.encode("utf-8")
176 blob += struct.pack("<I", len(b))
177 blob += b
178
179 off_merges = len(blob)
180
181 # Merges section: write (a, b) as two LP-strings per entry
182 for a, b in merges:
183 ab = a.encode("utf-8")
184 bb = b.encode("utf-8")
185 blob += struct.pack("<I", len(ab))
186 blob += ab
187 blob += struct.pack("<I", len(bb))
188 blob += bb
189
190 off_special = len(blob)
191
192 # Fill header fields
193 struct.pack_into(
194 "<IIIIII",
195 blob,
196 6 + 2,
197 vocab_size,
198 merges_count,
199 off_vocab,
200 off_merges,
201 off_special,
202 0,
203 )
204
205 out_path = os.path.join(md, "tokenizer.ie.bin")
206 with open(out_path, "wb") as f:
207 f.write(blob)
208
209 print(f"ok: wrote {out_path}")
210 print(f" vocab_size: {vocab_size} (base_vocab: {base_vocab_count}, added_tokens: {added_count})")
211 print(f" merges: {merges_count}")
212 print(f" offsets: vocab={off_vocab} merges={off_merges} special={off_special}")
