52static void join_path(char* out, size_t outsz, const char* dir, const char* leaf) {
53 if (!out || outsz == 0) return;
54 out[0] = '\0';
55 if (!leaf || !*leaf) return;
56 if (!dir || !*dir) {
57 snprintf(out, outsz, "%s", leaf);
58 return;
59 }
60 size_t n = strlen(dir);
61 if (n && dir[n - 1] == '/') snprintf(out, outsz, "%s%s", dir, leaf);
62 else snprintf(out, outsz, "%s/%s", dir, leaf);
63}

65static int open_and_touch(const char* json_path, const char* bin_path) {
66 ie_weights_t w;
67 memset(&w, 0, sizeof(w));
68
69 int rc = ie_weights_open(json_path, bin_path, &w);
70 if (rc != IE_IO_OK) {
71 IE_LOG_ERROR("ie_weights_open failed (json='%s', bin='%s') rc=%d errno=%d (%s)",
72 json_path, bin_path, rc, errno, strerror(errno));
73 return 1;
74 }
75
76 if (ie_weights_touch(&w) != 0) {
77 IE_LOG_ERROR("ie_weights_touch failed (json='%s', bin='%s') errno=%d (%s)",
78 json_path, bin_path, errno, strerror(errno));
79 ie_weights_close(&w);
80 return 2;
81 }
82
83 ie_weights_close(&w);
84 return 0;
85}
