61static void ie_set_affinity(const char *policy, unsigned tidx, unsigned nth, int use_affinity) {
62#ifdef __linux__
63 if (!use_affinity || !policy) return;
64
65 cpu_set_t set;
66 CPU_ZERO(&set);
67
68 long ncpu = sysconf(_SC_NPROCESSORS_ONLN);
69 if (ncpu <= 0) return;
70
71 unsigned cpu = tidx % (unsigned)ncpu; /* default "auto" */
72
73 if (strcmp(policy, "compact") == 0) {
74 cpu = tidx % (unsigned)ncpu;
75 } else if (strcmp(policy, "scatter") == 0) {
76 unsigned stride = (unsigned)(ncpu / (nth ? nth : 1)) + 1u;
77 cpu = (unsigned)((tidx * stride) % (unsigned)ncpu);
78 }
79
80 CPU_SET(cpu, &set);
81 (void)pthread_setaffinity_np(pthread_self(), sizeof(set), &set);
82#else
83 (void)policy; (void)tidx; (void)nth; (void)use_affinity;
84#endif
85}

91static void ie_tp_worker_run(struct ie_threadpool *tp) {
92 for (;;) {
93 pthread_mutex_lock(&tp->mu);
94 unsigned start = tp->next;
95 if (start >= tp->n) {
96 pthread_mutex_unlock(&tp->mu);
97 break;
98 }
99 unsigned end = start + tp->grainsize;
100 if (end > tp->n) end = tp->n;
101 tp->next = end;
102
103 ie_tp_for_body_fn body = tp->body;
104 void *ctx = tp->ctx;
105 pthread_mutex_unlock(&tp->mu);
106
107 if (body) body(ctx, start, end);
108 }
109}

111static void* ie_worker_main(void *arg) {
112 ie_worker_arg_t *wa = (ie_worker_arg_t*)arg;
113 struct ie_threadpool *tp = wa->tp;
114 unsigned tidx = wa->tidx;
115 free(wa);
116
117 ie_set_affinity(tp->affinity, tidx, tp->nth_created ? tp->nth_created : tp->nth, tp->use_affinity);
118
119 unsigned epoch_seen = 0;
120 for (;;) {
121 pthread_mutex_lock(&tp->mu);
122 while (!tp->shutdown && epoch_seen == tp->epoch) {
123 pthread_cond_wait(&tp->cv, &tp->mu);
124 }
125 if (tp->shutdown) {
126 pthread_mutex_unlock(&tp->mu);
127 return NULL;
128 }
129 epoch_seen = tp->epoch;
130 pthread_mutex_unlock(&tp->mu);
131
132 ie_tp_worker_run(tp);
133
134 pthread_mutex_lock(&tp->mu);
135 if (tp->active > 0) {
136 tp->active--;
137 if (tp->active == 0) pthread_cond_signal(&tp->done);
138 }
139 pthread_mutex_unlock(&tp->mu);
140 }

147ie_threadpool_t* ie_tp_create(unsigned nth, const char *affinity) {
148 if (nth < 1) nth = 1;
149
150 struct ie_threadpool *tp = (struct ie_threadpool*)calloc(1, sizeof(*tp));
151 if (!tp) return NULL;
152
153 tp->nth = nth;
154 tp->affinity = (affinity && *affinity) ? affinity : "auto";
155
156#ifdef __linux__
157 const char *env = getenv("IE_TP_USE_AFFINITY");
158 tp->use_affinity = (env && strcmp(env, "1") == 0) ? 1 : 0;
159#else
160 tp->use_affinity = 0;
161#endif
162
163 tp->threads = (pthread_t*)calloc(nth, sizeof(pthread_t));
164 if (!tp->threads) { free(tp); return NULL; }
165
166 pthread_mutex_init(&tp->mu, NULL);
167 pthread_cond_init(&tp->cv, NULL);
168 pthread_cond_init(&tp->done, NULL);
169
170 for (unsigned i = 0; i < nth; ++i) {
171 ie_worker_arg_t *wa = (ie_worker_arg_t*)malloc(sizeof(*wa));
172 if (!wa) break;
173 wa->tp = tp;
174 wa->tidx = i;
175 if (pthread_create(&tp->threads[tp->nth_created], NULL, ie_worker_main, wa) != 0) {
176 free(wa);
177 break;
178 }
179 tp->nth_created++;
180 }
181
182 return (ie_threadpool_t*)tp;
183}

205void ie_tp_parallel_for(ie_threadpool_t *h,
206 unsigned n,
207 unsigned grainsize,
208 ie_tp_for_body_fn body_fn,
209 void *ctx) {
210 if (n == 0 || !body_fn) return;
211 if (!h) {
212 body_fn(ctx, 0u, n);
213 return;
214 }
215
216 struct ie_threadpool *tp = (struct ie_threadpool*)h;
217 if (tp->nth_created <= 1) {
218 body_fn(ctx, 0u, n);
219 return;
220 }
221
222 unsigned min_chunk = (grainsize > 0 ? grainsize
223 : (n + tp->nth_created - 1u) / tp->nth_created);
224 if (min_chunk == 0) min_chunk = 1;
225
226 pthread_mutex_lock(&tp->mu);
227 tp->n = n;
228 tp->grainsize = min_chunk;
229 tp->body = body_fn;
230 tp->ctx = ctx;
231 tp->next = 0;
232 tp->active = tp->nth_created + 1u; /* include the calling thread */
233 tp->epoch++;
234 pthread_cond_broadcast(&tp->cv);
235 pthread_mutex_unlock(&tp->mu);
236
237 ie_tp_worker_run(tp);
238
239 pthread_mutex_lock(&tp->mu);
240 if (tp->active > 0) {
241 tp->active--;
242 if (tp->active == 0) pthread_cond_signal(&tp->done);
243 }
244 while (tp->active != 0) {
245 pthread_cond_wait(&tp->done, &tp->mu);
246 }
247 pthread_mutex_unlock(&tp->mu);
248}
