206ie_sparse_status_t ie_block_sparse_load(const char *path,
207 ie_block_sparse_matrix_t *out) {
208 if (!path || !out) {
209 return IE_SPARSE_ERR_ARGS;
210 }
211
212 ie_block_sparse_init(out);
213
214 FILE *f = fopen(path, "rb");
215 if (!f) {
216 return IE_SPARSE_ERR_OPEN;
217 }
218
219 ie_block_sparse_header_disk_t hdr;
220 const ie_sparse_status_t st_hdr = read_fully(&hdr, sizeof(hdr), f);
221 if (st_hdr != IE_SPARSE_OK) {
222 fclose(f);
223 return st_hdr;
224 }
225
226 /* Validate magic and version. */
227 if (memcmp(hdr.magic, "IEBS", 4) != 0 || hdr.version != 1u) {
228 fclose(f);
229 return IE_SPARSE_ERR_MAGIC;
230 }
231
232 /* Basic sanity checks on dimensions. */
233 if (hdr.rows == 0u || hdr.cols == 0u ||
234 hdr.block_rows == 0u || hdr.block_cols == 0u) {
235 fclose(f);
236 return IE_SPARSE_ERR_FORMAT;
237 }
238
239 const uint32_t n_block_rows =
240 ie_block_sparse_n_block_rows(hdr.rows, hdr.block_rows);
241 const uint32_t n_block_cols =
242 ie_block_sparse_n_block_cols(hdr.cols, hdr.block_cols);
243 const uint32_t nnzb = hdr.nnzb;
244
245 /* Guard against overflow in total values. */
246 const uint64_t block_elems =
247 (uint64_t)hdr.block_rows * (uint64_t)hdr.block_cols;
248 const uint64_t total_vals = block_elems * (uint64_t)nnzb;
249
250 if (n_block_rows == 0u || n_block_cols == 0u || block_elems == 0u) {
251 fclose(f);
252 return IE_SPARSE_ERR_FORMAT;
253 }
254
255 if (total_vals > (uint64_t)SIZE_MAX / sizeof(float)) {
256 fclose(f);
257 return IE_SPARSE_ERR_FORMAT;
258 }
259
260 /* Allocate arrays. */
261 uint32_t *row_ptr = (uint32_t *)malloc((size_t)(n_block_rows + 1u) *
262 sizeof(uint32_t));
263 uint32_t *col_idx = (uint32_t *)malloc((size_t)nnzb * sizeof(uint32_t));
264 float *values = NULL;
265
266 if (!row_ptr || (!col_idx && nnzb > 0u)) {
267 free(row_ptr);
268 free(col_idx);
269 fclose(f);
270 return IE_SPARSE_ERR_NOMEM;
271 }
272
273 if (nnzb > 0u) {
274 values = (float *)malloc((size_t)total_vals * sizeof(float));
275 if (!values) {
276 free(row_ptr);
277 free(col_idx);
278 fclose(f);
279 return IE_SPARSE_ERR_NOMEM;
280 }
281 }
282
283 /* Read arrays from file. */
284 ie_sparse_status_t st = IE_SPARSE_OK;
285
286 st = read_fully(row_ptr,
287 (size_t)(n_block_rows + 1u) * sizeof(uint32_t),
288 f);
289 if (st != IE_SPARSE_OK) {
290 goto fail;
291 }
292
293 if (nnzb > 0u) {
294 st = read_fully(col_idx, (size_t)nnzb * sizeof(uint32_t), f);
295 if (st != IE_SPARSE_OK) {
296 goto fail;
297 }
298
299 st = read_fully(values, (size_t)total_vals * sizeof(float), f);
300 if (st != IE_SPARSE_OK) {
301 goto fail;
302 }
303 }
304
305 fclose(f);
306
307 /* Commit to the output descriptor. */
308 out->rows = hdr.rows;
309 out->cols = hdr.cols;
310 out->block_rows = hdr.block_rows;
311 out->block_cols = hdr.block_cols;
312 out->n_block_rows = n_block_rows;
313 out->n_block_cols = n_block_cols;
314 out->nnzb = nnzb;
315 out->row_ptr = row_ptr;
316 out->col_idx = col_idx;
317 out->values = values;
318
319 return IE_SPARSE_OK;
320
321fail:
322 free(row_ptr);
323 free(col_idx);
324 free(values);
325 fclose(f);
326 ie_block_sparse_init(out);
327 return st;
328}
