268static int js_parse_string(const char **p, char **out) {
269 const char *s = js_skip_ws(*p);
270 if (*s != '"') return 0;
271 s++;
272
273 size_t cap = 64;
274 size_t len = 0;
275 char *buf = (char *)malloc(cap);
276 if (!buf) return 0;
277
278 while (*s && *s != '"') {
279 char c = *s++;
280 if (c == '\\') {
281 char e = *s++;
282 if (e == '"' || e == '\\' || e == '/') c = e;
283 else if (e == 'b') c = '\b';
284 else if (e == 'f') c = '\f';
285 else if (e == 'n') c = '\n';
286 else if (e == 'r') c = '\r';
287 else if (e == 't') c = '\t';
288 else {
289 free(buf);
290 return 0;
291 }
292 }
293 if (len + 1 >= cap) {
294 cap *= 2;
295 char *nb = (char *)realloc(buf, cap);
296 if (!nb) {
297 free(buf);
298 return 0;
299 }
300 buf = nb;
301 }
302 buf[len++] = c;
303 }
304 if (*s != '"') {
305 free(buf);
306 return 0;
307 }
308 s++;
309
310 buf[len] = '\0';
311 *out = buf;
312 *p = s;
313 return 1;
314}

316static int js_parse_i64(const char **p, int64_t *out) {
317 const char *s = js_skip_ws(*p);
318 int neg = 0;
319 if (*s == '-') { neg = 1; s++; }
320 if (*s < '0' || *s > '9') return 0;
321
322 int64_t v = 0;

327 s++;
328 }
329 *out = neg ? -v : v;
330 *p = s;
331 return 1;
332}

434ie_wdtype_t ie_weights_dedup_parse_dtype(const char *s) {
435 if (!s) return IE_WDTYPE_UNKNOWN;
436
437 /* Case-insensitive matching without locale. */
438 char tmp[16];
439 size_t n = strlen(s);
440 if (n >= sizeof(tmp)) n = sizeof(tmp) - 1;
441 for (size_t i = 0; i < n; i++) {
442 char c = s[i];
443 if (c >= 'A' && c <= 'Z') c = (char)(c - 'A' + 'a');
444 tmp[i] = c;
445 }
446 tmp[n] = '\0';
447
448 if (strcmp(tmp, "fp32") == 0) return IE_WDTYPE_FP32;
449 if (strcmp(tmp, "fp16") == 0) return IE_WDTYPE_FP16;
450 if (strcmp(tmp, "bf16") == 0) return IE_WDTYPE_BF16;
451 if (strcmp(tmp, "int8") == 0) return IE_WDTYPE_INT8;
452 if (strcmp(tmp, "int4") == 0) return IE_WDTYPE_INT4;
453
454 return IE_WDTYPE_UNKNOWN;
455}

781static ie_wdedup_status_t ie_parse_files_object(const char *json_obj,
782 ie_file_ref_t **out,
783 size_t *out_count) {
784 if (!out || !out_count) return IE_WDEDUP_EINVAL;
785 *out = NULL;
786 *out_count = 0;
787
788 const char *p = js_skip_ws(json_obj);
789 if (*p != '{') return IE_WDEDUP_EFORMAT;
790 p++;
791
792 size_t cap = 8;
793 size_t cnt = 0;
794 ie_file_ref_t *arr = (ie_file_ref_t *)calloc(cap, sizeof(*arr));
795 if (!arr) return IE_WDEDUP_ENOMEM;
796
797 p = js_skip_ws(p);
798 if (*p == '}') {
799 *out = arr;
800 *out_count = 0;
801 return IE_WDEDUP_OK;
802 }
803
804 for (;;) {
805 char *k = NULL;
806 if (!js_parse_string(&p, &k)) {
807 free(arr);
808 return IE_WDEDUP_EFORMAT;
809 }
810 p = js_skip_ws(p);
811 if (*p != ':') {
812 free(k);
813 for (size_t i = 0; i < cnt; i++) { free(arr[i].logical); free(arr[i].relpath); }
814 free(arr);
815 return IE_WDEDUP_EFORMAT;
816 }
817 p++;
818 p = js_skip_ws(p);
819
820 char *v = NULL;
821 if (!js_parse_string(&p, &v)) {
822 free(k);
823 for (size_t i = 0; i < cnt; i++) { free(arr[i].logical); free(arr[i].relpath); }
824 free(arr);
825 return IE_WDEDUP_EFORMAT;
826 }
827
828 if (cnt == cap) {
829 size_t oldcap = cap;
830 cap *= 2;
831 ie_file_ref_t *na = (ie_file_ref_t *)realloc(arr, cap * sizeof(*arr));
832 if (!na) {
833 free(k);
834 free(v);
835 for (size_t i = 0; i < cnt; i++) { free(arr[i].logical); free(arr[i].relpath); }
836 free(arr);
837 return IE_WDEDUP_ENOMEM;
838 }
839 arr = na;
840 memset(arr + oldcap, 0, (cap - oldcap) * sizeof(*arr));
841 }
842
843 arr[cnt].logical = k;
844 arr[cnt].relpath = v;
845 cnt++;
846
847 p = js_skip_ws(p);
848 if (*p == ',') { p++; p = js_skip_ws(p); continue; }
849 if (*p == '}') break;
850
851 for (size_t i = 0; i < cnt; i++) { free(arr[i].logical); free(arr[i].relpath); }
852 free(arr);
853 return IE_WDEDUP_EFORMAT;
854 }
855
856 *out = arr;
857 *out_count = cnt;
858 return IE_WDEDUP_OK;
859}

861static int ie_find_file_index(const ie_file_ref_t *refs, size_t n, const char *logical, uint32_t *out_idx) {
862 if (!refs || !logical || !out_idx) return 0;
863 for (size_t i = 0; i < n; i++) {
864 if (refs[i].logical && strcmp(refs[i].logical, logical) == 0) {
865 *out_idx = (uint32_t)i;
866 return 1;
867 }
868 }
869 return 0;
870}

872static int js_parse_blobref(const char **p, char **out_file, int64_t *out_off, int64_t *out_nb) {
873 if (!p || !*p || !out_file || !out_off || !out_nb) return 0;
874
875 const char *s = js_skip_ws(*p);
876 if (*s != '{') return 0;
877
878 const char *val = NULL;
879 if (!js_find_member_value(s, "file", &val)) return 0;
880 if (!js_parse_string(&val, out_file)) return 0;
881
882 val = NULL;
883 if (!js_find_member_value(s, "offset", &val)) { free(*out_file); *out_file = NULL; return 0; }
884 if (!js_parse_i64(&val, out_off)) { free(*out_file); *out_file = NULL; return 0; }
885
886 val = NULL;
887 if (!js_find_member_value(s, "nbytes", &val)) { free(*out_file); *out_file = NULL; return 0; }
888 if (!js_parse_i64(&val, out_nb)) { free(*out_file); *out_file = NULL; return 0; }
889
890 *p = s;
891 return js_skip_object(p);
892}

894static ie_wdedup_status_t ie_parse_one_tensor(const char **p,
895 const ie_file_ref_t *files,
896 size_t files_n,
897 ie_tensor_ref_t *out) {
898 memset(out, 0, sizeof(*out));
899 out->dtype = IE_WDTYPE_UNKNOWN;
900
901 const char *s = js_skip_ws(*p);
902 if (*s != '{') return IE_WDEDUP_EFORMAT;
903
904 const char *val = NULL;
905
906 val = NULL;
907 if (!js_find_member_value(s, "name", &val)) return IE_WDEDUP_EFORMAT;
908 if (!js_parse_string(&val, &out->name)) return IE_WDEDUP_EFORMAT;
909
910 char *dtype_s = NULL;
911 val = NULL;
912 if (!js_find_member_value(s, "dtype", &val)) { free(out->name); out->name = NULL; return IE_WDEDUP_EFORMAT; }
913 if (!js_parse_string(&val, &dtype_s)) { free(out->name); out->name = NULL; return IE_WDEDUP_EFORMAT; }
914 out->dtype = ie_weights_dedup_parse_dtype(dtype_s);
915 free(dtype_s);
916
917 /* In schema1, dtype must be known. */
918 if (out->dtype == IE_WDTYPE_UNKNOWN) { free(out->name); out->name = NULL; return IE_WDEDUP_EFORMAT; }
919
920 val = NULL;
921 if (!js_find_member_value(s, "shape", &val)) { free(out->name); out->name = NULL; return IE_WDEDUP_EFORMAT; }
922 if (!js_parse_shape(&val, out->shape, &out->rank)) { free(out->name); out->name = NULL; return IE_WDEDUP_EFORMAT; }
923
924 char *f_default = NULL, *f_mask = NULL, *f_exc = NULL;
925 int64_t off_default = 0, nb_default = 0;
926 int64_t off_mask = 0, nb_mask = 0;
927 int64_t off_exc = 0, nb_exc = 0;
928
929 val = NULL;
930 if (!js_find_member_value(s, "default", &val)) { free(out->name); out->name = NULL; return IE_WDEDUP_EFORMAT; }
931 if (!js_parse_blobref(&val, &f_default, &off_default, &nb_default)) { free(out->name); out->name = NULL; return IE_WDEDUP_EFORMAT; }
932
933 val = NULL;
934 if (!js_find_member_value(s, "mask", &val)) { free(out->name); out->name = NULL; free(f_default); return IE_WDEDUP_EFORMAT; }
935 if (!js_parse_blobref(&val, &f_mask, &off_mask, &nb_mask)) { free(out->name); out->name = NULL; free(f_default); return IE_WDEDUP_EFORMAT; }
936
937 val = NULL;
938 if (!js_find_member_value(s, "exceptions", &val)) { free(out->name); out->name = NULL; free(f_default); free(f_mask); return IE_WDEDUP_EFORMAT; }
939 if (!js_parse_blobref(&val, &f_exc, &off_exc, &nb_exc)) { free(out->name); out->name = NULL; free(f_default); free(f_mask); return IE_WDEDUP_EFORMAT; }
940
941 uint32_t idx_def = 0, idx_mask = 0, idx_exc = 0;
942 int ok_def = ie_find_file_index(files, files_n, f_default, &idx_def);
943 int ok_msk = ie_find_file_index(files, files_n, f_mask, &idx_mask);
944 int ok_exc = ie_find_file_index(files, files_n, f_exc, &idx_exc);
945
946 free(f_default);
947 free(f_mask);
948 free(f_exc);
949
950 if (!ok_def || !ok_msk || !ok_exc) {
951 free(out->name);
952 out->name = NULL;
953 return IE_WDEDUP_EFORMAT;
954 }
955
956 if (off_default < 0 || nb_default < 0 || off_mask < 0 || nb_mask < 0 || off_exc < 0 || nb_exc < 0) {
957 free(out->name);
958 out->name = NULL;
959 return IE_WDEDUP_ERANGE;
960 }
961
962 out->file_defaults = idx_def;
963 out->off_defaults = (uint64_t)off_default;
964 out->nbytes_defaults = (uint64_t)nb_default;
965
966 out->file_mask = idx_mask;
967 out->off_mask = (uint64_t)off_mask;
968 out->nbytes_mask = (uint64_t)nb_mask;
969
970 out->file_exceptions = idx_exc;
971 out->off_exceptions = (uint64_t)off_exc;
972 out->nbytes_exceptions = (uint64_t)nb_exc;
973
974 out->elem_size = ie_dtype_elem_size(out->dtype);
975 if (!ie_shape_elem_count(out->shape, out->rank, &out->elem_count)) {
976 free(out->name);
977 out->name = NULL;
978 return IE_WDEDUP_ERANGE;
979 }
980
981 *p = s;
982 if (!js_skip_object(p)) {
983 free(out->name);
984 out->name = NULL;
985 return IE_WDEDUP_EFORMAT;
986 }
987 return IE_WDEDUP_OK;
988}

990static ie_wdedup_status_t ie_parse_tensors_array(const char *json_val,
991 const ie_file_ref_t *files,
992 size_t files_n,
993 ie_tensor_ref_t **out,
994 size_t *out_count) {
995 if (!out || !out_count) return IE_WDEDUP_EINVAL;
996 *out = NULL;
997 *out_count = 0;
998
999 const char *p = js_skip_ws(json_val);
1000 if (*p != '[') return IE_WDEDUP_EFORMAT;
1001 p++;
1002 p = js_skip_ws(p);
1003
1004 size_t cap = 256;
1005 size_t cnt = 0;
1006 ie_tensor_ref_t *arr = (ie_tensor_ref_t *)calloc(cap, sizeof(*arr));
1007 if (!arr) return IE_WDEDUP_ENOMEM;
1008
1009 if (*p == ']') {
1010 *out = arr;
1011 *out_count = 0;
1012 return IE_WDEDUP_OK;
1013 }
1014
1015 for (;;) {
1016 if (cnt == cap) {
1017 size_t oldcap = cap;
1018 cap *= 2;
1019 ie_tensor_ref_t *na = (ie_tensor_ref_t *)realloc(arr, cap * sizeof(*arr));
1020 if (!na) {
1021 for (size_t i = 0; i < cnt; i++) free(arr[i].name);
1022 free(arr);
1023 return IE_WDEDUP_ENOMEM;
1024 }
1025 arr = na;
1026 memset(arr + oldcap, 0, (cap - oldcap) * sizeof(*arr));
1027 }
1028
1029 ie_wdedup_status_t st = ie_parse_one_tensor(&p, files, files_n, &arr[cnt]);
1030 if (st != IE_WDEDUP_OK) {
1031 for (size_t i = 0; i < cnt; i++) free(arr[i].name);
1032 free(arr);
1033 return st;
1034 }
1035 cnt++;
1036
1037 p = js_skip_ws(p);
1038 if (*p == ',') { p++; p = js_skip_ws(p); continue; }
1039 if (*p == ']') break;
1040
1041 for (size_t i = 0; i < cnt; i++) free(arr[i].name);
1042 free(arr);
1043 return IE_WDEDUP_EFORMAT;
1044 }
1045
1046 *out = arr;
1047 *out_count = cnt;
1048 return IE_WDEDUP_OK;
1049}

1055ie_wdedup_status_t ie_weights_dedup_open(ie_weights_dedup_t **out,
1056 const char *model_dir,
1057 const ie_weights_dedup_opts_t *opts) {
1058 if (!out || !model_dir) return IE_WDEDUP_EINVAL;
1059 *out = NULL;
1060
1061 ie_weights_dedup_t *h = (ie_weights_dedup_t *)calloc(1, sizeof(*h));
1062 if (!h) return IE_WDEDUP_ENOMEM;
1063 h->prefault_policy = opts ? opts->prefault_policy : 0;
1064 h->iebin.fd = -1;
1065
1066 h->model_dir = ie_strdup(model_dir);
1067 if (!h->model_dir) { free(h); return IE_WDEDUP_ENOMEM; }
1068
1069 /* Always attempt to parse model.dedup.json. If it is not schema1, fall back to schema2 diagnostics. */
1070 char *dedup_path = ie_path_join(model_dir, "model.dedup.json");
1071 if (!dedup_path) { ie_weights_dedup_close(&h); return IE_WDEDUP_ENOMEM; }
1072
1073 size_t dedup_n = 0;
1074 char *dedup_json = ie_read_file_text(dedup_path, &dedup_n);
1075 free(dedup_path);
1076 if (!dedup_json) {
1077 ie_wdedup_logf("open: missing model.dedup.json under model_dir=%s", model_dir);
1078 ie_weights_dedup_close(&h);
1079 return IE_WDEDUP_ENOENT;
1080 }
1081
1082 const char *files_val = NULL;
1083 const char *tensors_val = NULL;
1084 int has_schema1 = js_find_member_value(dedup_json, "files", &files_val) &&
1085 js_find_member_value(dedup_json, "tensors", &tensors_val);
1086
1087 if (has_schema1) {
1088 ie_file_ref_t *files = NULL;
1089 size_t files_n = 0;
1090 ie_wdedup_status_t st = ie_parse_files_object(files_val, &files, &files_n);
1091 if (st != IE_WDEDUP_OK) {
1092 ie_wdedup_logf("schema1: parse files failed status=%d (%s)", (int)st, ie_wdedup_status_str(st));
1093 free(dedup_json);
1094 ie_weights_dedup_close(&h);
1095 return st;
1096 }
1097
1098 h->files = (ie_mmap_file_t *)calloc(files_n, sizeof(*h->files));
1099 if (!h->files) {
1100 for (size_t i = 0; i < files_n; i++) { free(files[i].logical); free(files[i].relpath); }
1101 free(files);
1102 free(dedup_json);
1103 ie_weights_dedup_close(&h);
1104 return IE_WDEDUP_ENOMEM;
1105 }
1106 h->files_count = files_n;
1107
1108 ie_wdedup_status_t st_map = IE_WDEDUP_OK;
1109 for (size_t i = 0; i < files_n; i++) {
1110 char *full = ie_path_join(model_dir, files[i].relpath);
1111 if (!full) { st_map = IE_WDEDUP_ENOMEM; break; }
1112 st_map = ie_mmap_ro(&h->files[i], full, h->prefault_policy);
1113 if (st_map != IE_WDEDUP_OK) {
1114 ie_wdedup_logf("schema1: mmap file failed logical=%s rel=%s status=%d (%s)",
1115 files[i].logical ? files[i].logical : "(null)",
1116 files[i].relpath ? files[i].relpath : "(null)",
1117 (int)st_map, ie_wdedup_status_str(st_map));
1118 }
1119 free(full);
1120 if (st_map != IE_WDEDUP_OK) break;
1121 }
1122
1123 if (st_map == IE_WDEDUP_OK) {
1124 ie_wdedup_status_t st_t = ie_parse_tensors_array(tensors_val, files, files_n, &h->tensors, &h->tensors_count);
1125 if (st_t != IE_WDEDUP_OK) {
1126 ie_wdedup_logf("schema1: parse tensors failed status=%d (%s)", (int)st_t, ie_wdedup_status_str(st_t));
1127 st_map = st_t;
1128 } else if (h->tensors_count > 1) {
1129 qsort(h->tensors, h->tensors_count, sizeof(h->tensors[0]), ie_tensor_cmp_name);
1130 }
1131 }
1132
1133 for (size_t i = 0; i < files_n; i++) { free(files[i].logical); free(files[i].relpath); }
1134 free(files);
1135 free(dedup_json);
1136
1137 if (st_map != IE_WDEDUP_OK) {
1138 ie_weights_dedup_close(&h);
1139 return st_map;
1140 }
1141
1142 *out = h;
1143 return IE_WDEDUP_OK;
1144 }
1145
1146 /* ---------------------------------------------------------------------- */
1147 /* Schema2 fallback: parse model.ie.json tensors for detailed diagnostics. */
1148 /* This is where your current failure happens; this implementation is */
1149 /* strict about JSON structure but tolerant about dtype. */
1150 /* ---------------------------------------------------------------------- */
1151
1152 free(dedup_json);
1153
1154 char *iejson_path = ie_path_join(model_dir, "model.ie.json");
1155 if (!iejson_path) { ie_weights_dedup_close(&h); return IE_WDEDUP_ENOMEM; }
1156
1157 size_t iejson_n = 0;
1158 char *iejson = ie_read_file_text(iejson_path, &iejson_n);
1159 if (!iejson) {
1160 ie_wdedup_logf("schema2: failed to read %s errno=%d (%s)", iejson_path, errno, ie_wdedup_errno_str(errno));
1161 free(iejson_path);
1162 ie_weights_dedup_close(&h);
1163 return IE_WDEDUP_ENOENT;
1164 }
1165 free(iejson_path);
1166
1167 ie_wdedup_status_t st_ie = ie_parse_iejson_tensors(iejson, &h->iejson_tensors, &h->iejson_tensors_count);
1168 if (st_ie != IE_WDEDUP_OK) {
1169 ie_wdedup_logf("schema2: parse model.ie.json tensors failed status=%d (%s)", (int)st_ie, ie_wdedup_status_str(st_ie));
1170 free(iejson);
1171 ie_weights_dedup_close(&h);
1172 return st_ie;
1173 }
1174
1175 ie_wdedup_logf("schema2: parsed model.ie.json tensors=%zu", h->iejson_tensors_count);
1176
1177 /* Also mmap model.ie.bin so future schema2 can provide direct views if needed. */
1178 char *iebin_path = ie_path_join(model_dir, "model.ie.bin");
1179 if (!iebin_path) { free(iejson); ie_weights_dedup_close(&h); return IE_WDEDUP_ENOMEM; }
1180
1181 ie_wdedup_status_t st_bin = ie_mmap_ro(&h->iebin, iebin_path, h->prefault_policy);
1182 free(iebin_path);
1183 if (st_bin != IE_WDEDUP_OK) {
1184 ie_wdedup_logf("schema2: mmap model.ie.bin failed status=%d (%s)", (int)st_bin, ie_wdedup_status_str(st_bin));
1185 free(iejson);
1186 ie_weights_dedup_close(&h);
1187 return st_bin;
1188 }
1189
1190 /* For schema2, require the three blobs to exist (strict behavior). */
1191 char *p_def = ie_path_join(model_dir, "model.defaults.bin");
1192 char *p_msk = ie_path_join(model_dir, "model.masks.bin");
1193 char *p_exc = ie_path_join(model_dir, "model.exceptions.bin");
1194 if (!p_def || !p_msk || !p_exc) {
1195 free(p_def); free(p_msk); free(p_exc);
1196 free(iejson);
1197 ie_weights_dedup_close(&h);
1198 return IE_WDEDUP_ENOMEM;
1199 }
1200
1201 /* Map blobs into files[] so IE_WVIEW_DEDUP views can be added later. */
1202 h->files = (ie_mmap_file_t *)calloc(3, sizeof(*h->files));
1203 if (!h->files) {
1204 free(p_def); free(p_msk); free(p_exc);
1205 free(iejson);
1206 ie_weights_dedup_close(&h);
1207 return IE_WDEDUP_ENOMEM;
1208 }
1209 h->files_count = 3;
1210
1211 ie_wdedup_status_t st0 = ie_mmap_ro(&h->files[0], p_def, h->prefault_policy);
1212 ie_wdedup_status_t st1 = ie_mmap_ro(&h->files[1], p_msk, h->prefault_policy);
1213 ie_wdedup_status_t st2 = ie_mmap_ro(&h->files[2], p_exc, h->prefault_policy);
1214
1215 free(p_def); free(p_msk); free(p_exc);
1216 free(iejson);
1217
1218 if (st0 != IE_WDEDUP_OK || st1 != IE_WDEDUP_OK || st2 != IE_WDEDUP_OK) {
1219 ie_wdedup_logf("schema2: mmap blobs failed status=%d (%s)",
1220 (int)(st0 != IE_WDEDUP_OK ? st0 : (st1 != IE_WDEDUP_OK ? st1 : st2)),
1221 ie_wdedup_status_str(st0 != IE_WDEDUP_OK ? st0 : (st1 != IE_WDEDUP_OK ? st1 : st2)));
1222 ie_weights_dedup_close(&h);
1223 return (st0 != IE_WDEDUP_OK ? st0 : (st1 != IE_WDEDUP_OK ? st1 : st2));
1224 }
1225
1226 /* NOTE: schema2 mapping (iejson + tensor_map + groups) is not implemented here.
1227 * This open() now succeeds and enforces the presence of the three blobs, and
1228 * provides extremely detailed diagnostics for why model.ie.json parsing fails.
1229 * Weight views for schema2 are expected to be implemented on top of this (next step).
1230 */
1231
1232 *out = h;
1233 return IE_WDEDUP_OK;
1234}

1336size_t ie_weights_dedup_materialize(const ie_weight_view_t *view, void *dst, size_t dst_nbytes) {
1337 if (!view || !dst) return 0;
1338
1339 if (view->kind == IE_WVIEW_DIRECT) {
1340 if (!view->data) return 0;
1341 if (view->nbytes > dst_nbytes) return 0;
1342 memcpy(dst, view->data, view->nbytes);
1343 return view->nbytes;
1344 }
1345
1346 if (view->kind != IE_WVIEW_DEDUP) return 0;
1347 if (!view->defaults || !view->mask) return 0;
1348
1349 if (view->dtype != IE_WDTYPE_INT4) {
1350 if (view->elem_size == 0) return 0;
1351
1352 size_t need = view->elem_count * view->elem_size;
1353 if (need > dst_nbytes) return 0;
1354 if (view->defaults_nbytes < need) return 0;
1355
1356 memcpy(dst, view->defaults, need);
1357
1358 const uint8_t *exc = (const uint8_t *)view->exceptions;
1359 size_t exc_off = 0;
1360
1361 for (size_t i = 0; i < view->elem_count; i++) {

1364
1365 uint8_t *dst_b = (uint8_t *)dst + i * view->elem_size;
1366 memcpy(dst_b, exc + exc_off, view->elem_size);
1367 exc_off += view->elem_size;
1368 }
1369
1370 return need;
1371 }
1372
1373 /* INT4 nibble-packed */
1374 {
1375 size_t need = view->defaults_nbytes;
1376 if (need > dst_nbytes) return 0;
1377 if (view->defaults_nbytes == 0) return 0;
1378
1379 memcpy(dst, view->defaults, need);
1380
1381 const uint8_t *exc_bytes = (const uint8_t *)view->exceptions;
1382 size_t exc_nibbles = view->exceptions_nbytes * 2;
1383 size_t exc_idx = 0;
1384

1386 if (!ie_bitset_get(view->mask, i)) continue;

1390 ie_int4_set_nibble((uint8_t *)dst, i, v);
