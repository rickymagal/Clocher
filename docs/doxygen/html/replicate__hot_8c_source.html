83static void first_touch_write(void* p, size_t sz) {
84 volatile uint8_t* b = (volatile uint8_t*)p;
85 const size_t step = 64;
86 for (size_t i = 0; i < sz; i += step) b[i] = 0u;
87 if (sz) b[sz - 1] = 0u;

131int ie_hot_replicate_build(const ie_topology_t* topo,
132 int n_sockets,
133 const void* blob,
134 size_t blob_size,
135 ie_hot_replicas_t** out)
136{
137 if (!out || !blob || blob_size == 0) return -EINVAL;
138
139 int S = n_sockets;
140 if (S <= 0 && topo) S = ie_topology_sockets(topo);
141 if (S <= 0) S = 1;
142
143 ie_hot_replicas_t* set = (ie_hot_replicas_t*)calloc(1, sizeof(*set));
144 if (!set) return -ENOMEM;
145
146 set->n_sockets = S;
147 set->r = (ie_hot_replica_t*)calloc((size_t)S, sizeof(*set->r));
148 if (!set->r) { free(set); return -ENOMEM; }
149
150 int rc = 0;
151 for (int s = 0; s < S; ++s) {
152 /* Bind for first-touch locality (best-effort). */
153 int brc = ie_topology_bind_thread_to_socket(topo, s);
154 if (brc != 0) IE_LOG_WARN("binding to socket %d failed (%d)", s, brc);
155
156 void* buf = malloc(blob_size);
157 if (!buf) { rc = -ENOMEM; goto fail; }
158
159 first_touch_write(buf, blob_size);
160 memcpy(buf, blob, blob_size);
161 advise_willneed(buf, blob_size);
162
163 set->r[s].ptr = buf;
164 set->r[s].size = blob_size;
165 set->r[s].socket_id = s;
166 }
167
168 *out = set;
169 return 0;
170
171fail:
172 if (set) {
173 if (set->r) {
174 for (int s = 0; s < S; ++s) free(set->r[s].ptr);
175 free(set->r);
176 }
177 free(set);
178 }
179 return rc ? rc : -ENOMEM;
180}
