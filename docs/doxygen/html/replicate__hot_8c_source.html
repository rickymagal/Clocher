97static void first_touch_write(void* p, size_t sz) {

99 const size_t step = 64;
100 for (size_t i = 0; i < sz; i += step) b[i] = 0u;
101 if (sz) b[sz - 1] = 0u;

171int ie_hot_replicate_build(const ie_topology_t* topo,
172 int n_sockets,
173 const void* blob,
174 size_t blob_size,
175 ie_hot_replicas_t** out)
176{
177 if (!out || !blob || blob_size == 0) return -EINVAL;
178
179 int S = n_sockets;
180 if (S <= 0 && topo) S = ie_topology_sockets(topo);
181 if (S <= 0) S = 1;
182
183 ie_hot_replicas_t* set = (ie_hot_replicas_t*)calloc(1, sizeof(*set));
184 if (!set) return -ENOMEM;
185
186 set->n_sockets = S;
187 set->r = (ie_hot_replica_t*)calloc((size_t)S, sizeof(*set->r));
188 if (!set->r) { free(set); return -ENOMEM; }
189
190 int rc = 0;
191 for (int s = 0; s < S; ++s) {
192 int brc = ie_topology_bind_thread_to_socket(topo, s);
193 if (brc != 0) IE_LOG_WARN("binding to socket %d failed (%d)", s, brc);
194
195 void* buf = alloc_replica_first_touch(blob_size);
196 if (!buf) { rc = -ENOMEM; goto fail; }
197
198 memcpy(buf, blob, blob_size);
199 advise_willneed(buf, blob_size);
200 advise_sequential(buf, blob_size);
201
202 set->r[s].ptr = buf;
203 set->r[s].size = blob_size;
204 set->r[s].socket_id = s;
205 }
206
207 *out = set;
208 return 0;
209
210fail:
211 free_replicas_internal(set);
212 return rc ? rc : -ENOMEM;
213}
