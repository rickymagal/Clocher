25def pack_int4_rows(q_int8: np.ndarray) -> bytes:
26 rows, cols = q_int8.shape
27 if cols % 2:
28 q_int8 = np.pad(q_int8, ((0,0),(0,1)), mode="constant")
29 cols += 1
30 q16 = q_int8.astype(np.int16)
31 lo = (q16[:, 0::2] & 0xF).astype(np.uint8)
32 hi = (q16[:, 1::2] & 0xF).astype(np.uint8)
33 packed = (hi << 4) | lo
34 return packed.tobytes()
