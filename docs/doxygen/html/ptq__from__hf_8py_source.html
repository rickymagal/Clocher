4#!/usr/bin/env python3
5"""
6Hugging Face one-shot PTQ:
7- Downloads a model repo snapshot via huggingface_hub
8- Locates a checkpoint (safetensors or PyTorch .bin/.pth/.ckpt)
9- Loads state_dict, extracts the EXACT tensor by key
10- (Optional) transposes 2D
11- Exports row-major FP32 .bin
12- Runs weight-only PTQ (INT8 by default; INT4 with --wbits=4)
13
14Example:
15 make deps-ptq
16 python3 scripts/ptq_from_hf.py \
17 --repo facebook/opt-125m \
18 --key model.decoder.layers.0.self_attn.q_proj.weight \
19 --out-prefix out/qproj_int4 \
20 --wbits 4 --mode per_row --scale-dtype fp16
21"""

31def find_checkpoint(root: Path, prefer_file: str | None) -> Path:
32 """
33 Resolve a checkpoint file path inside a downloaded HF snapshot.
34 Prefers .safetensors, otherwise typical PyTorch filenames.
35 """
36 if prefer_file:
37 p = root / prefer_file
38 if p.exists():
39 return p
40 print(f"ERROR: --file '{prefer_file}' not found in snapshot", file=sys.stderr)
41 sys.exit(3)
42
43 candidates: list[Path] = []
44 for pat in ("*.safetensors", "*pytorch_model*.bin", "*.bin", "*.pt", "*.pth", "*.ckpt"):
45 candidates += list(root.glob(pat))
46 if not candidates:
47 print("ERROR: no checkpoint files found in repo snapshot", file=sys.stderr)
48 sys.exit(3)
49 candidates.sort(key=lambda p: (0 if p.suffix == ".safetensors" else 1, len(p.name)))
50 return candidates[0]

53def load_state_dict(ckpt_path: Path) -> dict:
54 """
55 Load a state_dict from .safetensors or PyTorch checkpoint-like files.
56 """
57 if ckpt_path.suffix == ".safetensors":
58 try:
59 from safetensors.torch import load_file as safe_load
60 return safe_load(str(ckpt_path))
61 except Exception as e:
62 print(f"ERROR: failed to load safetensors: {e}", file=sys.stderr)
63 sys.exit(2)
64
65 try:
66 import torch
67 except Exception:
68 print("ERROR: torch is required. pip install torch", file=sys.stderr)
69 sys.exit(2)
70
71 obj = torch.load(str(ckpt_path), map_location="cpu")
72 if hasattr(obj, "state_dict"):
73 return obj.state_dict()
74 if isinstance(obj, dict) and "state_dict" in obj:
75 return obj["state_dict"]
76 if isinstance(obj, dict):
77 return obj
78 print("ERROR: unsupported checkpoint format", file=sys.stderr)
79 sys.exit(2)

82def tensor_to_fp32_bin(t, out_bin: Path, transpose: bool) -> tuple[int, int]:
83 """
84 Export a tensor to row-major float32 .bin. Returns (rows, cols).
85 """
86 import numpy as np
87
88 # Torch tensor -> numpy
89 if hasattr(t, "detach"):
90 t = t.detach().cpu()
91 arr = t.to(dtype=t.dtype if str(t.dtype).startswith("float") else None).float().numpy()
92 else:
93 arr = t
94 if arr.dtype != np.float32:
95 arr = arr.astype(np.float32)
96
97 if arr.ndim != 2:
98 print(f"ERROR: selected tensor is {arr.ndim}D; expected 2D [rows, cols]", file=sys.stderr)
99 sys.exit(3)
100
101 if transpose:
102 arr = arr.T.copy()
103
104 rows, cols = int(arr.shape[0]), int(arr.shape[1])
105 out_bin.parent.mkdir(parents=True, exist_ok=True)
106 with open(out_bin, "wb") as f:
107 f.write(arr.tobytes(order="C"))
108 return rows, cols

116 ap.add_argument("--key", required=True, help="state_dict key to extract")

119 ap.add_argument("--mode", default="per_row", choices=["per_row", "per_tensor"])

123 ap.add_argument("--scale-dtype", default="fp16", choices=["fp16", "fp32"],

133 # 1) Download snapshot

174 # 5) Run PTQ with the TRUE rows/cols
