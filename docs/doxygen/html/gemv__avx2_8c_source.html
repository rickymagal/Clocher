39ie_gemv_f32_avx2_impl(const float *W, const float *x, float *y, size_t rows,
40 size_t cols, const float *bias, size_t blk_k) {
41 const size_t BK = (blk_k > 0 ? blk_k : cols);
42
43 for (size_t r = 0; r < rows; ++r) {
44 const float *wrow = W + r * cols;
45 float acc_scalar = 0.0f;
46
47 __m256 vacc0 = _mm256_setzero_ps();
48 __m256 vacc1 = _mm256_setzero_ps();
49 __m256 vacc2 = _mm256_setzero_ps();
50 __m256 vacc3 = _mm256_setzero_ps();
51
52 for (size_t k0 = 0; k0 < cols; k0 += BK) {
53 const size_t klen = (k0 + BK <= cols) ? BK : (cols - k0);
54 const float *wblk = wrow + k0;
55 const float *xblk = x + k0;
56
57 /* Prefetch next chunk (temporal hint) */
58 if (k0 + 2 * BK < cols) {
59 _mm_prefetch((const char *)(wblk + BK), _MM_HINT_T0);
60 _mm_prefetch((const char *)(xblk + BK), _MM_HINT_T0);
61 }
62
63 size_t k = 0;
64 for (; k + 32 <= klen; k += 32) {
65 __m256 w0 = _mm256_loadu_ps(wblk + k + 0);
66 __m256 x0 = _mm256_loadu_ps(xblk + k + 0);
67 vacc0 = _mm256_fmadd_ps(w0, x0, vacc0);
68
69 __m256 w1 = _mm256_loadu_ps(wblk + k + 8);
70 __m256 x1 = _mm256_loadu_ps(xblk + k + 8);
71 vacc1 = _mm256_fmadd_ps(w1, x1, vacc1);
72
73 __m256 w2 = _mm256_loadu_ps(wblk + k + 16);
74 __m256 x2 = _mm256_loadu_ps(xblk + k + 16);
75 vacc2 = _mm256_fmadd_ps(w2, x2, vacc2);
76
77 __m256 w3 = _mm256_loadu_ps(wblk + k + 24);
78 __m256 x3 = _mm256_loadu_ps(xblk + k + 24);
79 vacc3 = _mm256_fmadd_ps(w3, x3, vacc3);
80 }
81 for (; k + 8 <= klen; k += 8) {
82 __m256 wv = _mm256_loadu_ps(wblk + k);
83 __m256 xv = _mm256_loadu_ps(xblk + k);
84 vacc0 = _mm256_fmadd_ps(wv, xv, vacc0);
85 }
86 for (; k < klen; ++k) {
87 acc_scalar += wblk[k] * xblk[k];
88 }
89 }
90
91 /* Reduce four accumulators to one scalar */
92 __m256 t01 = _mm256_add_ps(vacc0, vacc1);
93 __m256 t23 = _mm256_add_ps(vacc2, vacc3);
94 __m256 t = _mm256_add_ps(t01, t23);
95 __m128 low = _mm256_castps256_ps128(t);
96 __m128 high = _mm256_extractf128_ps(t, 1);
97 __m128 sum = _mm_add_ps(low, high);
98 __m128 shf = _mm_movehdup_ps(sum);
99 sum = _mm_add_ps(sum, shf);
100 shf = _mm_movehl_ps(shf, sum);
101 sum = _mm_add_ss(sum, shf);
102
103 float acc = _mm_cvtss_f32(sum) + acc_scalar;
104 if (bias) acc += bias[r];
105 y[r] = acc;
106 }
107}

127ie_gemv_qi8_f32_avx2_impl(const float *W, const int8_t *x_q, float *y,
128 size_t rows, size_t cols, const float *bias,
129 size_t blk_k, ie_act_i8_params params,
130 int symmetric) {
131 (void)symmetric;
132#if defined(_MSC_VER)
133 float *x = (float *)_aligned_malloc(cols * sizeof(float), 64);
134 if (!x) return;
135#else
136 float *x = (float *)aligned_alloc(64, cols * sizeof(float));
137 if (!x) return;
138#endif
139
140 const float s = params.scale;
141 const int z = (int)params.zero_point;
142
143 size_t i = 0;
144 for (; i + 8 <= cols; i += 8) {
145 __m128i q = _mm_loadl_epi64((const __m128i *)(x_q + i)); /* 8x i8 */
146 __m256i qi = _mm256_cvtepi8_epi32(q); /* -> 8x i32 */
147 __m256 qf = _mm256_cvtepi32_ps(qi); /* -> 8x f32 */
148 __m256 zf = _mm256_set1_ps((float)z);
149 __m256 sf = _mm256_set1_ps(s);
150 __m256 xf = _mm256_mul_ps(sf, _mm256_sub_ps(qf, zf));
151 _mm256_storeu_ps(x + i, xf);
152 }
153 for (; i < cols; ++i) x[i] = s * ((int)x_q[i] - z);

157#if defined(_MSC_VER)
158 _aligned_free(x);
159#else
160 free(x);
161#endif
162}
