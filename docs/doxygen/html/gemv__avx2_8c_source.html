89__attribute__((target("avx2,fma"), used)) void
90ie_gemv_f32_avx2_impl(const float *W, const float *x, float *y, size_t rows,
91 size_t cols, const float *bias, size_t blk_k) {
92 if (!W || !x || !y || rows == 0 || cols == 0) return;
93
94 const size_t BK = (blk_k > 0 ? blk_k : cols);
95
96#if defined(__GNUC__) || defined(__clang__)
97 W = (const float *)__builtin_assume_aligned(W, 32);
98 x = (const float *)__builtin_assume_aligned(x, 32);
99 y = (float *)__builtin_assume_aligned(y, 32);
100#endif
101
102 for (size_t r = 0; r < rows; ++r) {
103 const float *wrow = W + r * cols;
104 float acc_scalar = 0.0f;
105
106 __m256 vacc0 = _mm256_setzero_ps();
107 __m256 vacc1 = _mm256_setzero_ps();
108 __m256 vacc2 = _mm256_setzero_ps();
109 __m256 vacc3 = _mm256_setzero_ps();
110
111 for (size_t k0 = 0; k0 < cols; k0 += BK) {
112 const size_t klen = (k0 + BK <= cols) ? BK : (cols - k0);
113 const float *wblk = wrow + k0;
114 const float *xblk = x + k0;
115
116 if (k0 + 2 * BK < cols) {
117 _mm_prefetch((const char *)(wblk + BK), _MM_HINT_T0);
118 _mm_prefetch((const char *)(xblk + BK), _MM_HINT_T0);
119 }
120
121 size_t k = 0;
122 for (; k + 32 <= klen; k += 32) {
123 __m256 w0 = _mm256_loadu_ps(wblk + k + 0);
124 __m256 x0 = _mm256_loadu_ps(xblk + k + 0);
125 vacc0 = _mm256_fmadd_ps(w0, x0, vacc0);
126
127 __m256 w1 = _mm256_loadu_ps(wblk + k + 8);
128 __m256 x1 = _mm256_loadu_ps(xblk + k + 8);
129 vacc1 = _mm256_fmadd_ps(w1, x1, vacc1);
130
131 __m256 w2 = _mm256_loadu_ps(wblk + k + 16);
132 __m256 x2 = _mm256_loadu_ps(xblk + k + 16);
133 vacc2 = _mm256_fmadd_ps(w2, x2, vacc2);
134
135 __m256 w3 = _mm256_loadu_ps(wblk + k + 24);
136 __m256 x3 = _mm256_loadu_ps(xblk + k + 24);
137 vacc3 = _mm256_fmadd_ps(w3, x3, vacc3);
138 }
139
140 for (; k + 8 <= klen; k += 8) {
141 __m256 wv = _mm256_loadu_ps(wblk + k);
142 __m256 xv = _mm256_loadu_ps(xblk + k);
143 vacc0 = _mm256_fmadd_ps(wv, xv, vacc0);
144 }
145
146 for (; k < klen; ++k) {
147 acc_scalar += wblk[k] * xblk[k];
148 }
149 }
150
151 __m256 t01 = _mm256_add_ps(vacc0, vacc1);
152 __m256 t23 = _mm256_add_ps(vacc2, vacc3);
153 __m256 t = _mm256_add_ps(t01, t23);
154 __m128 low = _mm256_castps256_ps128(t);
155 __m128 high = _mm256_extractf128_ps(t, 1);
156 __m128 sum = _mm_add_ps(low, high);
157 __m128 shf = _mm_movehdup_ps(sum);
158 sum = _mm_add_ps(sum, shf);
159 shf = _mm_movehl_ps(shf, sum);
160 sum = _mm_add_ss(sum, shf);
161
162 float acc = _mm_cvtss_f32(sum) + acc_scalar;
163 if (bias) acc += bias[r];
164 y[r] = acc;
165 }
166}

185__attribute__((target("avx2,fma"), used)) void
186ie_gemv_qi8_f32_avx2_impl(const float *W, const int8_t *x_q, float *y,
187 size_t rows, size_t cols, const float *bias,
188 size_t blk_k, ie_act_i8_params params,
189 int symmetric) {
190 (void)symmetric;
191 if (!W || !x_q || !y || rows == 0 || cols == 0) return;
192
193 float *x = (float *)ie_aligned_malloc_64(cols * sizeof(float));
194 if (!x) return;
195
196 const float s = params.scale;
197 const int z = (int)params.zero_point;
198
199 size_t i = 0;
200 for (; i + 8 <= cols; i += 8) {
201 __m128i q = _mm_loadl_epi64((const __m128i *)(x_q + i));
202 __m256i qi = _mm256_cvtepi8_epi32(q);
203 __m256 qf = _mm256_cvtepi32_ps(qi);
204 __m256 zf = _mm256_set1_ps((float)z);
205 __m256 sf = _mm256_set1_ps(s);
206 __m256 xf = _mm256_mul_ps(sf, _mm256_sub_ps(qf, zf));
207 _mm256_storeu_ps(x + i, xf);
208 }
209 for (; i < cols; ++i) x[i] = s * ((int)x_q[i] - z);
210
211 ie_gemv_f32_avx2_impl(W, x, y, rows, cols, bias, blk_k);
212
213 ie_aligned_free_64(x);
214}
