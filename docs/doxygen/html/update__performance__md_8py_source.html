4"""

6
7Key points:
8- Uses only the last 3 runs for each device (for aggregate sections).
9- Aggregations use Σ tokens / Σ time.
10- Includes Memory Details when present in per-run logs.
11- Includes a Deduplication section that reports *real* artifact sizes
12 by following symlinks (reports target sizes, not link lengths).
13- Adds a Per-Prompt CPU vs GPU section when per-prompt metrics are available
14 in the latest run object (supports several common field names).
15- Prints per-prompt token-ID verification status when present:
16 - token_id_check_ok (or variants)

114def _discover_cpu_model() -> Optional[str]:
115 try:
116 with open("/proc/cpuinfo", "r", encoding="utf-8") as f:
117 m = re.search(r"model name\s*:\s*(.+)", f.read())
118 if m:
119 return m.group(1).strip()
120 except Exception:
121 pass
122 return platform.processor() or platform.machine() or None

125def _discover_mem_total_gb() -> Optional[float]:
126 try:
127 with open("/proc/meminfo", "r", encoding="utf-8") as f:
128 m = re.search(r"MemTotal:\s+(\d+)\s+kB", f.read())
129 if m:
130 kb = float(m.group(1))
131 return kb * 1024.0 / 1_000_000_000.0
132 except Exception:
133 pass
134 return None

205def _agg(runs: List[Run]) -> Agg:
206 tokens = [int(r.get("tokens_generated", 0)) for r in runs]
207 wall = [float(r.get("wall_time_s", 0.0)) for r in runs]
208 p50 = [
209 float(r.get("latency_p50_ms", r.get("p50_ms", 0.0)))
210 for r in runs
211 if ("latency_p50_ms" in r or "p50_ms" in r)
212 ]
213 p95 = [
214 float(r.get("latency_p95_ms", r.get("p95_ms", 0.0)))
215 for r in runs
216 if ("latency_p95_ms" in r or "p95_ms" in r)
217 ]
218
219 tokens_sum = _sum_i(tokens)
220 wall_sum = _sum_f(wall)
221 tps_true = (float(tokens_sum) / wall_sum) if wall_sum > 0 else None
222
223 rss_mean, _, rss_max = _agg_numeric(runs, "rss_peak_mb")
224 pss_mean, _, pss_max = _agg_numeric(runs, "pss_peak_mb")
225 vms_mean, _, vms_max = _agg_numeric(runs, "vms_peak_mb")
226
227 rss_floor_mean, _, rss_floor_max = _agg_numeric(runs, "rss_floor_mb")
228 rss_delta_mean, _, rss_delta_max = _agg_numeric(runs, "rss_delta_mb")
229
230 minflt_sum = _agg_sum(runs, "minflt")
231 majflt_sum = _agg_sum(runs, "majflt")
232
233 swap_in_sum_mb = _agg_sum(runs, "swap_in_mb")
234 swap_out_sum_mb = _agg_sum(runs, "swap_out_mb")
235
236 psi_some_mean, _, psi_some_max = _agg_numeric(runs, "psi_mem_some_pct")
237 psi_full_mean, _, psi_full_max = _agg_numeric(runs, "psi_mem_full_pct")
238
239 numa_loc_mean, _, numa_loc_max = _agg_numeric(runs, "numa_locality_pct")
240
241 mem_avail_mb_mean, _, _ = _agg_numeric(runs, "mem_available_mb")
242 mem_avail_pct_mean, _, _ = _agg_numeric(runs, "mem_available_pct")
243
244 kvh = [int(r.get("kv_hits", 0)) for r in runs]
245 kvm = [int(r.get("kv_misses", 0)) for r in runs]
246
247 return {
248 "runs": len(runs),
249 "tokens_sum": tokens_sum,
250 "wall_sum": wall_sum,
251 "tps_true": tps_true,
252 "p50_mean": _mean(p50),
253 "p95_mean": _mean(p95),
254 "rss_mean": rss_mean,
255 "rss_max": rss_max,
256 "pss_mean": pss_mean,
257 "pss_max": pss_max,
258 "vms_mean": vms_mean,
259 "vms_max": vms_max,
260 "rss_floor_mean": rss_floor_mean,
261 "rss_floor_max": rss_floor_max,
262 "rss_delta_mean": rss_delta_mean,
263 "rss_delta_max": rss_delta_max,
264 "minflt_sum": minflt_sum,
265 "majflt_sum": majflt_sum,
266 "swap_in_sum_mb": swap_in_sum_mb,
267 "swap_out_sum_mb": swap_out_sum_mb,
268 "psi_some_mean": psi_some_mean,
269 "psi_some_max": psi_some_max,
270 "psi_full_mean": psi_full_mean,
271 "psi_full_max": psi_full_max,
272 "numa_locality_mean": numa_loc_mean,
273 "numa_locality_max": numa_loc_max,
274 "mem_available_mb_mean": mem_avail_mb_mean,
275 "mem_available_pct_mean": mem_avail_pct_mean,
276 "kv_hits": _sum_i(kvh),

355def _dedup_artifact_stats(model_dir: Optional[str], model_bin_size: Optional[int]) -> Dict[str, Any]:

380 def _one(p: str) -> Tuple[Optional[str], Optional[int]]:
381 try:
382 rp = os.path.realpath(p)
383 if not os.path.exists(rp):
384 return None, None
385 return rp, os.path.getsize(rp)
386 except Exception:
387 return None, None
388
389 d_rp, d_sz = _one(defaults)
390 m_rp, m_sz = _one(masks)
391 e_rp, e_sz = _one(exceptions)
392
393 out["defaults_realpath"] = d_rp
394 out["masks_realpath"] = m_rp
395 out["exceptions_realpath"] = e_rp
396
397 out["defaults_size"] = d_sz
398 out["masks_size"] = m_sz
399 out["exceptions_size"] = e_sz
400
401 sizes = [x for x in (d_sz, m_sz, e_sz) if isinstance(x, int)]
402 if sizes:
403 total = int(sum(sizes))
404 out["total_size"] = total
405 if model_bin_size and model_bin_size > 0:
406 out["ratio_vs_model_bin"] = float(total) / float(model_bin_size)
407 return out

741def _extract_token_check(rec: Optional[PromptRec]) -> Tuple[Optional[bool], Optional[int], Optional[int], Optional[int]]:
742 """
743 Returns (ok, step, engine_id, hf_id) best-effort.
744
745 Supported schemas:
746 - Flat:
747 - token_id_check_ok / token_ids_ok / ids_ok
748 - token_id_check_first_mismatch_step / first_mismatch_step / mismatch_step
749 - token_id_check_engine_id / engine_id / engine_next_id
750 - token_id_check_hf_id / hf_id / hf_next_id
751 - Nested:
752 - token_id_check: { ok: bool, first_mismatch: { step, engine_id, hf_id } }
753 - token_check / hf_check / id_check (similar shape)
754 """
755 if not rec or not isinstance(rec, dict):
756 return None, None, None, None
757
758 ok_keys = ("token_id_check_ok", "token_ids_ok", "ids_ok", "id_check_ok", "token_check_ok")
759 step_keys = ("token_id_check_first_mismatch_step", "first_mismatch_step", "mismatch_step", "diverge_step")
760 eng_keys = ("token_id_check_engine_id", "engine_id", "engine_next_id", "engine_token_id", "engine_next_token_id")
761 hf_keys = ("token_id_check_hf_id", "hf_id", "hf_next_id", "hf_token_id", "hf_next_token_id")
762
763 ok = None
764 for k in ok_keys:
765 ok = _b(rec, k)
766 if ok is not None:
767 break
768
769 step = None
770 for k in step_keys:
771 step = _i(rec, k)
772 if step is not None:
773 break
774
775 eng = None
776 for k in eng_keys:
777 eng = _i(rec, k)
778 if eng is not None:
779 break
780
781 hf = None
782 for k in hf_keys:
783 hf = _i(rec, k)
784 if hf is not None:
785 break
786
787 if ok is not None or step is not None or eng is not None or hf is not None:
788 return ok, step, eng, hf
789
790 for container_key in ("token_id_check", "token_check", "hf_check", "id_check", "verify_ids", "verify_hf_ids"):
791 v = rec.get(container_key)
792 if isinstance(v, dict):
793 ok2 = None
794 if "ok" in v:
795 vv = v.get("ok")
796 ok2 = vv if isinstance(vv, bool) else _b({"x": vv}, "x")
797 if ok2 is None:
798 for k in ok_keys:
799 if k in v:
800 vv = v.get(k)
801 ok2 = vv if isinstance(vv, bool) else _b({"x": vv}, "x")
802 if ok2 is not None:
803 break
804
805 fm = v.get("first_mismatch")
806 if not isinstance(fm, dict):
807 fm = v.get("mismatch")
808 if not isinstance(fm, dict):
809 fm = v.get("first_failure")
810 if not isinstance(fm, dict):
811 fm = None
812
813 step2 = None
814 eng2 = None
815 hf2 = None
816 if fm:
817 for k in ("step", "t", "index"):
818 step2 = _i(fm, k)
819 if step2 is not None:
820 break
821 for k in ("engine_id", "engine_next_id", "engine", "engine_token"):
822 eng2 = _i(fm, k)
823 if eng2 is not None:
824 break
825 for k in ("hf_id", "hf_next_id", "hf", "hf_token"):
826 hf2 = _i(fm, k)
827 if hf2 is not None:
828 break
829
830 return ok2, step2, eng2, hf2
831
832 return None, None, None, None
