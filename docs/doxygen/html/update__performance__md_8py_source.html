109def _discover_cpu_model() -> Optional[str]:
110 try:
111 with open("/proc/cpuinfo", "r", encoding="utf-8") as f:
112 m = re.search(r"model name\s*:\s*(.+)", f.read())
113 if m:
114 return m.group(1).strip()
115 except Exception:
116 pass
117 return platform.processor() or platform.machine() or None

120def _discover_mem_total_gb() -> Optional[float]:
121 try:
122 with open("/proc/meminfo", "r", encoding="utf-8") as f:
123 m = re.search(r"MemTotal:\s+(\d+)\s+kB", f.read())
124 if m:
125 kb = float(m.group(1))
126 return kb * 1024.0 / 1_000_000_000.0
127 except Exception:
128 pass
129 return None

152def _discover_git() -> Optional[str]:
153 try:
154 sha = subprocess.check_output(["git", "rev-parse", "--short", "HEAD"], stderr=subprocess.DEVNULL).decode().strip()

197def _agg(runs: List[Run]) -> Agg:
198 tokens = [int(r.get("tokens_generated", 0)) for r in runs]
199 wall = [float(r.get("wall_time_s", 0.0)) for r in runs]
200 p50 = [float(r.get("latency_p50_ms", r.get("p50_ms", 0.0))) for r in runs if ("latency_p50_ms" in r or "p50_ms" in r)]
201 p95 = [float(r.get("latency_p95_ms", r.get("p95_ms", 0.0))) for r in runs if ("latency_p95_ms" in r or "p95_ms" in r)]
202
203 tokens_sum = _sum_i(tokens)
204 wall_sum = _sum_f(wall)
205 tps_true = (float(tokens_sum) / wall_sum) if wall_sum > 0 else None
206
207 rss_mean, _, rss_max = _agg_numeric(runs, "rss_peak_mb")
208 pss_mean, _, pss_max = _agg_numeric(runs, "pss_peak_mb")
209 vms_mean, _, vms_max = _agg_numeric(runs, "vms_peak_mb")
210
211 rss_floor_mean, _, rss_floor_max = _agg_numeric(runs, "rss_floor_mb")
212 rss_delta_mean, _, rss_delta_max = _agg_numeric(runs, "rss_delta_mb")
213
214 minflt_sum = _agg_sum(runs, "minflt")
215 majflt_sum = _agg_sum(runs, "majflt")
216
217 swap_in_sum_mb = _agg_sum(runs, "swap_in_mb")
218 swap_out_sum_mb = _agg_sum(runs, "swap_out_mb")
219
220 psi_some_mean, _, psi_some_max = _agg_numeric(runs, "psi_mem_some_pct")
221 psi_full_mean, _, psi_full_max = _agg_numeric(runs, "psi_mem_full_pct")
222
223 numa_loc_mean, _, numa_loc_max = _agg_numeric(runs, "numa_locality_pct")
224
225 mem_avail_mb_mean, _, _ = _agg_numeric(runs, "mem_available_mb")
226 mem_avail_pct_mean, _, _ = _agg_numeric(runs, "mem_available_pct")
227
228 kvh = [int(r.get("kv_hits", 0)) for r in runs]
229 kvm = [int(r.get("kv_misses", 0)) for r in runs]
230
231 return {
232 "runs": len(runs),
233 "tokens_sum": tokens_sum,
234 "wall_sum": wall_sum,
235 "tps_true": tps_true,
236 "p50_mean": _mean(p50),
237 "p95_mean": _mean(p95),
238
239 "rss_mean": rss_mean,
240 "rss_max": rss_max,
241
242 "pss_mean": pss_mean,
243 "pss_max": pss_max,
244 "vms_mean": vms_mean,
245 "vms_max": vms_max,
246 "rss_floor_mean": rss_floor_mean,
247 "rss_floor_max": rss_floor_max,
248 "rss_delta_mean": rss_delta_mean,
249 "rss_delta_max": rss_delta_max,
250 "minflt_sum": minflt_sum,
251 "majflt_sum": majflt_sum,
252 "swap_in_sum_mb": swap_in_sum_mb,
253 "swap_out_sum_mb": swap_out_sum_mb,
254 "psi_some_mean": psi_some_mean,
255 "psi_some_max": psi_some_max,
256 "psi_full_mean": psi_full_mean,
257 "psi_full_max": psi_full_max,
258 "numa_locality_mean": numa_loc_mean,
259 "numa_locality_max": numa_loc_max,
260 "mem_available_mb_mean": mem_avail_mb_mean,
261 "mem_available_pct_mean": mem_avail_pct_mean,
262
263 "kv_hits": _sum_i(kvh),
264 "kv_misses": _sum_i(kvm),
265 }

324def _dedup_artifact_stats(model_dir: Optional[str], model_bin_size: Optional[int]) -> Dict[str, Any]:
325 out: Dict[str, Any] = {
326 "defaults_path": None,
327 "masks_path": None,
328 "exceptions_path": None,
329 "defaults_realpath": None,
330 "masks_realpath": None,

333 "masks_size": None,
334 "exceptions_size": None,

341 defaults = os.path.join(model_dir, "model.defaults.bin")
342 masks = os.path.join(model_dir, "model.masks.bin")
343 exceptions = os.path.join(model_dir, "model.exceptions.bin")

349 def _one(p: str) -> Tuple[Optional[str], Optional[int]]:
350 try:
351 rp = os.path.realpath(p)
352 if not os.path.exists(rp):
353 return None, None
354 return rp, os.path.getsize(rp)
355 except Exception:
356 return None, None
357
358 d_rp, d_sz = _one(defaults)
359 m_rp, m_sz = _one(masks)
360 e_rp, e_sz = _one(exceptions)
361
362 out["defaults_realpath"] = d_rp
363 out["masks_realpath"] = m_rp
364 out["exceptions_realpath"] = e_rp
365
366 out["defaults_size"] = d_sz
367 out["masks_size"] = m_sz
368 out["exceptions_size"] = e_sz
369
370 sizes = [x for x in (d_sz, m_sz, e_sz) if isinstance(x, int)]
371 if sizes:
372 total = int(sum(sizes))
373 out["total_size"] = total
374 if model_bin_size and model_bin_size > 0:
375 out["ratio_vs_model_bin"] = float(total) / float(model_bin_size)
376 return out

388 def _fmt_bytes(n: Optional[int]) -> str:

416 lines.append("- Artifact paths (best effort):")

424 bt = _bytes_touched(agg["tokens_sum"], bpt)
