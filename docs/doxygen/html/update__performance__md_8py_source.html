4 """
5 Generate docs/PERFORMANCE.md from strict benchmark JSON logs.
6 
7 Behavior:
8 - Accepts CPU and/or GPU JSONL files (each produced by scripts/true_tps_strict.sh).
9 - If only one side is provided, it still renders both sections using any existing
10  counterpart JSON found under build/ (best effort).
11 - Aggregations and the comparative table use ONLY the last 3 runs for each device.
12 - "Best true TPS" is the best aggregate TPS across the currently included CPU/GPU sections.
13 - "Run Parameters & Conditions" come from CLI overrides first, then CPU summary, then GPU summary.
14 - Spatial metrics include MB/token, total bytes touched, coverage vs model.bin, and effective bandwidth.
15 - System & Model Info is discovered locally (Linux-friendly).
16 
17 New in this version:
18 - Memory-Detail metrics support (aligned with monitoring/metrics_memory.toml):
19  Optional per-run fields are aggregated when present:
20  * pss_peak_mb, vms_peak_mb
21  * rss_floor_mb, rss_delta_mb
22  * minflt, majflt
23  * swap_in_mb, swap_out_mb
24  * psi_mem_some_pct, psi_mem_full_pct (PSI averages/percentiles)
25  * numa_locality_pct
26  * mem_available_mb, mem_available_pct
27  The renderer adds a "Memory Details" subsection under each device.
28 
29 Inputs (per device JSONL):
30 - Per-run rows: include tokens_generated, wall_time_s, tps_true, latency_p50_ms, latency_p95_ms,
31  rss_peak_mb, kv_hits, kv_misses. Optionally include new memory keys above.
32 - Final summary row: includes aggregated fields and configuration (threads, precision, prompts,
33  model_dir, etc.).

201 def _agg(runs: List[Run]) -> Agg:
202  tokens = [int(r.get("tokens_generated", 0)) for r in runs]
203  wall = [float(r.get("wall_time_s", 0.0)) for r in runs]
204  p50 = [float(r.get("latency_p50_ms", r.get("p50_ms", 0.0))) for r in runs if ("latency_p50_ms" in r or "p50_ms" in r)]
205  p95 = [float(r.get("latency_p95_ms", r.get("p95_ms", 0.0))) for r in runs if ("latency_p95_ms" in r or "p95_ms" in r)]
206 
207  # Core aggregates
208  tokens_sum = _sum_i(tokens)
209  wall_sum = _sum_f(wall)
210  tps_true = (float(tokens_sum) / wall_sum) if wall_sum > 0 else None
211 
212  # Memory aggregates (optional fields)

214  pss_mean, pss_min, pss_max = _agg_numeric(runs, "pss_peak_mb")
215  vms_mean, vms_min, vms_max = _agg_numeric(runs, "vms_peak_mb")

217  rss_floor_mean, _, rss_floor_max = _agg_numeric(runs, "rss_floor_mb")
218  rss_delta_mean, _, rss_delta_max = _agg_numeric(runs, "rss_delta_mb")
219

221  majflt_sum = _agg_sum(runs, "majflt")
222

231  mem_avail_mb_mean, _, _ = _agg_numeric(runs, "mem_available_mb")

256  "swap_in_sum_mb": swap_in_sum_mb, "swap_out_sum_mb": swap_out_sum_mb,
257  "psi_some_mean": psi_some_mean, "psi_some_max": psi_some_max,
