1#!/usr/bin/env python3
2"""
3Pack a Hugging Face checkpoint directory (PyTorch shards) into the
4Inference Engine container: model.ie.json + model.ie.bin.
5
6Supports mixed precision:
7 - FP32 (default): tensors are streamed and stored as row-major FP32.
8 - INT4 weight-only: tensors are copied from pre-packed INT4 artifacts and
9 annotated with a `quant` object in JSON.
10
11Inputs
12------
13Required:
14 --hf-dir : HF directory with config.json + pytorch_model-*-of-*.bin
15 --out-dir : Output directory for model.ie.json / model.ie.bin
16
17Optional (INT4 integration):
18 --q4-map : Path to a manifest that lists INT4 artifacts per tensor.
19 Accepted formats:
20 * JSON object: { "<name>": {...}, ... }
21 * JSON list : [ {"name": "...", ...}, ... ]
22 * JSONL : one JSON object per line
23 * Directory : read all *.json files and merge
24 Each item must provide:
25 - "name": state_dict key (exact)
26 - "int4_bin" or "packed_bin": path to packed INT4 bytes
27 - "scale_bin": path to per-row or per-tensor scales
28 - "per": "row" | "tensor"
29 - "rows", "cols": stored shape of the INT4 matrix
30 Optional fields (defaults shown):
31 - "scale_dtype": "fp16"
32 - "pack": "nibble_lohi"
33 - "zp": 0
34 - "symmetric": true
35
36 --align : Byte alignment for each tensor in the aggregate bin
37 (default: 64). Use 1 to disable alignment.
38"""

73def pass1_index(hf_dir: Path) -> Tuple[List[Path], List[str], Dict[str, Tuple[Tuple[int, ...], str, int]]]:
74 """Map key -> (shape, dtype_str, shard_idx) without holding tensors in memory."""
75 shards = find_shards(hf_dir)
76 key_meta: Dict[str, Tuple[Tuple[int, ...], str, int]] = {}
77 order: List[str] = []
78 for idx, sh in enumerate(shards):
79 sd = load_state_keys_from_shard(sh)
80 for k, v in sd.items():
81 if k in key_meta:
82 continue
83 if not hasattr(v, "shape"):
84 continue
85 shape = tuple(int(x) for x in v.shape)
86 dtype = str(getattr(v, "dtype", "float32")).replace("torch.", "")
87 key_meta[k] = (shape, dtype, idx)
88 order.append(k)
89 del sd
90 gc.collect()
91 return shards, order, key_meta

107def load_q4_map(path: Path) -> Dict[str, Dict[str, Any]]:
108 """
109 Load an INT4 manifest into a dict keyed by tensor name.
110
111 Canonical item keys:
112 name (required)
113 int4_bin / packed_bin (required) -> "int4_bin"
114 scale_bin (required)
115 per ("row"|"tensor", required)
116 rows (required), cols (required)
117 scale_dtype (optional, default "fp16")
118 pack (optional, default "nibble_lohi")
119 zp (optional, default 0)
120 symmetric (optional, default True)
121 """
122 if not path.exists():
123 die(f"--q4-map not found: {path}")
124
125 items: List[Dict[str, Any]] = []
126 if path.is_dir():
127 for jp in sorted(path.glob("*.json")):
128 try:
129 obj = _read_json(jp)
130 if isinstance(obj, list):
131 items.extend(obj)
132 else:
133 items.append(obj)
134 except Exception as e:
135 die(f"failed to read {jp}: {e}")
136 else:
137 suffix = path.suffix.lower()
138 try:
139 if suffix == ".jsonl":
140 items.extend(_read_jsonl(path))
141 else:
142 obj = _read_json(path)
143 if isinstance(obj, list):
144 items.extend(obj)
145 elif isinstance(obj, dict):
146 if "name" in obj:
147 items.append(obj)
148 else:
149 for name, spec in obj.items():
150 if isinstance(spec, dict):
151 spec = dict(spec)
152 spec["name"] = name
153 items.append(spec)
154 else:
155 die(f"--q4-map has unsupported top-level type: {type(obj)}")
156 except Exception as e:
157 die(f"failed to parse --q4-map: {e}")
158
159 q4: Dict[str, Dict[str, Any]] = {}
160 for it in items:
161 if not isinstance(it, dict) or "name" not in it:
162 die("every INT4 manifest item must be an object with a 'name'")
163 name = str(it["name"])
164 spec = dict(it)
165
166 if "int4_bin" not in spec and "packed_bin" in spec:
167 spec["int4_bin"] = spec.pop("packed_bin")
168 if "int4_bin" not in spec:
169 die(f"INT4 spec for '{name}' missing 'int4_bin' (or 'packed_bin')")
170 if "scale_bin" not in spec:
171 die(f"INT4 spec for '{name}' missing 'scale_bin'")
172 if "per" not in spec:
173 die(f"INT4 spec for '{name}' missing 'per' ('row'|'tensor')")
174 if "rows" not in spec or "cols" not in spec:
175 die(f"INT4 spec for '{name}' must include 'rows' and 'cols'")
176
177 spec.setdefault("scale_dtype", "fp16")
178 spec.setdefault("pack", "nibble_lohi")
179 spec.setdefault("zp", 0)
180 spec.setdefault("symmetric", True)
181
182 # Normalize paths
183 spec["int4_bin"] = str(Path(spec["int4_bin"]).resolve())
184 spec["scale_bin"] = str(Path(spec["scale_bin"]).resolve())
185 q4[name] = spec
186
187 log(f"Loaded INT4 manifest entries: {len(q4)}")
188 return q4

190def align_up(x: int, a: int) -> int:
191 if a <= 1:
192 return x
193 return (x + (a - 1)) & ~(a - 1)

195def compute_layout(
196 order: List[str],
197 key_meta: Dict[str, Tuple[Tuple[int, ...], str, int]],
198 q4_map: Dict[str, Dict[str, Any]] | None,
199 align: int,
200) -> Tuple[List[Dict[str, Any]], int]:
201 """Compute per-tensor layout (dtype, offset, nbytes) for the aggregate bin."""
202 tensors: List[Dict[str, Any]] = []
203 offset = 0
204 for k in order:
205 shape, _dtype, _si = key_meta[k]
206
207 if q4_map and k in q4_map:
208 spec = q4_map[k]
209 rows = int(spec["rows"])
210 cols = int(spec["cols"])
211 nbytes = rows * ((cols + 1) // 2) # packed bytes: 2 values per byte
212 offset = align_up(offset, align)
213 tensors.append({
214 "name": k,
215 "shape": [rows, cols],
216 "dtype": "int4",
217 "offset": offset,
218 "nbytes": nbytes,
219 "quant": {
220 "per": str(spec["per"]),
221 "scale_bin": spec["scale_bin"],
222 "scale_dtype": str(spec.get("scale_dtype", "fp16")),
223 "pack": str(spec.get("pack", "nibble_lohi")),
224 "zp": int(spec.get("zp", 0)),
225 "symmetric": bool(spec.get("symmetric", True))
226 }
227 })
228 offset += nbytes
229 else:
230 n_elem = 1
231 for d in shape:
232 n_elem *= int(d)
233 nbytes = n_elem * 4 # float32
234 offset = align_up(offset, align)
235 tensors.append({
236 "name": k,
237 "shape": list(shape),
238 "dtype": "fp32",
239 "offset": offset,
240 "nbytes": nbytes
241 })
242 offset += nbytes
243
244 return tensors, offset

246def stream_write_blob(
247 out_bin: Path,
248 shards: List[Path],
249 order: List[str],
250 key_meta: Dict[str, Tuple[Tuple[int, ...], str, int]],
251 tensors: List[Dict[str, Any]],
252 q4_map: Dict[str, Dict[str, Any]] | None,
253 total_bytes: int,
254) -> None:
255 """Create and fill model.ie.bin with FP32 and INT4 data in a single pass."""
256 import torch
257 out_bin.parent.mkdir(parents=True, exist_ok=True)
258
259 log(f"Writing model.ie.bin (~{total_bytes / (1024**3):.2f} GiB)")
260 with open(out_bin, "wb") as fout:
261 # Pre-size (best effort)
262 try:
263 if total_bytes > 0:
264 fout.seek(total_bytes - 1)
265 fout.write(b"\0")
266 fout.flush()
267 except Exception:
268 pass
269 fout.seek(0)
270
271 # Build shard -> keys
272 shard2keys: Dict[int, List[str]] = {}
273 for k in order:
274 _shape, _dtype, si = key_meta[k]
275 shard2keys.setdefault(si, []).append(k)
276
277 # Index tensor meta by name
278 meta_by_name = {t["name"]: t for t in tensors}
279
280 # First, write all INT4 tensors by direct copy (independent of shards)
281 if q4_map:
282 for name, spec in q4_map.items():
283 if name not in meta_by_name:
284 log(f"INT4 spec name not found among HF keys, skipping: {name}")
285 continue
286 tmeta = meta_by_name[name]
287 if tmeta.get("dtype") != "int4":
288 continue
289 srcp = Path(spec["int4_bin"])
290 if not srcp.is_file():
291 die(f"INT4 bytes file not found: {srcp}")
292 expected = int(tmeta["nbytes"])
293 with open(srcp, "rb") as fsrc:
294 data = fsrc.read()
295 if len(data) != expected:
296 die(f"INT4 bytes length mismatch for '{name}': got {len(data)}, want {expected}")
297 fout.seek(int(tmeta["offset"]))
298 fout.write(data)
299
300 # Then, stream remaining FP32 tensors grouped by shard
301 for si, sh in enumerate(shards):
302 sd = load_state_keys_from_shard(sh)
303 for k in shard2keys.get(si, []):
304 tmeta = meta_by_name[k]
305 if tmeta.get("dtype") == "int4":
306 continue
307 fout.seek(int(tmeta["offset"]))
308 t = sd[k]
309 ten = t.detach().to(dtype=torch.float32, device="cpu").contiguous()
310 fout.write(ten.numpy().tobytes(order="C"))
311 del t, ten
312 del sd
313 gc.collect()
