38static void softmax_inplace_f32(float *x, size_t n) {
39 if (!x || n == 0) return;
40
41 float mx = x[0];
42 for (size_t i = 1; i < n; ++i) {
43 if (x[i] > mx) mx = x[i];
44 }
45
46 float sum = 0.0f;
47 for (size_t i = 0; i < n; ++i) {
48 float v = expf(x[i] - mx);
49 x[i] = v;
50 sum += v;
51 }
52
53 if (sum > 0.0f) {
54 const float inv = 1.0f / sum;

56 } else {
57 const float inv = 1.0f / (float)n;
58 for (size_t i = 0; i < n; ++i) x[i] = inv;
59 }
60}
