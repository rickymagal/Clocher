196static bool read_file(const std::string& path, std::vector<uint8_t>& out) {
197 std::ifstream f(path, std::ios::binary);
198 if (!f) return false;
199 f.seekg(0, std::ios::end);
200 std::streampos sz = f.tellg();
201 if (sz <= 0) return false;
202 out.resize((size_t)sz);
203 f.seekg(0, std::ios::beg);
204 f.read(reinterpret_cast<char*>(out.data()), sz);
205 return f.good();
206}

215static int ensure_ze_ready() {
216 if (g_ctx && g_dev && g_q && g_cl) return 0;
217
218 ZE_GUARD(zeInit(0), "zeInit failed");
219
220 uint32_t nd = 0;
221 ZE_GUARD(zeDriverGet(&nd, nullptr), "zeDriverGet(count) failed");
222 if (nd == 0) { set_err("no Level Zero drivers found"); return -1; }
223 std::vector<ze_driver_handle_t> drivers(nd);
224 ZE_GUARD(zeDriverGet(&nd, drivers.data()), "zeDriverGet(list) failed");
225
226 ze_driver_handle_t drv = drivers[0];
227
228 uint32_t ndev = 0;
229 ZE_GUARD(zeDeviceGet(drv, &ndev, nullptr), "zeDeviceGet(count) failed");
230 if (ndev == 0) { set_err("no Level Zero devices found"); return -1; }
231 std::vector<ze_device_handle_t> devs(ndev);
232 ZE_GUARD(zeDeviceGet(drv, &ndev, devs.data()), "zeDeviceGet(list) failed");
233
234 /* pick first GPU if possible */
235 for (auto d : devs) {
236 ze_device_properties_t p = {ZE_STRUCTURE_TYPE_DEVICE_PROPERTIES};
237 zeDeviceGetProperties(d, &p);
238 if (p.type == ZE_DEVICE_TYPE_GPU) { g_dev = d; break; }
239 }
240 if (!g_dev) g_dev = devs[0]; /* fallback */
241
242 ze_context_desc_t cdesc = {ZE_STRUCTURE_TYPE_CONTEXT_DESC};
243 ZE_GUARD(zeContextCreate(drv, &cdesc, &g_ctx), "zeContextCreate failed");
244
245 /* find a compute queue group ordinal */
246 uint32_t ng = 0;
247 zeDeviceGetCommandQueueGroupProperties(g_dev, &ng, nullptr);
248 std::vector<ze_command_queue_group_properties_t> gps(
249 ng, {ZE_STRUCTURE_TYPE_COMMAND_QUEUE_GROUP_PROPERTIES});
250 zeDeviceGetCommandQueueGroupProperties(g_dev, &ng, gps.data());
251 uint32_t ord = 0;
252 for (uint32_t i = 0; i < ng; ++i) {
253 if (gps[i].flags & ZE_COMMAND_QUEUE_GROUP_PROPERTY_FLAG_COMPUTE) {
254 ord = i; break;
255 }
256 }
257
258 ze_command_queue_desc_t qd = {
259 ZE_STRUCTURE_TYPE_COMMAND_QUEUE_DESC, /* stype */
260 nullptr, /* pNext */
261 ord, /* ordinal */
262 0, /* index */
263 ZE_COMMAND_QUEUE_MODE_DEFAULT,
264 ZE_COMMAND_QUEUE_PRIORITY_NORMAL
265 };
266 ZE_GUARD(zeCommandQueueCreate(g_ctx, g_dev, &qd, &g_q), "zeCommandQueueCreate failed");
267
268 /* Immediate command list to reduce submission overhead. */
269 ze_command_list_desc_t ld = {
270 ZE_STRUCTURE_TYPE_COMMAND_LIST_DESC, nullptr, ord, 0
271 };
272 ZE_GUARD(zeCommandListCreateImmediate(g_ctx, g_dev, &ld, &g_cl),
273 "zeCommandListCreateImmediate failed");
274
275 return 0;
276}
