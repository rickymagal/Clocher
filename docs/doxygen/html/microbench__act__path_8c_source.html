32static void fill_gaussian(float* x, size_t n, unsigned seed) {
33 /* Boxâ€“Muller transform: two uniforms -> two normals */
34 srand(seed);
35 for (size_t i = 0; i + 1 < n; i += 2) {
36 float u1 = (float)rand() / (float)RAND_MAX;
37 float u2 = (float)rand() / (float)RAND_MAX;
38 float r = sqrtf(-2.0f * logf(1.0f - u1));
39 float th = 2.0f * (float)M_PI * u2;
40 x[i] = r * cosf(th);
41 x[i+1] = r * sinf(th);
42 }
43 if (n & 1) x[n-1] = 0.0f;
44}
