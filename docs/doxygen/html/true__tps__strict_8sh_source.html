144cpu_guard_preflight() {
145 if [[ "${DEVICE}" != "cpu" ]]; then
146 return 0
147 fi
148 if [[ "${STRICT_CPU_GUARD}" == "0" ]]; then
149 echo "[strict] WARN: STRICT_CPU_GUARD=0; skipping CPU preflight" >&2
150 return 0
151 fi
152 if ! command -v strace >/dev/null 2>&1; then
153 echo "[strict] WARN: strace not available; skipping CPU preflight" >&2
154 return 0
155 fi
156
157 local trace="${_tmpdir}/cpu_preflight.trace"
158 local out="${_tmpdir}/cpu_preflight.out"
159
160 set +e
161 strace -f -e trace=openat,mmap,read -o "${trace}" \
162 "${ENGINE_BIN}" \
163 --device cpu \
164 --model-dir "${MODEL_DIR}" \
165 --prompts-file "${PROMPTS}" \
166 --threads 1 \
167 --precision "${CLI_PREC}" \
168 --batch 1 \
169 --prefetch "${PREFETCH}" \
170 --pretranspose "${PRETRANSPOSE}" \
171 --affinity "${AFFINITY}" \
172 --max-new 1 \
173 --rounds 1 > "${out}" 2>&1
174 local rc=$?
175 set -e
176
177 # Extract FD(s) from: openat(... "model.ie.bin" ...) = <fd>
178 # POSIX awk only (no match(..., ..., arr)).
179 local fds
180 fds="$(awk '
181 /openat\‍(/ && /model\.ie\.bin"/ {
182 n=split($0, parts, "=")
183 if (n < 2) next
184 fd=parts[n]
185 gsub(/^[[:space:]]+/, "", fd)
186 gsub(/[[:space:]]+$/, "", fd)
187 if (fd ~ /^[0-9]+$/) print fd
188 }
189 ' "${trace}" | sort -n | uniq | tr '\n' ' ')"
190
191 if [[ -z "${fds// }" ]]; then
192 echo "[strict] ERROR: CPU preflight did not open model.ie.bin (trace=${trace}, out=${out}, rc=${rc})" >&2
193 exit 91
194 fi
195
196 local ok=0
197 for fd in ${fds}; do
198 if grep -Eq "mmap\\‍(.*,[[:space:]]*${fd}[[:space:]]*,[[:space:]]*[0-9]+" "${trace}" || \
199 grep -Eq "read\\‍(${fd}," "${trace}"; then
200 ok=1
201 break
202 fi
203 done
204
205 if [[ "${ok}" != "1" ]]; then
206 echo "[strict] ERROR: CPU preflight opened model.ie.bin but did not mmap/read it (fds=${fds})" >&2
207 echo "[strict] DEBUG: trace=${trace} out=${out} rc=${rc}" >&2
208 exit 92
209 fi
210
211 echo "[strict] CPU preflight: model.ie.bin mmap/read detected (OK)" >&2
212 return 0
213}
