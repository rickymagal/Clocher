2"""
3Prepare inputs for the lossless byte-level dedup extractor.
4
5This script converts:
6 - an IE weight manifest (model.ie*.json) into tensor_map.for_dedup.json
7 - a dedup pairing manifest (manifest.dedup.json) into groups.for_dedup.json
8
9It is intentionally tolerant of multiple IE JSON layouts:
10 - weights_bin vs bin
11 - tensors as a list vs a dict keyed by tensor name
12 - dtype spelled as torch.* or short forms (BF16/F16/F32/U8/I32)
13
14Outputs match tools/dedup_extract_int4.py:
15 groups: [{"default_index": int, "targets": [int, ...]}, ...]
16 tensor_map: {"weights_bin": "file.bin", "tensors": [{"index": i, "name": ..., "offset": ..., "nbytes": ...}, ...]}
17"""

54def _as_list_tensors(ie: Dict[str, Any]) -> Tuple[str, List[Dict[str, Any]]]:
55 """
56 Return (weights_bin_rel, tensors_list) where each tensor has:
57 name, dtype, shape, offset, nbytes
58 """
59 weights_bin = ie.get("weights_bin") or ie.get("bin")
60 if not isinstance(weights_bin, str) or not weights_bin:
61 weights_bin = "model.ie.bin"
62
63 tensors = ie.get("tensors")
64 if tensors is None:
65 raise SystemExit("ERROR: IE json missing 'tensors'")
66
67 out: List[Dict[str, Any]] = []
68 if isinstance(tensors, list):
69 for t in tensors:
70 if not isinstance(t, dict):
71 continue
72 out.append(dict(t))
73 elif isinstance(tensors, dict):
74 for name, meta in tensors.items():
75 if not isinstance(meta, dict):
76 continue
77 d = dict(meta)
78 d.setdefault("name", name)
79 out.append(d)
80 else:
81 raise SystemExit(f"ERROR: unsupported tensors container type: {type(tensors)}")
82
83 norm: List[Dict[str, Any]] = []
84 for t in out:
85 name = t.get("name")
86 dtype = t.get("dtype")
87 shape = t.get("shape")
88 offset = t.get("offset")
89 nbytes = t.get("nbytes")
90
91 if not isinstance(name, str) or not name:
92 continue
93 if not isinstance(dtype, str) or not dtype:
94 raise SystemExit(f"ERROR: tensor '{name}' missing dtype")
95 if not isinstance(shape, list) or not all(isinstance(x, int) for x in shape):
96 raise SystemExit(f"ERROR: tensor '{name}' has invalid shape")
97 if not isinstance(offset, int):
98 raise SystemExit(f"ERROR: tensor '{name}' missing offset")
99
100 if not isinstance(nbytes, int):
101 esz = _dtype_size(dtype)
102 elems = 1
103 for d in shape:
104 elems *= int(d)
105 nbytes = elems * esz
106
107 norm.append(
108 {
109 "name": name,
110 "dtype": dtype,
111 "shape": shape,
112 "offset": int(offset),
113 "nbytes": int(nbytes),
114 }
115 )
116
117 norm.sort(key=lambda x: (x["offset"], x["name"]))
118 for i, t in enumerate(norm):
119 t["index"] = i
120 return weights_bin, norm

143def _make_virtual_slices(base: Dict[str, Any], slice_count: int) -> List[Dict[str, Any]]:
144 """
145 Create virtual tensors representing slicing along dimension 0 of a fused tensor.
146
147 Slice size is derived from nbytes/slice_count (byte-accurate).
148 Slice shape is base.shape[1:].
149 """
150 base_shape = base["shape"]
151 if not base_shape or not isinstance(base_shape[0], int):
152 raise SystemExit(f"ERROR: base tensor '{base['name']}' has invalid shape")
153 if base_shape[0] != slice_count:
154 raise SystemExit(
155 f"ERROR: fused tensor '{base['name']}' shape[0]={base_shape[0]} "
156 f"does not match expected expert_count={slice_count}"
157 )
158
159 total_bytes = int(base["nbytes"])
160 if total_bytes % slice_count != 0:
161 raise SystemExit(
162 f"ERROR: fused tensor '{base['name']}' nbytes={total_bytes} is not divisible by expert_count={slice_count}"
163 )
164 slice_nbytes = total_bytes // slice_count
165
166 slice_shape = list(base_shape[1:])
167 if not slice_shape:
168 slice_shape = [1]
169
170 slices: List[Dict[str, Any]] = []
171 for i in range(slice_count):
172 slices.append(
173 {
174 "name": f"{base['name']}#slice{i}",
175 "dtype": base["dtype"],
176 "shape": slice_shape,
177 "offset": int(base["offset"]) + i * slice_nbytes,
178 "nbytes": slice_nbytes,
179 }
180 )
181 return slices

184def _prepare_groups_and_tensor_map(
185 ie_json_path: str,
186 manifest_path: str,
187 out_tensor_map_path: str,
188 out_groups_path: str,
189) -> None:
190 ie = _load_json(ie_json_path)
191 weights_bin, base_tensors = _as_list_tensors(ie)
192
193 name_to_index = _build_name_to_index(base_tensors)
194 compat_tensors = list(base_tensors)
195 compat_name_to_index = dict(name_to_index)
196
197 out_groups: List[Dict[str, Any]] = []
198
199 man = _load_json(manifest_path)
200 groups = man.get("groups", man)
201 if not isinstance(groups, list):
202 raise SystemExit("ERROR: manifest is neither a list nor an object with 'groups'")
203
204 def ensure_virtual_slice(base_name: str, slice_idx: int, expected_slices: int) -> int:
205 vname = f"{base_name}#slice{slice_idx}"
206 if vname in compat_name_to_index:
207 return compat_name_to_index[vname]
208 base_idx = name_to_index.get(base_name)
209 if base_idx is None:
210 raise SystemExit(f"ERROR: fused tensor '{base_name}' not found in IE json")
211 base = base_tensors[base_idx]
212 new_slices = _make_virtual_slices(base, expected_slices)
213 start_index = len(compat_tensors)
214 for j, s in enumerate(new_slices):
215 s["index"] = start_index + j
216 compat_tensors.append(s)
217 compat_name_to_index[s["name"]] = s["index"]
218 return compat_name_to_index[vname]
219
220 for g in groups:
221 if not isinstance(g, dict):
222 continue
223 kind = g.get("kind")
224 if kind == "kv_pair":
225 members = g.get("members")
226 if not isinstance(members, list) or len(members) < 2:
227 continue
228 member_indices: List[int] = []
229 for name in members:
230 if not isinstance(name, str):
231 continue
232 idx = name_to_index.get(name)
233 if idx is None:
234 raise SystemExit(f"ERROR: kv_pair member '{name}' not found in IE json")
235 member_indices.append(idx)
236 if len(member_indices) >= 2:
237 out_groups.append({"default_index": member_indices[0], "targets": member_indices[1:]})
238
239 elif kind == "expert_pair":
240 fused_name = g.get("fused_tensor")
241 expert_indices = g.get("expert_indices")
242 if not isinstance(fused_name, str) or not fused_name:
243 continue
244 if not isinstance(expert_indices, list) or len(expert_indices) < 2:
245 continue
246 base_idx = name_to_index.get(fused_name)
247 if base_idx is None:
248 raise SystemExit(f"ERROR: expert_pair fused_tensor '{fused_name}' not found in IE json")
249 base = base_tensors[base_idx]
250 expert_count = int(base["shape"][0])
251
252 slice_indices: List[int] = []
253 for ei in expert_indices:
254 if not isinstance(ei, int):
255 continue
256 slice_indices.append(ensure_virtual_slice(fused_name, ei, expert_count))
257 if len(slice_indices) >= 2:
258 out_groups.append({"default_index": slice_indices[0], "targets": slice_indices[1:]})
259
260 tensor_map = {
261 "weights_bin": weights_bin,
262 "tensors": [
263 {
264 "index": int(t["index"]),
265 "name": t["name"],
266 "offset": int(t["offset"]),
267 "nbytes": int(t["nbytes"]),
268 }
269 for t in compat_tensors
270 ],
271 }
272
273 _save_json(out_tensor_map_path, tensor_map)
274 _save_json(out_groups_path, out_groups)
275
276 print(f"Wrote: {out_tensor_map_path}")
277 print(f"Wrote: {out_groups_path}")
278 print(f"Tensors: base={len(base_tensors)} total(with slices)={len(compat_tensors)} groups={len(out_groups)}")
