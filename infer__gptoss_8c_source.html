761static void ie_softmax_inplace_f32(float *x, size_t n) {
762 if (!x || n == 0u) return;
763
764 float m = x[0];
765 for (size_t i = 1; i < n; ++i)
766 if (x[i] > m) m = x[i];
767
768 float sum = 0.0f;
769 for (size_t i = 0; i < n; ++i) {
770 x[i] = expf(x[i] - m);
771 sum += x[i];
772 }
773
774 const float inv = (sum > 0.0f) ? (1.0f / sum) : 1.0f;
775 for (size_t i = 0; i < n; ++i) x[i] *= inv;
776}

955IE_WEAK int ie_rope_apply_f32(float *q, float *k, size_t heads, size_t head_dim, uint32_t pos,
956 float theta) {
957 if (head_dim == 0u || (head_dim & 1u) != 0u) return -1;
958 if (!q && !k) return 0;
959
960 const size_t half = head_dim / 2u;
961 const float log_theta = logf(theta);
962
963 for (size_t h = 0; h < heads; ++h) {
964 float *vq = q ? (q + h * head_dim) : NULL;
965 float *vk = k ? (k + h * head_dim) : NULL;
966
967 for (size_t i = 0; i < half; ++i) {
968 const float exponent = -2.0f * (float)i / (float)head_dim;
969 const float inv_freq = expf(log_theta * exponent);
970 const float ang = (float)pos * inv_freq;
971 const float c = cosf(ang);
972 const float s = sinf(ang);
973
974 const size_t i0 = 2u * i;
975 const size_t i1 = i0 + 1u;
976
977 if (vq) {
978 const float x0 = vq[i0];
979 const float x1 = vq[i1];
980 vq[i0] = x0 * c - x1 * s;
981 vq[i1] = x0 * s + x1 * c;
982 }
983
984 if (vk) {
985 const float x0 = vk[i0];
986 const float x1 = vk[i1];
987 vk[i0] = x0 * c - x1 * s;
988 vk[i1] = x0 * s + x1 * c;
989 }
990 }
991 }
992
993 return 0;
994}

1000IE_WEAK int ie_attn_cpu_causal_f32(const float *Q, const float *K, const float *V,
1001 size_t seq_len, size_t n_heads, size_t head_dim,
1002 float inv_sqrt_d, float *out, float *scores) {
1003 if (!Q || !K || !V || !out || !scores) return -1;
1004 if (seq_len == 0u || n_heads == 0u || head_dim == 0u) return -2;
1005
1006 for (size_t h = 0; h < n_heads; ++h) {
1007 const float *Qh = Q + h * head_dim;
1008
1009 for (size_t t = 0; t < seq_len; ++t) {
1010 const size_t base = (t * n_heads + h) * head_dim;
1011 const float *Kh = K + base;
1012 scores[t] = ie_dot_f32(Qh, Kh, head_dim) * inv_sqrt_d;
1013 }
1014
1015 ie_softmax_inplace_f32(scores, seq_len);
1016
1017 float *Oh = out + h * head_dim;
1018 for (size_t d = 0; d < head_dim; ++d) Oh[d] = 0.0f;
1019
1020 for (size_t t = 0; t < seq_len; ++t) {
1021 const float w = scores[t];
1022 const size_t base = (t * n_heads + h) * head_dim;
1023 const float *Vh = V + base;
1024 for (size_t d = 0; d < head_dim; ++d) Oh[d] += w * Vh[d];
1025 }
1026 }
1027
1028 return 0;
1029}

1032 uint32_t n_heads, uint32_t n_kv_heads, uint32_t head_dim,
1033 float inv_sqrt_d, float *out, float *scores) {
1034 const uint32_t group_size = (n_kv_heads > 0) ? (n_heads / n_kv_heads) : 0;
1035
1036 for (uint32_t hq = 0; hq < n_heads; ++hq) {
1037 const uint32_t hk = (group_size > 0) ? (hq / group_size) : 0;
1038 const float *Qh = Q + (size_t)hq * (size_t)head_dim;
1039
1040 for (uint32_t t = 0; t < seq_len; ++t) {
1041 const size_t base =
1042 ((size_t)t * (size_t)n_kv_heads + (size_t)hk) * (size_t)head_dim;
1043 const float *Kh = K + base;
1044 scores[t] = ie_dot_f32(Qh, Kh, (size_t)head_dim) * inv_sqrt_d;
1045 }
1046
1047 ie_softmax_inplace_f32(scores, (size_t)seq_len);
1048
1049 float *Oh = out + (size_t)hq * (size_t)head_dim;
1050 for (uint32_t d = 0; d < head_dim; ++d) Oh[d] = 0.0f;
1051
1052 for (uint32_t t = 0; t < seq_len; ++t) {
1053 const float w = scores[t];
1054 const size_t base =
1055 ((size_t)t * (size_t)n_kv_heads + (size_t)hk) * (size_t)head_dim;
1056 const float *Vh = V + base;
1057 for (uint32_t d = 0; d < head_dim; ++d) Oh[d] += w * Vh[d];
1058 }
1059 }
1060}

1173 }

1175 if (ie_w_get_layer_fmt_view(impl, l, gu_blocks_fmts, 1, &v_gu_blocks, 1) != 0) return -6;
1176 if (ie_w_get_layer_fmt_view(impl, l, gu_scales_fmts, 1, &v_gu_scales, 1) != 0) return -7;

1186 }

1190 else if (v_gu_scales.desc->dtype == TENSOR_DTYPE_U8) gu_scale_bytes = 1u;
1191

1198 else if (v_gu_scales.desc->size_bytes == tmp_scales_2) gu_scale_bytes = 2u;
1199 else {
1200 IE_LOGE("moe: cannot infer gate_up scale_bytes layer=%u size_bytes=%llu (want %llu or %llu)",
1201 (unsigned)l,
1202 (unsigned long long)v_gu_scales.desc->size_bytes,
1203 (unsigned long long)tmp_scales_1,
1204 (unsigned long long)tmp_scales_2);
1205 return -11;
1206 }
1207 }
1208
1209 uint8_t dn_scale_bytes = 0u;
1210 if (v_dn_scales.desc->dtype == TENSOR_DTYPE_BF16) dn_scale_bytes = 2u;
1211 else if (v_dn_scales.desc->dtype == TENSOR_DTYPE_U8) dn_scale_bytes = 1u;
1212
1213 if (dn_scale_bytes == 0u) {
1214 uint64_t tmp_blocks = 0, tmp_scales_1 = 0, tmp_scales_2 = 0;
1215 if (ie_moe_expected_bytes(n_experts, d_model, d_ff, 1u, &tmp_blocks, &tmp_scales_1) != 0) return -12;
1216 if (ie_moe_expected_bytes(n_experts, d_model, d_ff, 2u, &tmp_blocks, &tmp_scales_2) != 0) return -12;
1217
1218 if (v_dn_scales.desc->size_bytes == tmp_scales_1) dn_scale_bytes = 1u;

1223 (unsigned long long)v_dn_scales.desc->size_bytes,
1224 (unsigned long long)tmp_scales_1,
1225 (unsigned long long)tmp_scales_2);

1228 }

1231 if (ie_moe_expected_bytes(n_experts, 2u * d_ff, d_model, gu_scale_bytes, &want_gu_blocks, &want_gu_scales) != 0)
1232 return -11;
1233
1234 uint64_t want_dn_blocks = 0, want_dn_scales = 0;
1235 if (ie_moe_expected_bytes(n_experts, d_model, d_ff, dn_scale_bytes, &want_dn_blocks, &want_dn_scales) != 0)
1236 return -12;
1237
1238 if (ie_require_size_eq(v_gu_blocks.desc, want_gu_blocks) != 0) return -13;
1239 if (ie_require_size_eq(v_gu_scales.desc, want_gu_scales) != 0) return -14;
1240 if (ie_require_size_eq(v_dn_blocks.desc, want_dn_blocks) != 0) return -15;
1241 if (ie_require_size_eq(v_dn_scales.desc, want_dn_scales) != 0) return -16;
1242
1243 if (v_gu_bias.desc) {
1244 if (ie_require_size_eq(v_gu_bias.desc,
1245 (uint64_t)n_experts * (uint64_t)(2u * d_ff) * 2u) != 0)
1246 return -17;
1247 }
1248 if (v_dn_bias.desc) {
1249 if (ie_require_size_eq(v_dn_bias.desc, (uint64_t)n_experts * (uint64_t)d_model * 2u) != 0)
1250 return -18;
1251 }
1252
1253 LW->moe.n_experts = n_experts;
1254 LW->moe.router_w = (const uint16_t *)v_router_w.ptr;
1255 LW->moe.router_b = (const uint16_t *)v_router_b.ptr;
1256 LW->moe.router_transposed = 0;
1257
1258 LW->moe.gate_up_blocks = (const uint8_t *)v_gu_blocks.ptr;
1259 LW->moe.gate_up_scales = (const uint8_t *)v_gu_scales.ptr;
1260 LW->moe.gate_up_scale_bytes = gu_scale_bytes;
1261 LW->moe.gate_up_bias = (const uint16_t *)v_gu_bias.ptr;
1262
1263 LW->moe.down_blocks = (const uint8_t *)v_dn_blocks.ptr;
1264 LW->moe.down_scales = (const uint8_t *)v_dn_scales.ptr;
1265 LW->moe.down_scale_bytes = dn_scale_bytes;
1266 LW->moe.down_bias = (const uint16_t *)v_dn_bias.ptr;
1267
1268 LW->moe.gate_up_blocks_stride = (size_t)(v_gu_blocks.desc->size_bytes / n_experts);
1269 LW->moe.gate_up_scales_stride = (size_t)(v_gu_scales.desc->size_bytes / n_experts);
1270 LW->moe.gate_up_bias_stride = v_gu_bias.desc ? (size_t)(v_gu_bias.desc->size_bytes / n_experts) : 0u;
1271
1272 LW->moe.down_blocks_stride = (size_t)(v_dn_blocks.desc->size_bytes / n_experts);
1273 LW->moe.down_scales_stride = (size_t)(v_dn_scales.desc->size_bytes / n_experts);
1274 LW->moe.down_bias_stride = v_dn_bias.desc ? (size_t)(v_dn_bias.desc->size_bytes / n_experts) : 0u;
1275
1276 if (impl->max_experts < n_experts) impl->max_experts = n_experts;
1277
1278 IE_LOGI("moe: layer=%u n_experts=%u gu_scale_bytes=%u dn_scale_bytes=%u",
1279 (unsigned)l, (unsigned)n_experts, (unsigned)gu_scale_bytes, (unsigned)dn_scale_bytes);
1280
1281 return 0;
1282}

1359static int ie_forward_one_token(struct ie_gptoss_infer_impl *impl, ie_kv_cache *kv_layers,
1360 uint32_t token_id, uint32_t pos, float *out_logits) {
1361 static int bf16_trace_done = 0;
1362 if (!impl || !kv_layers || !out_logits) return -1;
1363
1364 const ie_gptoss_hparams_t *hp = impl->hp;
1365 if (!hp) return -2;
1366
1367 const uint32_t d_model = hp->d_model;
1368 const uint32_t d_head = hp->d_head;
1369 const uint32_t n_heads = hp->n_heads;
1370 const uint32_t n_kv_heads = hp->n_kv_heads;
1371 const uint32_t vocab = hp->vocab_size;
1372 const uint32_t n_layers = hp->n_layers;
1373
1374 if (!impl->w_embed_bf16 || !impl->w_norm_bf16 || !impl->w_lm_bf16) return -3;
1375 if (token_id >= vocab) return -4;
1376 if (pos >= hp->max_seq) return -5;
1377 if ((d_head & 1u) != 0u) return -6;
1378
1379 {
1380 const uint16_t *emb = impl->w_embed_bf16 + (size_t)token_id * (size_t)d_model;
1381 for (uint32_t i = 0; i < d_model; ++i) impl->x[i] = ie_bf16_to_f32(emb[i]);
1382 }
1383
1384 const float inv_sqrt_d = 1.0f / sqrtf((float)d_head);
1385 const size_t q_dim = (size_t)n_heads * (size_t)d_head;
1386 const size_t kv_dim = (size_t)n_kv_heads * (size_t)d_head;
1387 const size_t d_ff = (size_t)hp->d_ff;
1388
1389 const float rope_theta_eff = impl->rope_theta * impl->rope_scale;
1390
1391 for (uint32_t l = 0; l < n_layers; ++l) {
1392 const ie_gptoss_layer_w_t *W = &impl->layers[l];
1393 ie_kv_cache *kv = &kv_layers[l];
1394
1395 if (kv->storage != IE_KV_STORAGE_F32) {
1396 IE_LOGE("kv storage mismatch: layer=%u storage=%d (want %d)",
1397 (unsigned)l, (int)kv->storage, (int)IE_KV_STORAGE_F32);
1398 return -7;
1399 }
1400 if ((uint32_t)kv->heads != n_kv_heads) {
1401 IE_LOGE("kv heads mismatch: layer=%u heads=%zu want=%u",
1402 (unsigned)l, kv->heads, (unsigned)n_kv_heads);
1403 return -8;
1404 }
1405 if ((uint32_t)kv->head_dim != d_head) {
1406 IE_LOGE("kv head_dim mismatch: layer=%u head_dim=%zu want=%u",
1407 (unsigned)l, kv->head_dim, (unsigned)d_head);
1408 return -9;
1409 }
1410
1411 /* Allow KV max_seq to be smaller than hp->max_seq: normal short-context inference. */
1412 if ((uint32_t)kv->max_seq == 0u) {
1413 IE_LOGE("kv max_seq invalid: layer=%u max_seq=%zu", (unsigned)l, kv->max_seq);
1414 return -10;
1415 }
1416 if ((uint32_t)kv->max_seq > hp->max_seq) {
1417 IE_LOGE("kv max_seq exceeds model max_seq: layer=%u max_seq=%zu hp_seq=%u",
1418 (unsigned)l, kv->max_seq, (unsigned)hp->max_seq);
1419 return -10;
1420 }
1421 if (pos >= (uint32_t)kv->max_seq) {
1422 IE_LOGE("kv max_seq too small for pos: layer=%u pos=%u max_seq=%zu",
1423 (unsigned)l, (unsigned)pos, kv->max_seq);
1424 return -10;
1425 }
1426
1427 if (W->ln1_w_f32) {
1428 if (ie_rmsnorm_cpu_f32(impl->x, W->ln1_w_f32, (size_t)d_model, impl->rms_eps, impl->x1) != 0) {
1429 IE_LOGE("rmsnorm1 failed: layer=%u pos=%u", (unsigned)l, (unsigned)pos);
1430 return -11;
1431 }
1432 } else if (ie_rmsnorm_cpu_f32_bf16w(impl->x, W->ln1_w, (size_t)d_model, impl->rms_eps, impl->x1) != 0) {
1433 IE_LOGE("rmsnorm1 failed: layer=%u pos=%u", (unsigned)l, (unsigned)pos);
1434 return -11;
1435 }
1436 if (ie_debug_nan_enabled_() &&
1437 ie_check_finite_f32_(impl->x1, (size_t)d_model, "rmsnorm1", l, pos) != 0) {
1438 return -90;
1439 }
1440
1441 const int trace_on = ie_trace_bf16_enabled_() && !bf16_trace_done;
1442 if (trace_on && !W->q_q4_blocks && !W->q_w_f32) {
1443 fprintf(stderr, "[TRACE] bf16 gemv: layer=%u op=q rows=%zu cols=%zu\n",
1444 (unsigned)l, q_dim, (size_t)d_model);
1445 }
1446 if (W->q_q4_blocks && W->q_q4_scales) {
1447 if (ie_gemv_q4_0_f32(W->q_q4_blocks, W->q_q4_scales, W->q_q4_scale_bytes,
1448 impl->x1, impl->q, q_dim, (size_t)d_model, W->q_b) != 0) {
1449 return -12;
1450 }
1451 if (!W->q_b && W->q_b_f32) ie_add_bias_f32(impl->q, W->q_b_f32, q_dim);
1452 } else if (W->q_w_f32) {
1453 ie_gemv_f32(W->q_w_f32, impl->x1, impl->q, q_dim, (size_t)d_model, W->q_b_f32, 1);
1454 } else {
1455 if (ie_gemv_bf16_f32(W->q_w, impl->x1, impl->q, q_dim, (size_t)d_model, W->q_b) != 0) return -12;
1456 }
1457 if (trace_on && !W->k_q4_blocks && !W->k_w_f32) {
1458 fprintf(stderr, "[TRACE] bf16 gemv: layer=%u op=k rows=%zu cols=%zu\n",
1459 (unsigned)l, kv_dim, (size_t)d_model);
1460 }
1461 if (W->k_q4_blocks && W->k_q4_scales) {
1462 if (ie_gemv_q4_0_f32(W->k_q4_blocks, W->k_q4_scales, W->k_q4_scale_bytes,
1463 impl->x1, impl->k, kv_dim, (size_t)d_model, W->k_b) != 0) {
1464 return -13;
1465 }
1466 if (!W->k_b && W->k_b_f32) ie_add_bias_f32(impl->k, W->k_b_f32, kv_dim);
1467 } else if (W->k_w_f32) {
1468 ie_gemv_f32(W->k_w_f32, impl->x1, impl->k, kv_dim, (size_t)d_model, W->k_b_f32, 1);
1469 } else {
1470 if (ie_gemv_bf16_f32(W->k_w, impl->x1, impl->k, kv_dim, (size_t)d_model, W->k_b) != 0) return -13;
1471 }
1472 if (trace_on && !W->v_q4_blocks && !W->v_w_f32) {
1473 fprintf(stderr, "[TRACE] bf16 gemv: layer=%u op=v rows=%zu cols=%zu\n",
1474 (unsigned)l, kv_dim, (size_t)d_model);
1475 }
1476 if (W->v_q4_blocks && W->v_q4_scales) {
1477 if (ie_gemv_q4_0_f32(W->v_q4_blocks, W->v_q4_scales, W->v_q4_scale_bytes,
1478 impl->x1, impl->v, kv_dim, (size_t)d_model, W->v_b) != 0) {
1479 return -14;
1480 }
1481 if (!W->v_b && W->v_b_f32) ie_add_bias_f32(impl->v, W->v_b_f32, kv_dim);
1482 } else if (W->v_w_f32) {
1483 ie_gemv_f32(W->v_w_f32, impl->x1, impl->v, kv_dim, (size_t)d_model, W->v_b_f32, 1);
1484 } else {
1485 if (ie_gemv_bf16_f32(W->v_w, impl->x1, impl->v, kv_dim, (size_t)d_model, W->v_b) != 0) return -14;
1486 }
1487
1488 if (ie_rope_apply_f32(impl->q, NULL, (size_t)n_heads, (size_t)d_head, pos, rope_theta_eff) != 0) return -15;
1489 if (ie_rope_apply_f32(NULL, impl->k, (size_t)n_kv_heads, (size_t)d_head, pos, rope_theta_eff) != 0) return -16;
1490
1491 if (ie_kv_store_token_f32(kv, (size_t)pos, impl->k, impl->v) != 0) {
1492 IE_LOGE("kv store failed: layer=%u pos=%u", (unsigned)l, (unsigned)pos);
1493 return -17;
1494 }
1495
1496 {
1497 /* KV reuse instrumentation: per layer, this token attends over (pos) cached tokens. */
1498 ie_kv_add_hits((uint64_t)pos);
1499 ie_kv_add_misses(1u);
1500
1501 const float *Kbase = (const float *)kv->K;
1502 const float *Vbase = (const float *)kv->V;
1503
1504 if (n_kv_heads == n_heads) {
1505 if (ie_attn_cpu_causal_f32(impl->q, Kbase, Vbase, (size_t)(pos + 1u),
1506 (size_t)n_heads, (size_t)d_head, inv_sqrt_d,
1507 impl->attn_out, impl->scores) != 0) {
1508 IE_LOGE("attn kernel failed: layer=%u pos=%u", (unsigned)l, (unsigned)pos);
1509 return -18;
1510 }
1511 } else {
1512 ie_attn_causal_gqa_f32(impl->q, Kbase, Vbase, pos + 1u, n_heads, n_kv_heads, d_head,
1513 inv_sqrt_d, impl->attn_out, impl->scores);
1514 }
1515 }
1516
1517 if (trace_on && !W->o_q4_blocks && !W->o_w_f32) {
1518 fprintf(stderr, "[TRACE] bf16 gemv: layer=%u op=o rows=%zu cols=%zu\n",
1519 (unsigned)l, (size_t)d_model, q_dim);
1520 }
1521 if (W->o_q4_blocks && W->o_q4_scales) {
1522 if (ie_gemv_q4_0_f32(W->o_q4_blocks, W->o_q4_scales, W->o_q4_scale_bytes,
1523 impl->attn_out, impl->x2, (size_t)d_model, q_dim, W->o_b) != 0) {
1524 return -19;
1525 }
1526 if (!W->o_b && W->o_b_f32) ie_add_bias_f32(impl->x2, W->o_b_f32, (size_t)d_model);
1527 } else if (W->o_w_f32) {
1528 ie_gemv_f32(W->o_w_f32, impl->attn_out, impl->x2, (size_t)d_model, q_dim, W->o_b_f32, 1);
1529 } else {
1530 if (ie_gemv_bf16_f32(W->o_w, impl->attn_out, impl->x2, (size_t)d_model, q_dim, W->o_b) != 0) return -19;
1531 }
1532 for (uint32_t i = 0; i < d_model; ++i) impl->x[i] += impl->x2[i];
1533
1534 if (W->ln2_w_f32) {
1535 if (ie_rmsnorm_cpu_f32(impl->x, W->ln2_w_f32, (size_t)d_model, impl->rms_eps, impl->x1) != 0) {
1536 IE_LOGE("rmsnorm2 failed: layer=%u pos=%u", (unsigned)l, (unsigned)pos);
1537 return -20;
1538 }
1539 } else if (ie_rmsnorm_cpu_f32_bf16w(impl->x, W->ln2_w, (size_t)d_model, impl->rms_eps, impl->x1) != 0) {
1540 IE_LOGE("rmsnorm2 failed: layer=%u pos=%u", (unsigned)l, (unsigned)pos);
1541 return -20;
1542 }
1543
1544 {
1545 const ie_moe_w_t *M = &W->moe;
1546 const uint32_t n_exp = M->n_experts;
1547 if (n_exp == 0u || !impl->router_logits) return -21;
1548
1549 if (trace_on) {
1550 fprintf(stderr, "[TRACE] bf16 gemv: layer=%u op=router rows=%u cols=%u\n",
1551 (unsigned)l, (unsigned)n_exp, (unsigned)d_model);
1552 }
1553 if (M->router_w_f32) {
1554 ie_gemv_f32(M->router_w_f32, impl->x1, impl->router_logits,
1555 (size_t)n_exp, (size_t)d_model, M->router_b_f32, 1);
1556 } else if (ie_gemv_bf16_f32(M->router_w, impl->x1, impl->router_logits,
1557 (size_t)n_exp, (size_t)d_model, M->router_b) != 0) {
1558 IE_LOGE("router gemv failed: layer=%u pos=%u n_experts=%u",
1559 (unsigned)l, (unsigned)pos, (unsigned)n_exp);
1560 return -22;
1561 }
1562 if (ie_debug_nan_enabled_() &&
1563 ie_check_finite_f32_(impl->router_logits, (size_t)n_exp, "router_logits", l, pos) != 0) {
1564 return -90;
1565 }
1566
1567 const uint32_t topk = ie_u32_min(impl->moe_topk, n_exp);
1568 if (topk == 0u) return -23;
1569
1570 uint32_t idx[IE_GPTOSS_MOE_TOPK_DEFAULT] = {0, 0, 0, 0};
1571 float val[IE_GPTOSS_MOE_TOPK_DEFAULT] = {-INFINITY, -INFINITY, -INFINITY, -INFINITY};
1572
1573 for (uint32_t e = 0; e < n_exp; ++e) {
1574 const float v = impl->router_logits[e];
1575 for (uint32_t k = 0; k < topk; ++k) {
1576 if (v > val[k]) {
1577 for (uint32_t j = topk - 1u; j > k; --j) {
1578 val[j] = val[j - 1u];
1579 idx[j] = idx[j - 1u];
1580 }
1581 val[k] = v;
1582 idx[k] = e;
1583 break;
1584 }
1585 }
1586 }
1587
1588 ie_softmax_inplace_f32(val, (size_t)topk);
1589
1590 for (uint32_t i = 0; i < d_model; ++i) impl->x2[i] = 0.0f;
1591
1592 for (uint32_t sel = 0; sel < topk; ++sel) {
1593 const uint32_t ex = idx[sel];
1594 const float p = val[sel];
1595 if (!(p > 0.0f)) continue;
1596
1597 const uint8_t *gu_b = M->gate_up_blocks + ex * M->gate_up_blocks_stride;
1598 const uint8_t *gu_s = M->gate_up_scales + ex * M->gate_up_scales_stride;
1599 const uint16_t *gu_bias =
1600 M->gate_up_bias ? (const uint16_t *)((const uint8_t *)M->gate_up_bias + ex * M->gate_up_bias_stride)
1601 : NULL;
1602
1603 const uint8_t *dn_b = M->down_blocks + ex * M->down_blocks_stride;
1604 const uint8_t *dn_s = M->down_scales + ex * M->down_scales_stride;
1605 const uint16_t *dn_bias =
1606 M->down_bias ? (const uint16_t *)((const uint8_t *)M->down_bias + ex * M->down_bias_stride) : NULL;
1607
1608 if (ie_gemv_q4_0_f32(gu_b, gu_s, M->gate_up_scale_bytes,
1609 impl->x1, impl->mlp_up,
1610 2u * d_ff, (size_t)d_model, gu_bias) != 0) {
1611 IE_LOGE("moe gate_up gemv failed: layer=%u pos=%u ex=%u",
1612 (unsigned)l, (unsigned)pos, (unsigned)ex);
1613 return -24;
1614 }
1615 if (ie_debug_nan_enabled_() &&
1616 ie_check_finite_f32_(impl->mlp_up, 2u * d_ff, "mlp_up", l, pos) != 0) {
1617 return -90;
1618 }
1619
1620 for (size_t i = 0; i < d_ff; ++i) {
1621 const float g = ie_silu_f32(impl->mlp_up[i]);
1622 const float u = impl->mlp_up[d_ff + i];
1623 impl->mlp_gate[i] = g * u;
1624 }
1625 if (ie_debug_nan_enabled_() &&
1626 ie_check_finite_f32_(impl->mlp_gate, d_ff, "mlp_gate", l, pos) != 0) {
1627 return -90;
1628 }
1629
1630 if (ie_gemv_q4_0_f32(dn_b, dn_s, M->down_scale_bytes,
1631 impl->mlp_gate, impl->attn_out,
1632 (size_t)d_model, d_ff, dn_bias) != 0) {
1633 IE_LOGE("moe down gemv failed: layer=%u pos=%u ex=%u",
1634 (unsigned)l, (unsigned)pos, (unsigned)ex);
1635 return -25;
1636 }
1637 if (ie_debug_nan_enabled_() &&
1638 ie_check_finite_f32_(impl->attn_out, (size_t)d_model, "mlp_down", l, pos) != 0) {
1639 return -90;
1640 }
1641
1642 for (uint32_t i = 0; i < d_model; ++i) impl->x2[i] += p * impl->attn_out[i];
1643 }
1644 }
1645
1646 for (uint32_t i = 0; i < d_model; ++i) impl->x[i] += impl->x2[i];
1647 }
1648
1649 if (impl->w_norm_f32) {
1650 if (ie_rmsnorm_cpu_f32(impl->x, impl->w_norm_f32, (size_t)d_model, impl->rms_eps, impl->x1) != 0)
1651 return -26;
1652 } else if (ie_rmsnorm_cpu_f32_bf16w(impl->x, impl->w_norm_bf16, (size_t)d_model, impl->rms_eps, impl->x1) != 0) {
1653 return -26;
1654 }
1655
1656 {
1657 const int trace_on = ie_trace_bf16_enabled_() && !bf16_trace_done;
1658 if (trace_on && !impl->w_lm_q4_blocks && !impl->w_lm_f32) {
1659 fprintf(stderr, "[TRACE] bf16 gemv: op=lm_head rows=%u cols=%u\n",
1660 (unsigned)vocab, (unsigned)d_model);
1661 bf16_trace_done = 1;
1662 }
1663 }
1664 if (impl->w_lm_q4_blocks && impl->w_lm_q4_scales) {
1665 if (ie_gemv_q4_0_f32(impl->w_lm_q4_blocks, impl->w_lm_q4_scales,
1666 impl->w_lm_q4_scale_bytes, impl->x1, out_logits,
1667 (size_t)vocab, (size_t)d_model, NULL) != 0)
1668 return -27;
1669 } else if (impl->w_lm_f32) {
1670 ie_gemv_f32(impl->w_lm_f32, impl->x1, out_logits, (size_t)vocab, (size_t)d_model, NULL, 0);
1671 } else {
1672 if (ie_gemv_bf16_f32(impl->w_lm_bf16, impl->x1, out_logits, (size_t)vocab, (size_t)d_model, NULL) != 0)
1673 return -27;
1674 }
1675
1676 if (getenv("IE_DEBUG_TOPK")) {
1677 uint32_t K = ie_env_u32_clamped("IE_DEBUG_TOPK", 10u, 1u, 100u);
1678 for (uint32_t i = 0; i < K; ++i) {
1679 float maxv = -INFINITY;
1680 int maxi = -1;
1681 for (uint32_t j = 0; j < vocab; ++j) {
1682 if (out_logits[j] > maxv) {
1683 maxv = out_logits[j];
1684 maxi = (int)j;
1685 }
1686 }
1687 if (maxi < 0) break;
1688 fprintf(stderr, "[DBG] top%u id=%d val=%g\n", (unsigned)i, maxi, (double)maxv);
1689 out_logits[(uint32_t)maxi] = -INFINITY;
1690 }
1691 }
1692
1693 return 0;
1694}
