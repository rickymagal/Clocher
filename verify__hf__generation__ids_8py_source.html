4"""
5Verify engine-generated token IDs against a Hugging Face reference at the ID level.
6
7This is designed to work *before* decode/vocab alignment is fixed.
8
9How it works:
10- The engine produces a JSONL "trace" with (at minimum):
11 1) One "prompt" record per prompt:
12 {"kind":"prompt","prompt_index":0,"prompt_token_ids":[...], "prompt":"..."}
13 2) One "gen_step" record per generation step:
14 {"kind":"gen_step","prompt_index":0,"step":0,"next_id":123}
15 Optionally include:
16 {"topk_ids":[...], "topk_logits":[...]} (aligned arrays)
17
18- For each prompt, we run the HF model in greedy mode step-by-step starting from
19 the exact prompt_token_ids logged by the engine, and compare:
20 - next-token ID at every step
21 - optional top-k IDs, and (optionally) logits within a tolerance
22
23Outputs:
24- A JSON report (one object) with per-prompt results and an overall OK flag.
25"""
26

62def _read_jsonl(path: str) -> List[Dict[str, Any]]:
63 out: List[Dict[str, Any]] = []
64 with open(path, "r", encoding="utf-8", errors="ignore") as f:
65 for line in f:
66 s = line.strip()
67 if not s:
68 continue
69 try:
70 out.append(json.loads(s))
71 except json.JSONDecodeError:
72 continue
73 return out

76def _load_trace(path: str) -> List[PromptTrace]:
77 objs = _read_jsonl(path)
78 prompts: Dict[int, PromptTrace] = {}
79
80 for o in objs:
81 kind = str(o.get("kind", ""))
82 if kind == "prompt":
83 pi = int(o["prompt_index"])
84 ptxt = o.get("prompt")
85 ids = o.get("prompt_token_ids")
86 if not isinstance(ids, list) or not all(isinstance(x, int) for x in ids):
87 raise ValueError(f"Trace prompt record missing/invalid prompt_token_ids (prompt_index={pi})")
88 prompts[pi] = PromptTrace(
89 prompt_index=pi,
90 prompt_text=str(ptxt) if isinstance(ptxt, str) else None,
91 prompt_token_ids=[int(x) for x in ids],
92 steps=[],
93 )
94 elif kind == "gen_step":
95 pi = int(o["prompt_index"])
96 st = int(o["step"])
97 nid = int(o["next_id"])
98 topk_ids = o.get("topk_ids")
99 topk_logits = o.get("topk_logits")
100
101 tk_ids: Optional[List[int]] = None
102 tk_logits: Optional[List[float]] = None
103
104 if isinstance(topk_ids, list) and all(isinstance(x, int) for x in topk_ids):
105 tk_ids = [int(x) for x in topk_ids]
106 if isinstance(topk_logits, list) and all(isinstance(x, (int, float)) for x in topk_logits):
107 tk_logits = [float(x) for x in topk_logits]
108
109 if pi not in prompts:
110 # Allow traces that emit steps before the prompt record by creating a stub.
111 prompts[pi] = PromptTrace(prompt_index=pi, prompt_text=None, prompt_token_ids=[], steps=[])
112
113 prompts[pi].steps.append(Step(step=st, next_id=nid, topk_ids=tk_ids, topk_logits=tk_logits))
114
115 out = list(prompts.values())
116 out.sort(key=lambda p: p.prompt_index)
117
118 for p in out:
119 p.steps.sort(key=lambda s: s.step)
120 if not p.prompt_token_ids:
121 raise ValueError(f"Missing prompt_token_ids for prompt_index={p.prompt_index} (no 'prompt' record?)")
122
123 return out

146def _topk_from_logits(logits: torch.Tensor, k: int) -> Tuple[List[int], List[float]]:
147 # logits: [vocab]
148 vals, idx = torch.topk(logits, k)
149 ids = [int(x) for x in idx.tolist()]
150 lgs = [float(x) for x in vals.tolist()]
151 return ids, lgs

154def _compare_topk(
155 engine_ids: Optional[List[int]],
156 engine_logits: Optional[List[float]],
157 hf_ids: List[int],
158 hf_logits: List[float],
159 tol: Optional[float],
160) -> Tuple[bool, Optional[Dict[str, Any]]]:
161 if engine_ids is None:
162 return True, None # nothing to compare
163 if engine_ids != hf_ids:
164 return False, {"reason": "topk_ids_mismatch", "engine_topk_ids": engine_ids, "hf_topk_ids": hf_ids}
165 if tol is None or engine_logits is None:
166 return True, None
167 if len(engine_logits) != len(hf_logits):
168 return False, {"reason": "topk_logits_len_mismatch", "engine_topk_logits": engine_logits, "hf_topk_logits": hf_logits}
169 diffs = [abs(float(a) - float(b)) for a, b in zip(engine_logits, hf_logits)]
170 mx = max(diffs) if diffs else 0.0
171 if mx > float(tol):
172 return False, {"reason": "topk_logits_tol_exceeded", "max_abs_diff": mx, "tol": float(tol)}
173 return True, None

176def _check_prompt_tokens(
177 tokenizer: Any,
178 prompt_text: str,
179 prompt_ids: List[int],
180) -> Tuple[bool, Optional[Dict[str, Any]]]:
181 hf_ids = tokenizer.encode(prompt_text, add_special_tokens=False)
182 if hf_ids == prompt_ids:
183 return True, None
184 # Report first mismatch and lengths for quick debugging.
185 min_len = min(len(hf_ids), len(prompt_ids))
186 mismatch_at = None
187 for i in range(min_len):
188 if int(hf_ids[i]) != int(prompt_ids[i]):
189 mismatch_at = i
190 break
191 return False, {
192 "reason": "prompt_token_ids_mismatch",
193 "engine_len": len(prompt_ids),
194 "hf_len": len(hf_ids),
195 "mismatch_index": mismatch_at,
196 "engine_id": int(prompt_ids[mismatch_at]) if mismatch_at is not None else None,
197 "hf_id": int(hf_ids[mismatch_at]) if mismatch_at is not None else None,
198 }

201def _verify_prompt(
202 model: Any,
203 prompt_ids: List[int],
204 steps: List[Step],
205 topk: int,
206 tol: Optional[float],
207 device: torch.device,
208) -> Dict[str, Any]:
209 # Greedy next-token check, step-by-step.
210 # We keep a running list of ids so we don't depend on decode.
211 gen: List[int] = []
212 first_mismatch: Optional[Dict[str, Any]] = None
213
214 with torch.no_grad():
215 for s in steps:
216 # Build input ids = prompt + generated_so_far
217 ids = prompt_ids + gen
218 inp = torch.tensor([ids], dtype=torch.long, device=device)
219 out = model(input_ids=inp)
220 logits = out.logits[0, -1, :] # [vocab]
221
222 hf_next = int(torch.argmax(logits).item())
223 if hf_next != int(s.next_id):
224 first_mismatch = {
225 "step": int(s.step),
226 "engine_next_id": int(s.next_id),
227 "hf_next_id": hf_next,
228 }
229 break
230
231 # Optional top-k comparison (IDs and optionally logits)
232 if topk > 0 and (s.topk_ids is not None or s.topk_logits is not None):
233 hf_ids, hf_lgs = _topk_from_logits(logits, topk)
234 ok, detail = _compare_topk(s.topk_ids, s.topk_logits, hf_ids, hf_lgs, tol)
235 if not ok:
236 first_mismatch = {
237 "step": int(s.step),
238 "engine_next_id": int(s.next_id),
239 "hf_next_id": hf_next,
240 "topk_detail": detail,
241 }
242 break
243
244 gen.append(hf_next)
245
246 return {
247 "ok": first_mismatch is None,
248 "steps_checked": len(gen) if first_mismatch is None else int(first_mismatch["step"]),
249 "first_mismatch": first_mismatch,
250 }

267def main() -> int:
268 args = _parse()
269 device = _device_from_str(args.device)
270 dtype = _dtype_from_str(args.dtype)
271
272 traces = _load_trace(args.trace_jsonl)
273 if args.max_prompts and args.max_prompts > 0:
274 traces = traces[: int(args.max_prompts)]
275
276 # Tokenizer not strictly required for ID-level checks, but loading it ensures
277 # the HF folder is complete and config is readable.
278 tok = AutoTokenizer.from_pretrained(args.hf_dir, use_fast=True)
279
280 # Load model
281 # - On CUDA, default to half precision to reduce VRAM unless the user requests otherwise.
282 # - On CPU, float32 is safest (bf16 may also work on some CPUs).
283 load_kwargs: Dict[str, Any] = {"torch_dtype": dtype}
284 if device.type == "cuda":
285 # Let transformers place layers automatically across GPUs if available.
286 load_kwargs["device_map"] = "auto"
287 model = AutoModelForCausalLM.from_pretrained(args.hf_dir, **load_kwargs)
288 model.eval()
289
290 # If we are on CPU without device_map, explicitly move.
291 if device.type == "cpu":
292 model.to(device)
293
294 results: List[Dict[str, Any]] = []
295 overall_ok = True
296 first_fail: Optional[Dict[str, Any]] = None
297
298 for pt in traces:
299 prompt_ok = True
300 prompt_mismatch: Optional[Dict[str, Any]] = None
301 if args.check_prompt:
302 if not pt.prompt_text:
303 prompt_ok = False
304 prompt_mismatch = {"reason": "prompt_text_missing"}
305 else:
306 prompt_ok, prompt_mismatch = _check_prompt_tokens(tok, pt.prompt_text, pt.prompt_token_ids)
307
308 r = _verify_prompt(
309 model=model,
310 prompt_ids=pt.prompt_token_ids,
311 steps=pt.steps,
312 topk=int(args.topk),
313 tol=args.logits_tol,
314 device=device,
315 )
316 rec = {
317 "prompt_index": pt.prompt_index,
318 "prompt_len": len(pt.prompt_token_ids),
319 "gen_steps": len(pt.steps),
320 "prompt_ok": prompt_ok,
321 "prompt_mismatch": prompt_mismatch,
322 **r,
323 }
324 results.append(rec)
325 if (not rec["ok"] or not prompt_ok) and overall_ok:
326 overall_ok = False
327 if not prompt_ok:
328 first_fail = {"prompt_index": pt.prompt_index, **(prompt_mismatch or {})}
329 else:
330 first_fail = {"prompt_index": pt.prompt_index, **(rec.get("first_mismatch") or {})}
331
332 out = {
333 "ok": overall_ok,
334 "prompts_verified": len(results),
335 "first_failure": first_fail,
336 "per_prompt": results,
337 }
338
339 os.makedirs(os.path.dirname(args.out) or ".", exist_ok=True)
340 with open(args.out, "w", encoding="utf-8", newline="\n") as f:
341 json.dump(out, f, indent=2, sort_keys=True)
342
343 print(f"[verify_hf_generation_ids] ok={overall_ok} prompts={len(results)} out={args.out}")
344 return 0 if overall_ok else 1
